57      /** Class name and their corresponding parent module name.. */
58      private static final Map<Class<?>, String> CLASS_TO_PARENT_MODULE = Map.ofEntries(
59          Map.entry(AbstractCheck.class, TreeWalker.class.getSimpleName()),
60          Map.entry(TreeWalkerFilter.class, TreeWalker.class.getSimpleName()),
61          Map.entry(AbstractFileSetCheck.class, Checker.class.getSimpleName()),
62          Map.entry(Filter.class, Checker.class.getSimpleName()),
63          Map.entry(BeforeExecutionFileFilter.class, Checker.class.getSimpleName())

72      /**
73       * Get string values of the message keys from the given check class.
74       *
75       * @param module class to examine.
76       * @return a set of checkstyle's module message keys.
77       * @throws MacroExecutionException if extraction of message keys fails.
78       */
79      public static Set<String> getMessageKeys(Class<?> module)
80              throws MacroExecutionException {
81          final Set<Field> messageKeyFields = getCheckMessageKeys(module);
82          // We use a TreeSet to sort the message keys alphabetically
83          final Set<String> messageKeys = new TreeSet<>();
84          for (Field field : messageKeyFields) {
85              messageKeys.add(getFieldValue(field, module));
86          }
87          return messageKeys;
88      }

90      /**
91       * Gets the check's messages keys.
92       *
93       * @param module class to examine.
94       * @return a set of checkstyle's module message fields.
95       * @throws MacroExecutionException if the attempt to read a protected class fails.
96       * @noinspection ChainOfInstanceofChecks
97       * @noinspectionreason ChainOfInstanceofChecks - We will deal with this at
98       *                     <a href="https://github.com/checkstyle/checkstyle/issues/13500">13500</a>
99       *
100      */
101     private static Set<Field> getCheckMessageKeys(Class<?> module)
102             throws MacroExecutionException {
103         try {
104             final Set<Field> checkstyleMessages = new HashSet<>();
105 
106             // get all fields from current class
107             final Field[] fields = module.getDeclaredFields();
108 
109             for (Field field : fields) {
110                 if (field.getName().startsWith("MSG_")) {
111                     checkstyleMessages.add(field);
112                 }
113             }
114 
115             // deep scan class through hierarchy
116             final Class<?> superModule = module.getSuperclass();
117 
118             if (superModule != null) {
119                 checkstyleMessages.addAll(getCheckMessageKeys(superModule));
120             }
121 
122             // special cases that require additional classes
123             if (module == RegexpMultilineCheck.class) {
124                 checkstyleMessages.addAll(getCheckMessageKeys(Class
125                     .forName("com.puppycrawl.tools.checkstyle.checks.regexp.MultilineDetector")));
126             }
127             else if (module == RegexpSinglelineCheck.class
128                     || module == RegexpSinglelineJavaCheck.class) {
129                 checkstyleMessages.addAll(getCheckMessageKeys(Class
130                     .forName("com.puppycrawl.tools.checkstyle.checks.regexp.SinglelineDetector")));
131             }
132 
133             return checkstyleMessages;
134         }
135         catch (ClassNotFoundException ex) {
136             final String message = String.format(Locale.ROOT, "Couldn't find class: %s",
137                     module.getName());
138             throw new MacroExecutionException(message, ex);
139         }
140     }

142     /**
143      * Returns the value of the given field.
144      *
145      * @param field the field.
146      * @param instance the instance of the module.
147      * @return the value of the field.
148      * @throws MacroExecutionException if the value could not be retrieved.
149      */
150     private static String getFieldValue(Field field, Object instance)
151             throws MacroExecutionException {
152         try {
153             // required for package/private classes
154             field.trySetAccessible();
155             return field.get(instance).toString();
156         }
157         catch (IllegalAccessException ex) {
158             throw new MacroExecutionException("Couldn't get field value", ex);
159         }
160     }

162     /**
163      * Returns the instance of the module with the given name.
164      *
165      * @param moduleName the name of the module.
166      * @return the instance of the module.
167      * @throws MacroExecutionException if the module could not be created.
168      */
169     public static Object getModuleInstance(String moduleName) throws MacroExecutionException {
170         final ModuleFactory factory = getPackageObjectFactory();
171         try {
172             return factory.createModule(moduleName);
173         }
174         catch (CheckstyleException ex) {
175             throw new MacroExecutionException("Couldn't find class: " + moduleName, ex);
176         }
177     }

179     /**
180      * Returns the default PackageObjectFactory with the default package names.
181      *
182      * @return the default PackageObjectFactory.
183      * @throws MacroExecutionException if the PackageObjectFactory cannot be created.
184      */
185     private static PackageObjectFactory getPackageObjectFactory() throws MacroExecutionException {
186         try {
187             final ClassLoader cl = ViolationMessagesMacro.class.getClassLoader();
188             final Set<String> packageNames = PackageNamesLoader.getPackageNames(cl);
189             return new PackageObjectFactory(packageNames, cl);
190         }
191         catch (CheckstyleException ex) {
192             throw new MacroExecutionException("Couldn't load checkstyle modules", ex);
193         }
194     }

196     /**
197      * Construct a string with a leading newline character and followed by
198      * the given amount of spaces. We use this method only to match indentation in
199      * regular xdocs and have minimal diff when parsing the templates.
200      * This method exists until
201      * <a href="https://github.com/checkstyle/checkstyle/issues/13426">13426</a>
202      *
203      * @param amountOfSpaces the amount of spaces to add after the newline.
204      * @return the constructed string.
205      */
206     public static String getNewlineAndIndentSpaces(int amountOfSpaces) {
207         return System.lineSeparator() + " ".repeat(amountOfSpaces);
208     }

210     /**
211      * Gets xdocs template file paths. These are files ending with .xml.template.
212      * This method will be changed to gather .xml once
213      * <a href="https://github.com/checkstyle/checkstyle/issues/13426">#13426</a> is resolved.
214      *
215      * @return a set of xdocs template file paths.
216      * @throws MacroExecutionException if an I/O error occurs.
217      */
218     public static Set<Path> getXdocsTemplatesFilePaths() throws MacroExecutionException {
219         final Path directory = Paths.get("src/xdocs");
220         try (Stream<Path> stream = Files.find(directory, Integer.MAX_VALUE,
221                 (path, attr) -> {
222                     return attr.isRegularFile()
223                             && path.toString().endsWith(".xml.template");
224                 })) {
225             return stream.collect(Collectors.toSet());
226         }
227         catch (IOException ioException) {
228             throw new MacroExecutionException("Failed to find xdocs templates", ioException);
229         }
230     }

232     /**
233      * Returns the parent module name for the given module class. Returns either
234      * "TreeWalker" or "Checker". Returns null if the module class is null.
235      *
236      * @param moduleClass the module class.
237      * @return the parent module name as a string.
238      * @throws MacroExecutionException if the parent module cannot be found.
239      */
240     public static String getParentModule(Class<?> moduleClass)
241                 throws MacroExecutionException {
242         String parentModuleName = "";
243         Class<?> parentClass = moduleClass.getSuperclass();
244 
245         while (parentClass != null) {
246             parentModuleName = CLASS_TO_PARENT_MODULE.get(parentClass);
247             if (parentModuleName != null) {
248                 break;
249             }
250             parentClass = parentClass.getSuperclass();
251         }
252 
253         // If parent class is not found, check interfaces
254         if (parentModuleName == null || parentModuleName.isEmpty()) {
255             final Class<?>[] interfaces = moduleClass.getInterfaces();
256             for (Class<?> interfaceClass : interfaces) {
257                 parentModuleName = CLASS_TO_PARENT_MODULE.get(interfaceClass);
258                 if (parentModuleName != null) {
259                     break;
260                 }
261             }
262         }
263 
264         if (parentModuleName == null || parentModuleName.isEmpty()) {
265             final String message = String.format(Locale.ROOT,
266                     "Failed to find parent module for %s", moduleClass.getSimpleName());
267             throw new MacroExecutionException(message);
268         }
269 
270         return parentModuleName;
271     }

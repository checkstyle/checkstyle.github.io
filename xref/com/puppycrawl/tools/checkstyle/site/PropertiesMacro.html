48  /**
49   * A macro that inserts a table of properties for the given checkstyle module.
50   */
51  @Component(role = Macro.class, hint = "properties")
52  public class PropertiesMacro extends AbstractMacro {
53  
54      /** Represents the relative path to the property types XML. */
55      private static final String PROPERTY_TYPES_XML = "property_types.xml";
56  
57      /** Represents the format string for constructing URLs with two placeholders. */
58      private static final String URL_F = "%s#%s";
59  
60      /** Reflects start of a code segment. */
61      private static final String CODE_START = "<code>";
62  
63      /** Reflects end of a code segment. */
64      private static final String CODE_END = "</code>";
65  
66      /** A newline with 10 spaces of indentation. */
67      private static final String INDENT_LEVEL_10 = SiteUtil.getNewlineAndIndentSpaces(10);
68      /** A newline with 12 spaces of indentation. */
69      private static final String INDENT_LEVEL_12 = SiteUtil.getNewlineAndIndentSpaces(12);
70      /** A newline with 14 spaces of indentation. */
71      private static final String INDENT_LEVEL_14 = SiteUtil.getNewlineAndIndentSpaces(14);
72      /** A newline with 16 spaces of indentation. */
73      private static final String INDENT_LEVEL_16 = SiteUtil.getNewlineAndIndentSpaces(16);
74      /** A newline with 18 spaces of indentation. */
75      private static final String INDENT_LEVEL_18 = SiteUtil.getNewlineAndIndentSpaces(18);
76      /** A newline with 20 spaces of indentation. */
77      private static final String INDENT_LEVEL_20 = SiteUtil.getNewlineAndIndentSpaces(20);

79      /**
80       * This property is used to change the existing properties for javadoc.
81       * Tokens always present at the end of all properties.
82      */
83      private static final String TOKENS_PROPERTY = SiteUtil.TOKENS;
84  
85      /** The name of the current module being processed. */
86      private static String currentModuleName = "";
87  
88      /** The file of the current module being processed. */
89      private static File currentModuleFile = new File("");

91      @Override
92      public void execute(Sink sink, MacroRequest request) throws MacroExecutionException {
93          // until https://github.com/checkstyle/checkstyle/issues/13426
94          if (!(sink instanceof XdocSink)) {
95              throw new MacroExecutionException("Expected Sink to be an XdocSink.");
96          }
97  
98          final String modulePath = (String) request.getParameter("modulePath");
99  
100         configureGlobalProperties(modulePath);
101 
102         writePropertiesTable((XdocSink) sink);
103     }
104 
105     /**
106      * Configures the global properties for the current module.
107      *
108      * @param modulePath the path of the current module processed.
109      */
110     private static void configureGlobalProperties(String modulePath) {
111         final File moduleFile = new File(modulePath);
112         currentModuleFile = moduleFile;
113         currentModuleName = CommonUtil.getFileNameWithoutExtension(moduleFile.getName());
114     }

116     /**
117      * Writes the properties table for the given module. Expects that the module has been processed
118      * with the ClassAndPropertiesSettersJavadocScraper before calling this method.
119      *
120      * @param sink the sink to write to.
121      * @throws MacroExecutionException if an error occurs during writing.
122      */
123     private static void writePropertiesTable(XdocSink sink)
124             throws MacroExecutionException {
125         sink.table();
126         sink.setInsertNewline(false);
127         sink.tableRows(null, false);
128         sink.rawText(INDENT_LEVEL_12);
129         writeTableHeaderRow(sink);
130         writeTablePropertiesRows(sink);
131         sink.rawText(INDENT_LEVEL_10);
132         sink.tableRows_();
133         sink.table_();
134         sink.setInsertNewline(true);
135     }

137     /**
138      * Writes the table header row with 5 columns - name, description, type, default value, since.
139      *
140      * @param sink sink to write to.
141      */
142     private static void writeTableHeaderRow(Sink sink) {
143         sink.tableRow();
144         writeTableHeaderCell(sink, "name");
145         writeTableHeaderCell(sink, "description");
146         writeTableHeaderCell(sink, "type");
147         writeTableHeaderCell(sink, "default value");
148         writeTableHeaderCell(sink, "since");
149         sink.rawText(INDENT_LEVEL_12);
150         sink.tableRow_();
151     }

153     /**
154      * Writes a table header cell with the given text.
155      *
156      * @param sink sink to write to.
157      * @param text the text to write.
158      */
159     private static void writeTableHeaderCell(Sink sink, String text) {
160         sink.rawText(INDENT_LEVEL_14);
161         sink.tableHeaderCell();
162         sink.text(text);
163         sink.tableHeaderCell_();
164     }

166     /**
167      * Writes the rows of the table with the 5 columns - name, description, type, default value,
168      * since. Each row corresponds to a property of the module.
169      *
170      * @param sink sink to write to.
171      * @throws MacroExecutionException if an error occurs during writing.
172      */
173     private static void writeTablePropertiesRows(Sink sink)
174             throws MacroExecutionException {
175         final Object instance = SiteUtil.getModuleInstance(currentModuleName);
176         final Class<?> clss = instance.getClass();
177 
178         final Set<String> properties = SiteUtil.getPropertiesForDocumentation(clss, instance);
179         final Map<String, DetailNode> propertiesJavadocs = SiteUtil
180                 .getPropertiesJavadocs(properties, currentModuleName, currentModuleFile);
181 
182         final List<String> orderedProperties = orderProperties(properties);
183 
184         for (String property : orderedProperties) {
185             final DetailNode propertyJavadoc = propertiesJavadocs.get(property);
186             final DetailNode currentModuleJavadoc = propertiesJavadocs.get(currentModuleName);
187             writePropertyRow(sink, property, propertyJavadoc, instance, currentModuleJavadoc);
188         }
189     }
190 
191     /**
192      * Reorder properties to always have the 'tokens' property last (if present).
193      *
194      * @param properties module properties.
195      * @return Collection of ordered properties.
196      *
197      */
198     private static List<String> orderProperties(Set<String> properties) {
199 
200         final List<String> orderProperties = new LinkedList<>(properties);
201 
202         if (orderProperties.remove(TOKENS_PROPERTY)) {
203             orderProperties.add(TOKENS_PROPERTY);
204         }
205         return List.copyOf(orderProperties);
206 
207     }

209     /**
210      * Writes a table row with 5 columns for the given property - name, description, type,
211      * default value, since.
212      *
213      * @param sink sink to write to.
214      * @param propertyName the name of the property.
215      * @param propertyJavadoc the Javadoc of the property.
216      * @param instance the instance of the module.
217      * @param moduleJavadoc the Javadoc of the module.
218      * @throws MacroExecutionException if an error occurs during writing.
219      */
220     private static void writePropertyRow(Sink sink, String propertyName,
221                                          DetailNode propertyJavadoc, Object instance,
222                                             DetailNode moduleJavadoc)
223             throws MacroExecutionException {
224         final Field field = SiteUtil.getField(instance.getClass(), propertyName);
225 
226         sink.rawText(INDENT_LEVEL_12);
227         sink.tableRow();
228 
229         writePropertyNameCell(sink, propertyName);
230         writePropertyDescriptionCell(sink, propertyName, propertyJavadoc);
231         writePropertyTypeCell(sink, propertyName, field, instance);
232         writePropertyDefaultValueCell(sink, propertyName, field, instance);
233         writePropertySinceVersionCell(
234                 sink, propertyName, moduleJavadoc, propertyJavadoc);
235 
236         sink.rawText(INDENT_LEVEL_12);
237         sink.tableRow_();
238     }

273     /**
274      * Writes a table cell with the property type.
275      *
276      * @param sink sink to write to.
277      * @param propertyName the name of the property.
278      * @param field the field of the property.
279      * @param instance the instance of the module.
280      * @throws MacroExecutionException if link to the property_types.html file cannot be
281      *                                 constructed.
282      */
283     private static void writePropertyTypeCell(Sink sink, String propertyName,
284                                               Field field, Object instance)
285             throws MacroExecutionException {
286         sink.rawText(INDENT_LEVEL_14);
287         sink.tableCell();
288 
289         if (SiteUtil.TOKENS.equals(propertyName)) {
290             final AbstractCheck check = (AbstractCheck) instance;
291             if (check.getRequiredTokens().length == 0
292                     && Arrays.equals(check.getAcceptableTokens(), TokenUtil.getAllTokenIds())) {
293                 sink.text("set of any supported");
294                 writeLink(sink);
295             }
296             else {
297                 final List<String> configurableTokens = SiteUtil
298                         .getDifference(check.getAcceptableTokens(),
299                                 check.getRequiredTokens())
300                         .stream()
301                         .map(TokenUtil::getTokenName)
302                         .collect(Collectors.toList());
303                 sink.text("subset of tokens");
304 
305                 writeTokensList(sink, configurableTokens, SiteUtil.PATH_TO_TOKEN_TYPES);
306             }
307         }
308         else if (SiteUtil.JAVADOC_TOKENS.equals(propertyName)) {
309             final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
310             final List<String> configurableTokens = SiteUtil
311                     .getDifference(check.getAcceptableJavadocTokens(),
312                             check.getRequiredJavadocTokens())
313                     .stream()
314                     .map(JavadocUtil::getTokenName)
315                     .collect(Collectors.toList());
316             sink.text("subset of javadoc tokens");
317             writeTokensList(sink, configurableTokens, SiteUtil.PATH_TO_JAVADOC_TOKEN_TYPES);
318         }
319         else {
320             final String type = SiteUtil.getType(field, propertyName, currentModuleName, instance);
321             if (PropertyType.TOKEN_ARRAY.getDescription().equals(type)) {
322                 processLinkForTokenTypes(sink);
323             }
324             else {
325                 final String relativePathToPropertyTypes =
326                         SiteUtil.getLinkToDocument(currentModuleName, PROPERTY_TYPES_XML);
327                 final String escapedType = type
328                         .replace("[", ".5B")
329                         .replace("]", ".5D");
330 
331                 final String url =
332                         String.format(Locale.ROOT, URL_F, relativePathToPropertyTypes, escapedType);
333 
334                 sink.link(url);
335                 sink.text(type);
336                 sink.link_();
337             }
338         }
339         sink.tableCell_();
340     }

342     /**
343      * Writes a formatted link for "TokenTypes" to the given sink.
344      *
345      * @param sink The output target where the link is written.
346      * @throws MacroExecutionException If an error occurs during the link processing.
347      */
348     private static void processLinkForTokenTypes(Sink sink)
349             throws MacroExecutionException {
350         final String link =
351                 SiteUtil.getLinkToDocument(currentModuleName, SiteUtil.PATH_TO_TOKEN_TYPES);
352 
353         sink.text("subset of tokens ");
354         sink.link(link);
355         sink.text("TokenTypes");
356         sink.link_();
357     }

359     /**
360      * Write a link when all types of token supported.
361      *
362      * @param sink sink to write to.
363      * @throws MacroExecutionException if link cannot be constructed.
364      */
365     private static void writeLink(Sink sink)
366             throws MacroExecutionException {
367         sink.rawText(INDENT_LEVEL_16);
368         final String link =
369                 SiteUtil.getLinkToDocument(currentModuleName, SiteUtil.PATH_TO_TOKEN_TYPES);
370         sink.link(link);
371         sink.rawText(INDENT_LEVEL_20);
372         sink.text(SiteUtil.TOKENS);
373         sink.link_();
374         sink.rawText(INDENT_LEVEL_14);
375     }

377     /**
378      * Write a list of tokens with links to the tokenTypesLink file.
379      *
380      * @param sink sink to write to.
381      * @param tokens the list of tokens to write.
382      * @param tokenTypesLink the link to the token types file.
383      * @throws MacroExecutionException if link to the tokenTypesLink file cannot be constructed.
384      */
385     private static void writeTokensList(Sink sink, List<String> tokens, String tokenTypesLink)
386             throws MacroExecutionException {
387         for (int index = 0; index < tokens.size(); index++) {
388             final String token = tokens.get(index);
389             sink.rawText(INDENT_LEVEL_16);
390             if (index != 0) {
391                 sink.text(SiteUtil.COMMA_SPACE);
392             }
393             writeLinkToToken(sink, tokenTypesLink, token);
394         }
395         if (tokens.isEmpty()) {
396             sink.rawText(CODE_START);
397             sink.text("empty");
398             sink.rawText(CODE_END);
399         }
400         else {
401             sink.rawText(INDENT_LEVEL_18);
402             sink.text(SiteUtil.DOT);
403             sink.rawText(INDENT_LEVEL_14);
404         }
405     }

407     /**
408      * Writes a link to the given token.
409      *
410      * @param sink sink to write to.
411      * @param document the document to link to.
412      * @param tokenName the name of the token.
413      * @throws MacroExecutionException if link to the document file cannot be constructed.
414      */
415     private static void writeLinkToToken(Sink sink, String document, String tokenName)
416             throws MacroExecutionException {
417         final String link = SiteUtil.getLinkToDocument(currentModuleName, document)
418                         + "#" + tokenName;
419         sink.link(link);
420         sink.rawText(INDENT_LEVEL_20);
421         sink.text(tokenName);
422         sink.link_();
423     }

425     /**
426      * Writes a table cell with the property default value.
427      *
428      * @param sink sink to write to.
429      * @param propertyName the name of the property.
430      * @param field the field of the property.
431      * @param instance the instance of the module.
432      * @throws MacroExecutionException if an error occurs during retrieval of the default value.
433      */
434     private static void writePropertyDefaultValueCell(Sink sink, String propertyName,
435                                                       Field field, Object instance)
436             throws MacroExecutionException {
437         sink.rawText(INDENT_LEVEL_14);
438         sink.tableCell();
439 
440         if (SiteUtil.TOKENS.equals(propertyName)) {
441             final AbstractCheck check = (AbstractCheck) instance;
442             if (check.getRequiredTokens().length == 0
443                     && Arrays.equals(check.getDefaultTokens(), TokenUtil.getAllTokenIds())) {
444                 sink.text(SiteUtil.TOKEN_TYPES);
445             }
446             else {
447                 final List<String> configurableTokens = SiteUtil
448                         .getDifference(check.getDefaultTokens(),
449                                 check.getRequiredTokens())
450                         .stream()
451                         .map(TokenUtil::getTokenName)
452                         .collect(Collectors.toList());
453                 writeTokensList(sink, configurableTokens, SiteUtil.PATH_TO_TOKEN_TYPES);
454             }
455         }
456         else if (SiteUtil.JAVADOC_TOKENS.equals(propertyName)) {
457             final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
458             final List<String> configurableTokens = SiteUtil
459                     .getDifference(check.getDefaultJavadocTokens(),
460                             check.getRequiredJavadocTokens())
461                     .stream()
462                     .map(JavadocUtil::getTokenName)
463                     .collect(Collectors.toList());
464             writeTokensList(sink, configurableTokens, SiteUtil.PATH_TO_JAVADOC_TOKEN_TYPES);
465         }
466         else {
467             final String defaultValue = SiteUtil.getDefaultValue(
468                     propertyName, field, instance, currentModuleName);
469             sink.rawText(CODE_START);
470             sink.text(defaultValue);
471             sink.rawText(CODE_END);
472         }
473 
474         sink.tableCell_();
475     }

20  package com.puppycrawl.tools.checkstyle.checks.imports;
21  
22  /**
23   * Represents the strategy when none of the rules (allow/disallow tags) match
24   * inside subpackage and import-control tag of ImportControlCheck config.
25   *
26   * @see ImportControlCheck
27   */
28  public enum MismatchStrategy {
29  
30      /**
31       * In a case if matching allow/disallow rule was not found,
32       * decision should be delegated to the parent and continue checking.
33       */
34      DELEGATE_TO_PARENT,
35  
36      /**
37       * In a case if matching allow/disallow rule was not found,
38       * import will be allowed by default.
39       */
40      ALLOWED,
41  
42      /**
43       * In a case if matching allow/disallow rule was not found,
44       * import will be disallowed by default.
45       */
46      DISALLOWED,
47  
48  }

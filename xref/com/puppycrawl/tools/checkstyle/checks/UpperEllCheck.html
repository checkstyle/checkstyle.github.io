20  package com.puppycrawl.tools.checkstyle.checks;
21  
22  import com.puppycrawl.tools.checkstyle.StatelessCheck;
23  import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
24  import com.puppycrawl.tools.checkstyle.api.DetailAST;
25  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
26  import com.puppycrawl.tools.checkstyle.utils.CommonUtils;
27  
28  /**
29   * <p>Checks that long constants are defined with an upper ell.
30   * That is <span class="code">'L'</span> and not
31   * <span class="code">'l'</span>. This is in accordance to the Java Language
32   * Specification, <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-3.html#jls-3.10.1">
33   * Section 3.10.1</a>.
34   * </p>
35   * <p>
36   * Rationale: The letter <span class="code">l</span> looks a lot
37   * like the number <span class="code">1</span>.
38   * </p>
39   *
40   * <p>Examples
41   * <p class="body">
42   * To configure the check:
43   *
44   * </p>
45   * <pre class="body">
46   * &lt;module name=&quot;UpperEll&quot;/&gt;
47   * </pre>
48   *
49   * @author Oliver Burn
50   */
51  @StatelessCheck
52  public class UpperEllCheck extends AbstractCheck {
53  
54      /**
55       * A key is pointing to the warning message text in "messages.properties"
56       * file.
57       */
58      public static final String MSG_KEY = "upperEll";
59  
60      @Override
61      public int[] getDefaultTokens() {
62          return getRequiredTokens();
63      }
64  
65      @Override
66      public int[] getAcceptableTokens() {
67          return getRequiredTokens();
68      }
69  
70      @Override
71      public int[] getRequiredTokens() {
72          return new int[] {TokenTypes.NUM_LONG};
73      }
74  
75      @Override
76      public void visitToken(DetailAST ast) {
77          if (CommonUtils.endsWithChar(ast.getText(), 'l')) {
78              log(ast.getLineNo(),
79                  ast.getColumnNo() + ast.getText().length() - 1,
80                  MSG_KEY);
81          }
82      }
83  
84  }

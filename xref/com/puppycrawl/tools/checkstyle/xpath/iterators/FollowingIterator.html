20  package com.puppycrawl.tools.checkstyle.xpath.iterators;
21  
22  import net.sf.saxon.om.AxisInfo;
23  import net.sf.saxon.om.NodeInfo;
24  import net.sf.saxon.tree.iter.AxisIterator;
25  
26  /**
27   * Recursive-free implementation of the following axis iterator.
28   */
29  public class FollowingIterator implements AxisIterator {
30      /**
31       * Ancestor axis iterator.
32       */
33      private final AxisIterator ancestorEnum;
34      /**
35       * Following-sibling axis iterator.
36       */
37      private AxisIterator siblingEnum;
38      /**
39       * Descendant axis iterator.
40       */
41      private AxisIterator descendantEnum;
42  
43      /**
44       * Create an iterator over the "following" axis.
45       *
46       * @param start the initial context node.
47       */
48      public FollowingIterator(NodeInfo start) {
49          ancestorEnum = start.iterateAxis(AxisInfo.ANCESTOR);
50          siblingEnum = start.iterateAxis(AxisInfo.FOLLOWING_SIBLING);
51      }
52  
53      /**
54       * Get the next item in the sequence.
55       *
56       * @return the next Item. If there are no more nodes, return null.
57       */
58      @Override
59      public NodeInfo next() {
60          NodeInfo result = null;
61  
62          do {
63              if (descendantEnum != null) {
64                  result = descendantEnum.next();
65              }
66  
67              if (result == null && siblingEnum != null) {
68                  result = siblingEnum.next();
69                  if (result == null) {
70                      siblingEnum = null;
71                  }
72                  else {
73                      descendantEnum = result.iterateAxis(AxisInfo.DESCENDANT);
74                  }
75              }
76  
77              if (result == null) {
78                  final NodeInfo parent = ancestorEnum.next();
79                  if (parent == null) {
80                      break;
81                  }
82                  siblingEnum = parent.iterateAxis(AxisInfo.FOLLOWING_SIBLING);
83              }
84          } while (result == null);
85          return result;
86      }
87  }

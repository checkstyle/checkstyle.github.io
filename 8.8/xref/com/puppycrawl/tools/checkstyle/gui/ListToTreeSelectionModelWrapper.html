20  package com.puppycrawl.tools.checkstyle.gui;
21  
22  import javax.swing.ListSelectionModel;
23  import javax.swing.tree.DefaultTreeSelectionModel;
24  import javax.swing.tree.TreePath;
25  
26  /**
27   * ListToTreeSelectionModelWrapper extends DefaultTreeSelectionModel
28   * to listen for changes in the ListSelectionModel it maintains. Once
29   * a change in the ListSelectionModel happens, the paths are updated
30   * in the DefaultTreeSelectionModel.
31   *
32   * @author unknown
33   */
34  class ListToTreeSelectionModelWrapper extends DefaultTreeSelectionModel {
35  
36      private static final long serialVersionUID = 2267930983939339510L;
37      /** TreeTable to perform updates on. */
38      private final TreeTable treeTable;
39      /** Set to true when we are updating the ListSelectionModel. */
40      private boolean updatingListSelectionModel;
41  
42      /**
43       * Constructor to initialise treeTable.
44       * @param jTreeTable TreeTable to perform updates on.
45       */
46      ListToTreeSelectionModelWrapper(TreeTable jTreeTable) {
47          treeTable = jTreeTable;
48          getListSelectionModel().addListSelectionListener(event -> {
49              updateSelectedPathsFromSelectedRows();
50          });
51      }
52  
53      /**
54       * Returns the list selection model. ListToTreeSelectionModelWrapper
55       * listens for changes to this model and updates the selected paths
56       * accordingly.
57       *
58       * @return the list selection model
59       */
60      protected final ListSelectionModel getListSelectionModel() {
61          return listSelectionModel;
62      }
63  
64      /**
65       * This is overridden to set {@code updatingListSelectionModel}
66       * and message super. This is the only place DefaultTreeSelectionModel
67       * alters the ListSelectionModel.
68       */
69      @Override
70      public void resetRowSelection() {
71          if (!updatingListSelectionModel) {
72              updatingListSelectionModel = true;
73              try {
74                  super.resetRowSelection();
75              }
76              finally {
77                  updatingListSelectionModel = false;
78              }
79          }
80          // Notice how we don't message super if
81          // updatingListSelectionModel is true. If
82          // updatingListSelectionModel is true, it implies the
83          // ListSelectionModel has already been updated and the
84          // paths are the only thing that needs to be updated.
85      }
86  
87      /**
88       * If {@code updatingListSelectionModel} is false, this will
89       * reset the selected paths from the selected rows in the list
90       * selection model.
91       */
92      private void updateSelectedPathsFromSelectedRows() {
93          if (!updatingListSelectionModel) {
94              updatingListSelectionModel = true;
95              try {
96                  // This is way expensive, ListSelectionModel needs an
97                  // enumerator for iterating.
98                  final int min = listSelectionModel.getMinSelectionIndex();
99                  final int max = listSelectionModel.getMaxSelectionIndex();
100 
101                 clearSelection();
102                 if (min != -1 && max != -1) {
103                     for (int counter = min; counter <= max; counter++) {
104                         updateSelectedPathIfRowIsSelected(counter);
105                     }
106                 }
107             }
108             finally {
109                 updatingListSelectionModel = false;
110             }
111         }
112     }
113 
114     /**
115      * If the row at given index is selected, selected paths are updated.
116      * @param counter number of row.
117      */
118     private void updateSelectedPathIfRowIsSelected(int counter) {
119         if (listSelectionModel.isSelectedIndex(counter)) {
120             final TreePath selPath = treeTable.getTree().getPathForRow(counter);
121 
122             if (selPath != null) {
123                 addSelectionPath(selPath);
124             }
125         }
126     }
127 
128 }

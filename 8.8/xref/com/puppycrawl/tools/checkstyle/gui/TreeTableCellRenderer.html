20  package com.puppycrawl.tools.checkstyle.gui;
21  
22  import java.awt.Component;
23  import java.awt.Graphics;
24  
25  import javax.swing.JTable;
26  import javax.swing.JTree;
27  import javax.swing.UIManager;
28  import javax.swing.table.TableCellRenderer;
29  import javax.swing.tree.DefaultTreeCellRenderer;
30  import javax.swing.tree.TreeCellRenderer;
31  import javax.swing.tree.TreeModel;
32  
33  /**
34   * A TreeCellRenderer that displays a JTree.
35   * @author Lars KÃ¼hne
36   */
37  class TreeTableCellRenderer extends JTree implements
38          TableCellRenderer {
39  
40      /**
41       * Serial ID.
42       */
43      private static final long serialVersionUID = 4324031590789321581L;
44  
45      /** Tree table to render. */
46      private final TreeTable treeTable;
47  
48      /** Last table/tree row asked to renderer. */
49      private int visibleRow;
50  
51      /**
52       * Creates a new instance.
53       * @param treeTable tree table to render.
54       * @param model Tree model.
55       */
56      TreeTableCellRenderer(TreeTable treeTable, TreeModel model) {
57          super(model);
58          this.treeTable = treeTable;
59      }
60  
61      /**
62       * UpdateUI is overridden to set the colors of the Tree's renderer
63       * to match that of the table.
64       */
65      @Override
66      public void updateUI() {
67          super.updateUI();
68          // Make the tree's cell renderer use the table's cell selection
69          // colors.
70          final TreeCellRenderer tcr = getCellRenderer();
71          if (tcr instanceof DefaultTreeCellRenderer) {
72              final DefaultTreeCellRenderer renderer = (DefaultTreeCellRenderer) tcr;
73              // For 1.1 uncomment this, 1.2 has a bug that will cause an
74              // exception to be thrown if the border selection color is
75              // null.
76              // renderer.setBorderSelectionColor(null);
77              renderer.setTextSelectionColor(UIManager.getColor("Table.selectionForeground"));
78              renderer.setBackgroundSelectionColor(UIManager.getColor("Table.selectionBackground"));
79          }
80      }
81  
82      /**
83       * Sets the row height of the tree, and forwards the row height to
84       * the table.
85       */
86      @Override
87      public void setRowHeight(int newRowHeight) {
88          if (newRowHeight > 0) {
89              super.setRowHeight(newRowHeight);
90              if (treeTable != null
91                      && treeTable.getRowHeight() != newRowHeight) {
92                  treeTable.setRowHeight(getRowHeight());
93              }
94          }
95      }
96  
97      /**
98       * This is overridden to set the height to match that of the JTable.
99       */
100     @Override
101     public void setBounds(int x, int y, int w, int h) {
102         super.setBounds(x, 0, w, treeTable.getHeight());
103     }
104 
105     /**
106      * Subclassed to translate the graphics such that the last visible
107      * row will be drawn at 0,0.
108      */
109     @Override
110     public void paint(Graphics graph) {
111         graph.translate(0, -visibleRow * getRowHeight());
112         super.paint(graph);
113     }
114 
115     /**
116      * TreeCellRenderer method. Overridden to update the visible row.
117      * @see TableCellRenderer
118      */
119     @Override
120     public Component getTableCellRendererComponent(JTable table,
121             Object value,
122             boolean isSelected,
123             boolean hasFocus,
124             int row, int column) {
125         if (isSelected) {
126             setBackground(table.getSelectionBackground());
127         }
128         else {
129             setBackground(table.getBackground());
130         }
131 
132         visibleRow = row;
133         return this;
134     }
135 
136 }

20  package com.puppycrawl.tools.checkstyle.api;
21  
22  import java.io.Serializable;
23  import java.util.Map;
24  
25  /**
26   * A Configuration is used to configure a Configurable component.  The general
27   * idea of Configuration/Configurable was taken from <a target="_top"
28   * href="http://avalon.apache.org/closed.html">Jakarta's Avalon framework</a>.
29   * @author lkuehne
30   */
31  public interface Configuration extends Serializable {
32  
33      /**
34       * The set of attribute names.
35       * @return The set of attribute names, never null.
36       */
37      String[] getAttributeNames();
38  
39      /**
40       * The attribute value for an attribute name.
41       * @param name the attribute name
42       * @return the value that is associated with name
43       * @throws CheckstyleException if name is not a valid attribute name
44       */
45      String getAttribute(String name) throws CheckstyleException;
46  
47      /**
48       * The set of child configurations.
49       * @return The set of child configurations, never null.
50       */
51      Configuration[] getChildren();
52  
53      /**
54       * The name of this configuration.
55       * @return The name of this configuration.
56       */
57      String getName();
58  
59      /**
60       * Returns an unmodifiable map instance containing the custom messages
61       * for this configuration.
62       * @return unmodifiable map containing custom messages
63       */
64      Map<String, String> getMessages();
65  
66  }

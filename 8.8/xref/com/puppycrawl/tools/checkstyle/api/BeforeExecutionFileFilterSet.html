20  package com.puppycrawl.tools.checkstyle.api;
21  
22  import java.util.Collections;
23  import java.util.HashSet;
24  import java.util.Set;
25  
26  /**
27   * A before execution file filter set applies filters to events.
28   * If a before execution file filter in the set rejects an event, then the
29   * event is rejected. Otherwise, the event is accepted.
30   * @author Richard Veach
31   */
32  public final class BeforeExecutionFileFilterSet
33      implements BeforeExecutionFileFilter {
34  
35      /** Filter set. */
36      private final Set<BeforeExecutionFileFilter> beforeExecutionFileFilters = new HashSet<>();
37  
38      /**
39       * Adds a Filter to the set.
40       * @param filter the Filter to add.
41       */
42      public void addBeforeExecutionFileFilter(BeforeExecutionFileFilter filter) {
43          beforeExecutionFileFilters.add(filter);
44      }
45  
46      /**
47       * Removes filter.
48       * @param filter filter to remove.
49       */
50      public void removeBeforeExecutionFileFilter(BeforeExecutionFileFilter filter) {
51          beforeExecutionFileFilters.remove(filter);
52      }
53  
54      /**
55       * Returns the Filters of the filter set.
56       * @return the Filters of the filter set.
57       */
58      public Set<BeforeExecutionFileFilter> getBeforeExecutionFileFilters() {
59          return Collections.unmodifiableSet(beforeExecutionFileFilters);
60      }
61  
62      @Override
63      public String toString() {
64          return beforeExecutionFileFilters.toString();
65      }
66  
67      @Override
68      public boolean accept(String uri) {
69          boolean result = true;
70          for (BeforeExecutionFileFilter filter : beforeExecutionFileFilters) {
71              if (!filter.accept(uri)) {
72                  result = false;
73                  break;
74              }
75          }
76          return result;
77      }
78  
79      /** Clears the BeforeExecutionFileFilterSet. */
80      public void clear() {
81          beforeExecutionFileFilters.clear();
82      }
83  
84  }

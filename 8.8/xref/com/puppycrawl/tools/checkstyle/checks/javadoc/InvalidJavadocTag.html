20  package com.puppycrawl.tools.checkstyle.checks.javadoc;
21  
22  /**
23   * Value object for storing data about an invalid Javadoc validTags.
24   * @author Oliver Burn
25   */
26  public final class InvalidJavadocTag {
27  
28      /** The line in which the invalid tag occurs. */
29      private final int line;
30      /** The column in which the invalid tag occurs. */
31      private final int col;
32      /** The name of the invalid tag. */
33      private final String name;
34  
35      /**
36       * Creates an instance.
37       * @param line the line of the tag
38       * @param col the column of the tag
39       * @param name the name of the invalid tag
40       */
41      public InvalidJavadocTag(int line, int col, String name) {
42          this.line = line;
43          this.col = col;
44          this.name = name;
45      }
46  
47      /**
48       *  Getter for line field.
49       *  @return line field
50       */
51      public int getLine() {
52          return line;
53      }
54  
55      /**
56       *  Getter for col field.
57       *  @return col field
58       */
59      public int getCol() {
60          return col;
61      }
62  
63      /**
64       *  Getter for name field.
65       *  @return name field
66       */
67      public String getName() {
68          return name;
69      }
70  
71  }

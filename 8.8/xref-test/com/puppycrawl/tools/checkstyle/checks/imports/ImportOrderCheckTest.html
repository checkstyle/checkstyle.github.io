671     /**
672      * Creates MOCK lexical token and returns AST node for this token.
673      * @param tokenType type of token
674      * @param tokenText text of token
675      * @param tokenFileName file name of token
676      * @param tokenRow token position in a file (row)
677      * @param tokenColumn token position in a file (column)
678      * @return AST node for the token
679      */
680     private static DetailAST mockAST(final int tokenType, final String tokenText,
681             final String tokenFileName, final int tokenRow, final int tokenColumn) {
682         final CommonHiddenStreamToken tokenImportSemi = new CommonHiddenStreamToken();
683         tokenImportSemi.setType(tokenType);
684         tokenImportSemi.setText(tokenText);
685         tokenImportSemi.setLine(tokenRow);
686         tokenImportSemi.setColumn(tokenColumn);
687         tokenImportSemi.setFilename(tokenFileName);
688         final DetailAST astSemi = new DetailAST();
689         astSemi.initialize(tokenImportSemi);
690         return astSemi;
691     }

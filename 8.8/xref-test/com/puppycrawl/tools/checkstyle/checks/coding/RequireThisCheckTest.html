361     @Test
362     public void testUnusedMethod() throws Exception {
363         final DetailAST ident = new DetailAST();
364         ident.setText("testName");
365 
366         final Class<?> cls = Class.forName(RequireThisCheck.class.getName() + "$CatchFrame");
367         final Constructor<?> constructor = cls.getDeclaredConstructors()[0];
368         constructor.setAccessible(true);
369         final Object o = constructor.newInstance(null, ident);
370 
371         Assert.assertEquals("expected ident token", ident,
372                 TestUtil.getClassDeclaredMethod(cls, "getFrameNameIdent").invoke(o));
373     }

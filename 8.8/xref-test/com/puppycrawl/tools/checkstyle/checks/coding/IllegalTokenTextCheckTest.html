47              throws Exception {

49              createModuleConfig(IllegalTokenTextCheck.class);

51          checkConfig.addAttribute("format", "a href");

53          final String[] expected = {
54              "24:28: " + getCheckMessage(MSG_KEY, "a href"),

56          verify(checkConfig, getPath("InputIllegalTokenTextTokens.java"), expected);

128     @Test
129     public void testOrderOfProperties() throws Exception {
130         // pure class must be used as configuration doesn't guarantee order of
131         // attributes
132         final IllegalTokenTextCheck check = new IllegalTokenTextCheck();
133         check.setFormat("test");
134         check.setIgnoreCase(true);
135         final Pattern actual = (Pattern) TestUtil.getClassDeclaredField(
136                 IllegalTokenTextCheck.class, "format").get(check);
137         Assert.assertEquals("should match", Pattern.CASE_INSENSITIVE, actual.flags());
138         Assert.assertEquals("should match", "test", actual.pattern());
139     }

141     @Test
142     public void testAcceptableTokensMakeSense() {
143         final int expectedTokenTypesTotalNumber = 169;
144         Assert.assertEquals("Total number of TokenTypes has changed, acceptable tokens in"
145                 + " IllegalTokenTextCheck need to be reconsidered.",
146             expectedTokenTypesTotalNumber, TokenUtils.getTokenTypesTotalNumber());
147 
148         final IllegalTokenTextCheck check = new IllegalTokenTextCheck();
149         final int[] allowedTokens = check.getAcceptableTokens();
150         final List<Integer> tokenTypesWithMutableText = Arrays.asList(
151             TokenTypes.NUM_DOUBLE,
152             TokenTypes.NUM_FLOAT,
153             TokenTypes.NUM_INT,
154             TokenTypes.NUM_LONG,
155             TokenTypes.IDENT,
156             TokenTypes.COMMENT_CONTENT,
157             TokenTypes.STRING_LITERAL,
158             TokenTypes.CHAR_LITERAL
159         );
160         for (int tokenType : allowedTokens) {
161             Assert.assertTrue(TokenUtils.getTokenName(tokenType) + " should not be allowed"
162                 + " in this check as its text is a constant (IllegalTokenCheck should be used for"
163                 + " such cases).", tokenTypesWithMutableText.contains(tokenType));
164         }
165     }

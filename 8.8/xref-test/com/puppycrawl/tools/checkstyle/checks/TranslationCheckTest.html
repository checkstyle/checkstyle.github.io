143     @Test
144     public void testLogOutput() throws Exception {
145         final DefaultConfiguration checkConfig = createModuleConfig(TranslationCheck.class);
146         checkConfig.addAttribute("requiredTranslations", "ja,de");
147         checkConfig.addAttribute("baseName", "^InputTranslation.*$");
148         final Checker checker = createChecker(checkConfig);
149         checker.setBasedir(getPath(""));
150         final ByteArrayOutputStream out = new ByteArrayOutputStream();
151         final XMLLogger logger = new XMLLogger(out, AutomaticBean.OutputStreamOptions.NONE);
152         checker.addListener(logger);
153 
154         final String defaultProps = getPath("InputTranslationCheckFireErrors.properties");
155         final String translationProps = getPath("InputTranslationCheckFireErrors_de.properties");
156 
157         final File[] propertyFiles = {
158             new File(defaultProps),
159             new File(translationProps),
160         };
161 
162         final String line = "0: ";
163         final String firstErrorMessage = getCheckMessage(MSG_KEY_MISSING_TRANSLATION_FILE,
164                 "InputTranslationCheckFireErrors_ja.properties");
165         final String secondErrorMessage = getCheckMessage(MSG_KEY, "anotherKey");
166 
167         verify(checker, propertyFiles, ImmutableMap.of(
168             ":0", Collections.singletonList(" " + firstErrorMessage),
169             "InputTranslationCheckFireErrors_de.properties",
170                 Collections.singletonList(line + secondErrorMessage)));
171 
172         verifyXml(getPath("ExpectedTranslationLog.xml"), out, (expected, actual) -> {
173             // order is not always maintained here for an unknown reason.
174             // File names can appear in different orders depending on the OS and VM.
175             // This ensures we pick up the correct file based on its name and the
176             // number of children it has.
177             return !"file".equals(expected.getNodeName())
178                     || expected.getAttributes().getNamedItem("name").getNodeValue()
179                             .equals(actual.getAttributes().getNamedItem("name").getNodeValue())
180                     && XmlUtil.getChildrenElements(expected).size() == XmlUtil
181                             .getChildrenElements(actual).size();
182         }, firstErrorMessage, secondErrorMessage);
183     }

199     @Test
200     @SuppressWarnings("unchecked")
201     public void testLogIoExceptionFileNotFound() throws Exception {
202         //I can't put wrong file here. Checkstyle fails before check started.
203         //I saw some usage of file or handling of wrong file in Checker, or somewhere
204         //in checks running part. So I had to do it with reflection to improve coverage.
205         final TranslationCheck check = new TranslationCheck();
206         final DefaultConfiguration checkConfig = createModuleConfig(TranslationCheck.class);
207         check.configure(checkConfig);
208         final Checker checker = createChecker(checkConfig);
209         final SeverityLevelCounter counter = new SeverityLevelCounter(SeverityLevel.ERROR);
210         checker.addListener(counter);
211         check.setMessageDispatcher(checker);
212 
213         final Method loadKeys =
214             check.getClass().getDeclaredMethod("getTranslationKeys", File.class);
215         loadKeys.setAccessible(true);
216         final Set<String> keys = (Set<String>) loadKeys.invoke(check, new File(""));
217         assertTrue("Translation keys should be empty when File is not found", keys.isEmpty());
218         assertEquals("Invalid error count", 1, counter.getCount());
219     }

221     @Test
222     public void testLogIoException() throws Exception {
223         //I can't put wrong file here. Checkstyle fails before check started.
224         //I saw some usage of file or handling of wrong file in Checker, or somewhere
225         //in checks running part. So I had to do it with reflection to improve coverage.
226         final TranslationCheck check = new TranslationCheck();
227         final DefaultConfiguration checkConfig = createModuleConfig(TranslationCheck.class);
228         final MessageDispatcher dispatcher = mock(MessageDispatcher.class);
229         check.configure(checkConfig);
230         check.setMessageDispatcher(dispatcher);
231 
232         final Method logIoException = check.getClass().getDeclaredMethod("logIoException",
233                 IOException.class,
234                 File.class);
235         logIoException.setAccessible(true);
236         final File file = new File("");
237         logIoException.invoke(check, new IOException("test exception"), file);
238 
239         Mockito.verify(dispatcher, times(1)).fireErrors(any(String.class), captor.capture());
240         final String actual = captor.getValue().first().getMessage();
241         assertThat("Invalid message: " + actual, actual, endsWith("test exception"));
242     }

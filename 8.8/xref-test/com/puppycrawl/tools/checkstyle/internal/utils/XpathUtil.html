20  package com.puppycrawl.tools.checkstyle.internal.utils;
21  
22  import java.util.List;
23  
24  import com.puppycrawl.tools.checkstyle.xpath.AbstractNode;
25  import net.sf.saxon.om.Item;
26  import net.sf.saxon.sxpath.XPathDynamicContext;
27  import net.sf.saxon.sxpath.XPathEvaluator;
28  import net.sf.saxon.sxpath.XPathExpression;
29  import net.sf.saxon.trans.XPathException;
30  
31  /**
32   * XpathUtil.
33   * @noinspection ClassOnlyUsedInOnePackage
34   */
35  public final class XpathUtil {
36  
37      private XpathUtil() {
38      }
39  
40      /**
41       * Returns list of nodes matching xpath expression given node context.
42       * @param xpath Xpath expression
43       * @param rootNode {@code NodeInfo} node context
44       * @return list of nodes matching xpath expression given node context
45       */
46      public static List<Item> getXpathItems(String xpath, AbstractNode rootNode)
47              throws XPathException {
48          final XPathEvaluator xpathEvaluator = new XPathEvaluator();
49          final XPathExpression xpathExpression = xpathEvaluator.createExpression(xpath);
50          final XPathDynamicContext xpathDynamicContext = xpathExpression
51                  .createDynamicContext(rootNode);
52          return xpathExpression.evaluate(xpathDynamicContext);
53      }
54  
55  }

361     private static void validateAllCheckTokensAreReferencedInConfigFile(String configName,
362             Configuration configuration, Map<String, Set<String>> tokensToIgnore) throws Exception {
363         final ModuleFactory moduleFactory = TestUtil.getPackageObjectFactory();
364         final Set<Configuration> configChecks = ConfigurationUtil.getChecks(configuration);
365 
366         final Map<String, Set<String>> configCheckTokens = new HashMap<>();
367         final Map<String, Set<String>> checkTokens = new HashMap<>();
368 
369         for (Configuration checkConfig : configChecks) {
370             final String checkName = checkConfig.getName();
371             final Object instance;
372 
373             try {
374                 instance = moduleFactory.createModule(checkName);
375             }
376             catch (CheckstyleException ex) {
377                 throw new CheckstyleException("Couldn't find check: " + checkName, ex);
378             }
379 
380             if (instance instanceof AbstractCheck) {
381                 final AbstractCheck check = (AbstractCheck) instance;
382 
383                 Set<String> configTokens = configCheckTokens.get(checkName);
384 
385                 if (configTokens == null) {
386                     configTokens = new HashSet<>();
387 
388                     configCheckTokens.put(checkName, configTokens);
389 
390                     // add all overridden tokens
391                     final Set<String> overrideTokens = tokensToIgnore.get(checkName);
392 
393                     if (overrideTokens != null) {
394                         configTokens.addAll(overrideTokens);
395                     }
396 
397                     configTokens.addAll(CheckUtil.getTokenNameSet(check.getRequiredTokens()));
398                     checkTokens.put(checkName,
399                             CheckUtil.getTokenNameSet(check.getAcceptableTokens()));
400                 }
401 
402                 try {
403                     configTokens.addAll(Arrays.asList(checkConfig.getAttribute("tokens").trim()
404                             .split(",\\s*")));
405                 }
406                 catch (CheckstyleException ex) {
407                     // no tokens defined, so it is using default
408                     configTokens.addAll(CheckUtil.getTokenNameSet(check.getDefaultTokens()));
409                 }
410             }
411         }
412 
413         for (Entry<String, Set<String>> entry : checkTokens.entrySet()) {
414             Assert.assertEquals("'" + entry.getKey()
415                     + "' should have all acceptable tokens from check in " + configName
416                     + " config or specify an override to ignore the specific tokens",
417                     entry.getValue(), configCheckTokens.get(entry.getKey()));
418         }
419     }

463     @Test
464     public void testAllCheckstyleMessages() throws Exception {
465         final Map<String, List<String>> usedMessages = new TreeMap<>();
466 
467         // test validity of messages from modules
468         for (Class<?> module : CheckUtil.getCheckstyleModules()) {
469             for (Field message : CheckUtil.getCheckMessages(module)) {
470                 Assert.assertEquals(module.getSimpleName() + "." + message.getName()
471                         + " should be 'public static final'", Modifier.PUBLIC | Modifier.STATIC
472                         | Modifier.FINAL, message.getModifiers());
473 
474                 // below is required for package/private classes
475                 if (!message.isAccessible()) {
476                     message.setAccessible(true);
477                 }
478 
479                 verifyCheckstyleMessage(usedMessages, module, message);
480             }
481         }
482 
483         // test properties for messages not used by checks
484         for (Entry<String, List<String>> entry : usedMessages.entrySet()) {
485             final Properties pr = new Properties();
486             pr.load(AllChecksTest.class.getResourceAsStream(
487                     "/" + entry.getKey().replace('.', '/') + "/messages.properties"));
488 
489             for (Object key : pr.keySet()) {
490                 // hidden exception messages
491                 if ("translation.wrongLanguageCode".equals(key)) {
492                     continue;
493                 }
494 
495                 Assert.assertTrue("property '" + key + "' isn't used by any check in package '"
496                         + entry.getKey() + "'", entry.getValue().contains(key.toString()));
497             }
498         }
499     }

501     private static void verifyCheckstyleMessage(Map<String, List<String>> usedMessages,
502             Class<?> module, Field message) throws Exception {
503         final String messageString = message.get(null).toString();
504         final String packageName = module.getPackage().getName();
505         List<String> packageMessages = usedMessages.get(packageName);
506 
507         if (packageMessages == null) {
508             packageMessages = new ArrayList<>();
509             usedMessages.put(packageName, packageMessages);
510         }
511 
512         packageMessages.add(messageString);
513 
514         for (Locale locale : ALL_LOCALES) {
515             String result = null;
516 
517             try {
518                 result = CheckUtil.getCheckMessage(module, locale, messageString);
519             }
520             catch (IllegalArgumentException ex) {
521                 Assert.fail(module.getSimpleName() + " with the message '" + messageString
522                         + "' in locale '" + locale.getLanguage() + "' failed with: "
523                         + ex.getClass().getSimpleName() + " - " + ex.getMessage());
524             }
525 
526             Assert.assertNotNull(
527                     module.getSimpleName() + " should have text for the message '"
528                             + messageString + "' in locale " + locale.getLanguage() + "'",
529                     result);
530             Assert.assertFalse(
531                     module.getSimpleName() + " should have non-empty text for the message '"
532                             + messageString + "' in locale '" + locale.getLanguage() + "'",
533                     result.trim().isEmpty());
534             Assert.assertFalse(
535                     module.getSimpleName() + " should have non-TODO text for the message '"
536                             + messageString + "' in locale " + locale.getLanguage() + "'",
537                     !"todo.match".equals(messageString)
538                             && result.trim().startsWith("TODO"));
539         }
540     }

542     /**
543      * Checks that an array is a subset of other array.
544      * @param array to check whether it is a subset.
545      * @param arrayToCheckIn array to check in.
546      * @return {@code true} if all elements in {@code array} are in {@code arrayToCheckIn}.
547      */
548     private static boolean isSubset(int[] array, int... arrayToCheckIn) {
549         Arrays.sort(arrayToCheckIn);
550         boolean result = true;
551         for (final int element : array) {
552             if (Arrays.binarySearch(arrayToCheckIn, element) < 0) {
553                 result = false;
554                 break;
555             }
556         }
557         return result;
558     }

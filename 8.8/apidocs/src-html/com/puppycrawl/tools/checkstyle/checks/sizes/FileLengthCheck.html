020package com.puppycrawl.tools.checkstyle.checks.sizes;
021
022import java.io.File;
023
024import com.puppycrawl.tools.checkstyle.StatelessCheck;
025import com.puppycrawl.tools.checkstyle.api.AbstractFileSetCheck;
026import com.puppycrawl.tools.checkstyle.api.FileText;
027
028/**
029 * <p>
030 * Checks for long source files.
031 * </p>
032 * <p>
033 * Rationale: If a source file becomes very long it is hard to understand.
034 * Therefore long classes should usually be refactored into several
035 * individual classes that focus on a specific task.
036 * </p>
037 * <p>
038 * The default maximum file length is 2000 lines. To change the maximum
039 * number of lines, set property max.
040 * </p>
041 * <p>
042 * An example of how to configure the check is:
043 * </p>
044 * <pre>
045 * &lt;module name="FileLength"/&gt;
046 * </pre>
047 * <p>
048 * An example of how to configure the check so that it accepts files with at
049 * most 1500 lines is:
050 * </p>
051 * <pre>
052 * &lt;module name="FileLength"&gt;
053 *    &lt;property name="max" value="1500"/&gt;
054 * &lt;/module&gt;
055 * </pre>
056 * @author Lars KÃ¼hne
057 */
058@StatelessCheck
059public class FileLengthCheck extends AbstractFileSetCheck {
060
061    /**
062     * A key is pointing to the warning message text in "messages.properties"
063     * file.
064     */
065    public static final String MSG_KEY = "maxLen.file";
066
067    /** Default maximum number of lines. */
068    private static final int DEFAULT_MAX_LINES = 2000;
069
070    /** The maximum number of lines. */
071    private int max = DEFAULT_MAX_LINES;
072
073    @Override
074    protected void processFiltered(File file, FileText fileText) {
075        if (fileText.size() > max) {
076            log(1, MSG_KEY, fileText.size(), max);
077        }
078    }
079
080    /**
081     * Sets the maximum length of a Java source file.
082     * @param length the maximum length of a Java source file
083     */
084    public void setMax(int length) {
085        max = length;
086    }
087
088}

020package com.puppycrawl.tools.checkstyle.checks.metrics;
021
022import com.puppycrawl.tools.checkstyle.api.TokenTypes;
023
024/**
025 * The number of other classes a given class relies on. Also the square
026 * of this has been shown to indicate the amount of maintenance required
027 * in functional programs (on a file basis) at least.
028 *
029 * @author <a href="mailto:simon@redhillconsulting.com.au">Simon Harris</a>
030 * @author o_sukhodolsky
031 */
032public final class ClassFanOutComplexityCheck extends AbstractClassCouplingCheck {
033
034    /**
035     * A key is pointing to the warning message text in "messages.properties"
036     * file.
037     */
038    public static final String MSG_KEY = "classFanOutComplexity";
039
040    /** Default value of max value. */
041    private static final int DEFAULT_MAX = 20;
042
043    /** Creates new instance of this check. */
044    public ClassFanOutComplexityCheck() {
045        super(DEFAULT_MAX);
046    }
047
048    @Override
049    public int[] getRequiredTokens() {
050        return new int[] {
051            TokenTypes.PACKAGE_DEF,
052            TokenTypes.IMPORT,
053            TokenTypes.CLASS_DEF,
054            TokenTypes.INTERFACE_DEF,
055            TokenTypes.ENUM_DEF,
056            TokenTypes.TYPE,
057            TokenTypes.LITERAL_NEW,
058            TokenTypes.LITERAL_THROWS,
059            TokenTypes.ANNOTATION_DEF,
060        };
061    }
062
063    @Override
064    public int[] getAcceptableTokens() {
065        return getRequiredTokens();
066    }
067
068    // -@cs[SimpleAccessorNameNotation] Override methods from base class.
069    // Issue: https://github.com/sevntu-checkstyle/sevntu.checkstyle/issues/166
070    @Override
071    protected String getLogMessageId() {
072        return MSG_KEY;
073    }
074
075}

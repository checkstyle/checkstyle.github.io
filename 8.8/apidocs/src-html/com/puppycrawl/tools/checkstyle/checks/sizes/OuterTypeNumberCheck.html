020package com.puppycrawl.tools.checkstyle.checks.sizes;
021
022import com.puppycrawl.tools.checkstyle.FileStatefulCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.TokenTypes;
026
027/**
028 * Checks for the number of defined types at the "outer" level.
029 * @author oliverb
030 */
031@FileStatefulCheck
032public class OuterTypeNumberCheck extends AbstractCheck {
033
034    /**
035     * A key is pointing to the warning message text in "messages.properties"
036     * file.
037     */
038    public static final String MSG_KEY = "maxOuterTypes";
039
040    /** The maximum allowed number of outer types. */
041    private int max = 1;
042    /** Tracks the current depth in types. */
043    private int currentDepth;
044    /** Tracks the number of outer types found. */
045    private int outerNum;
046
047    @Override
048    public int[] getDefaultTokens() {
049        return getRequiredTokens();
050    }
051
052    @Override
053    public int[] getAcceptableTokens() {
054        return getRequiredTokens();
055    }
056
057    @Override
058    public int[] getRequiredTokens() {
059        return new int[] {TokenTypes.CLASS_DEF, TokenTypes.INTERFACE_DEF,
060            TokenTypes.ENUM_DEF, TokenTypes.ANNOTATION_DEF, };
061    }
062
063    @Override
064    public void beginTree(DetailAST ast) {
065        currentDepth = 0;
066        outerNum = 0;
067    }
068
069    @Override
070    public void finishTree(DetailAST ast) {
071        if (max < outerNum) {
072            log(ast, MSG_KEY, outerNum, max);
073        }
074    }
075
076    @Override
077    public void visitToken(DetailAST ast) {
078        if (currentDepth == 0) {
079            outerNum++;
080        }
081        currentDepth++;
082    }
083
084    @Override
085    public void leaveToken(DetailAST ast) {
086        currentDepth--;
087    }
088
089    /**
090     * Sets the maximum allowed number of outer types.
091     * @param max the new number.
092     */
093    public void setMax(int max) {
094        this.max = max;
095    }
096
097}

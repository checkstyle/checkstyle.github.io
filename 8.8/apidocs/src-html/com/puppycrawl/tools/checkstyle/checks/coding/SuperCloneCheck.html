020package com.puppycrawl.tools.checkstyle.checks.coding;
021
022import com.puppycrawl.tools.checkstyle.StatelessCheck;
023
024/**
025 * <p>
026 * Checks that an overriding clone() method invokes super.clone().
027 * Does not check native methods, as they have no possible java defined implementation.
028 * </p>
029 * <p>
030 * Reference:<a
031 * href="https://docs.oracle.com/javase/8/docs/api/java/lang/Object.html#clone--">
032 * Object.clone</a>.
033 * </p>
034 * <p>
035 * An example of how to configure the check is:
036 * </p>
037 * <pre>
038 * &lt;module name="SuperClone"/&gt;
039 * </pre>
040 * @author Rick Giles
041 */
042@StatelessCheck
043public class SuperCloneCheck extends AbstractSuperCheck {
044
045    @Override
046    protected String getMethodName() {
047        return "clone";
048    }
049
050}

020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for do...while blocks.
027 *
028 * @author jrichard
029 */
030public class DoWhileHandler extends BlockParentHandler {
031
032    /**
033     * Construct an instance of this handler with the given indentation check,
034     * abstract syntax tree, and parent handler.
035     *
036     * @param indentCheck   the indentation check
037     * @param ast           the abstract syntax tree
038     * @param parent        the parent handler
039     */
040    public DoWhileHandler(IndentationCheck indentCheck,
041            DetailAST ast, AbstractExpressionHandler parent) {
042        super(indentCheck, "do..while", ast, parent);
043    }
044
045    /**
046     * Check the indentation level of the while and conditional expression.
047     */
048    private void checkWhileExpr() {
049        // check while statement alone
050
051        final DetailAST whileAst = getMainAst().findFirstToken(TokenTypes.DO_WHILE);
052
053        if (isOnStartOfLine(whileAst)
054                && !getIndent().isAcceptable(expandedTabsColumnNo(whileAst))) {
055            logError(whileAst, "while", expandedTabsColumnNo(whileAst));
056        }
057
058        // check condition alone
059
060        final DetailAST condAst = getMainAst().findFirstToken(TokenTypes.LPAREN).getNextSibling();
061
062        checkExpressionSubtree(condAst, getIndent(), false, false);
063    }
064
065    @Override
066    public void checkIndentation() {
067        super.checkIndentation();
068        checkWhileExpr();
069    }
070
071}

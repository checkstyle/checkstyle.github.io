020package com.puppycrawl.tools.checkstyle.checks.coding;
021
022import com.puppycrawl.tools.checkstyle.FileStatefulCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.TokenTypes;
026import com.puppycrawl.tools.checkstyle.utils.CheckUtils;
027
028/**
029 * Restricts nested if-else blocks to a specified depth (default = 1).
030 *
031 * @author <a href="mailto:simon@redhillconsulting.com.au">Simon Harris</a>
032 */
033@FileStatefulCheck
034public final class NestedIfDepthCheck extends AbstractCheck {
035
036    /**
037     * A key is pointing to the warning message text in "messages.properties"
038     * file.
039     */
040    public static final String MSG_KEY = "nested.if.depth";
041
042    /** Maximum allowed nesting depth. */
043    private int max = 1;
044    /** Current nesting depth. */
045    private int depth;
046
047    /**
048     * Setter for maximum allowed nesting depth.
049     * @param max maximum allowed nesting depth.
050     */
051    public void setMax(int max) {
052        this.max = max;
053    }
054
055    @Override
056    public int[] getDefaultTokens() {
057        return getRequiredTokens();
058    }
059
060    @Override
061    public int[] getAcceptableTokens() {
062        return getRequiredTokens();
063    }
064
065    @Override
066    public int[] getRequiredTokens() {
067        return new int[] {TokenTypes.LITERAL_IF};
068    }
069
070    @Override
071    public void beginTree(DetailAST rootAST) {
072        depth = 0;
073    }
074
075    @Override
076    public void visitToken(DetailAST literalIf) {
077        if (!CheckUtils.isElseIf(literalIf)) {
078            if (depth > max) {
079                log(literalIf, MSG_KEY, depth, max);
080            }
081            ++depth;
082        }
083    }
084
085    @Override
086    public void leaveToken(DetailAST literalIf) {
087        if (!CheckUtils.isElseIf(literalIf)) {
088            --depth;
089        }
090    }
091
092}

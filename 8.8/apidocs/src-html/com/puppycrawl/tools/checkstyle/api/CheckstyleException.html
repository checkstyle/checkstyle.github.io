020package com.puppycrawl.tools.checkstyle.api;
021
022/**
023 * Represents an error condition within Checkstyle.
024 *
025 * @author Oliver Burn
026 * @noinspection CheckedExceptionClass
027 */
028public class CheckstyleException extends Exception {
029
030    /** For Serialisation that will never happen. */
031    private static final long serialVersionUID = -3517342299748221108L;
032
033    /**
034     * Creates a new {@code CheckstyleException} instance.
035     *
036     * @param message a {@code String} value
037     */
038    public CheckstyleException(String message) {
039        super(message);
040    }
041
042    /**
043     * Creates a new {@code CheckstyleException} instance
044     * that was caused by another exception.
045     *
046     * @param message a message that explains this exception
047     * @param cause the Exception that is wrapped by this exception
048     */
049    public CheckstyleException(String message, Throwable cause) {
050        super(message, cause);
051    }
052
053}

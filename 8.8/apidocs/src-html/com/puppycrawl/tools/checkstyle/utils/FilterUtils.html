020package com.puppycrawl.tools.checkstyle.utils;
021
022import java.io.IOException;
023import java.io.InputStream;
024import java.net.URI;
025import java.net.URL;
026
027import com.puppycrawl.tools.checkstyle.api.CheckstyleException;
028
029/**
030 * Utility methods for suppression filters.
031 *
032 * @author Timur Tibeyev.
033 */
034public final class FilterUtils {
035
036    /** Stop instances being created. **/
037    private FilterUtils() {
038    }
039
040    /**
041     * Checks if suppression source with given fileName exists.
042     * @param fileName name of the suppressions file.
043     * @return true if suppression file exists, otherwise false
044     */
045    public static boolean isFileExists(String fileName) {
046        boolean suppressionSourceExists = true;
047        InputStream sourceInput = null;
048        try {
049            final URI uriByFilename = CommonUtils.getUriByFilename(fileName);
050            final URL url = uriByFilename.toURL();
051            sourceInput = url.openStream();
052        }
053        catch (CheckstyleException | IOException ignored) {
054            suppressionSourceExists = false;
055        }
056        finally {
057            suppressionSourceExists = closeQuietlyWithResult(sourceInput, suppressionSourceExists);
058        }
059        return suppressionSourceExists;
060    }
061
062    /**
063     * Close input.
064     * This method is required till https://github.com/cobertura/cobertura/issues/170
065     * @param sourceInput stream to close
066     * @param suppressionSourceExists previous state of flag
067     * @return result of close operation
068     */
069    private static boolean closeQuietlyWithResult(InputStream sourceInput,
070                                                  boolean suppressionSourceExists) {
071        boolean closed = suppressionSourceExists;
072        if (sourceInput != null) {
073            try {
074                sourceInput.close();
075            }
076            catch (IOException ignored) {
077                closed = false;
078            }
079        }
080        return closed;
081    }
082
083}

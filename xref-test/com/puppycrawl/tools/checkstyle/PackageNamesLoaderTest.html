41  /**
42   * Tests loading of package names from XML files.
43   *
44   * @noinspection ClassLoaderInstantiation
45   * @noinspectionreason ClassLoaderInstantiation - Custom class loader is needed to pass URLs to
46   *      pretend these are loaded from the classpath though we can't add/change the files for
47   *      testing. The class loader is nested in this class, so the custom class loader we
48   *      are using is safe.
49   */
50  public class PackageNamesLoaderTest extends AbstractPathTestSupport {
51  
52      @Override
53      protected String getPackageLocation() {
54          return "com/puppycrawl/tools/checkstyle/packagenamesloader";
55      }

57      @Test
58      public void testDefault()
59              throws CheckstyleException {
60          final Set<String> packageNames = PackageNamesLoader
61                  .getPackageNames(Thread.currentThread()
62                          .getContextClassLoader());
63          assertWithMessage("pkgNames.length.")
64              .that(packageNames)
65              .isEmpty();
66      }

78      @Test
79      public void testPackagesFile() throws Exception {
80          final Enumeration<URL> enumeration = Collections.enumeration(Collections.singleton(
81                  new File(getPath("InputPackageNamesLoaderFile.xml")).toURI().toURL()));
82  
83          final Set<String> actualPackageNames = PackageNamesLoader
84                  .getPackageNames(new TestUrlsClassLoader(enumeration));
85          final String[] expectedPackageNames = {
86              "com.puppycrawl.tools.checkstyle",
87              "com.puppycrawl.tools.checkstyle.checks",
88              "com.puppycrawl.tools.checkstyle.checks.annotation",
89              "com.puppycrawl.tools.checkstyle.checks.blocks",
90              "com.puppycrawl.tools.checkstyle.checks.coding",
91              "com.puppycrawl.tools.checkstyle.checks.design",
92              "com.puppycrawl.tools.checkstyle.checks.header",
93              "com.puppycrawl.tools.checkstyle.checks.imports",
94              "com.puppycrawl.tools.checkstyle.checks.indentation",
95              "com.puppycrawl.tools.checkstyle.checks.javadoc",
96              "com.puppycrawl.tools.checkstyle.checks.metrics",
97              "com.puppycrawl.tools.checkstyle.checks.modifier",
98              "com.puppycrawl.tools.checkstyle.checks.naming",
99              "com.puppycrawl.tools.checkstyle.checks.regexp",
100             "com.puppycrawl.tools.checkstyle.checks.sizes",
101             "com.puppycrawl.tools.checkstyle.checks.whitespace",
102             "com.puppycrawl.tools.checkstyle.filefilters",
103             "com.puppycrawl.tools.checkstyle.filters",
104         };
105 
106         assertWithMessage("Invalid package names length.")
107             .that(actualPackageNames)
108             .hasSize(expectedPackageNames.length);
109         final Set<String> checkstylePackagesSet =
110                 new HashSet<>(Arrays.asList(expectedPackageNames));
111         assertWithMessage("Invalid names set.")
112             .that(actualPackageNames)
113             .isEqualTo(checkstylePackagesSet);
114     }

176     @Test
177     public void testPackagesWithIoException() throws Exception {
178         final URLConnection urlConnection = new URLConnection(null) {
179             @Override
180             public void connect() {
181                 // no code
182             }
183 
184             @Override
185             public InputStream getInputStream() {
186                 return null;
187             }
188         };
189         final URL url = new URL("test", null, 0, "", new URLStreamHandler() {
190             @Override
191             protected URLConnection openConnection(URL u) {
192                 return urlConnection;
193             }
194         });
195 
196         final Enumeration<URL> enumeration = Collections.enumeration(Collections.singleton(url));
197 
198         try {
199             PackageNamesLoader.getPackageNames(new TestUrlsClassLoader(enumeration));
200             assertWithMessage("CheckstyleException is expected").fail();
201         }
202         catch (CheckstyleException ex) {
203             assertWithMessage("Invalid exception cause class")
204                     .that(ex)
205                     .hasCauseThat()
206                     .isInstanceOf(IOException.class);
207             assertWithMessage("Invalid exception message")
208                     .that(ex)
209                     .hasMessageThat()
210                     .isNotEqualTo("unable to get package file resources");
211         }
212     }

231     /**
232      * Mocked ClassLoader for testing URL loading.
233      *
234      * @noinspection CustomClassloader
235      * @noinspectionreason CustomClassloader - needed to pass URLs to pretend these are loaded
236      *      from the classpath though we can't add/change the files for testing
237      */
238     private static final class TestUrlsClassLoader extends ClassLoader {
239 
240         private final Enumeration<URL> urls;
241 
242         private TestUrlsClassLoader(Enumeration<URL> urls) {
243             this.urls = urls;
244         }
245 
246         @Override
247         public Enumeration<URL> getResources(String name) {
248             return urls;
249         }
250     }
251 
252     /**
253      * Mocked ClassLoader for testing exceptions.
254      *
255      * @noinspection CustomClassloader
256      * @noinspectionreason CustomClassloader - needed to throw an exception to
257      *      test a catch statement
258      */
259     private static final class TestIoExceptionClassLoader extends ClassLoader {
260         @Override
261         public Enumeration<URL> getResources(String name) throws IOException {
262             throw new IOException("test");
263         }
264     }

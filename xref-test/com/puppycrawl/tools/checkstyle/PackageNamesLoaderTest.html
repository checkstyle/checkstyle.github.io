85      @Test
86      @SuppressWarnings("unchecked")
87      public void testPackagesFile() throws Exception {
88          mockStatic(Closeables.class);
89          doNothing().when(Closeables.class);
90          Closeables.closeQuietly(any(InputStream.class));
91  
92          final Method processFileMethod = PackageNamesLoader.class.getDeclaredMethod("processFile",
93                  URL.class, PackageNamesLoader.class);
94          processFileMethod.setAccessible(true);
95          final Constructor<PackageNamesLoader> constructor = PackageNamesLoader.class
96                  .getDeclaredConstructor();
97          constructor.setAccessible(true);
98          final PackageNamesLoader namesLoader = constructor.newInstance();
99          final URL input = new File(getPath("InputPackageNamesLoaderFile.xml")).toURI().toURL();
100 
101         processFileMethod.invoke(null, input, namesLoader);
102 
103         final Field packageNamesField = PackageNamesLoader.class.getDeclaredField("packageNames");
104         packageNamesField.setAccessible(true);
105 
106         final Set<String> actualPackageNames = (Set<String>) packageNamesField.get(namesLoader);
107         final String[] expectedPackageNames = {
108             "com.puppycrawl.tools.checkstyle",
109             "com.puppycrawl.tools.checkstyle.checks",
110             "com.puppycrawl.tools.checkstyle.checks.annotation",
111             "com.puppycrawl.tools.checkstyle.checks.blocks",
112             "com.puppycrawl.tools.checkstyle.checks.coding",
113             "com.puppycrawl.tools.checkstyle.checks.design",
114             "com.puppycrawl.tools.checkstyle.checks.header",
115             "com.puppycrawl.tools.checkstyle.checks.imports",
116             "com.puppycrawl.tools.checkstyle.checks.indentation",
117             "com.puppycrawl.tools.checkstyle.checks.javadoc",
118             "com.puppycrawl.tools.checkstyle.checks.metrics",
119             "com.puppycrawl.tools.checkstyle.checks.modifier",
120             "com.puppycrawl.tools.checkstyle.checks.naming",
121             "com.puppycrawl.tools.checkstyle.checks.regexp",
122             "com.puppycrawl.tools.checkstyle.checks.sizes",
123             "com.puppycrawl.tools.checkstyle.checks.whitespace",
124             "com.puppycrawl.tools.checkstyle.filefilters",
125             "com.puppycrawl.tools.checkstyle.filters",
126         };
127 
128         assertEquals("Invalid package names length.", expectedPackageNames.length,
129             actualPackageNames.size());
130         final Set<String> checkstylePackagesSet =
131                 new HashSet<>(Arrays.asList(expectedPackageNames));
132         assertEquals("Invalid names set.", checkstylePackagesSet, actualPackageNames);
133 
134         verifyStatic(times(1));
135         Closeables.closeQuietly(any(InputStream.class));
136     }

138     @Test
139     @SuppressWarnings("unchecked")
140     public void testPackagesWithDots() throws Exception {
141         final Constructor<PackageNamesLoader> constructor =
142                 PackageNamesLoader.class.getDeclaredConstructor();
143         constructor.setAccessible(true);
144         final PackageNamesLoader loader = constructor.newInstance();
145 
146         final Attributes attributes = mock(Attributes.class);
147         when(attributes.getValue("name")).thenReturn("coding.");
148         loader.startElement("", "", "package", attributes);
149         loader.endElement("", "", "package");
150 
151         final Field field = PackageNamesLoader.class.getDeclaredField("packageNames");
152         field.setAccessible(true);
153         final Set<String> list = (Set<String>) field.get(loader);
154         assertEquals("Invalid package name", "coding.", list.iterator().next());
155     }

160         final URLConnection mockConnection = Mockito.mock(URLConnection.class);

163

171         when(classLoader.getResources("checkstyle_packages.xml")).thenReturn(enumeration);

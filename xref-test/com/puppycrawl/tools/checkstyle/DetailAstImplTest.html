644     private static List<File> getAllFiles(File dir) {
645         final List<File> result = new ArrayList<>();
646 
647         dir.listFiles(file -> {
648             if (file.isDirectory()) {
649                 result.addAll(getAllFiles(file));
650             }
651             else if (file.getName().endsWith(".java")
652                     && !NO_ROOT_FILES.contains(file.getName())) {
653                 result.add(file);
654             }
655             return false;
656         });
657 
658         return result;
659     }

661     private static boolean checkTree(final String filename, final DetailAST root) {
662         DetailAST curNode = root;
663         DetailAST parent = null;
664         DetailAST prev = null;
665         while (curNode != null) {
666             checkNode(curNode, parent, prev, filename, root);
667             DetailAST toVisit = curNode.getFirstChild();
668             if (toVisit == null) {
669                 while (curNode != null && toVisit == null) {
670                     toVisit = curNode.getNextSibling();
671                     if (toVisit == null) {
672                         curNode = curNode.getParent();
673                         if (curNode != null) {
674                             parent = curNode.getParent();
675                         }
676                     }
677                     else {
678                         prev = curNode;
679                         curNode = toVisit;
680                     }
681                 }
682             }
683             else {
684                 parent = curNode;
685                 curNode = toVisit;
686                 prev = null;
687             }
688         }
689 
690         return true;
691     }

693     private static void checkNode(final DetailAST node,
694                                   final DetailAST parent,
695                                   final DetailAST prev,
696                                   final String filename,
697                                   final DetailAST root) {
698         final Object[] params = {
699             node, parent, prev, filename, root,
700         };
701         final MessageFormat badParentFormatter = new MessageFormat(
702                 "Bad parent node={0} parent={1} filename={3} root={4}", Locale.ROOT);
703         final String badParentMsg = badParentFormatter.format(params);
704         assertWithMessage(badParentMsg)
705             .that(node.getParent())
706             .isEqualTo(parent);
707         final MessageFormat badPrevFormatter = new MessageFormat(
708                 "Bad prev node={0} prev={2} parent={1} filename={3} root={4}", Locale.ROOT);
709         final String badPrevMsg = badPrevFormatter.format(params);
710         assertWithMessage(badPrevMsg)
711             .that(node.getPreviousSibling())
712             .isEqualTo(prev);

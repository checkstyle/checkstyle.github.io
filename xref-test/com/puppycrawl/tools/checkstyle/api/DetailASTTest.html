64      private static Method getSetParentMethod() throws Exception {
65          final Class<DetailAST> detailAstClass = DetailAST.class;
66          final Method setParentMethod =
67              detailAstClass.getDeclaredMethod("setParent", DetailAST.class);
68          setParentMethod.setAccessible(true);
69          return setParentMethod;
70      }

72      @Test
73      public void testGetChildCount() throws Exception {
74          final DetailAST root = new DetailAST();
75          final DetailAST firstLevelA = new DetailAST();
76          final DetailAST firstLevelB = new DetailAST();
77          final DetailAST secondLevelA = new DetailAST();
78  
79          root.setFirstChild(firstLevelA);
80  
81          final Method setParentMethod = getSetParentMethod();
82          setParentMethod.invoke(firstLevelA, root);
83          firstLevelA.setFirstChild(secondLevelA);
84          firstLevelA.setNextSibling(firstLevelB);
85  
86          setParentMethod.invoke(firstLevelB, root);
87  
88          setParentMethod.invoke(secondLevelA, root);
89  
90          assertEquals("Invalid child count", 0, secondLevelA.getChildCount());
91          assertEquals("Invalid child count", 0, firstLevelB.getChildCount());
92          assertEquals("Invalid child count", 1, firstLevelA.getChildCount());
93          assertEquals("Invalid child count", 2, root.getChildCount());
94          assertEquals("Invalid child count", 2, root.getChildCount());
95  
96          assertNull("Previous sibling should be null", root.getPreviousSibling());
97          assertNull("Previous sibling should be null", firstLevelA.getPreviousSibling());
98          assertNull("Previous sibling should be null", secondLevelA.getPreviousSibling());
99          assertEquals("Invalid previous sibling", firstLevelA, firstLevelB.getPreviousSibling());
100     }

163     @Test
164     public void testClearBranchTokenTypes() throws Exception {
165         final DetailAST parent = new DetailAST();
166         final DetailAST child = new DetailAST();
167         parent.setFirstChild(child);
168 
169         final List<Consumer<DetailAST>> clearBranchTokenTypesMethods = Arrays.asList(
170                 child::setFirstChild,
171                 child::setNextSibling,
172                 child::addPreviousSibling,
173                 child::addNextSibling,
174                 child::addChild,
175             ast -> {
176                 try {
177                     Whitebox.invokeMethod(child, "setParent", ast);
178                 }
179                 // -@cs[IllegalCatch] Cannot avoid catching it.
180                 catch (Exception exception) {
181                     throw new IllegalStateException(exception);
182                 }
183             }
184         );
185 
186         for (Consumer<DetailAST> method : clearBranchTokenTypesMethods) {
187             final BitSet branchTokenTypes = Whitebox.invokeMethod(parent, "getBranchTokenTypes");
188             method.accept(null);
189             final BitSet branchTokenTypes2 = Whitebox.invokeMethod(parent, "getBranchTokenTypes");
190             assertEquals("Branch token types are not equal", branchTokenTypes, branchTokenTypes2);
191             assertNotSame("Branch token types should not be the same",
192                     branchTokenTypes, branchTokenTypes2);
193         }
194     }

196     @Test
197     public void testClearChildCountCache() {
198         final DetailAST parent = new DetailAST();
199         final DetailAST child = new DetailAST();
200         parent.setFirstChild(child);
201 
202         final List<Consumer<DetailAST>> clearChildCountCacheMethods = Arrays.asList(
203                 child::setNextSibling,
204                 child::addPreviousSibling,
205                 child::addNextSibling
206         );
207 
208         for (Consumer<DetailAST> method : clearChildCountCacheMethods) {
209             final int startCount = parent.getChildCount();
210             method.accept(null);
211             final int intermediateCount = Whitebox.getInternalState(parent, "childCount");
212             final int finishCount = parent.getChildCount();
213             assertEquals("Child count has changed", startCount, finishCount);
214             assertEquals("Invalid child count", Integer.MIN_VALUE, intermediateCount);
215         }
216 
217         final int startCount = child.getChildCount();
218         child.addChild(null);
219         final int intermediateCount = Whitebox.getInternalState(child, "childCount");
220         final int finishCount = child.getChildCount();
221         assertEquals("Child count has changed", startCount, finishCount);
222         assertEquals("Invalid child count", Integer.MIN_VALUE, intermediateCount);
223     }

276     private static void checkDir(File dir) throws Exception {
277         final File[] files = dir.listFiles(file -> {
278             return (file.getName().endsWith(".java")
279                 || file.isDirectory())
280                 && !file.getName().endsWith("InputGrammar.java");
281         });
282         for (File file : files) {
283             if (file.isFile()) {
284                 checkFile(file.getCanonicalPath());
285             }
286             else if (file.isDirectory()) {
287                 checkDir(file);
288             }
289         }
290     }
291 
292     private static void checkFile(String filename) throws Exception {
293         final DetailAST rootAST =
294             JavaParser.parseFile(new File(filename), JavaParser.Options.WITHOUT_COMMENTS);
295         if (rootAST != null) {
296             checkTree(filename, rootAST);
297         }
298     }
299 
300     private static void checkTree(final String filename, final DetailAST root) {
301         DetailAST curNode = root;
302         DetailAST parent = null;
303         DetailAST prev = null;
304         while (curNode != null) {
305             checkNode(curNode, parent, prev, filename, root);
306             DetailAST toVisit = curNode.getFirstChild();
307             if (toVisit == null) {
308                 while (curNode != null && toVisit == null) {
309                     toVisit = curNode.getNextSibling();
310                     if (toVisit == null) {
311                         curNode = curNode.getParent();
312                         if (curNode != null) {
313                             parent = curNode.getParent();
314                         }
315                     }
316                     else {
317                         prev = curNode;
318                         curNode = toVisit;
319                     }
320                 }
321             }
322             else {
323                 parent = curNode;
324                 curNode = toVisit;
325                 prev = null;
326             }
327         }
328     }
329 
330     private static void checkNode(final DetailAST node,
331                                   final DetailAST parent,
332                                   final DetailAST prev,
333                                   final String filename,
334                                   final DetailAST root) {
335         final Object[] params = {
336             node, parent, prev, filename, root,
337         };
338         final MessageFormat badParentFormatter = new MessageFormat(
339                 "Bad parent node={0} parent={1} filename={3} root={4}", Locale.ROOT);
340         final String badParentMsg = badParentFormatter.format(params);
341         assertEquals(badParentMsg, parent, node.getParent());
342         final MessageFormat badPrevFormatter = new MessageFormat(
343                 "Bad prev node={0} prev={2} parent={1} filename={3} root={4}", Locale.ROOT);
344         final String badPrevMsg = badPrevFormatter.format(params);
345         assertEquals(badPrevMsg, prev, node.getPreviousSibling());
346     }

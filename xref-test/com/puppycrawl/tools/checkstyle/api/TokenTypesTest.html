20  package com.puppycrawl.tools.checkstyle.api;
21  
22  import static com.puppycrawl.tools.checkstyle.internal.utils.TestUtil.isUtilsClassHasPrivateConstructor;
23  import static org.junit.Assert.assertEquals;
24  import static org.junit.Assert.assertTrue;
25  
26  import org.junit.Test;
27  
28  import com.puppycrawl.tools.checkstyle.utils.TokenUtils;
29  
30  public class TokenTypesTest {
31  
32      @Test
33      public void testGetShortDescription() {
34          assertEquals("short description for EQUAL",
35                  "The <code>==</code> (equal) operator.",
36                  TokenUtils.getShortDescription("EQUAL"));
37  
38          assertEquals("short description for LAND",
39                  "The <code>&&</code> (conditional AND) operator.",
40                  TokenUtils.getShortDescription("LAND"));
41  
42          assertEquals("short description for LCURLY",
43                  "A left curly brace (<code>{</code>).",
44                  TokenUtils.getShortDescription("LCURLY"));
45  
46          assertEquals("short description for SR_ASSIGN",
47                  "The <code>>>=</code> (signed right shift assignment)",
48                  TokenUtils.getShortDescription("SR_ASSIGN"));
49  
50          assertEquals("short description for SL",
51                  "The <code><<</code> (shift left) operator.",
52                  TokenUtils.getShortDescription("SL"));
53  
54          assertEquals("short description for BSR",
55                  "The <code>>>></code> (unsigned shift right) operator.",
56                  TokenUtils.getShortDescription("BSR"));
57      }
58  
59      @Test
60      public void testIsProperUtilsClass() throws ReflectiveOperationException {
61          assertTrue("Constructor is not private",
62                  isUtilsClassHasPrivateConstructor(TokenTypes.class, true));
63      }
64  
65  }

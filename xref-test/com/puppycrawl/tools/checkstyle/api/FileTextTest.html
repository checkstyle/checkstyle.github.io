50      @Test
51      public void testUnsupportedCharset() throws IOException {
52          // just to make UT coverage 100%
53          final String charsetName = "STRANGE_CHARSET";
54          try {
55              final Object test = new FileText(new File("any name"), charsetName);
56              fail("UnsupportedEncodingException is expected but got " + test);
57          }
58          catch (IllegalStateException ex) {
59              assertEquals("Unsupported charset: " + charsetName, ex.getMessage(),
60                      "Invalid exception message");
61          }
62      }

64      @Test
65      public void testFileNotFound() throws IOException {
66          final String charsetName = StandardCharsets.ISO_8859_1.name();
67          try {
68              final Object test = new FileText(new File("any name"), charsetName);
69              fail("FileNotFoundException is expected but got " + test);
70          }
71          catch (FileNotFoundException ex) {
72              assertEquals("any name (No such file or directory)", ex.getMessage(),
73                      "Invalid exception message");
74          }

77      @Test
78      public void testSupportedCharset() throws IOException {
79          final String charsetName = StandardCharsets.ISO_8859_1.name();
80          final FileText fileText = new FileText(new File(getPath("InputFileTextImportControl.xml")),
81                  charsetName);
82          assertEquals(charsetName, fileText.getCharset().name(), "Invalid charset name");

85      @Test
86      public void testLineColumnBeforeCopyConstructor() throws IOException {
87          final String charsetName = StandardCharsets.ISO_8859_1.name();
88          final FileText fileText = new FileText(new File(getPath("InputFileTextImportControl.xml")),
89                  charsetName);
90          final LineColumn lineColumn = fileText.lineColumn(100);
91          final FileText copy = new FileText(fileText);
92          assertNotNull(Whitebox.getInternalState(copy, "lineBreaks"), "LineBreaks not copied");
93          final LineColumn actual = copy.lineColumn(100);
94          assertEquals(lineColumn, actual, "Invalid linecolumn");
95      }
96  
97      @Test
98      public void testLineColumnAfterCopyConstructor() throws IOException {
99          final Charset charset = StandardCharsets.ISO_8859_1;
100         final String filepath = getPath("InputFileTextImportControl.xml");
101         final FileText fileText = new FileText(new File(filepath), charset.name());
102         final FileText copy = new FileText(fileText);
103         assertNull(Whitebox.getInternalState(copy, "lineBreaks"), "LineBreaks not null");
104         final LineColumn lineColumn = copy.lineColumn(100);
105         assertEquals(3, lineColumn.getLine(), "Invalid line");
106         if (CheckUtil.CRLF.equals(CheckUtil.getLineSeparatorForFile(filepath, charset))) {
107             assertEquals(44, lineColumn.getColumn(), "Invalid column");
108         }
109         else {
110             assertEquals(46, lineColumn.getColumn(), "Invalid column");
111         }
112     }

114     @Test
115     public void testLineColumnAtTheStartOfFile() throws IOException {
116         final String charsetName = StandardCharsets.ISO_8859_1.name();
117         final FileText fileText = new FileText(new File(getPath("InputFileTextImportControl.xml")),
118                 charsetName);
119         final FileText copy = new FileText(fileText);
120         final LineColumn lineColumn = copy.lineColumn(0);
121         assertEquals(1, lineColumn.getLine(), "Invalid line");
122         assertEquals(0, lineColumn.getColumn(), "Invalid column");
123     }

129                 lines);

133     @Test
134     public void testFindLineBreaks() throws Exception {
135         final FileText fileText = new FileText(new File("fileName"), Arrays.asList("1", "2"));
136 
137         assertArrayEquals(new int[] {0, 2, 4},
138                 Whitebox.invokeMethod(fileText, "findLineBreaks"), "Invalid line breaks");
139 
140         final FileText fileText2 = new FileText(new File("fileName"), Arrays.asList("1", "2"));
141         Whitebox.setInternalState(fileText2, "fullText", "1\n2");
142 
143         assertArrayEquals(new int[] {0, 2, 3},
144                 Whitebox.invokeMethod(fileText2, "findLineBreaks"), "Invalid line breaks");
145     }

147     /**
148      * Reflection is the only way to test that a field is cached since we can't
149      * access the field directly or receive notice when the field is
150      * initialized.
151      *
152      * @throws Exception if there is an error.
153      */
154     @Test
155     public void testFindLineBreaksCache() throws Exception {
156         final FileText fileText = new FileText(new File("fileName"), Collections.emptyList());
157         final int[] lineBreaks = {5};
158         Whitebox.setInternalState(fileText, "lineBreaks", lineBreaks);
159         // produces NPE if used
160         Whitebox.setInternalState(fileText, "fullText", (Object) null);
161 
162         assertArrayEquals(lineBreaks,
163                 Whitebox.invokeMethod(fileText, "findLineBreaks"), "Invalid line breaks");
164     }

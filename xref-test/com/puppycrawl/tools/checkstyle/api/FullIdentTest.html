20  package com.puppycrawl.tools.checkstyle.api;
21  
22  import java.io.File;
23  import java.nio.charset.StandardCharsets;
24  
25  import org.junit.Assert;
26  import org.junit.Test;
27  
28  import com.puppycrawl.tools.checkstyle.AbstractModuleTestSupport;
29  import com.puppycrawl.tools.checkstyle.DetailAstImpl;
30  import com.puppycrawl.tools.checkstyle.JavaParser;
31  
32  public class FullIdentTest extends AbstractModuleTestSupport {
33  
34      @Override
35      protected String getPackageLocation() {
36          return "com/puppycrawl/tools/checkstyle/api/fullident/";
37      }
38  
39      @Test
40      public void testToString() {
41          final DetailAstImpl ast = new DetailAstImpl();
42          ast.setType(TokenTypes.LITERAL_NEW);
43          ast.setColumnNo(14);
44          ast.setLineNo(15);
45          ast.setText("MyTest");
46  
47          final FullIdent indent = FullIdent.createFullIdent(ast);
48          Assert.assertEquals("Invalid full indent", "MyTest[15x14]", indent.toString());
49          Assert.assertEquals("Invalid text", "MyTest", indent.getText());
50          Assert.assertEquals("Invalid line", 15, indent.getLineNo());
51          Assert.assertEquals("Invalid column", 14, indent.getColumnNo());
52      }
53  
54      @Test
55      public void testCreateFullIdentBelow() {
56          final DetailAST ast = new DetailAstImpl();
57  
58          final FullIdent indent = FullIdent.createFullIdentBelow(ast);
59          Assert.assertEquals("Invalid full indent", "", indent.getText());
60      }
61  
62      @Test
63      public void testGetDetailAst() throws Exception {
64          final FileText testFileText = new FileText(
65                  new File(getPath("InputFullIdentTestArrayType.java")).getAbsoluteFile(),
66                  System.getProperty("file.encoding", StandardCharsets.UTF_8.name()));
67          final DetailAST packageDefinitionNode = JavaParser.parse(new FileContents(testFileText));
68          final DetailAST packageName = packageDefinitionNode.getFirstChild().getNextSibling();
69          final FullIdent ident = FullIdent.createFullIdent(packageName);
70          Assert.assertEquals("Invalid full indent", "com[1x8]", ident.getDetailAst().toString());
71      }
72  
73      @Test
74      public void testNonValidCoordinatesWithNegative() {
75          final FullIdent fullIdent = prepareFullIdentWithCoordinates(14, 15);
76          Assert.assertEquals("Invalid full indent", "MyTest.MyTestik[15x14]", fullIdent.toString());
77      }
78  
79      @Test
80      public void testNonValidCoordinatesWithZero() {
81          final FullIdent fullIdent = prepareFullIdentWithCoordinates(0, 0);
82          Assert.assertEquals("Invalid full indent", "MyTest.MyTestik[15x14]", fullIdent.toString());
83      }
84  
85      @Test
86      public void testWithArrayCreateFullIdentWithArrayDeclare() throws Exception {
87          final FileText testFileText = new FileText(
88                  new File(getPath("InputFullIdentTestArrayType.java")).getAbsoluteFile(),
89                  System.getProperty("file.encoding", StandardCharsets.UTF_8.name()));
90          final DetailAST packageDefinitionNode = JavaParser.parse(new FileContents(testFileText));
91          final DetailAST arrayDeclarator = packageDefinitionNode.getNextSibling()
92                  .findFirstToken(TokenTypes.OBJBLOCK)
93                  .findFirstToken(TokenTypes.VARIABLE_DEF)
94                  .findFirstToken(TokenTypes.TYPE)
95                  .getFirstChild();
96          final FullIdent ident = FullIdent.createFullIdent(arrayDeclarator);
97          Assert.assertEquals("Invalid full indent", "int[][][5x12]", ident.toString());
98      }
99  
100     private static FullIdent prepareFullIdentWithCoordinates(int columnNo, int lineNo) {
101         final DetailAstImpl ast = new DetailAstImpl();
102         ast.setType(TokenTypes.DOT);
103         ast.setColumnNo(1);
104         ast.setLineNo(2);
105         ast.setText("Root");
106 
107         final DetailAstImpl ast2 = new DetailAstImpl();
108         ast2.setType(TokenTypes.LE);
109         ast2.setColumnNo(columnNo);
110         ast2.setLineNo(lineNo);
111         ast2.setText("MyTestik");
112 
113         final DetailAstImpl ast1 = new DetailAstImpl();
114         ast1.setType(TokenTypes.LITERAL_NEW);
115         ast1.setColumnNo(14);
116         ast1.setLineNo(15);
117         ast1.setText("MyTest");
118 
119         ast.addChild(ast1);
120         ast.addChild(ast2);
121 
122         return FullIdent.createFullIdent(ast);
123     }

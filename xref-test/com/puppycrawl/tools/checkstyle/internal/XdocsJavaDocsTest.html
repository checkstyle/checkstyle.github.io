171     private static void examineCheckSection(ModuleFactory moduleFactory, String fileName,
172             String sectionName, Node section) throws Exception {
173         final Object instance;
174 
175         try {
176             instance = moduleFactory.createModule(sectionName);
177         }
178         catch (CheckstyleException ex) {
179             throw new CheckstyleException(fileName + " couldn't find class: " + sectionName, ex);
180         }
181 
182         CHECK_TEXT.clear();
183         CHECK_PROPERTIES.clear();
184         CHECK_PROPERTY_DOC.clear();
185         checkName = sectionName;

187         examineCheckSectionChildren(section);

189         final List<File> files = new ArrayList<>();
190         files.add(new File("src/main/java/" + instance.getClass().getName().replace(".", "/")
191                 + ".java"));
192 
193         checker.process(files);
194     }

196     private static void examineCheckSectionChildren(Node section) {
197         for (Node subSection : XmlUtil.getChildrenElements(section)) {
198             if (!"subsection".equals(subSection.getNodeName())) {
199                 final String text = getNodeText(subSection);
200                 if (text.startsWith("Since Checkstyle")) {
201                     CHECK_TEXT.put("since", text.substring(17));
202                 }
203                 continue;
204             }
205 
206             final String subSectionName = XmlUtil.getNameAttributeOfNode(subSection);
207 
208             examineCheckSubSection(subSection, subSectionName);
209         }
210     }

212     private static void examineCheckSubSection(Node subSection, String subSectionName) {
213         switch (subSectionName) {
214             case "Description":
215             case "Examples":
216             case "Notes":
217             case "Rule Description":
218                 CHECK_TEXT.put(subSectionName, getNodeText(subSection).replace("\r", ""));
219                 break;
220             case "Properties":
221                 populateProperties(subSection);
222                 CHECK_TEXT.put(subSectionName, createPropertiesText());
223                 break;
224             case "Example of Usage":
225             case "Violation Messages":
226                 CHECK_TEXT.put(subSectionName,
227                         createViolationMessagesText(getViolationMessages(subSection)));
228                 break;
229             case "Package":
230             case "Parent Module":
231                 CHECK_TEXT.put(subSectionName, createParentText(subSection));
232                 break;
233             default:
234                 break;
235         }
236     }

264     private static void populateProperties(Node subSection) {
265         boolean skip = true;
266 
267         // if the first child is a wrapper element instead of the first table row containing
268         // the table headset
269         //   element to populate properties for to the current elements first child
270         Node child = XmlUtil.getFirstChildElement(subSection);
271         if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
272                 && "wrapper".equals(child.getAttributes().getNamedItem("class")
273                 .getTextContent())) {
274             child = XmlUtil.getFirstChildElement(child);
275         }
276         for (Node row : XmlUtil.getChildrenElements(child)) {
277             if (skip) {
278                 skip = false;
279                 continue;
280             }
281             CHECK_PROPERTIES.add(new ArrayList<>(XmlUtil.getChildrenElements(row)));
282         }

285     private static String createPropertiesText() {
286         final StringBuilder result = new StringBuilder(100);
287 
288         result.append("\n<ul>");
289 
290         for (List<Node> property : CHECK_PROPERTIES) {
291             final String propertyName = getNodeText(property.get(0));
292 
293             result.append("\n<li>\nProperty {@code ");
294             result.append(propertyName);
295             result.append("} - ");
296 
297             final String temp = getNodeText(property.get(1));
298 
299             result.append(temp);
300             CHECK_PROPERTY_DOC.put(propertyName, temp);
301 
302             String typeText = "java.lang.String[]";
303             final String propertyType = property.get(2).getTextContent();
304             final boolean isSpecialAllTokensType = propertyType.contains("set of any supported");
305             final boolean isPropertyTokenType = isSpecialAllTokensType
306                     || propertyType.contains("subset of tokens")
307                     || propertyType.contains("subset of javadoc tokens");
308             if (!isPropertyTokenType) {
309                 final String typeName =
310                         getCorrectNodeBasedOnPropertyType(property).getTextContent().trim();
311                 typeText = FULLY_QUALIFIED_CLASS_NAMES.get(typeName).getTypeName();
312             }
313             if (isSpecialAllTokensType) {
314                 typeText = "anyTokenTypesSet";
315             }
316             result.append(" Type is {@code ").append(typeText).append("}.");
317 
318             if (!isSpecialAllTokensType) {
319                 final String validationType = getValidationType(isPropertyTokenType, propertyName);
320                 if (validationType != null) {
321                     result.append(validationType);
322                 }
323             }
324 
325             result.append(getDefaultValueOfType(propertyName, isSpecialAllTokensType));
326 
327             result.append(emptyStringArrayDefaultValue(property.get(3), isPropertyTokenType));
328 
329             if (result.charAt(result.length() - 1) != '.') {
330                 result.append('.');
331             }
332 
333             result.append("\n</li>");
334         }
335 
336         result.append("\n</ul>");
337 
338         return result.toString();
339     }

388     private static String getNodeText(Node node) {
389         final StringBuilder result = new StringBuilder(20);
390 
391         for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling()) {
392             if (child.getNodeType() == Node.TEXT_NODE) {
393                 for (String temp : child.getTextContent().split("\n")) {
394                     final String text = temp.trim();
395 
396                     if (!text.isEmpty()) {
397                         if (shouldAppendSpace(result, text.charAt(0))) {
398                             result.append(' ');
399                         }
400 
401                         result.append(text);
402                     }
403                 }
404             }
405             else {
406                 if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
407                         && "wrapper".equals(child.getAttributes().getNamedItem("class")
408                         .getNodeValue())) {
409                     appendNodeText(result, XmlUtil.getFirstChildElement(child));
410                 }
411                 else {
412                     appendNodeText(result, child);
413                 }
414             }
415         }
416 
417         return result.toString();
418     }

420     // -@cs[CyclomaticComplexity] No simple way to split this apart.
421     private static void appendNodeText(StringBuilder result, Node node) {
422         final String name = transformXmlToJavaDocName(node.getNodeName());
423         final boolean list = "ol".equals(name) || "ul".equals(name);
424         final boolean newLineOpenBefore = list || "p".equals(name) || "pre".equals(name)
425                 || "li".equals(name);
426         final boolean newLineOpenAfter = newLineOpenBefore && !list;
427         final boolean newLineClose = newLineOpenAfter || list;
428         final boolean sanitize = "pre".equals(name);
429         final boolean changeToTag = "code".equals(name);
430 
431         if (newLineOpenBefore) {
432             result.append('\n');
433         }
434         else if (shouldAppendSpace(result, '<')) {
435             result.append(' ');
436         }
437 
438         if (changeToTag) {
439             result.append("{@");
440             result.append(name);
441             result.append(' ');
442         }
443         else {
444             result.append('<');
445             result.append(name);
446             result.append(getAttributeText(name, node.getAttributes()));
447             result.append('>');
448         }
449 
450         if (newLineOpenAfter) {
451             result.append('\n');
452         }
453 
454         if (sanitize) {
455             result.append(XmlUtil.sanitizeXml(node.getTextContent()));
456         }
457         else {
458             result.append(getNodeText(node));
459         }
460 
461         if (newLineClose) {
462             result.append('\n');
463         }
464 
465         if (changeToTag) {
466             result.append('}');
467         }
468         else {
469             result.append("</");
470             result.append(name);
471             result.append('>');
472         }
473     }

475     private static boolean shouldAppendSpace(StringBuilder text, char firstCharToAppend) {
476         final boolean result;
477 
478         if (text.length() == 0) {
479             result = false;
480         }
481         else {
482             final char last = text.charAt(text.length() - 1);
483 
484             result = (firstCharToAppend == '@'
485                     || Character.getType(firstCharToAppend) == Character.DASH_PUNCTUATION
486                     || Character.getType(last) == Character.OTHER_PUNCTUATION
487                     || Character.isAlphabetic(last)
488                     || Character.isAlphabetic(firstCharToAppend)) && !Character.isWhitespace(last);
489         }
490 
491         return result;
492     }

494     private static String transformXmlToJavaDocName(String name) {
495         final String result;
496 
497         if ("source".equals(name)) {
498             result = "pre";
499         }
500         else if ("h4".equals(name)) {
501             result = "p";
502         }
503         else {
504             result = name;
505         }
506 
507         return result;

510     private static String getAttributeText(String nodeName, NamedNodeMap attributes) {
511         final StringBuilder result = new StringBuilder(20);
512 
513         for (int i = 0; i < attributes.getLength(); i++) {
514             result.append(' ');
515 
516             final Node attribute = attributes.item(i);
517             final String attrName = attribute.getNodeName();
518             final String attrValue;
519 
520             if ("a".equals(nodeName) && "href".equals(attrName)) {
521                 String value = attribute.getNodeValue();
522 
523                 assertWithMessage("links starting with '#' aren't supported: " + value)
524                     .that(value.charAt(0))
525                     .isNotEqualTo('#');
526 
527                 if (value.contains("://")) {
528                     attrValue = value;
529                 }
530                 else {
531                     if (value.charAt(0) == '/') {
532                         value = value.substring(1);
533                     }
534 
535                     attrValue = "https://checkstyle.org/" + value;
536                 }
537             }
538             else {
539                 attrValue = attribute.getNodeValue();
540             }
541 
542             result.append(attrName);
543             result.append("=\"");
544             result.append(attrValue);
545             result.append('"');
546         }
547 
548         return result.toString();
549     }

551     public static class JavaDocCapture extends AbstractCheck {
552         private static final Pattern SETTER_PATTERN = Pattern.compile("^set[A-Z].*");
553 
554         @Override
555         public boolean isCommentNodesRequired() {
556             return true;
557         }
558 
559         @Override
560         public int[] getRequiredTokens() {
561             return new int[] {
562                 TokenTypes.BLOCK_COMMENT_BEGIN,
563             };
564         }
565 
566         @Override
567         public int[] getDefaultTokens() {
568             return getRequiredTokens();
569         }
570 
571         @Override
572         public int[] getAcceptableTokens() {
573             return getRequiredTokens();
574         }

576         @Override
577         public void visitToken(DetailAST ast) {
578             if (JavadocUtil.isJavadocComment(ast)) {
579                 final DetailAST parentNode = getParent(ast);
580 
581                 switch (parentNode.getType()) {
582                     case TokenTypes.CLASS_DEF:
583                         visitClass(ast);
584                         break;
585                     case TokenTypes.METHOD_DEF:
586                         visitMethod(ast, parentNode);
587                         break;
588                     case TokenTypes.VARIABLE_DEF:
589                         visitField(ast, parentNode);
590                         break;
591                     case TokenTypes.CTOR_DEF:
592                     case TokenTypes.ENUM_DEF:
593                     case TokenTypes.ENUM_CONSTANT_DEF:
594                         // ignore
595                         break;
596                     default:
597                         assertWithMessage(
598                                 "Unknown token '" + TokenUtil.getTokenName(parentNode.getType())
599                                         + "': " + ast.getLineNo()).fail();
600                         break;
601                 }
602             }
603         }
604 
605         private static DetailAST getParent(DetailAST node) {
606             DetailAST result = node.getParent();
607             int type = result.getType();
608 
609             while (type == TokenTypes.MODIFIERS || type == TokenTypes.ANNOTATION) {
610                 result = result.getParent();
611                 type = result.getType();
612             }
613 
614             return result;
615         }
616 
617         private static void visitClass(DetailAST node) {
618             String violationMessagesText = CHECK_TEXT.get("Violation Messages");
619 
620             if (checkName.endsWith("Filter") || "SuppressWarningsHolder".equals(checkName)) {
621                 violationMessagesText = "";
622             }
623 
624             if (ScopeUtil.isInScope(node, Scope.PUBLIC)) {
625                 assertWithMessage(checkName + "'s class-level JavaDoc")
626                     .that(getJavaDocText(node))
627                     .isEqualTo(CHECK_TEXT.get("Description")
628                         + CHECK_TEXT.computeIfAbsent("Rule Description", unused -> "")
629                         + CHECK_TEXT.computeIfAbsent("Notes", unused -> "")
630                         + CHECK_TEXT.computeIfAbsent("Properties", unused -> "")
631                         + CHECK_TEXT.get("Examples")
632                         + CHECK_TEXT.get("Parent Module")
633                         + violationMessagesText + " @since "
634                         + CHECK_TEXT.get("since"));
635             }
636         }
637 
638         private static void visitField(DetailAST node, DetailAST parentNode) {
639             if (ScopeUtil.isInScope(parentNode, Scope.PUBLIC)) {
640                 final String propertyName = parentNode.findFirstToken(TokenTypes.IDENT).getText();
641                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
642 
643                 if (propertyDoc != null) {
644                     assertWithMessage(checkName + "'s class field-level JavaDoc for "
645                             + propertyName)
646                         .that(getJavaDocText(node))
647                         .isEqualTo(makeFirstUpper(propertyDoc));
648                 }
649             }
650         }
651 
652         private static void visitMethod(DetailAST node, DetailAST parentNode) {
653             if (ScopeUtil.isInScope(node, Scope.PUBLIC) && isSetterMethod(parentNode)) {
654                 final String propertyUpper = parentNode.findFirstToken(TokenTypes.IDENT)
655                         .getText().substring(3);
656                 final String propertyName = makeFirstLower(propertyUpper);
657                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
658 
659                 if (propertyDoc != null) {
660                     final String javaDoc = getJavaDocText(node);
661 
662                     assertWithMessage(checkName + "'s class method-level JavaDoc for "
663                             + propertyName)
664                         .that(javaDoc.substring(0, javaDoc.indexOf(" @param")))
665                         .isEqualTo("Setter to " + makeFirstLower(propertyDoc));
666                 }
667             }
668         }
669 
670         /**
671          * Returns whether an AST represents a setter method. This is similar to
672          * {@link CheckUtil#isSetterMethod(DetailAST)} except this doesn't care
673          * about the number of children in the method.
674          *
675          * @param ast the AST to check with.
676          * @return whether the AST represents a setter method.
677          */
678         private static boolean isSetterMethod(DetailAST ast) {
679             boolean setterMethod = false;
680 
681             if (ast.getType() == TokenTypes.METHOD_DEF) {
682                 final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
683                 final String name = type.getNextSibling().getText();
684                 final boolean matchesSetterFormat = SETTER_PATTERN.matcher(name).matches();
685                 final boolean voidReturnType = type.findFirstToken(TokenTypes.LITERAL_VOID) != null;
686 
687                 final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
688                 final boolean singleParam = params.getChildCount(TokenTypes.PARAMETER_DEF) == 1;
689 
690                 if (matchesSetterFormat && voidReturnType && singleParam) {
691                     final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
692 
693                     setterMethod = slist != null;
694                 }
695             }
696             return setterMethod;
697         }

699         private static String getJavaDocText(DetailAST node) {
700             final String text = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<document>\n"
701                     + node.getFirstChild().getText().replaceAll("(^|\\r?\\n)\\s*\\* ?", "\n")
702                             .replaceAll("\\n?@noinspection.*\\r?\\n[^@]*", "\n")
703                             .trim() + "\n</document>";
704             String result = null;
705 
706             try {
707                 result = getNodeText(XmlUtil.getRawXml(checkName, text, text).getFirstChild())
708                         .replace("\r", "");
709             }
710             catch (ParserConfigurationException ex) {
711                 assertWithMessage("Exception: " + ex.getClass() + " - " + ex.getMessage()).fail();
712             }
713 
714             return result;
715         }

717         private static String makeFirstUpper(String str) {
718             final char ch = str.charAt(0);
719             final String result;
720 
721             if (Character.isLowerCase(ch)) {
722                 result = Character.toUpperCase(ch) + str.substring(1);
723             }
724             else {
725                 result = str;
726             }
727 
728             return result;
729         }

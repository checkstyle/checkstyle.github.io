128         final ModuleFactory moduleFactory = TestUtil.getPackageObjectFactory();

199     private static void examineCheckSubSection(Node subSection, String subSectionName) {
200         switch (subSectionName) {
201             case "Description":
202             case "Examples":
203             case "Notes":
204             case "Rule Description":
205                 CHECK_TEXT.put(subSectionName, getNodeText(subSection).replace("\r", ""));
206                 break;
207             case "Properties":
208                 populateProperties(subSection);
209                 CHECK_TEXT.put(subSectionName, createPropertiesText());
210                 break;
211             case "Example of Usage":
212             case "Violation Messages":
213                 CHECK_TEXT.put(subSectionName,
214                         createViolationMessagesText(getViolationMessages(subSection)));
215                 break;
216             case "Package":
217             case "Parent Module":
218                 CHECK_TEXT.put(subSectionName, createParentText(subSection));
219                 break;
220             default:
221                 break;
222         }
223     }

251     private static void populateProperties(Node subSection) {
252         boolean skip = true;
253 
254         // if the first child is a wrapper element instead of the first table row containing
255         // the table headset
256         //   element to populate properties for to the current elements first child
257         Node child = XmlUtil.getFirstChildElement(subSection);
258         if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
259                 && "wrapper".equals(child.getAttributes().getNamedItem("class")
260                 .getTextContent())) {
261             child = XmlUtil.getFirstChildElement(child);
262         }
263         for (Node row : XmlUtil.getChildrenElements(child)) {
264             if (skip) {
265                 skip = false;
266                 continue;
267             }
268             CHECK_PROPERTIES.add(new ArrayList<>(XmlUtil.getChildrenElements(row)));
269         }

272     private static String createPropertiesText() {
273         final StringBuilder result = new StringBuilder(100);
274 
275         result.append("\n<ul>");
276 
277         for (List<Node> property : CHECK_PROPERTIES) {
278             final String propertyName = getNodeText(property.get(0));
279 
280             result.append("\n<li>\nProperty {@code ");
281             result.append(propertyName);
282             result.append("} - ");
283 
284             final String temp = getNodeText(property.get(1));
285 
286             result.append(temp);
287             CHECK_PROPERTY_DOC.put(propertyName, temp);
288 
289             String typeText = "java.lang.String[]";
290             final String propertyType = property.get(2).getTextContent();
291             final boolean isSpecialAllTokensType = propertyType.contains("set of any supported");
292             final boolean isPropertyTokenType = isSpecialAllTokensType
293                     || propertyType.contains("subset of tokens")
294                     || propertyType.contains("subset of javadoc tokens");
295             if (!isPropertyTokenType) {
296                 final String typeName =
297                         getCorrectNodeBasedOnPropertyType(property).getTextContent().trim();
298                 typeText = FULLY_QUALIFIED_CLASS_NAMES.get(typeName).getTypeName();
299             }
300             if (isSpecialAllTokensType) {
301                 typeText = "anyTokenTypesSet";
302             }
303             result.append(" Type is {@code ").append(typeText).append("}.");
304 
305             if (!isSpecialAllTokensType) {
306                 final String validationType = getValidationType(isPropertyTokenType, propertyName);
307                 if (validationType != null) {
308                     result.append(validationType);
309                 }
310             }
311 
312             result.append(getDefaultValueOfType(propertyName, isSpecialAllTokensType));
313 
314             result.append(emptyStringArrayDefaultValue(property.get(3), isPropertyTokenType));
315 
316             if (result.charAt(result.length() - 1) != '.') {
317                 result.append('.');
318             }
319 
320             result.append("\n</li>");
321         }
322 
323         result.append("\n</ul>");
324 
325         return result.toString();
326     }

375     private static String getNodeText(Node node) {
376         final StringBuilder result = new StringBuilder(20);
377 
378         for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling()) {
379             if (child.getNodeType() == Node.TEXT_NODE) {
380                 for (String temp : child.getTextContent().split("\n")) {
381                     final String text = temp.trim();
382 
383                     if (!text.isEmpty()) {
384                         if (shouldAppendSpace(result, text.charAt(0))) {
385                             result.append(' ');
386                         }
387 
388                         result.append(text);
389                     }
390                 }
391             }
392             else {
393                 if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
394                         && "wrapper".equals(child.getAttributes().getNamedItem("class")
395                         .getNodeValue())) {
396                     appendNodeText(result, XmlUtil.getFirstChildElement(child));
397                 }
398                 else {
399                     appendNodeText(result, child);
400                 }
401             }
402         }
403 
404         return result.toString();
405     }

407     // -@cs[CyclomaticComplexity] No simple way to split this apart.
408     private static void appendNodeText(StringBuilder result, Node node) {
409         final String name = transformXmlToJavaDocName(node.getNodeName());
410         final boolean list = "ol".equals(name) || "ul".equals(name);
411         final boolean newLineOpenBefore = list || "p".equals(name) || "pre".equals(name)
412                 || "li".equals(name);
413         final boolean newLineOpenAfter = newLineOpenBefore && !list;
414         final boolean newLineClose = newLineOpenAfter || list;
415         final boolean sanitize = "pre".equals(name);
416         final boolean changeToTag = "code".equals(name);
417 
418         if (newLineOpenBefore) {
419             result.append('\n');
420         }
421         else if (shouldAppendSpace(result, '<')) {
422             result.append(' ');
423         }
424 
425         if (changeToTag) {
426             result.append("{@");
427             result.append(name);
428             result.append(' ');
429         }
430         else {
431             result.append('<');
432             result.append(name);
433             result.append(getAttributeText(name, node.getAttributes()));
434             result.append('>');
435         }
436 
437         if (newLineOpenAfter) {
438             result.append('\n');
439         }
440 
441         if (sanitize) {
442             result.append(XmlUtil.sanitizeXml(node.getTextContent()));
443         }
444         else {
445             result.append(getNodeText(node));
446         }
447 
448         if (newLineClose) {
449             result.append('\n');
450         }
451 
452         if (changeToTag) {
453             result.append('}');
454         }
455         else {
456             result.append("</");
457             result.append(name);
458             result.append('>');
459         }
460     }

462     private static boolean shouldAppendSpace(StringBuilder text, char firstCharToAppend) {
463         final boolean result;
464 
465         if (text.length() == 0) {
466             result = false;
467         }
468         else {
469             final char last = text.charAt(text.length() - 1);
470 
471             result = (firstCharToAppend == '@'
472                     || Character.getType(firstCharToAppend) == Character.DASH_PUNCTUATION
473                     || Character.getType(last) == Character.OTHER_PUNCTUATION
474                     || Character.isAlphabetic(last)
475                     || Character.isAlphabetic(firstCharToAppend)) && !Character.isWhitespace(last);
476         }
477 
478         return result;
479     }

482         final String result;

497     private static String getAttributeText(String nodeName, NamedNodeMap attributes) {
498         final StringBuilder result = new StringBuilder(20);
499 
500         for (int i = 0; i < attributes.getLength(); i++) {
501             result.append(' ');
502 
503             final Node attribute = attributes.item(i);
504             final String attrName = attribute.getNodeName();
505             final String attrValue;
506 
507             if ("a".equals(nodeName) && "href".equals(attrName)) {
508                 final String value = attribute.getNodeValue();
509 
510                 assertWithMessage("links starting with '#' aren't supported: " + value)
511                     .that(value.charAt(0))
512                     .isNotEqualTo('#');
513 
514                 attrValue = getLinkValue(value);
515             }
516             else {
517                 attrValue = attribute.getNodeValue();
518             }
519 
520             result.append(attrName);
521             result.append("=\"");
522             result.append(attrValue);
523             result.append('"');
524         }
525 
526         return result.toString();
527     }

529     private static String getLinkValue(String initialValue) {
530         String value = initialValue;
531         final String attrValue;
532         if (value.contains("://")) {
533             attrValue = value;
534         }
535         else {
536             if (value.charAt(0) == '/') {
537                 value = value.substring(1);
538             }
539 
540             // Relative links to DTDs are prohibited, so we don't try to resolve them
541             if (!initialValue.startsWith("/dtds")) {
542                 value = currentXdocPath
543                         .getParent()
544                         .resolve(Paths.get(value))
545                         .normalize()
546                         .toString()
547                         .replaceAll("src[\\\\/]xdocs[\\\\/]", "")
548                         .replaceAll("\\\\", "/");
549             }
550 
551             attrValue = "https://checkstyle.org/" + value;
552         }
553         return attrValue;
554     }

581         @Override
582         public void visitToken(DetailAST ast) {
583             if (JavadocUtil.isJavadocComment(ast)) {
584                 final DetailAST parentNode = getParent(ast);
585 
586                 switch (parentNode.getType()) {
587                     case TokenTypes.CLASS_DEF:
588                         visitClass(ast);
589                         break;
590                     case TokenTypes.METHOD_DEF:
591                         visitMethod(ast, parentNode);
592                         break;
593                     case TokenTypes.VARIABLE_DEF:
594                         visitField(ast, parentNode);
595                         break;
596                     case TokenTypes.CTOR_DEF:
597                     case TokenTypes.ENUM_DEF:
598                     case TokenTypes.ENUM_CONSTANT_DEF:
599                         // ignore
600                         break;
601                     default:
602                         assertWithMessage(
603                                 "Unknown token '" + TokenUtil.getTokenName(parentNode.getType())
604                                         + "': " + ast.getLineNo()).fail();
605                         break;
606                 }
607             }
608         }

610         private static DetailAST getParent(DetailAST node) {
611             DetailAST result = node.getParent();
612             int type = result.getType();
613 
614             while (type == TokenTypes.MODIFIERS || type == TokenTypes.ANNOTATION) {
615                 result = result.getParent();
616                 type = result.getType();
617             }
618 
619             return result;
620         }

622         private static void visitClass(DetailAST node) {
623             String violationMessagesText = CHECK_TEXT.get("Violation Messages");
624 
625             if (checkName.endsWith("Filter") || "SuppressWarningsHolder".equals(checkName)) {
626                 violationMessagesText = "";
627             }
628 
629             if (ScopeUtil.isInScope(node, Scope.PUBLIC)) {
630                 final String expected = CHECK_TEXT.get("Description")
631                         + CHECK_TEXT.computeIfAbsent("Rule Description", unused -> "")
632                         + CHECK_TEXT.computeIfAbsent("Notes", unused -> "")
633                         + CHECK_TEXT.computeIfAbsent("Properties", unused -> "")
634                         + CHECK_TEXT.get("Parent Module")
635                         + violationMessagesText + " @since "
636                         + CHECK_TEXT.get("since");
637 
638                 assertWithMessage(checkName + "'s class-level JavaDoc")
639                     .that(getJavaDocText(node))
640                     .isEqualTo(expected);
641             }
642         }
643 
644         private static void visitField(DetailAST node, DetailAST parentNode) {
645             if (ScopeUtil.isInScope(parentNode, Scope.PUBLIC)) {
646                 final String propertyName = parentNode.findFirstToken(TokenTypes.IDENT).getText();
647                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
648 
649                 if (propertyDoc != null) {
650                     assertWithMessage(checkName + "'s class field-level JavaDoc for "
651                             + propertyName)
652                         .that(getJavaDocText(node))
653                         .isEqualTo(makeFirstUpper(propertyDoc));
654                 }
655             }
656         }
657 
658         private static void visitMethod(DetailAST node, DetailAST parentNode) {
659             if (ScopeUtil.isInScope(node, Scope.PUBLIC) && isSetterMethod(parentNode)) {
660                 final String propertyUpper = parentNode.findFirstToken(TokenTypes.IDENT)
661                         .getText().substring(3);
662                 final String propertyName = makeFirstLower(propertyUpper);
663                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
664 
665                 if (propertyDoc != null) {
666                     final String javaDoc = getJavaDocText(node);
667 
668                     assertWithMessage(checkName + "'s class method-level JavaDoc for "
669                             + propertyName)
670                         .that(javaDoc.substring(0, javaDoc.indexOf(" @param")))
671                         .isEqualTo("Setter to " + makeFirstLower(propertyDoc));
672                 }
673             }
674         }

676         /**
677          * Returns whether an AST represents a setter method. This is similar to
678          * {@link MissingJavadocMethodCheck#isSetterMethod(DetailAST)} except this doesn't care
679          * about the number of children in the method.
680          *
681          * @param ast the AST to check with.
682          * @return whether the AST represents a setter method.
683          */
684         private static boolean isSetterMethod(DetailAST ast) {
685             boolean setterMethod = false;
686 
687             if (ast.getType() == TokenTypes.METHOD_DEF) {
688                 final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
689                 final String name = type.getNextSibling().getText();
690                 final boolean matchesSetterFormat = SETTER_PATTERN.matcher(name).matches();
691                 final boolean voidReturnType = type.findFirstToken(TokenTypes.LITERAL_VOID) != null;
692 
693                 final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
694                 final boolean singleParam = params.getChildCount(TokenTypes.PARAMETER_DEF) == 1;
695 
696                 if (matchesSetterFormat && voidReturnType && singleParam) {
697                     final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
698 
699                     setterMethod = slist != null;
700                 }
701             }
702             return setterMethod;
703         }

705         private static String getJavaDocText(DetailAST node) {
706             final String text = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<document>\n"
707                     + node.getFirstChild().getText().replaceAll("(^|\\r?\\n)\\s*\\* ?", "\n")
708                             .replaceAll("\\n?@noinspection.*\\r?\\n[^@]*", "\n")
709                             .trim() + "\n</document>";
710             String result = null;

712             try {

714                         .replace("\r", "");
715             }
716             catch (ParserConfigurationException ex) {
717                 assertWithMessage("Exception: " + ex.getClass() + " - " + ex.getMessage()).fail();
718             }
719 
720             return result;
721         }

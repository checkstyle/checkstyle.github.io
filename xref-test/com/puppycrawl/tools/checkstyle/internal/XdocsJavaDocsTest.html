133     public void testAllCheckSectionJavaDocs() throws Exception {
134         final ModuleFactory moduleFactory = TestUtil.getPackageObjectFactory();
135 
136         for (Path path : XdocUtil.getXdocsConfigFilePaths(XdocUtil.getXdocsFilePaths())) {
137             final File file = path.toFile();
138             final String fileName = file.getName();
139 
140             if ("config_system_properties.xml".equals(fileName)) {
141                 continue;
142             }
143 
144             final String input = new String(Files.readAllBytes(path), UTF_8);
145             final Document document = XmlUtil.getRawXml(fileName, input, input);
146             final NodeList sources = document.getElementsByTagName("section");
147 
148             for (int position = 0; position < sources.getLength(); position++) {
149                 final Node section = sources.item(position);
150                 final String sectionName = XmlUtil.getNameAttributeOfNode(section);
151 
152                 if ("Content".equals(sectionName) || "Overview".equals(sectionName)) {
153                     continue;
154                 }
155 
156                 examineCheckSection(moduleFactory, fileName, sectionName, section);
157             }
158         }
159     }

161     private static void examineCheckSection(ModuleFactory moduleFactory, String fileName,
162             String sectionName, Node section) throws Exception {
163         final Object instance;
164 
165         try {
166             instance = moduleFactory.createModule(sectionName);
167         }
168         catch (CheckstyleException ex) {
169             throw new CheckstyleException(fileName + " couldn't find class: " + sectionName, ex);
170         }
171 
172         CHECK_TEXT.clear();
173         CHECK_PROPERTIES.clear();
174         CHECK_PROPERTY_DOC.clear();
175         checkName = sectionName;
176 
177         examineCheckSectionChildren(section);
178 
179         final List<File> files = new ArrayList<>();
180         files.add(new File("src/main/java/" + instance.getClass().getName().replace(".", "/")
181                 + ".java"));
182 
183         checker.process(files);
184     }

186     private static void examineCheckSectionChildren(Node section) {
187         for (Node subSection : XmlUtil.getChildrenElements(section)) {
188             if (!"subsection".equals(subSection.getNodeName())) {
189                 final String text = getNodeText(subSection);
190                 if (text.startsWith("Since Checkstyle")) {
191                     CHECK_TEXT.put("since", text.substring(17));
192                 }
193                 continue;
194             }
195 
196             final String subSectionName = XmlUtil.getNameAttributeOfNode(subSection);
197 
198             examineCheckSubSection(subSection, subSectionName);
199         }
200     }

202     private static void examineCheckSubSection(Node subSection, String subSectionName) {
203         switch (subSectionName) {
204             case "Description":
205             case "Examples":
206             case "Notes":
207             case "Rule Description":
208                 CHECK_TEXT.put(subSectionName, getNodeText(subSection).replace("\r", ""));
209                 break;
210             case "Properties":
211                 populateProperties(subSection);
212                 CHECK_TEXT.put(subSectionName, createPropertiesText());
213                 break;
214             case "Example of Usage":
215             case "Violation Messages":
216                 CHECK_TEXT.put(subSectionName,
217                         createViolationMessagesText(getViolationMessages(subSection)));
218                 break;
219             case "Package":
220             case "Parent Module":
221                 CHECK_TEXT.put(subSectionName, createParentText(subSection));
222                 break;
223             default:
224                 break;
225         }
226     }

228     private static List<String> getViolationMessages(Node subsection) {
229         final Node child = XmlUtil.getFirstChildElement(subsection);
230         final List<String> violationMessages = new ArrayList<>();
231         for (Node row : XmlUtil.getChildrenElements(child)) {
232             violationMessages.add(row.getTextContent().trim());
233         }
234         return violationMessages;
235     }

254     private static void populateProperties(Node subSection) {
255         boolean skip = true;
256 
257         // if the first child is a wrapper element instead of the first table row containing
258         // the table head
259         //   set element to populate properties for to the current elements first child
260         Node child = XmlUtil.getFirstChildElement(subSection);
261         if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
262                 && "wrapper".equals(child.getAttributes().getNamedItem("class")
263                 .getTextContent())) {
264             child = XmlUtil.getFirstChildElement(child);
265         }
266         for (Node row : XmlUtil.getChildrenElements(child)) {
267             if (skip) {
268                 skip = false;
269                 continue;
270             }
271             CHECK_PROPERTIES.add(new ArrayList<>(XmlUtil.getChildrenElements(row)));
272         }
273     }

275     private static String createPropertiesText() {
276         final StringBuilder result = new StringBuilder(100);
277 
278         result.append("\n<ul>");
279 
280         for (List<Node> property : CHECK_PROPERTIES) {
281             final String propertyName = getNodeText(property.get(0));
282 
283             result.append("\n<li>\nProperty {@code ");
284             result.append(propertyName);
285             result.append("} - ");
286 
287             final String temp = getNodeText(property.get(1));
288 
289             result.append(temp);
290             CHECK_PROPERTY_DOC.put(propertyName, temp);
291 
292             String typeText = "int[]";
293             if (!property.get(2).getTextContent().contains("subset of tokens")) {
294                 final Node typeNode;
295                 if (property.get(2).getFirstChild().getFirstChild() == null) {
296                     typeNode = property.get(2).getFirstChild().getNextSibling();
297                 }
298                 else {
299                     typeNode = property.get(2).getFirstChild().getFirstChild();
300                 }
301                 final String typeName = typeNode.getTextContent().trim();
302                 typeText = FULLY_QUALIFIED_CLASS_NAMES.get(typeName).getTypeName();
303             }
304             result.append(" Type is {@code ").append(typeText).append("}.");
305 
306             if (propertyName.endsWith("token") || propertyName.endsWith("tokens")) {
307                 result.append(" Default value is: ");
308             }
309             else {
310                 result.append(" Default value is ");
311             }
312 
313             result.append(getNodeText(property.get(3)));
314 
315             if (result.charAt(result.length() - 1) != '.') {
316                 result.append('.');
317             }
318 
319             result.append("\n</li>");
320         }
321 
322         result.append("\n</ul>");
323 
324         return result.toString();
325     }

327     private static String getNodeText(Node node) {
328         final StringBuilder result = new StringBuilder(20);
329 
330         for (Node child = node.getFirstChild(); child != null; child = child.getNextSibling()) {
331             if (child.getNodeType() == Node.TEXT_NODE) {
332                 for (String temp : child.getTextContent().split("\n")) {
333                     final String text = temp.trim();
334 
335                     if (!text.isEmpty()) {
336                         if (shouldAppendSpace(result, text.charAt(0))) {
337                             result.append(' ');
338                         }
339 
340                         result.append(text);
341                     }
342                 }
343             }
344             else {
345                 if (child.hasAttributes() && child.getAttributes().getNamedItem("class") != null
346                         && "wrapper".equals(child.getAttributes().getNamedItem("class")
347                         .getNodeValue())) {
348                     appendNodeText(result, XmlUtil.getFirstChildElement(child));
349                 }
350                 else {
351                     appendNodeText(result, child);
352                 }
353             }
354         }
355 
356         return result.toString();
357     }

359     // -@cs[CyclomaticComplexity] No simple way to split this apart.
360     private static void appendNodeText(StringBuilder result, Node node) {
361         final String name = transformXmlToJavaDocName(node.getNodeName());
362         final boolean list = "ol".equals(name) || "ul".equals(name);
363         final boolean newLineOpenBefore = list || "p".equals(name) || "pre".equals(name)
364                 || "li".equals(name);
365         final boolean newLineOpenAfter = newLineOpenBefore && !list;
366         final boolean newLineClose = newLineOpenAfter || list;
367         final boolean sanitize = "pre".equals(name);
368         final boolean changeToTag = "code".equals(name);
369 
370         if (newLineOpenBefore) {
371             result.append('\n');
372         }
373         else if (shouldAppendSpace(result, '<')) {
374             result.append(' ');
375         }
376 
377         if (changeToTag) {
378             result.append("{@");
379             result.append(name);
380             result.append(' ');
381         }
382         else {
383             result.append('<');
384             result.append(name);
385             result.append(getAttributeText(name, node.getAttributes()));
386             result.append('>');
387         }
388 
389         if (newLineOpenAfter) {
390             result.append('\n');
391         }
392 
393         if (sanitize) {
394             result.append(XmlUtil.sanitizeXml(node.getTextContent()));
395         }
396         else {
397             result.append(getNodeText(node));
398         }
399 
400         if (newLineClose) {
401             result.append('\n');
402         }
403 
404         if (changeToTag) {
405             result.append('}');
406         }
407         else {
408             result.append("</");
409             result.append(name);
410             result.append('>');
411         }
412     }

414     private static boolean shouldAppendSpace(StringBuilder text, char firstCharToAppend) {
415         final boolean result;
416 
417         if (text.length() == 0) {
418             result = false;
419         }
420         else {
421             final char last = text.charAt(text.length() - 1);
422 
423             result = (firstCharToAppend == '@'
424                     || Character.getType(firstCharToAppend) == Character.DASH_PUNCTUATION
425                     || Character.getType(last) == Character.OTHER_PUNCTUATION
426                     || Character.isAlphabetic(last)
427                     || Character.isAlphabetic(firstCharToAppend)) && !Character.isWhitespace(last);
428         }
429 
430         return result;
431     }

433     private static String transformXmlToJavaDocName(String name) {
434         final String result;
435 
436         if ("source".equals(name)) {
437             result = "pre";
438         }
439         else if ("h4".equals(name)) {
440             result = "p";
441         }
442         else {
443             result = name;
444         }
445 
446         return result;
447     }

449     private static String getAttributeText(String nodeName, NamedNodeMap attributes) {
450         final StringBuilder result = new StringBuilder(20);
451 
452         for (int i = 0; i < attributes.getLength(); i++) {
453             result.append(' ');
454 
455             final Node attribute = attributes.item(i);
456             final String attrName = attribute.getNodeName();
457             final String attrValue;
458 
459             if ("a".equals(nodeName) && "href".equals(attrName)) {
460                 String value = attribute.getNodeValue();
461 
462                 assertNotEquals('#', value.charAt(0),
463                         "links starting with '#' aren't supported: " + value);
464 
465                 if (value.contains("://")) {
466                     attrValue = value;
467                 }
468                 else {
469                     if (value.charAt(0) == '/') {
470                         value = value.substring(1);
471                     }
472 
473                     attrValue = "https://checkstyle.org/" + value;
474                 }
475             }
476             else {
477                 attrValue = attribute.getNodeValue();
478             }
479 
480             result.append(attrName);
481             result.append("=\"");
482             result.append(attrValue);
483             result.append('"');
484         }
485 
486         return result.toString();
487     }

489     public static class JavaDocCapture extends AbstractCheck {
490         private static final Pattern SETTER_PATTERN = Pattern.compile("^set[A-Z].*");
491 
492         @Override
493         public boolean isCommentNodesRequired() {
494             return true;
495         }
496 
497         @Override
498         public int[] getRequiredTokens() {
499             return new int[] {
500                 TokenTypes.BLOCK_COMMENT_BEGIN,
501             };
502         }
503 
504         @Override
505         public int[] getDefaultTokens() {
506             return getRequiredTokens();
507         }
508 
509         @Override
510         public int[] getAcceptableTokens() {
511             return getRequiredTokens();
512         }
513 
514         @Override
515         public void visitToken(DetailAST ast) {
516             if (JavadocUtil.isJavadocComment(ast)) {
517                 final DetailAST parentNode = getParent(ast);
518 
519                 switch (parentNode.getType()) {
520                     case TokenTypes.CLASS_DEF:
521                         visitClass(ast);
522                         break;
523                     case TokenTypes.METHOD_DEF:
524                         visitMethod(ast, parentNode);
525                         break;
526                     case TokenTypes.VARIABLE_DEF:
527                         visitField(ast, parentNode);
528                         break;
529                     case TokenTypes.CTOR_DEF:
530                     case TokenTypes.ENUM_DEF:
531                     case TokenTypes.ENUM_CONSTANT_DEF:
532                         // ignore
533                         break;
534                     default:
535                         fail("Unknown token '" + TokenUtil.getTokenName(parentNode.getType())
536                                 + "': " + ast.getLineNo());
537                         break;
538                 }
539             }
540         }
541 
542         private static DetailAST getParent(DetailAST node) {
543             DetailAST result = node.getParent();
544             int type = result.getType();
545 
546             while (type == TokenTypes.MODIFIERS || type == TokenTypes.ANNOTATION) {
547                 result = result.getParent();
548                 type = result.getType();
549             }
550 
551             return result;
552         }
553 
554         private static void visitClass(DetailAST node) {
555             String violationMessagesText = CHECK_TEXT.get("Violation Messages");
556 
557             if (checkName.endsWith("Filter") || "SuppressWarningsHolder".equals(checkName)) {
558                 violationMessagesText = "";
559             }
560 
561             if (ScopeUtil.isInScope(node, Scope.PUBLIC)) {
562                 assertEquals(CHECK_TEXT.get("Description")
563                         + CHECK_TEXT.computeIfAbsent("Rule Description", unused -> "")
564                         + CHECK_TEXT.computeIfAbsent("Notes", unused -> "")
565                         + CHECK_TEXT.computeIfAbsent("Properties", unused -> "")
566                         + CHECK_TEXT.get("Examples")
567                         + CHECK_TEXT.get("Parent Module")
568                         + violationMessagesText + " @since "
569                         + CHECK_TEXT.get("since"), getJavaDocText(node),
570                         checkName + "'s class-level JavaDoc");
571             }
572         }
573 
574         private static void visitField(DetailAST node, DetailAST parentNode) {
575             if (ScopeUtil.isInScope(parentNode, Scope.PUBLIC)) {
576                 final String propertyName = parentNode.findFirstToken(TokenTypes.IDENT).getText();
577                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
578 
579                 if (propertyDoc != null) {
580                     assertEquals(makeFirstUpper(propertyDoc), getJavaDocText(node),
581                             checkName + "'s class field-level JavaDoc for " + propertyName);
582                 }
583             }
584         }
585 
586         private static void visitMethod(DetailAST node, DetailAST parentNode) {
587             if (ScopeUtil.isInScope(node, Scope.PUBLIC) && isSetterMethod(parentNode)) {
588                 final String propertyUpper = parentNode.findFirstToken(TokenTypes.IDENT)
589                         .getText().substring(3);
590                 final String propertyName = makeFirstLower(propertyUpper);
591                 final String propertyDoc = CHECK_PROPERTY_DOC.get(propertyName);
592 
593                 if (propertyDoc != null) {
594                     final String javaDoc = getJavaDocText(node);
595 
596                     assertEquals("Setter to " + makeFirstLower(propertyDoc),
597                             javaDoc.substring(0, javaDoc.indexOf(" @param")),
598                             checkName + "'s class method-level JavaDoc for " + propertyName);
599                 }
600             }
601         }
602 
603         /**
604          * Returns whether an AST represents a setter method. This is similar to
605          * {@link CheckUtil#isSetterMethod(DetailAST)} except this doesn't care
606          * about the number of children in the method.
607          *
608          * @param ast the AST to check with.
609          * @return whether the AST represents a setter method.
610          */
611         private static boolean isSetterMethod(DetailAST ast) {
612             boolean setterMethod = false;
613 
614             if (ast.getType() == TokenTypes.METHOD_DEF) {
615                 final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
616                 final String name = type.getNextSibling().getText();
617                 final boolean matchesSetterFormat = SETTER_PATTERN.matcher(name).matches();
618                 final boolean voidReturnType = type.findFirstToken(TokenTypes.LITERAL_VOID) != null;
619 
620                 final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
621                 final boolean singleParam = params.getChildCount(TokenTypes.PARAMETER_DEF) == 1;
622 
623                 if (matchesSetterFormat && voidReturnType && singleParam) {
624                     final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
625 
626                     setterMethod = slist != null;
627                 }
628             }
629             return setterMethod;
630         }

632         private static String getJavaDocText(DetailAST node) {
633             final String text = "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n<document>\n"
634                     + node.getFirstChild().getText().replaceAll("(^|\\r?\\n)\\s*\\* ?", "\n")
635                             .replaceAll("\\n@noinspection.*\\r?\\n", "\n")
636                             .trim() + "\n</document>";
637             String result = null;
638 
639             try {
640                 result = getNodeText(XmlUtil.getRawXml(checkName, text, text).getFirstChild())
641                         .replace("\r", "");
642             }
643             catch (ParserConfigurationException ex) {
644                 fail("Exception: " + ex.getClass() + " - " + ex.getMessage());
645             }
646 
647             return result;
648         }

650         private static String makeFirstUpper(String str) {
651             final char ch = str.charAt(0);
652             final String result;
653 
654             if (Character.isLowerCase(ch)) {
655                 result = Character.toUpperCase(ch) + str.substring(1);
656             }
657             else {
658                 result = str;
659             }
660 
661             return result;
662         }
663 
664         private static String makeFirstLower(String str) {
665             return Character.toLowerCase(str.charAt(0)) + str.substring(1);
666         }

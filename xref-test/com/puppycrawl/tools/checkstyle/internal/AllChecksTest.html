390     private static void validateAllCheckTokensAreReferencedInConfigFile(String configName,
391             Configuration configuration, Map<String, Set<String>> tokensToIgnore,
392             boolean defaultTokensMustBeExplicit) throws Exception {
393         final ModuleFactory moduleFactory = TestUtil.getPackageObjectFactory();
394         final Set<Configuration> configChecks = ConfigurationUtil.getChecks(configuration);
395 
396         final Map<String, Set<String>> configCheckTokens = new HashMap<>();
397         final Map<String, Set<String>> checkTokens = new HashMap<>();
398 
399         for (Configuration checkConfig : configChecks) {
400             final String checkName = checkConfig.getName();
401             final Object instance;
402 
403             try {
404                 instance = moduleFactory.createModule(checkName);
405             }
406             catch (CheckstyleException ex) {
407                 throw new CheckstyleException("Couldn't find check: " + checkName, ex);
408             }
409             final AbstractCheck check;
410             if (instance instanceof AbstractCheck
411                     && !isAllTokensAcceptable((AbstractCheck) instance)) {
412                 check = (AbstractCheck) instance;
413             }
414             else {
415                 // we can not have in our config test for all tokens
416                 continue;
417             }
418 
419             Set<String> configTokens = configCheckTokens.get(checkName);
420 
421             if (configTokens == null) {
422                 configTokens = new HashSet<>();
423 
424                 configCheckTokens.put(checkName, configTokens);
425 
426                 // add all overridden tokens
427                 final Set<String> overrideTokens = tokensToIgnore.get(checkName);
428 
429                 if (overrideTokens != null) {
430                     configTokens.addAll(overrideTokens);
431                 }
432 
433                 configTokens.addAll(CheckUtil.getTokenNameSet(check.getRequiredTokens()));
434                 checkTokens.put(checkName,
435                         CheckUtil.getTokenNameSet(check.getAcceptableTokens()));
436             }
437 
438             try {
439                 configTokens.addAll(Arrays.asList(checkConfig.getProperty("tokens").trim()
440                         .split(",\\s*")));
441             }
442             catch (CheckstyleException ex) {
443                 // no tokens defined, so it is using default
444                 if (defaultTokensMustBeExplicit) {
445                     validateDefaultTokens(checkConfig, check, configTokens);
446                 }
447                 else {
448                     configTokens.addAll(CheckUtil.getTokenNameSet(check.getDefaultTokens()));
449                 }
450             }
451         }
452         for (Entry<String, Set<String>> entry : checkTokens.entrySet()) {
453             final Set<String> actual = configCheckTokens.get(entry.getKey());
454             assertWithMessage("'" + entry.getKey()
455                     + "' should have all acceptable tokens from check in " + configName
456                     + " config or specify an override to ignore the specific tokens")
457                 .that(actual)
458                 .isEqualTo(entry.getValue());
459         }
460     }

462     private static boolean isAllTokensAcceptable(AbstractCheck check) {
463         return Arrays.equals(check.getAcceptableTokens(), TokenUtil.getAllTokenIds());
464     }
465 
466     private static void validateDefaultTokens(Configuration checkConfig, AbstractCheck check,
467                                               Set<String> configTokens) {
468 
469         final BitSet defaultTokensSet = TokenUtil.asBitSet(check.getDefaultTokens());
470         final BitSet requiredTokensSet = TokenUtil.asBitSet(check.getRequiredTokens());
471 
472         if (defaultTokensSet.equals(requiredTokensSet)) {
473             configTokens.addAll(
474                     CheckUtil.getTokenNameSet(check.getDefaultTokens()));
475         }
476         else {
477             assertWithMessage("All default tokens should be used in config for "
478                     + checkConfig.getName()).fail();
479         }
480     }

484         final Set<String> checkstyleModulesNames = CheckUtil.getSimpleNames(CheckUtil

577     private static void verifyCheckstyleMessage(Map<String, List<String>> usedMessages,
578             Class<?> module, Field message) throws Exception {
579         final String messageString = message.get(null).toString();
580         final String packageName = module.getPackage().getName();
581         final List<String> packageMessages =
582                 usedMessages.computeIfAbsent(packageName, key -> new ArrayList<>());
583 
584         packageMessages.add(messageString);
585 
586         for (Locale locale : ALL_LOCALES) {
587             String result = null;
588 
589             try {
590                 result = CheckUtil.getCheckMessage(module, locale, messageString);
591             }
592             // -@cs[IllegalCatch] There is no other way to deliver filename that was used
593             catch (Exception ex) {
594                 assertWithMessage(module.getSimpleName() + " with the message '" + messageString
595                         + "' in locale '" + locale.getLanguage() + "' failed with: "
596                         + ex.getClass().getSimpleName() + " - " + ex.getMessage()).fail();
597             }
598 
599             assertWithMessage(module.getSimpleName() + " should have text for the message '"
600                     + messageString + "' in locale " + locale.getLanguage() + "'")
601                 .that(result)
602                 .isNotNull();
603             assertWithMessage("%s should have non-empty text for the message '%s' in locale '%s'",
604                             module.getSimpleName(), messageString, locale.getLanguage())
605                     .that(result.trim())
606                     .isNotEmpty();
607             assertWithMessage(
608                     module.getSimpleName() + " should have non-TODO text for the message '"
609                             + messageString + "' in locale " + locale.getLanguage() + "'")
610                                     .that(!"todo.match".equals(messageString)
611                                             && result.trim().startsWith("TODO"))
612                                     .isFalse();
613         }
614     }

616     /**
617      * Checks that an array is a subset of other array.
618      *
619      * @param array to check whether it is a subset.
620      * @param arrayToCheckIn array to check in.
621      * @return {@code true} if all elements in {@code array} are in {@code arrayToCheckIn}.
622      */
623     private static boolean isSubset(int[] array, int... arrayToCheckIn) {
624         Arrays.sort(arrayToCheckIn);
625         boolean result = true;
626         for (final int element : array) {
627             if (Arrays.binarySearch(arrayToCheckIn, element) < 0) {
628                 result = false;
629                 break;
630             }
631         }
632         return result;
633     }

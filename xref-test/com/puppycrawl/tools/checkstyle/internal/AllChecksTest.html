377     private static void validateAllCheckTokensAreReferencedInConfigFile(String configName,
378             Configuration configuration, Map<String, Set<String>> tokensToIgnore,
379             boolean defaultTokensMustBeExplicit) throws Exception {
380         final ModuleFactory moduleFactory = TestUtil.getPackageObjectFactory();
381         final Set<Configuration> configChecks = ConfigurationUtil.getChecks(configuration);
382 
383         final Map<String, Set<String>> configCheckTokens = new HashMap<>();
384         final Map<String, Set<String>> checkTokens = new HashMap<>();
385 
386         for (Configuration checkConfig : configChecks) {
387             final String checkName = checkConfig.getName();
388             final Object instance;
389 
390             try {
391                 instance = moduleFactory.createModule(checkName);
392             }
393             catch (CheckstyleException ex) {
394                 throw new CheckstyleException("Couldn't find check: " + checkName, ex);
395             }
396 
397             if (instance instanceof AbstractCheck) {
398                 final AbstractCheck check = (AbstractCheck) instance;
399 
400                 Set<String> configTokens = configCheckTokens.get(checkName);
401 
402                 if (configTokens == null) {
403                     configTokens = new HashSet<>();
404 
405                     configCheckTokens.put(checkName, configTokens);
406 
407                     // add all overridden tokens
408                     final Set<String> overrideTokens = tokensToIgnore.get(checkName);
409 
410                     if (overrideTokens != null) {
411                         configTokens.addAll(overrideTokens);
412                     }
413 
414                     configTokens.addAll(CheckUtil.getTokenNameSet(check.getRequiredTokens()));
415                     checkTokens.put(checkName,
416                             CheckUtil.getTokenNameSet(check.getAcceptableTokens()));
417                 }
418 
419                 try {
420                     configTokens.addAll(Arrays.asList(checkConfig.getAttribute("tokens").trim()
421                             .split(",\\s*")));
422                 }
423                 catch (CheckstyleException ex) {
424                     // no tokens defined, so it is using default
425                     if (defaultTokensMustBeExplicit) {
426                         validateDefaultTokens(checkConfig, check, configTokens);
427                     }
428                     else {
429                         configTokens.addAll(CheckUtil.getTokenNameSet(check.getDefaultTokens()));
430                     }
431                 }
432             }
433         }
434 
435         for (Entry<String, Set<String>> entry : checkTokens.entrySet()) {
436             final Set<String> actual = configCheckTokens.get(entry.getKey());
437             assertEquals(entry.getValue(), actual,
438                     "'" + entry.getKey() + "' should have all acceptable tokens from check in "
439                     + configName + " config or specify an override to ignore the specific tokens");
440         }
441     }

443     private static void validateDefaultTokens(Configuration checkConfig, AbstractCheck check,
444                                               Set<String> configTokens) {
445         if (Arrays.equals(check.getDefaultTokens(), check.getRequiredTokens())) {
446             configTokens.addAll(
447                     CheckUtil.getTokenNameSet(check.getDefaultTokens()));
448         }
449         else {
450             fail("All default tokens should be used in config for "
451                     + checkConfig.getName());
452         }
453     }

511                                 + " should be 'public static final'");
512 
513                 // below is required for package/private classes
514                 if (!message.isAccessible()) {
515                     message.setAccessible(true);
516                 }
517 
518                 verifyCheckstyleMessage(usedMessages, module, message);
519             }
520         }
521 
522         // test properties for messages not used by checks
523         for (Entry<String, List<String>> entry : usedMessages.entrySet()) {
524             final Properties pr = new Properties();
525             pr.load(AllChecksTest.class.getResourceAsStream(
526                     "/" + entry.getKey().replace('.', '/') + "/messages.properties"));
527 
528             for (Object key : pr.keySet()) {
529                 // hidden exception messages
530                 if ("translation.wrongLanguageCode".equals(key)) {
531                     continue;
532                 }
533 
534                 assertTrue(entry.getValue().contains(key.toString()),
535                         "property '" + key + "' isn't used by any check in package '"
536                         + entry.getKey() + "'");
537             }
538         }
539     }

541     private static void verifyCheckstyleMessage(Map<String, List<String>> usedMessages,
542             Class<?> module, Field message) throws Exception {
543         final String messageString = message.get(null).toString();
544         final String packageName = module.getPackage().getName();
545         final List<String> packageMessages =
546                 usedMessages.computeIfAbsent(packageName, key -> new ArrayList<>());
547 
548         packageMessages.add(messageString);
549 
550         for (Locale locale : ALL_LOCALES) {
551             String result = null;
552 
553             try {
554                 result = CheckUtil.getCheckMessage(module, locale, messageString);
555             }
556             // -@cs[IllegalCatch] There is no other way to deliver filename that was used
557             catch (Exception ex) {
558                 fail(module.getSimpleName() + " with the message '" + messageString
559                         + "' in locale '" + locale.getLanguage() + "' failed with: "
560                         + ex.getClass().getSimpleName() + " - " + ex.getMessage());
561             }
562 
563             assertNotNull(result, module.getSimpleName() + " should have text for the message '"
564                     + messageString + "' in locale " + locale.getLanguage() + "'");
565             assertFalse(result.trim().isEmpty(), module.getSimpleName()
566                     + " should have non-empty text for the message '"
567                     + messageString + "' in locale '" + locale.getLanguage() + "'");
568             assertFalse(!"todo.match".equals(messageString) && result.trim().startsWith("TODO"),
569                     module.getSimpleName()
570                     + " should have non-TODO text for the message '"
571                     + messageString + "' in locale " + locale.getLanguage() + "'");
572         }
573     }

575     /**
576      * Checks that an array is a subset of other array.
577      *
578      * @param array to check whether it is a subset.
579      * @param arrayToCheckIn array to check in.
580      * @return {@code true} if all elements in {@code array} are in {@code arrayToCheckIn}.
581      */
582     private static boolean isSubset(int[] array, int... arrayToCheckIn) {
583         Arrays.sort(arrayToCheckIn);
584         boolean result = true;
585         for (final int element : array) {
586             if (Arrays.binarySearch(arrayToCheckIn, element) < 0) {
587                 result = false;
588                 break;
589             }
590         }
591         return result;
592     }

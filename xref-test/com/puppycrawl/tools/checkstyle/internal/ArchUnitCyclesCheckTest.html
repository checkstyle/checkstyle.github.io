20  package com.puppycrawl.tools.checkstyle.internal;
21  
22  import static com.tngtech.archunit.library.dependencies.SlicesRuleDefinition.slices;
23  
24  import org.junit.jupiter.api.Test;
25  
26  import com.tngtech.archunit.core.domain.JavaClasses;
27  import com.tngtech.archunit.core.importer.ClassFileImporter;
28  import com.tngtech.archunit.core.importer.ImportOption;
29  import com.tngtech.archunit.lang.ArchRule;
30  import com.tngtech.archunit.library.freeze.FreezingArchRule;
31  
32  public class ArchUnitCyclesCheckTest {
33  
34      /**
35       * This test checks that
36       * <a href="https://www.archunit.org/userguide/html/000_Index.html#_slices">slices</a> should
37       * be free of
38       * <a href="https://www.archunit.org/userguide/html/000_Index.html#_cycle_checks">cycles</a>.
39       * In this particular test, each slice is a different package. Currently the violations from
40       * this test are frozen using
41       * <a href="https://www.archunit.org/userguide/html/000_Index.html#_freezing_arch_rules">
42       * FreezingArchRule</a>.
43       *
44       * <p>Whenever a new cycle is introduced or removed, the test will pass successfully but the
45       * frozen violations will be updated. It is highly recommended to avoid new cycles and, it
46       * is preferred to remove them. In both cases commit the changes for further discussion.
47       *
48       * <p>The frozen violations are present in {@code config/archunit-store} directory.
49       */
50      @Test
51      public void testSlicesShouldBeFreeOfCycles() {
52          final JavaClasses importedClasses = new ClassFileImporter()
53              .withImportOption(ImportOption.Predefined.DO_NOT_INCLUDE_TESTS)
54              .importPackages("com.puppycrawl.tools.checkstyle");
55  
56          final ArchRule slicesShouldBeFreeOfCycles = FreezingArchRule.freeze(
57              slices()
58              .matching("com.puppycrawl.tools.checkstyle.(**)")
59              .should().beFreeOfCycles());
60  
61          slicesShouldBeFreeOfCycles.check(importedClasses);
62      }
63  }

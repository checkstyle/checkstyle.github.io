46      /**
47       * Verifies that utils class has private constructor and invokes it to satisfy code coverage.
48       *
49       * @param utilClass class to test for c-tor
50       * @param checkConstructorIsPrivate flag to skip check for private visibility, it is useful
51       *                                  for Classes that are mocked by PowerMockRunner that make
52       *                                  private c-tors as public
53       * @return true if constructor is expected.
54       * @noinspection BooleanParameter
55       */
56      public static boolean isUtilsClassHasPrivateConstructor(final Class<?> utilClass,
57                                                               boolean checkConstructorIsPrivate)
58              throws ReflectiveOperationException {
59          final Constructor<?> constructor = utilClass.getDeclaredConstructor();
60          final boolean result;
61          if (checkConstructorIsPrivate && !Modifier.isPrivate(constructor.getModifiers())) {
62              result = false;
63          }
64          else {
65              constructor.setAccessible(true);
66              constructor.newInstance();
67              result = true;
68          }
69          return result;
70      }

72      /**
73       * Retrieves the specified field by it's name in the class or it's direct super.
74       *
75       * @param clss The class to retrieve the field for.
76       * @param fieldName The name of the field to retrieve.
77       * @return The class' field.
78       * @throws NoSuchFieldException if the requested field cannot be found in the class.
79       */
80      public static Field getClassDeclaredField(Class<?> clss, String fieldName)
81              throws NoSuchFieldException {
82          final Optional<Field> classField = Arrays.stream(clss.getDeclaredFields())
83                  .filter(field -> fieldName.equals(field.getName())).findFirst();
84          final Field resultField;
85          if (classField.isPresent()) {
86              resultField = classField.get();
87          }
88          else {
89              resultField = clss.getSuperclass().getDeclaredField(fieldName);
90          }
91          resultField.setAccessible(true);
92          return resultField;
93      }

95      /**
96       * Retrieves the specified method by it's name in the class or it's direct super.
97       *
98       * @param clss The class to retrieve the field for.
99       * @param methodName The name of the method to retrieve.
100      * @return The class' field.
101      * @throws NoSuchMethodException if the requested method cannot be found in the class.
102      */
103     public static Method getClassDeclaredMethod(Class<?> clss, String methodName)
104             throws NoSuchMethodException {
105         final Optional<Method> classMethod = Arrays.stream(clss.getDeclaredMethods())
106                 .filter(method -> methodName.equals(method.getName())).findFirst();
107         final Method resultMethod;
108         if (classMethod.isPresent()) {
109             resultMethod = classMethod.get();
110         }
111         else {
112             resultMethod = clss.getSuperclass().getDeclaredMethod(methodName);
113         }
114         resultMethod.setAccessible(true);
115         return resultMethod;
116     }

118     /**
119      * Checks if stateful field is cleared during {@link AbstractCheck#beginTree} in check.
120      *
121      * @param check      check object which field is to be verified
122      * @param astToVisit ast to pass into check methods
123      * @param fieldName  name of the field to be checked
124      * @param isClear    function for checking field state
125      * @return {@code true} if state of the field is cleared
126      * @throws NoSuchFieldException   if there is no field with the
127      *                                {@code fieldName} in the {@code check}
128      * @throws IllegalAccessException if the field is inaccessible
129      */
130     public static boolean isStatefulFieldClearedDuringBeginTree(AbstractCheck check,
131                                                                 DetailAST astToVisit,
132                                                                 String fieldName,
133                                                                 Predicate<Object> isClear)
134             throws NoSuchFieldException, IllegalAccessException {
135         check.beginTree(astToVisit);
136         check.visitToken(astToVisit);
137         check.beginTree(null);
138         final Field resultField = getClassDeclaredField(check.getClass(), fieldName);
139         return isClear.test(resultField.get(check));
140     }
141 
142     /**
143      * Checks if stateful field is cleared during {@link AutomaticBean}'s finishLocalSetup.
144      *
145      * @param filter filter object which field is to be verified
146      * @param event event to pass into filter methods
147      * @param fieldName name of the field to be checked
148      * @param isClear function for checking field state
149      * @return {@code true} if state of the field is cleared
150      * @throws Exception if there was an error.
151      */
152     public static boolean isStatefulFieldClearedDuringLocalSetup(
153             TreeWalkerFilter filter, TreeWalkerAuditEvent event,
154             String fieldName, Predicate<Object> isClear) throws Exception {
155         filter.accept(event);
156         getClassDeclaredMethod(filter.getClass(), "finishLocalSetup").invoke(filter);
157         final Field resultField = getClassDeclaredField(filter.getClass(), fieldName);
158         return isClear.test(resultField.get(filter));
159     }

172     /**
173      * Finds node of specified type among root children, siblings, siblings children
174      * on any deep level.
175      *
176      * @param root      DetailAST
177      * @param predicate predicate
178      * @return {@link Optional} of {@link DetailAST} node which matches the predicate.
179      */
180     public static Optional<DetailAST> findTokenInAstByPredicate(DetailAST root,
181                                                                 Predicate<DetailAST> predicate) {
182         DetailAST curNode = root;
183         while (!predicate.test(curNode)) {
184             DetailAST toVisit = curNode.getFirstChild();
185             while (curNode != null && toVisit == null) {
186                 toVisit = curNode.getNextSibling();
187                 if (toVisit == null) {
188                     curNode = curNode.getParent();
189                 }
190             }
191 
192             if (curNode == toVisit || curNode == root.getParent()) {
193                 curNode = null;
194                 break;
195             }
196 
197             curNode = toVisit;
198         }
199         return Optional.ofNullable(curNode);
200     }

202     /**
203      * <p>
204      * Returns the JDK version as a number that is easy to compare.
205      * </p>
206      * <p>
207      * For JDK "1.8" it will be 8; for JDK "11" it will be 11.
208      * </p>
209      *
210      * @return JDK version as integer
211      */
212     public static int getJdkVersion() {
213         String version = System.getProperty("java.specification.version");
214         if (version.startsWith("1.")) {
215             version = version.substring(2);
216         }
217         return Integer.parseInt(version);
218     }

220     /**
221      * <p>
222      * Adjusts the expected number of flushes for tests that call {@link OutputStream#close} method.
223      * </p>
224      * <p>
225      * After <a href="https://bugs.openjdk.java.net/browse/JDK-8220477">JDK-8220477</a>
226      * there is one additional flush from {@code sun.nio.cs.StreamEncoder#implClose}.
227      * </p>
228      *
229      * @param flushCount flush count to adjust
230      * @return adjusted flush count
231      */
232     public static int adjustFlushCountForOutputStreamClose(int flushCount) {
233         int result = flushCount;
234         if (getJdkVersion() >= 13) {
235             ++result;
236         }
237         return result;
238     }

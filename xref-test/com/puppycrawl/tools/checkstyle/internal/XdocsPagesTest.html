266         final Pattern endOfSentence = Pattern.compile("(.*?\\.)\\s", Pattern.DOTALL);
267         final Map<String, String> summaries = readSummaries();
268 
269         for (Path path : XdocUtil.getXdocsConfigFilePaths(XdocUtil.getXdocsFilePaths())) {

272                     || "config_filefilters.xml".equals(fileName)
273                     || "config_filters.xml".equals(fileName)) {
274                 continue;

278             final Document document = XmlUtil.getRawXml(fileName, input, input);
279             final NodeList sources = document.getElementsByTagName("subsection");
280 
281             for (int position = 0; position < sources.getLength(); position++) {
282                 final Node section = sources.item(position);
283                 final String sectionName = XmlUtil.getNameAttributeOfNode(section);
284                 if (!"Description".equals(sectionName)) {
285                     continue;
286                 }
287 
288                 final String checkName = XmlUtil.getNameAttributeOfNode(section.getParentNode());
289                 final Matcher matcher = endOfSentence.matcher(section.getTextContent());
290                 assertWithMessage(

292                         + checkName + " in the file \"" + fileName + "\" should end with a period")
293                     .that(matcher.find());

295                 assertWithMessage("The summary for check " + checkName

297                         + " should match the first sentence of the \"Description\" subsection"
298                         + " for this check in the file \"" + fileName + "\"")

301             }

398     private static String buildXml(String unserializedSource) throws IOException {
399         // not all examples come with the full xml structure
400         String code = unserializedSource
401             // don't corrupt our own cachefile
402             .replace("target/cachefile", "target/cachefile-test");
403 
404         if (!hasFileSetClass(code)) {
405             code = "<module name=\"TreeWalker\">\n" + code + "\n</module>";
406         }
407         if (!code.contains("name=\"Checker\"")) {
408             code = "<module name=\"Checker\">\n" + code + "\n</module>";
409         }
410         if (!code.startsWith("<?xml")) {
411             final String dtdPath = new File(
412                     "src/main/resources/com/puppycrawl/tools/checkstyle/configuration_1_3.dtd")
413                     .getCanonicalPath();
414 
415             code = "<?xml version=\"1.0\"?>\n<!DOCTYPE module PUBLIC "
416                     + "\"-//Checkstyle//DTD Checkstyle Configuration 1.3//EN\" \"" + dtdPath
417                     + "\">\n" + code;
418         }
419         return code;
420     }

435     private static boolean isValidCheckstyleXml(String fileName, String code,
436                                                 String unserializedSource)
437             throws IOException, CheckstyleException {
438         // can't process non-existent examples, or out of context snippets
439         if (!code.contains("com.mycompany") && !code.contains("checkstyle-packages")
440                 && !code.contains("MethodLimit") && !code.contains("<suppress ")
441                 && !code.contains("<suppress-xpath ")
442                 && !code.contains("<import-control ")
443                 && !unserializedSource.startsWith("<property ")
444                 && !unserializedSource.startsWith("<taskdef ")) {
445             // validate checkstyle structure and contents
446             try {
447                 final Properties properties = new Properties();
448 
449                 properties.setProperty("checkstyle.header.file",
450                         new File("config/java.header").getCanonicalPath());
451 
452                 final PropertiesExpander expander = new PropertiesExpander(properties);
453                 final Configuration config = ConfigurationLoader.loadConfiguration(new InputSource(
454                         new StringReader(code)), expander, IgnoredModulesOptions.EXECUTE);
455                 final Checker checker = new Checker();
456 
457                 try {
458                     final ClassLoader moduleClassLoader = Checker.class.getClassLoader();
459                     checker.setModuleClassLoader(moduleClassLoader);
460                     checker.configure(config);
461                 }
462                 finally {
463                     checker.destroy();
464                 }
465             }
466             catch (CheckstyleException ex) {
467                 throw new CheckstyleException(fileName + " has invalid Checkstyle xml ("
468                         + ex.getMessage() + "): " + unserializedSource, ex);
469             }
470         }
471         return true;
472     }

544     private static void validateCheckSection(ModuleFactory moduleFactory, String fileName,
545             String sectionName, Node section) throws Exception {
546         final Object instance;
547 
548         try {
549             instance = moduleFactory.createModule(sectionName);
550         }
551         catch (CheckstyleException ex) {
552             throw new CheckstyleException(fileName + " couldn't find class: " + sectionName, ex);
553         }
554 
555         int subSectionPos = 0;
556         for (Node subSection : XmlUtil.getChildrenElements(section)) {
557             if (subSectionPos == 0 && "p".equals(subSection.getNodeName())) {
558                 validateSinceDescriptionSection(fileName, sectionName, subSection);
559                 continue;
560             }
561 
562             final String subSectionName = XmlUtil.getNameAttributeOfNode(subSection);
563 
564             // can be in different orders, and completely optional
565             if ("Notes".equals(subSectionName)
566                     || "Rule Description".equals(subSectionName)
567                     || "Metadata".equals(subSectionName)) {
568                 continue;
569             }
570 
571             // optional sections that can be skipped if they have nothing to report
572             if (subSectionPos == 1 && !"Properties".equals(subSectionName)) {
573                 validatePropertySection(fileName, sectionName, null, instance);
574                 subSectionPos++;
575             }
576             if (subSectionPos == 4 && !"Violation Messages".equals(subSectionName)) {
577                 validateViolationSection(fileName, sectionName, null, instance);
578                 subSectionPos++;
579             }
580 
581             assertEquals(getSubSectionName(subSectionPos), subSectionName,
582                     fileName + " section '" + sectionName + "' should be in order");
583 
584             switch (subSectionPos) {
585                 case 0:
586                     validateDescriptionSection(fileName, sectionName, subSection);
587                     break;
588                 case 1:
589                     validatePropertySection(fileName, sectionName, subSection, instance);
590                     break;
591                 case 3:
592                     validateUsageExample(fileName, sectionName, subSection);
593                     break;
594                 case 4:
595                     validateViolationSection(fileName, sectionName, subSection, instance);
596                     break;
597                 case 5:
598                     validatePackageSection(fileName, sectionName, subSection, instance);
599                     break;
600                 case 6:
601                     validateParentSection(fileName, sectionName, subSection);
602                     break;
603                 case 2:
604                 default:
605                     break;
606             }
607 
608             subSectionPos++;
609         }
610 
611         if ("Checker".equals(sectionName)) {
612             assertTrue(subSectionPos >= 6, fileName + " section '" + sectionName
613                     + "' should contain up to 'Package' sub-section");
614         }
615         else {
616             assertTrue(subSectionPos >= 7, fileName + " section '" + sectionName
617                     + "' should contain up to 'Parent' sub-section");
618         }
619     }

720     private static void fixCapturedProperties(String sectionName, Object instance, Class<?> clss,
721             Set<String> properties) {
722         // remove global properties that don't need documentation
723         if (hasParentModule(sectionName)) {
724             if (AbstractJavadocCheck.class.isAssignableFrom(clss)) {
725                 properties.removeAll(JAVADOC_CHECK_PROPERTIES);
726 
727                 // override
728                 properties.add("violateExecutionOnNonTightHtml");
729             }
730             else if (AbstractCheck.class.isAssignableFrom(clss)) {
731                 properties.removeAll(CHECK_PROPERTIES);
732             }
733         }
734         if (AbstractFileSetCheck.class.isAssignableFrom(clss)) {
735             properties.removeAll(FILESET_PROPERTIES);
736 
737             // override
738             properties.add("fileExtensions");
739         }
740 
741         // remove undocumented properties
742         new HashSet<>(properties).stream()
743             .filter(prop -> UNDOCUMENTED_PROPERTIES.contains(clss.getSimpleName() + "." + prop))
744             .forEach(properties::remove);
745 
746         if (AbstractCheck.class.isAssignableFrom(clss)) {
747             final AbstractCheck check = (AbstractCheck) instance;
748 
749             final int[] acceptableTokens = check.getAcceptableTokens();
750             Arrays.sort(acceptableTokens);
751             final int[] defaultTokens = check.getDefaultTokens();
752             Arrays.sort(defaultTokens);
753             final int[] requiredTokens = check.getRequiredTokens();
754             Arrays.sort(requiredTokens);
755 
756             if (!Arrays.equals(acceptableTokens, defaultTokens)
757                     || !Arrays.equals(acceptableTokens, requiredTokens)) {
758                 properties.add("tokens");
759             }
760         }
761 
762         if (AbstractJavadocCheck.class.isAssignableFrom(clss)) {
763             final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
764 
765             final int[] acceptableJavadocTokens = check.getAcceptableJavadocTokens();
766             Arrays.sort(acceptableJavadocTokens);
767             final int[] defaultJavadocTokens = check.getDefaultJavadocTokens();
768             Arrays.sort(defaultJavadocTokens);
769             final int[] requiredJavadocTokens = check.getRequiredJavadocTokens();
770             Arrays.sort(requiredJavadocTokens);
771 
772             if (!Arrays.equals(acceptableJavadocTokens, defaultJavadocTokens)
773                     || !Arrays.equals(acceptableJavadocTokens, requiredJavadocTokens)) {
774                 properties.add("javadocTokens");
775             }
776         }

1624     private static String[] validateRuleNameOrder(String fileName, String lastRuleName,
1625                                                   String[] lastRuleNumberParts, String ruleName) {
1626         final String[] ruleNumberParts = ruleName.split(" ", 2)[0].split("\\.");
1627 
1628         if (lastRuleName != null) {
1629             final int ruleNumberPartsAmount = ruleNumberParts.length;
1630             final int lastRuleNumberPartsAmount = lastRuleNumberParts.length;
1631             final String outOfOrderReason = fileName + " rule '" + ruleName
1632                     + "' is out of order compared to '" + lastRuleName + "'";
1633             boolean lastRuleNumberPartWasEqual = false;
1634             int partIndex;
1635             for (partIndex = 0; partIndex < ruleNumberPartsAmount; partIndex++) {
1636                 if (lastRuleNumberPartsAmount <= partIndex) {
1637                     // equal up to here and last rule has less parts,
1638                     // thus order is correct, stop comparing
1639                     break;
1640                 }
1641 
1642                 final String ruleNumberPart = ruleNumberParts[partIndex];
1643                 final String lastRuleNumberPart = lastRuleNumberParts[partIndex];
1644                 final boolean ruleNumberPartsAreNumeric = IntStream.concat(
1645                         ruleNumberPart.chars(),
1646                         lastRuleNumberPart.chars()
1647                 ).allMatch(Character::isDigit);
1648 
1649                 if (ruleNumberPartsAreNumeric) {
1650                     final int numericRuleNumberPart = parseInt(ruleNumberPart);
1651                     final int numericLastRuleNumberPart = parseInt(lastRuleNumberPart);
1652                     assertThat(outOfOrderReason,
1653                             numericRuleNumberPart < numericLastRuleNumberPart,
1654                             describedAs("'%0' should not be less than '%1'",
1655                                     is(false),
1656                                     numericRuleNumberPart, numericLastRuleNumberPart));
1657                 }
1658                 else {
1659                     assertThat(outOfOrderReason,
1660                             ruleNumberPart.compareToIgnoreCase(lastRuleNumberPart) < 0,
1661                             describedAs("'%0' should not be less than '%1'",
1662                                     is(false),
1663                                     ruleNumberPart, lastRuleNumberPart));
1664                 }
1665                 lastRuleNumberPartWasEqual = ruleNumberPart.equalsIgnoreCase(lastRuleNumberPart);
1666                 if (!lastRuleNumberPartWasEqual) {
1667                     // number part is not equal but properly ordered,
1668                     // thus order is correct, stop comparing
1669                     break;
1670                 }
1671             }
1672             if (ruleNumberPartsAmount == partIndex && lastRuleNumberPartWasEqual) {
1673                 if (lastRuleNumberPartsAmount == partIndex) {
1674                     fail(fileName + " rule '" + ruleName + "' and rule '"
1675                             + lastRuleName + "' have the same rule number");
1676                 }
1677                 else {
1678                     fail(outOfOrderReason);
1679                 }
1680             }
1681         }
1682 
1683         return ruleNumberParts;
1684     }

75       * {@link InlineConfigParser#VIOLATION_WITH_EXPLANATION_PATTERN}.

346     /**
347      * Parses the input file provided.
348      *
349      * @param inputFilePath the input file path.
350      * @param setFilteredViolations flag to set filtered violations.
351      * @throws Exception if unable to read file or file not formatted properly.
352      */
353     private static TestInputConfiguration parse(String inputFilePath,
354                                                 boolean setFilteredViolations) throws Exception {
355         final TestInputConfiguration.Builder testInputConfigBuilder =
356                 new TestInputConfiguration.Builder();
357         final Path filePath = Paths.get(inputFilePath);
358         final List<String> lines = readFile(filePath);
359         try {
360             setModules(testInputConfigBuilder, inputFilePath, lines);
361         }
362         catch (Exception ex) {
363             throw new CheckstyleException("Config comment not specified properly in "
364                     + inputFilePath, ex);
365         }
366         try {
367             setViolations(testInputConfigBuilder, lines, setFilteredViolations);
368         }
369         catch (CheckstyleException ex) {
370             throw new CheckstyleException(ex.getMessage() + " in " + inputFilePath, ex);
371         }
372         return testInputConfigBuilder.build();
373     }

375     public static List<TestInputViolation> getViolationsFromInputFile(String inputFilePath)
376             throws Exception {
377         final TestInputConfiguration.Builder testInputConfigBuilder =
378                 new TestInputConfiguration.Builder();
379         final Path filePath = Paths.get(inputFilePath);
380         final List<String> lines = readFile(filePath);
381 
382         try {
383             for (int lineNo = 0; lineNo < lines.size(); lineNo++) {
384                 setViolations(testInputConfigBuilder, lines, false, lineNo, true);
385             }
386         }
387         catch (CheckstyleException ex) {
388             throw new CheckstyleException(ex.getMessage() + " in " + inputFilePath, ex);
389         }
390 
391         return testInputConfigBuilder.build().getViolations();
392     }

399     /**
400      * Parse the input file with configuration in xml header.
401      *
402      * @param inputFilePath the input file path.
403      * @throws Exception if unable to parse the xml header
404      */
405     public static TestInputConfiguration parseWithXmlHeader(String inputFilePath)
406             throws Exception {
407 
408         final Path filePath = Paths.get(inputFilePath);
409         final List<String> lines = readFile(filePath);
410         if (!checkIsXmlConfig(lines)) {
411             throw new CheckstyleException("Config cannot be parsed as xml.");
412         }
413 
414         final List<String> inlineConfig = getInlineConfig(lines);
415         final String stringXmlConfig = LATEST_DTD + String.join("", inlineConfig);
416         final InputSource inputSource = new InputSource(new StringReader(stringXmlConfig));
417         final Configuration xmlConfig = ConfigurationLoader.loadConfiguration(
418                 inputSource, new PropertiesExpander(System.getProperties()),
419                 ConfigurationLoader.IgnoredModulesOptions.EXECUTE
420         );
421         final String configName = xmlConfig.getName();
422         if (!"Checker".equals(configName)) {
423             throw new CheckstyleException(
424                     "First module should be Checker, but was " + configName);
425         }
426 
427         final TestInputConfiguration.Builder testInputConfigBuilder =
428                 new TestInputConfiguration.Builder();
429         testInputConfigBuilder.setXmlConfiguration(xmlConfig);
430         try {
431             setViolations(testInputConfigBuilder, lines, false);
432         }
433         catch (CheckstyleException ex) {
434             throw new CheckstyleException(ex.getMessage() + " in " + inputFilePath, ex);
435         }
436         return testInputConfigBuilder.buildWithXmlConfiguration();
437     }

439     /**
440      * Check whether a file provides xml configuration.
441      *
442      * @param lines lines of the file
443      * @return true if a file provides xml configuration, otherwise false.
444      */
445     private static boolean checkIsXmlConfig(List<String> lines) {
446         return "/*xml".equals(lines.get(0));
447     }
448 
449     private static void setModules(TestInputConfiguration.Builder testInputConfigBuilder,
450                                    String inputFilePath, List<String> lines)
451             throws Exception {
452         if (!lines.get(0).startsWith("/*")) {
453             throw new CheckstyleException("Config not specified on top."
454                 + "Please see other inputs for examples of what is required.");
455         }
456 
457         final List<String> inlineConfig = getInlineConfig(lines);
458 
459         if (checkIsXmlConfig(lines)) {
460             final String stringXmlConfig = LATEST_DTD + String.join("", inlineConfig);
461             final InputSource inputSource = new InputSource(new StringReader(stringXmlConfig));
462             final Configuration xmlConfig = ConfigurationLoader.loadConfiguration(
463                 inputSource, new PropertiesExpander(System.getProperties()),
464                     ConfigurationLoader.IgnoredModulesOptions.EXECUTE
465             );
466             final String configName = xmlConfig.getName();
467             if (!"Checker".equals(configName)) {
468                 throw new CheckstyleException(
469                         "First module should be Checker, but was " + configName);
470             }
471             handleXmlConfig(testInputConfigBuilder, inputFilePath, xmlConfig.getChildren());
472         }
473         else {
474             handleKeyValueConfig(testInputConfigBuilder, inputFilePath, inlineConfig);
475         }
476     }

478     private static List<String> getInlineConfig(List<String> lines) {
479         return lines.stream()
480                 .skip(1)
481                 .takeWhile(line -> !line.startsWith("*/"))
482                 .collect(Collectors.toUnmodifiableList());
483     }
484 
485     private static void handleXmlConfig(TestInputConfiguration.Builder testInputConfigBuilder,
486                                         String inputFilePath,
487                                         Configuration... modules)
488             throws CheckstyleException {
489 
490         for (Configuration module: modules) {
491             final String moduleName = module.getName();
492             if ("TreeWalker".equals(moduleName)) {
493                 handleXmlConfig(testInputConfigBuilder, inputFilePath, module.getChildren());
494             }
495             else {

499                 setProperties(inputFilePath, module, moduleInputConfigBuilder);
500                 testInputConfigBuilder.addChildModule(moduleInputConfigBuilder.build());
501             }
502         }

523     private static String getFullyQualifiedClassName(String filePath, String moduleName)
524             throws CheckstyleException {
525         // This is a hack until https://github.com/checkstyle/checkstyle/issues/13845
526         final Map<String, String> moduleMappings = new HashMap<>();
527         moduleMappings.put("ParameterNumber",
528                 "com.puppycrawl.tools.checkstyle.checks.sizes.ParameterNumberCheck");
529         moduleMappings.put("SuppressWarningsHolder",
530                 "com.puppycrawl.tools.checkstyle.checks.SuppressWarningsHolder");
531         moduleMappings.put("SuppressWarningsFilter",
532                 "com.puppycrawl.tools.checkstyle.filters.SuppressWarningsFilter");
533         moduleMappings.put("MemberName",
534                 "com.puppycrawl.tools.checkstyle.checks.naming.MemberNameCheck");
535         moduleMappings.put("ConstantName",
536                 "com.puppycrawl.tools.checkstyle.checks.naming.ConstantNameCheck");
537         moduleMappings.put("NoWhitespaceAfter",
538                 "com.puppycrawl.tools.checkstyle.checks.whitespace.NoWhitespaceAfterCheck");
539         moduleMappings.put("SummaryJavadoc",
540                 "com.puppycrawl.tools.checkstyle.checks.javadoc.SummaryJavadocCheck");
541         moduleMappings.put("LineLength",
542                 "com.puppycrawl.tools.checkstyle.checks.sizes.LineLengthCheck");
543         moduleMappings.put("ParameterName",
544                 "com.puppycrawl.tools.checkstyle.checks.naming.ParameterNameCheck");
545         moduleMappings.put("MethodName",
546                 "com.puppycrawl.tools.checkstyle.checks.naming.MethodNameCheck");
547         moduleMappings.put("SuppressionXpathSingleFilter",
548                 "com.puppycrawl.tools.checkstyle.filters.SuppressionXpathSingleFilter");
549 
550         String fullyQualifiedClassName;
551         if (moduleMappings.containsKey(moduleName)) {
552             fullyQualifiedClassName = moduleMappings.get(moduleName);
553         }
554         else if (moduleName.startsWith("com.")) {
555             fullyQualifiedClassName = moduleName;
556         }
557         else {
558             final String path = SLASH_PATTERN.matcher(filePath).replaceAll(".");
559             final int endIndex = path.lastIndexOf(moduleName.toLowerCase(Locale.ROOT));
560             if (endIndex == -1) {
561                 throw new CheckstyleException("Unable to resolve module name: " + moduleName
562                 + ". Please check for spelling errors or specify fully qualified class name.");
563             }
564             final int beginIndex = path.indexOf("com.puppycrawl");
565             fullyQualifiedClassName = path.substring(beginIndex, endIndex) + moduleName;
566             if (!fullyQualifiedClassName.endsWith("Filter")) {
567                 fullyQualifiedClassName += "Check";
568             }
569         }
570         return fullyQualifiedClassName;
571     }

592     private static String getResolvedPath(String fileValue, String inputFilePath) {
593         final String resolvedFilePath;
594         if (fileValue.startsWith("(resource)")) {
595             resolvedFilePath =
596                     getResourcePath(fileValue.substring(fileValue.indexOf(')') + 1),
597                             inputFilePath);
598         }
599         else if (fileValue.startsWith("(uri)")) {
600             resolvedFilePath =
601                     getUriPath(fileValue.substring(fileValue.indexOf(')') + 1), inputFilePath);
602         }
603         else {
604             resolvedFilePath = getFilePath(fileValue, inputFilePath);
605         }
606         return resolvedFilePath;
607     }

625     private static void setProperties(ModuleInputConfiguration.Builder inputConfigBuilder,
626                                       String inputFilePath,
627                                       List<String> lines,
628                                       int beginLineNo)
629                     throws IOException {
630         final StringBuilder stringBuilder = new StringBuilder(128);
631         int lineNo = beginLineNo;
632         for (String line = lines.get(lineNo); !line.isEmpty() && !"*/".equals(line);
633                 ++lineNo, line = lines.get(lineNo)) {
634             stringBuilder.append(line).append('\n');
635         }
636         final Properties properties = new Properties();
637         properties.load(new StringReader(stringBuilder.toString()));
638         for (final Map.Entry<Object, Object> entry : properties.entrySet()) {
639             final String key = entry.getKey().toString();
640             final String value = entry.getValue().toString();
641             if (key.startsWith("message.")) {
642                 inputConfigBuilder.addModuleMessage(key.substring(8), value);
643             }
644             else if (value.startsWith("(file)")) {
645                 final String fileName = value.substring(value.indexOf(')') + 1);
646                 final String filePath = getResolvedPath(fileName, inputFilePath);
647                 inputConfigBuilder.addNonDefaultProperty(key, filePath);
648             }
649             else if (value.startsWith("(default)")) {
650                 final String defaultValue = value.substring(value.indexOf(')') + 1);
651                 if (NULL_STRING.equals(defaultValue)) {
652                     inputConfigBuilder.addDefaultProperty(key, null);
653                 }
654                 else {
655                     inputConfigBuilder.addDefaultProperty(key, defaultValue);
656                 }
657             }
658             else {
659                 if (NULL_STRING.equals(value)) {
660                     inputConfigBuilder.addNonDefaultProperty(key, null);
661                 }
662                 else {
663                     inputConfigBuilder.addNonDefaultProperty(key, value);
664                 }
665             }
666         }
667     }

669     private static void setProperties(String inputFilePath, Configuration module,
670                                       ModuleInputConfiguration.Builder moduleInputConfigBuilder)
671             throws CheckstyleException {
672         final String[] getPropertyNames = module.getPropertyNames();
673         for (final String propertyName : getPropertyNames) {
674             final String propertyValue = module.getProperty(propertyName);
675 
676             if ("file".equals(propertyName)) {
677                 final String filePath = getResolvedPath(propertyValue, inputFilePath);
678                 moduleInputConfigBuilder.addNonDefaultProperty(propertyName, filePath);
679             }
680             else {
681                 if (NULL_STRING.equals(propertyValue)) {
682                     moduleInputConfigBuilder.addNonDefaultProperty(propertyName, null);
683                 }
684                 else {
685                     moduleInputConfigBuilder.addNonDefaultProperty(propertyName, propertyValue);
686                 }
687             }
688         }
689 
690         final Map<String, String> messages = module.getMessages();
691         for (final Map.Entry<String, String> entry : messages.entrySet()) {
692             final String key = entry.getKey();
693             final String value = entry.getValue();
694             moduleInputConfigBuilder.addModuleMessage(key, value);
695         }
696     }

699                                       List<String> lines, boolean useFilteredViolations)
700             throws CheckstyleException {

711     /**
712      * Sets the violations.
713      *
714      * @param inputConfigBuilder the input file path.
715      * @param lines all the lines in the file.
716      * @param useFilteredViolations flag to set filtered violations.
717      * @param lineNo current line.
718      * @noinspection IfStatementWithTooManyBranches
719      * @noinspectionreason IfStatementWithTooManyBranches - complex logic of violation
720      *      parser requires giant if/else
721      * @throws CheckstyleException if violation message is not specified
722      */
723     // -@cs[ExecutableStatementCount] splitting this method is not reasonable.
724     // -@cs[JavaNCSS] splitting this method is not reasonable.
725     // -@cs[CyclomaticComplexity] splitting this method is not reasonable.
726     private static void setViolations(TestInputConfiguration.Builder inputConfigBuilder,
727                                       List<String> lines, boolean useFilteredViolations,
728                                       int lineNo, boolean specifyViolationMessage)
729             throws CheckstyleException {
730         final Matcher violationMatcher =
731                 VIOLATION_PATTERN.matcher(lines.get(lineNo));
732         final Matcher violationAboveMatcher =
733                 VIOLATION_ABOVE_PATTERN.matcher(lines.get(lineNo));
734         final Matcher violationBelowMatcher =
735                 VIOLATION_BELOW_PATTERN.matcher(lines.get(lineNo));
736         final Matcher violationAboveWithExplanationMatcher =
737                 VIOLATION_ABOVE_WITH_EXPLANATION_PATTERN.matcher(lines.get(lineNo));
738         final Matcher violationBelowWithExplanationMatcher =
739                 VIOLATION_BELOW_WITH_EXPLANATION_PATTERN.matcher(lines.get(lineNo));
740         final Matcher violationWithExplanationMatcher =
741                 VIOLATION_WITH_EXPLANATION_PATTERN.matcher(lines.get(lineNo));
742         final Matcher multipleViolationsMatcher =
743                 MULTIPLE_VIOLATIONS_PATTERN.matcher(lines.get(lineNo));
744         final Matcher multipleViolationsAboveMatcher =
745                 MULTIPLE_VIOLATIONS_ABOVE_PATTERN.matcher(lines.get(lineNo));
746         final Matcher multipleViolationsBelowMatcher =
747                 MULTIPLE_VIOLATIONS_BELOW_PATTERN.matcher(lines.get(lineNo));
748         final Matcher violationSomeLinesAboveMatcher =
749                 VIOLATION_SOME_LINES_ABOVE_PATTERN.matcher(lines.get(lineNo));
750         final Matcher violationSomeLinesBelowMatcher =
751                 VIOLATION_SOME_LINES_BELOW_PATTERN.matcher(lines.get(lineNo));
752         final Matcher violationsAboveMatcherWithMessages =
753                 VIOLATIONS_ABOVE_PATTERN_WITH_MESSAGES.matcher(lines.get(lineNo));
754         final Matcher violationsSomeLinesAboveMatcher =
755                 VIOLATIONS_SOME_LINES_ABOVE_PATTERN.matcher(lines.get(lineNo));
756         final Matcher violationsSomeLinesBelowMatcher =
757                 VIOLATIONS_SOME_LINES_BELOW_PATTERN.matcher(lines.get(lineNo));
758         final Matcher violationsDefault =
759                 VIOLATION_DEFAULT.matcher(lines.get(lineNo));
760         if (violationMatcher.matches()) {
761             final String violationMessage = violationMatcher.group(1);
762             final int violationLineNum = lineNo + 1;
763             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
764                     violationLineNum);
765             inputConfigBuilder.addViolation(violationLineNum, violationMessage);
766         }
767         else if (violationAboveMatcher.matches()) {
768             final String violationMessage = violationAboveMatcher.group(1);
769             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage, lineNo);
770             inputConfigBuilder.addViolation(lineNo, violationMessage);
771         }
772         else if (violationBelowMatcher.matches()) {
773             final String violationMessage = violationBelowMatcher.group(1);
774             final int violationLineNum = lineNo + 2;
775             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
776                     violationLineNum);
777             inputConfigBuilder.addViolation(violationLineNum, violationMessage);
778         }
779         else if (violationAboveWithExplanationMatcher.matches()) {
780             final String violationMessage = violationAboveWithExplanationMatcher.group(1);
781             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage, lineNo);
782             inputConfigBuilder.addViolation(lineNo, violationMessage);
783         }
784         else if (violationBelowWithExplanationMatcher.matches()) {
785             final String violationMessage = violationBelowWithExplanationMatcher.group(1);
786             final int violationLineNum = lineNo + 2;
787             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
788                     violationLineNum);
789             inputConfigBuilder.addViolation(violationLineNum, violationMessage);
790         }
791         else if (violationWithExplanationMatcher.matches()) {
792             final int violationLineNum = lineNo + 1;
793             inputConfigBuilder.addViolation(violationLineNum, null);
794         }
795         else if (violationSomeLinesAboveMatcher.matches()) {
796             final String violationMessage = violationSomeLinesAboveMatcher.group(2);
797             final int linesAbove = Integer.parseInt(violationSomeLinesAboveMatcher.group(1)) - 1;
798             final int violationLineNum = lineNo - linesAbove;
799             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
800                     violationLineNum);
801             inputConfigBuilder.addViolation(violationLineNum, violationMessage);
802         }
803         else if (violationSomeLinesBelowMatcher.matches()) {
804             final String violationMessage = violationSomeLinesBelowMatcher.group(2);
805             final int linesBelow = Integer.parseInt(violationSomeLinesBelowMatcher.group(1)) + 1;
806             final int violationLineNum = lineNo + linesBelow;
807             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
808                     violationLineNum);
809             inputConfigBuilder.addViolation(violationLineNum, violationMessage);
810         }
811         else if (violationsAboveMatcherWithMessages.matches()) {
812             inputConfigBuilder.addViolations(
813                 getExpectedViolationsForSpecificLine(
814                     lines, lineNo, lineNo, violationsAboveMatcherWithMessages));
815         }
816         else if (violationsSomeLinesAboveMatcher.matches()) {
817             inputConfigBuilder.addViolations(
818                 getExpectedViolations(
819                     lines, lineNo, violationsSomeLinesAboveMatcher, true));
820         }
821         else if (violationsSomeLinesBelowMatcher.matches()) {
822             inputConfigBuilder.addViolations(
823                     getExpectedViolations(
824                             lines, lineNo, violationsSomeLinesBelowMatcher, false));
825         }
826         else if (multipleViolationsMatcher.matches()) {
827             Collections
828                     .nCopies(Integer.parseInt(multipleViolationsMatcher.group(1)), lineNo + 1)
829                     .forEach(actualLineNumber -> {
830                         inputConfigBuilder.addViolation(actualLineNumber, null);
831                     });
832         }
833         else if (multipleViolationsAboveMatcher.matches()) {
834             Collections
835                     .nCopies(Integer.parseInt(multipleViolationsAboveMatcher.group(1)), lineNo)
836                     .forEach(actualLineNumber -> {
837                         inputConfigBuilder.addViolation(actualLineNumber, null);
838                     });
839         }
840         else if (multipleViolationsBelowMatcher.matches()) {
841             Collections
842                     .nCopies(Integer.parseInt(multipleViolationsBelowMatcher.group(1)),
843                             lineNo + 2)
844                     .forEach(actualLineNumber -> {
845                         inputConfigBuilder.addViolation(actualLineNumber, null);
846                     });
847         }
848         else if (useFilteredViolations) {
849             setFilteredViolation(inputConfigBuilder, lineNo + 1,
850                     lines.get(lineNo), specifyViolationMessage);
851         }
852         else if (violationsDefault.matches()) {
853             final int violationLineNum = lineNo + 1;
854             inputConfigBuilder.addViolation(violationLineNum, null);
855         }
856     }

858     private static List<TestInputViolation> getExpectedViolationsForSpecificLine(
859                                               List<String> lines, int lineNo, int violationLineNum,
860                                               Matcher matcher) {
861         final List<TestInputViolation> results = new ArrayList<>();
862 
863         final int expectedMessageCount =
864             Integer.parseInt(matcher.group(1));
865         for (int index = 1; index <= expectedMessageCount; index++) {
866             final String lineWithMessage = lines.get(lineNo + index);
867             final Matcher messageMatcher = VIOLATION_MESSAGE_PATTERN.matcher(lineWithMessage);
868             if (messageMatcher.find()) {
869                 final String violationMessage = messageMatcher.group(1);
870                 results.add(new TestInputViolation(violationLineNum, violationMessage));
871             }
872         }
873         if (results.size() != expectedMessageCount) {
874             final String message = String.format(Locale.ROOT,
875                 "Declared amount of violation messages at line %s is %s but found %s",
876                 lineNo + 1, expectedMessageCount, results.size());
877             throw new IllegalStateException(message);
878         }
879         return results;
880     }

882     private static List<TestInputViolation> getExpectedViolations(
883                                               List<String> lines, int lineNo,
884                                               Matcher matcher, boolean isAbove) {
885         final int violationLine =
886             Integer.parseInt(matcher.group(2));
887         final int violationLineNum;
888         if (isAbove) {
889             violationLineNum = lineNo - violationLine + 1;
890         }
891         else {
892             violationLineNum = lineNo + violationLine + 1;
893         }
894         return getExpectedViolationsForSpecificLine(lines,
895             lineNo, violationLineNum, matcher);
896     }

898     private static void setFilteredViolation(TestInputConfiguration.Builder inputConfigBuilder,
899                                              int lineNo, String line,
900                                              boolean specifyViolationMessage)
901             throws CheckstyleException {
902         final Matcher violationMatcher =
903                 FILTERED_VIOLATION_PATTERN.matcher(line);
904         final Matcher violationAboveMatcher =
905                 FILTERED_VIOLATION_ABOVE_PATTERN.matcher(line);
906         final Matcher violationBelowMatcher =
907                 FILTERED_VIOLATION_BELOW_PATTERN.matcher(line);
908         if (violationMatcher.matches()) {
909             final String violationMessage = violationMatcher.group(1);
910             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage, lineNo);
911             inputConfigBuilder.addFilteredViolation(lineNo, violationMessage);
912         }
913         else if (violationAboveMatcher.matches()) {
914             final String violationMessage = violationAboveMatcher.group(1);
915             final int violationLineNum = lineNo - 1;
916             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
917                     violationLineNum);
918             inputConfigBuilder.addFilteredViolation(violationLineNum, violationMessage);
919         }
920         else if (violationBelowMatcher.matches()) {
921             final String violationMessage = violationBelowMatcher.group(1);
922             final int violationLineNum = lineNo + 1;
923             checkWhetherViolationSpecified(specifyViolationMessage, violationMessage,
924                     violationLineNum);
925             inputConfigBuilder.addFilteredViolation(violationLineNum, violationMessage);
926         }
927     }

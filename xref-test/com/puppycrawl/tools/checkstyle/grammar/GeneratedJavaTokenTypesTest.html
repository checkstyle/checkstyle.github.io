47       * <p>
48       * New tokens must be added onto the end of the list with new numbers, and
49       * old tokens must remain and keep their current numbering. Old token
50       * numberings are not allowed to change.
51       * </p>

55       * into the compiled Java code. This loses all connections with the original

58       * </p>

731     /**
732      * This test was created to make sure that new tokens are added to the 'tokens'
733      * block in the lexer grammar. If a new token is not added at the end of the list,
734      * it will become "mixed in" with the unused tokens and cause
735      * Collections#lastIndexOfSubList to return a -1 and fail the test.
736      */
737     @Test
738     public void testTokenHasBeenAddedToTokensBlockInLexerGrammar() {
739         final VocabularyImpl vocabulary = (VocabularyImpl) JavaLanguageLexer.VOCABULARY;
740         final String[] nullableSymbolicNames = vocabulary.getSymbolicNames();
741         final List<String> allTokenNames = Arrays.stream(nullableSymbolicNames)
742                 .filter(Objects::nonNull)
743                 .collect(Collectors.toList());
744 
745         // Since the following tokens are not declared in the 'tokens' block,
746         // they will always appear last in the list of symbolic names provided
747         // by the vocabulary.
748         final List<String> unusedTokenNames = List.of(
749                 // reserved keywords that are not part of the language
750                 "LITERAL_CONST", "LITERAL_GOTO",
751 
752                 // Lexer tokens that are not part of our API (they are used as components of
753                 // parser rules, but the token name is changed).
754                 "DECIMAL_LITERAL_LONG", "DECIMAL_LITERAL", "HEX_LITERAL_LONG",
755                 "HEX_LITERAL", "OCT_LITERAL_LONG", "OCT_LITERAL", "BINARY_LITERAL_LONG",
756                 "BINARY_LITERAL"
757         );
758 
759         // Get the starting index of the sublist of tokens, or -1 if sublist
760         // is not present.
761         final int lastIndexOfSublist =
762                 Collections.lastIndexOfSubList(allTokenNames, unusedTokenNames);
763         final int expectedNumberOfUsedTokens = allTokenNames.size() - unusedTokenNames.size();
764         final String message = "New tokens must be added to the 'tokens' block in the"
765                 + " lexer grammar.";
766 
767         assertWithMessage(message)
768                 .that(expectedNumberOfUsedTokens)
769                 .isEqualTo(lastIndexOfSublist);
770     }

772     /**
773      * Checks that a given field is 'public static final int'.
774      *
775      * @param field field to verify type and visibility of
776      * @return true if field is declared as 'public static final int'
777      */
778     private static boolean isPublicStaticFinalInt(Field field) {
779         final Class<?> fieldType = field.getType();
780         final int mods = field.getModifiers();
781         return fieldType.equals(Integer.TYPE)
782                 && Modifier.isPublic(mods)
783                 && Modifier.isStatic(mods)
784                 && Modifier.isFinal(mods);
785     }

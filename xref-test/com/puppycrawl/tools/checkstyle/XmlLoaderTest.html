31  import org.junit.Test;

34  import org.xml.sax.XMLReader;

36  import com.sun.org.apache.xerces.internal.impl.Constants;
37  
38  public class XmlLoaderTest {
39  
40      private static final String NAMESPACES_FEATURE =
41              Constants.SAX_FEATURE_PREFIX + Constants.NAMESPACES_FEATURE;
42  
43      @Test
44      public void testParserConfiguredSuccessfully() throws Exception {
45          final DummyLoader dummyLoader = new DummyLoader(new HashMap<>(1));
46          final XMLReader parser = Whitebox.getInternalState(dummyLoader, "parser");
47          assertTrue("Invalid feature state", parser.getFeature(NAMESPACES_FEATURE));
48          assertEquals("Invalid entity resolver", dummyLoader, parser.getEntityResolver());
49      }
50  
51      @Test
52      public void testIsProperUtilsClass() throws ReflectiveOperationException {
53          assertTrue("Constructor is not private", isUtilsClassHasPrivateConstructor(
54                  XmlLoader.FeaturesForVerySecureJavaInstallations.class, true));
55      }
56  
57      private static final class DummyLoader extends XmlLoader {
58  
59          DummyLoader(Map<String, String> publicIdToResourceNameMap)
60                  throws SAXException, ParserConfigurationException {
61              super(publicIdToResourceNameMap);
62          }
63  
64      }

66  }

88      /**
89       * Creates {@link Checker} instance based on the given {@link Configuration} instance.
90       *
91       * @param moduleConfig {@link Configuration} instance.
92       * @return {@link Checker} instance based on the given {@link Configuration} instance.
93       * @throws Exception if an exception occurs during checker configuration.
94       */
95      protected final Checker createChecker(Configuration moduleConfig)
96              throws Exception {
97          final String moduleName = moduleConfig.getName();
98          final Checker checker = new Checker();
99          checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
100 
101         if (ROOT_MODULE_NAME.equals(moduleName)) {
102             checker.configure(moduleConfig);
103         }
104         else {
105             configureChecker(checker, moduleConfig);
106         }
107 
108         checker.addListener(getBriefUtLogger());
109         return checker;
110     }

112     /**
113      * Configures the {@code checker} instance with {@code moduleConfig}.
114      *
115      * @param checker {@link Checker} instance.
116      * @param moduleConfig {@link Configuration} instance.
117      * @throws Exception if an exception occurs during configuration.
118      */
119     protected void configureChecker(Checker checker, Configuration moduleConfig) throws Exception {
120         final Class<?> moduleClass = Class.forName(moduleConfig.getName());
121 
122         if (ModuleReflectionUtil.isCheckstyleTreeWalkerCheck(moduleClass)
123                 || ModuleReflectionUtil.isTreeWalkerFilterModule(moduleClass)) {
124             final Configuration config = createTreeWalkerConfig(moduleConfig);
125             checker.configure(config);
126         }
127         else {
128             final Configuration config = createRootConfig(moduleConfig);
129             checker.configure(config);
130         }
131     }

134      * Creates {@link DefaultConfiguration} for the {@link TreeWalker}
135      * based on the given {@link Configuration} instance.
136      *
137      * @param config {@link Configuration} instance.
138      * @return {@link DefaultConfiguration} for the {@link TreeWalker}
139      *     based on the given {@link Configuration} instance.
140      */
141     protected static DefaultConfiguration createTreeWalkerConfig(Configuration config) {
142         final DefaultConfiguration rootConfig =
143                 new DefaultConfiguration(ROOT_MODULE_NAME);
144         final DefaultConfiguration twConf = createModuleConfig(TreeWalker.class);
145         // make sure that the tests always run with this charset
146         rootConfig.addProperty("charset", StandardCharsets.UTF_8.name());
147         rootConfig.addChild(twConf);
148         twConf.addChild(config);
149         return rootConfig;
150     }

191     /**
192      * Performs verification of the file with the given file path using specified configuration
193      * and the array of expected messages. Also performs verification of the config with filters
194      * specified in the input file.
195      *
196      * @param filePath file path to verify.
197      * @param expectedUnfiltered an array of expected unfiltered config.
198      * @param expectedFiltered an array of expected config with filters.
199      * @throws Exception if exception occurs during verification process.
200      */
201     protected final void verifyFilterWithInlineConfigParser(String filePath,
202                                                             String[] expectedUnfiltered,
203                                                             String... expectedFiltered)
204             throws Exception {
205         final TestInputConfiguration testInputConfiguration =
206                 InlineConfigParser.parseWithFilteredViolations(filePath);
207         final DefaultConfiguration configWithoutFilters =
208                 testInputConfiguration.createConfigurationWithoutFilters();
209         final List<TestInputViolation> violationsWithoutFilters =
210                 new ArrayList<>(testInputConfiguration.getViolations());
211         violationsWithoutFilters.addAll(testInputConfiguration.getFilteredViolations());
212         Collections.sort(violationsWithoutFilters);
213         verifyViolations(configWithoutFilters, filePath, violationsWithoutFilters);
214         verify(configWithoutFilters, filePath, expectedUnfiltered);
215         final DefaultConfiguration configWithFilters =
216                 testInputConfiguration.createConfiguration();
217         verifyViolations(configWithFilters, filePath, testInputConfiguration.getViolations());
218         verify(configWithFilters, filePath, expectedFiltered);
219     }

323     /**
324      *  Performs verification of the given files against the array of
325      *  expected messages using the provided {@link Checker} instance.
326      *
327      *  @param checker {@link Checker} instance.
328      *  @param processedFiles list of files to verify.
329      *  @param messageFileName message file name.
330      *  @param expected an array of expected messages.
331      *  @throws Exception if exception occurs during verification process.
332      */
333     protected void verify(Checker checker,
334                           File[] processedFiles,
335                           String messageFileName,
336                           String... expected)
337             throws Exception {
338         final Map<String, List<String>> expectedViolations = new HashMap<>();
339         expectedViolations.put(messageFileName, Arrays.asList(expected));
340         verify(checker, processedFiles, expectedViolations);
341     }

343     /**
344      * Performs verification of the given files.
345      *
346      * @param checker {@link Checker} instance
347      * @param processedFiles files to process.
348      * @param expectedViolations a map of expected violations per files.
349      * @throws Exception if exception occurs during verification process.
350      */
351     protected final void verify(Checker checker,
352                           File[] processedFiles,
353                           Map<String, List<String>> expectedViolations)
354             throws Exception {
355         stream.flush();
356         stream.reset();
357         final List<File> theFiles = new ArrayList<>();
358         Collections.addAll(theFiles, processedFiles);
359         final int errs = checker.process(theFiles);
360 
361         // process each of the lines
362         final Map<String, List<String>> actualViolations = getActualViolations(errs);
363         final Map<String, List<String>> realExpectedViolations =
364                 Maps.filterValues(expectedViolations, input -> !input.isEmpty());
365 
366         assertWithMessage("Files with expected violations and actual violations differ.")
367             .that(actualViolations.keySet())
368             .isEqualTo(realExpectedViolations.keySet());
369 
370         realExpectedViolations.forEach((fileName, violationList) -> {
371             assertWithMessage("Violations for %s differ.", fileName)
372                 .that(actualViolations.get(fileName))
373                 .containsExactlyElementsIn(violationList);
374         });
375 
376         checker.destroy();
377     }

380      * Runs 'verifyWithInlineConfigParser' with limited stack size and time duration.
381      *
382      * @param fileName file name to verify.
383      * @param expected an array of expected messages.
384      * @throws Exception if exception occurs during verification process.
385      */
386     protected final void verifyWithLimitedResources(String fileName, String... expected)

388         // We return null here, which gives us a result to make an assertion about

398     /**
399      * Executes given config on a list of files only. Does not verify violations.
400      *
401      * @param config check configuration
402      * @param filenames names of files to process
403      * @throws Exception if there is a problem during checker configuration
404      */
405     protected final void execute(Configuration config, String... filenames) throws Exception {
406         final Checker checker = createChecker(config);
407         final List<File> files = Arrays.stream(filenames)
408                 .map(File::new)
409                 .collect(Collectors.toList());
410         checker.process(files);
411         checker.destroy();
412     }

414     /**
415      * Performs verification of violation lines.
416      *
417      * @param config parsed config.
418      * @param file file path.
419      * @param testInputViolations List of TestInputViolation objects.
420      * @throws Exception if exception occurs during verification process.
421      */
422     private void verifyViolations(Configuration config,
423                                   String file,
424                                   List<TestInputViolation> testInputViolations)
425             throws Exception {
426         final List<String> actualViolations = getActualViolationsForFile(config, file);
427         final List<Integer> actualViolationLines = actualViolations.stream()
428                 .map(violation -> violation.substring(0, violation.indexOf(':')))
429                 .map(Integer::valueOf)
430                 .collect(Collectors.toList());
431         final List<Integer> expectedViolationLines = testInputViolations.stream()
432                 .map(TestInputViolation::getLineNo)
433                 .collect(Collectors.toList());
434         assertWithMessage("Violation lines for %s differ.", file)
435                 .that(actualViolationLines)
436                 .isEqualTo(expectedViolationLines);
437         for (int index = 0; index < actualViolations.size(); index++) {
438             assertWithMessage("Actual and expected violations differ.")
439                     .that(actualViolations.get(index))
440                     .matches(testInputViolations.get(index).toRegex());
441         }
442     }

444     /**
445      * Tests the file with the check config.
446      *
447      * @param config check configuration.
448      * @param file input file path.
449      * @return list of actual violations.
450      * @throws Exception if exception occurs during verification process.
451      */
452     private List<String> getActualViolationsForFile(Configuration config,
453                                                     String file) throws Exception {
454         stream.flush();
455         stream.reset();
456         final List<File> files = Collections.singletonList(new File(file));
457         final Checker checker = createChecker(config);
458         final Map<String, List<String>> actualViolations =
459                 getActualViolations(checker.process(files));
460         checker.destroy();
461         return actualViolations.getOrDefault(file, new ArrayList<>());
462     }

464     /**
465      * Returns the actual violations for each file that has been checked against {@link Checker}.
466      * Each file is mapped to their corresponding violation messages. Reads input stream for these
467      * messages using instance of {@link InputStreamReader}.
468      *
469      * @param errorCount count of errors after checking set of files against {@link Checker}.
470      * @return a {@link Map} object containing file names and the corresponding violation messages.
471      * @throws IOException exception can occur when reading input stream.
472      */
473     private Map<String, List<String>> getActualViolations(int errorCount) throws IOException {
474         // process each of the lines
475         try (ByteArrayInputStream inputStream =
476                 new ByteArrayInputStream(stream.toByteArray());
477             LineNumberReader lnr = new LineNumberReader(
478                 new InputStreamReader(inputStream, StandardCharsets.UTF_8))) {
479             final Map<String, List<String>> actualViolations = new HashMap<>();
480             for (String line = lnr.readLine(); line != null && lnr.getLineNumber() <= errorCount;
481                  line = lnr.readLine()) {
482                 // have at least 2 characters before the splitting colon,
483                 // to not split after the drive letter on Windows
484                 final String[] actualViolation = line.split("(?<=.{2}):", 2);
485                 final String actualViolationFileName = actualViolation[0];
486                 final String actualViolationMessage = actualViolation[1];
487 
488                 actualViolations
489                         .computeIfAbsent(actualViolationFileName, key -> new ArrayList<>())
490                         .add(actualViolationMessage);
491             }
492 
493             return actualViolations;
494         }
495     }

523     /**
524      * Gets the check message 'as is' from appropriate 'messages.properties'
525      * file.
526      *
527      * @param messageBundle the bundle name.
528      * @param messageKey the key of message in 'messages.properties' file.
529      * @param arguments the arguments of message in 'messages.properties' file.
530      * @return The message of the check with the arguments applied.
531      */
532     private static String internalGetCheckMessage(
533             String messageBundle, String messageKey, Object... arguments) {
534         final ResourceBundle resourceBundle = ResourceBundle.getBundle(
535                 messageBundle,
536                 Locale.getDefault(),
537                 Thread.currentThread().getContextClassLoader(),
538                 new Utf8Control());
539         final String pattern = resourceBundle.getString(messageKey);
540         final MessageFormat formatter = new MessageFormat(pattern, Locale.ROOT);
541         return formatter.format(arguments);
542     }

554     /**
555      * Returns message bundles for a class by providing class name.
556      *
557      * @param className name of the class.
558      * @return message bundles containing package name.
559      */
560     private static String getMessageBundle(String className) {
561         final String messageBundle;
562         final String messages = "messages";
563         final int endIndex = className.lastIndexOf('.');
564         if (endIndex < 0) {
565             messageBundle = messages;
566         }
567         else {
568             final String packageName = className.substring(0, endIndex);
569             messageBundle = packageName + "." + messages;
570         }
571         return messageBundle;
572     }

574     /**
575      * Remove suppressed violation messages from actual violation messages.
576      *
577      * @param actualViolations actual violation messages
578      * @param suppressedViolations suppressed violation messages
579      * @return an array of actual violation messages minus suppressed violation messages
580      */
581     protected static String[] removeSuppressed(String[] actualViolations,
582                                                String... suppressedViolations) {
583         final List<String> actualViolationsList =
584             Arrays.stream(actualViolations).collect(Collectors.toCollection(ArrayList::new));
585         actualViolationsList.removeAll(Arrays.asList(suppressedViolations));
586         return actualViolationsList.toArray(CommonUtil.EMPTY_STRING_ARRAY);
587     }

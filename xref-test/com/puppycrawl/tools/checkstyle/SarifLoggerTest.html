59      /**
60       * Executes the common logging steps for SARIF logging tests.
61       * This method mimic of natural execution at is done by Checker.
62       *
63       * <p>
64       * It starts file auditing, adds an error, and completes the audit process.
65       * </p>
66       *
67       * @param instance  the current instance of {@code SarifLoggerTest}, used for context
68       * @param logger    the {@code SarifLogger} instance to log errors
69       * @param fileName  the name of the file being audited
70       * @param violation the {@code Violation} object containing error details
71       */
72      public final void executeLogger(
73          SarifLoggerTest instance, SarifLogger logger, String fileName, Violation violation) {
74          final AuditEvent event = new AuditEvent(this, "Test.java", violation);
75          logger.fileStarted(event);
76          logger.addError(event);
77          logger.fileFinished(event);
78          logger.auditFinished(null);
79      }

81      @Test
82      public void testEscape() {
83          final String[][] encodings = {
84              {"\"", "\\\""},
85              {"\\", "\\\\"},
86              {"\b", "\\b"},
87              {"\f", "\\f"},
88              {"\n", "\\n"},
89              {"\r", "\\r"},
90              {"\t", "\\t"},
91              {"/", "\\/"},
92              {"\u0010", "\\u0010"},
93              {"\u001E", "\\u001E"},
94              {"\u001F", "\\u001F"},
95              {" ", " "},
96              {"bar1234", "bar1234"},
97          };
98          for (String[] encoding : encodings) {
99              final String encoded = SarifLogger.escape(encoding[0]);
100             assertWithMessage("\"" + encoding[0] + "\"")
101                 .that(encoded)
102                 .isEqualTo(encoding[1]);
103         }
104     }

460     /**
461      * Tests {@code ClassNotFoundException} handling in {@code getMessages} method.
462      * Uses reflection to inject fake module metadata with non-existent class
463      * to trigger the exception. Real checkstyle modules cannot be used as
464      * all are on the classpath during testing.
465      */
466     @Test
467     public void testGetMessagesWithClassNotFoundException() throws Exception {
468         final SarifLogger logger = new SarifLogger(outStream, OutputStreamOptions.CLOSE);
469 
470         final ModuleDetails fakeModule = new ModuleDetails();
471         fakeModule.setName("FakeCheck");
472         fakeModule.setFullQualifiedName("com.fake.NonExistentCheck");
473         fakeModule.setDescription("Fake check for testing");
474         fakeModule.addToViolationMessages("fake.message.key");
475 
476         @SuppressWarnings("unchecked")
477         final Map<Object, ModuleDetails> ruleMetadata =
478                 TestUtil.getInternalState(logger, "ruleMetadata", Map.class);
479         final Class<?> ruleKeyClass = TestUtil.getInnerClassType(SarifLogger.class, "RuleKey");
480         final Constructor<?> constructor =
481                 ruleKeyClass.getDeclaredConstructor(String.class, String.class);
482         constructor.setAccessible(true);
483         final Object ruleKey = constructor.newInstance("com.fake.NonExistentCheck", null);
484         ruleMetadata.put(ruleKey, fakeModule);
485 
486         logger.auditFinished(null);
487 
488         verifyContent(getPath("ExpectedSarifLoggerClassNotFoundException.sarif"), outStream);
489     }

523     private static void verifyContent(
524             String expectedOutputFile,
525             ByteArrayOutputStream actualOutputStream) throws IOException {
526         final String expectedContent = readFile(expectedOutputFile);
527         final String actualContent =
528                 toLfLineEnding(actualOutputStream.toString(StandardCharsets.UTF_8));
529         assertWithMessage("sarif content should match")
530             .that(actualContent)
531             .isEqualTo(expectedContent);
532     }

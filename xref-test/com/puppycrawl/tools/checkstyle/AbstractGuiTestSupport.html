20  package com.puppycrawl.tools.checkstyle;
21  
22  import static org.junit.jupiter.api.Assumptions.assumeFalse;
23  
24  import java.awt.Component;
25  import java.awt.Container;
26  import java.awt.GraphicsEnvironment;
27  
28  import org.junit.jupiter.api.BeforeEach;
29  
30  /**
31   * Abstract base class for testing GUI components.
32   */
33  public abstract class AbstractGuiTestSupport extends AbstractPathTestSupport {
34  
35      /**
36       * Validates the graphics environment.
37       */
38      @BeforeEach
39      public void validateGraphicsEnvironment() {
40          final boolean isHeadless = GraphicsEnvironment.isHeadless();
41          assumeFalse(isHeadless, "This test is incompatible with headless environment");
42      }
43  
44      /**
45       * Helper method to find a component by its name.
46       *
47       * @param root the root component to start search
48       * @param name the name of component to find
49       * @param <T> the type of component to find
50       * @return the component if found, {@code null} otherwise
51       * @noinspection unchecked
52       * @noinspectionreason unchecked - we know that any component is OK to typecast to T
53       */
54      protected static <T extends Component> T findComponentByName(Component root, String name) {
55          Component result = null;
56          if (name.equals(root.getName())) {
57              result = root;
58          }
59          else if (root instanceof Container) {
60              final Component[] children = ((Container) root).getComponents();
61              for (Component component : children) {
62                  result = findComponentByName(component, name);
63                  if (result != null) {
64                      break;
65                  }
66              }
67          }
68          return (T) result;
69      }
70  }

52      /**
53       * If a visit method is not overridden, we should explain why we do not 'visit' the
54       * parse tree at this node and construct an AST. Reasons could include that we have
55       * no terminal symbols (tokens) in the corresponding production rule, or that
56       * we handle the construction of this particular AST in its parent node. If we
57       * have a production rule where we have terminal symbols (tokens), but we do not build
58       * an AST from tokens in the rule context, the rule is extraneous.
59       */

64              "visitStat",

99      @Test
100     public void testAllVisitMethodsAreOverridden() {
101         final Method[] baseVisitMethods = JavaLanguageParserBaseVisitor
102                 .class.getDeclaredMethods();
103         final Method[] visitMethods = JavaAstVisitor.class.getDeclaredMethods();
104 
105         final Set<String> filteredBaseVisitMethodNames = Arrays.stream(baseVisitMethods)
106                 .filter(method -> !VISIT_METHODS_NOT_OVERRIDDEN.contains(method.getName()))
107                 .filter(method -> method.getName().contains("visit"))
108                 .filter(method -> method.getModifiers() == Modifier.PUBLIC)
109                 .map(Method::getName)
110                 .collect(Collectors.toSet());
111 
112         final ImmutableSet<String> filteredVisitMethodNames = Arrays.stream(visitMethods)
113                 .filter(method -> method.getName().contains("visit"))
114                 // remove overridden 'visit' method from ParseTreeVisitor interface in
115                 // JavaAstVisitor
116                 .filter(method -> !"visit".equals(method.getName()))
117                 .filter(method -> method.getModifiers() == Modifier.PUBLIC)
118                 .map(Method::getName)
119                 .collect(toImmutableSet());
120 
121         final String message = "Visit methods in 'JavaLanguageParserBaseVisitor' generated from "
122                 + "production rules and labeled alternatives in 'JavaLanguageParser.g4' should "
123                 + "be overridden in 'JavaAstVisitor' or be added to 'VISIT_METHODS_NOT_OVERRIDDEN' "
124                 + "with comment explaining why.";
125 
126         assertWithMessage(message)
127                 .that(filteredVisitMethodNames)
128                 .containsExactlyElementsIn(filteredBaseVisitMethodNames);
129     }

131     @Test
132     public void testOrderOfVisitMethodsAndProductionRules() throws Exception {
133         // Order of BaseVisitor's generated 'visit' methods match the order of
134         // production rules in 'JavaLanguageParser.g4'.
135         final String baseVisitorFilename = "target/generated-sources/antlr/com/puppycrawl"
136                 + "/tools/checkstyle/grammar/java/JavaLanguageParserBaseVisitor.java";
137         final DetailAST baseVisitorAst = JavaParser.parseFile(new File(baseVisitorFilename),
138                             JavaParser.Options.WITHOUT_COMMENTS);
139 
140         final String visitorFilename = "src/main/java/com/puppycrawl/tools/checkstyle"
141                 + "/JavaAstVisitor.java";
142         final DetailAST visitorAst = JavaParser.parseFile(new File(visitorFilename),
143                             JavaParser.Options.WITHOUT_COMMENTS);
144 
145         final List<String> orderedBaseVisitorMethodNames =
146                 getOrderedVisitMethodNames(baseVisitorAst);
147         final List<String> orderedVisitorMethodNames =
148                 getOrderedVisitMethodNames(visitorAst);
149 
150         orderedBaseVisitorMethodNames.removeAll(VISIT_METHODS_NOT_OVERRIDDEN);
151 
152         // remove overridden 'visit' method from ParseTreeVisitor interface in JavaAstVisitor
153         orderedVisitorMethodNames.remove("visit");
154 
155         assertWithMessage("Visit methods in 'JavaAstVisitor' should appear in same order as "
156                 + "production rules and labeled alternatives in 'JavaLanguageParser.g4'.")
157                 .that(orderedVisitorMethodNames)
158                 .containsExactlyElementsIn(orderedBaseVisitorMethodNames)
159                 .inOrder();
160     }

162     /**
163      * Finds all {@code visit...} methods in a source file, and collects
164      * the method names into a list. This method counts on the simple structure
165      * of 'JavaAstVisitor' and 'JavaLanguageParserBaseVisitor'.
166      *
167      * @param root the root of the AST to extract method names from
168      * @return list of all {@code visit...} method names
169      */
170     private static List<String> getOrderedVisitMethodNames(DetailAST root) {
171         final List<String> orderedVisitMethodNames = new ArrayList<>();
172 
173         DetailAST classDef = root.getFirstChild();
174         while (classDef.getType() != TokenTypes.CLASS_DEF) {
175             classDef = classDef.getNextSibling();
176         }
177 
178         final DetailAST objBlock = classDef.findFirstToken(TokenTypes.OBJBLOCK);
179         DetailAST objBlockChild = objBlock.findFirstToken(TokenTypes.METHOD_DEF);
180         while (objBlockChild != null) {
181             if (isVisitMethod(objBlockChild)) {
182                 orderedVisitMethodNames.add(objBlockChild
183                         .findFirstToken(TokenTypes.IDENT)
184                         .getText());
185             }
186             objBlockChild = objBlockChild.getNextSibling();
187         }
188         return orderedVisitMethodNames;
189     }
190 
191     /**
192      * Checks if given AST is a visit method.
193      *
194      * @param objBlockChild AST to check
195      * @return true if AST is a visit method
196      */
197     private static boolean isVisitMethod(DetailAST objBlockChild) {
198         return objBlockChild.getType() == TokenTypes.METHOD_DEF
199                 && objBlockChild.findFirstToken(TokenTypes.IDENT).getText().contains("visit");
200     }

211     /**
212      * This test exists to kill surviving mutation from pitest removing expression AST building
213      * optimization in {@link JavaAstVisitor#visitBinOp(JavaLanguageParser.BinOpContext)}.
214      * We do not use {@link JavaParser#parse(FileContents)} here due to DFA clearing hack.
215      *
216      * <p>
217      * Reason: we have iterative expression AST building to avoid stackoverflow
218      * in {@link JavaAstVisitor#visitBinOp(JavaLanguageParser.BinOpContext)}. In actual
219      * generated parser, we avoid stackoverflow thanks to the left recursive expression
220      * rule (eliminating unnecessary recursive calls to hierarchical expression production rules).
221      * However, ANTLR's ParserATNSimulator has no such optimization. So, the number of recursive
222      * calls to ParserATNSimulator#closure when calling ParserATNSimulator#clearDFA causes a
223      * StackOverflow error. We avoid this by using the single argument constructor (thus not
224      * forcing DFA clearing) in this test.
225      * </p>
226      *
227      * @throws Exception if input file does not exist
228      */
229 
230     @Test
231     public void testNoStackOverflowOnDeepStringConcat() throws Exception {
232         final File file =
233                 new File(getPath("InputJavaAstVisitorNoStackOverflowOnDeepStringConcat.java"));
234         final FileText fileText = new FileText(file, StandardCharsets.UTF_8.name());
235         final FileContents contents = new FileContents(fileText);
236 
237         final String fullText = contents.getText().getFullText().toString();
238         final CharStream codePointCharStream = CharStreams.fromString(fullText);
239         final JavaLanguageLexer lexer = new JavaLanguageLexer(codePointCharStream, true);
240         lexer.setCommentListener(contents);
241 
242         final CommonTokenStream tokenStream = new CommonTokenStream(lexer);
243         final JavaLanguageParser parser = new JavaLanguageParser(tokenStream);
244 
245         final JavaLanguageParser.CompilationUnitContext compilationUnit = parser.compilationUnit();
246 
247         // We restrict execution to use limited resources here, so that we can
248         // kill surviving pitest mutation from removal of nested binary operation
249         // optimization in JavaAstVisitor#visitBinOp. Limited resources (small stack size)
250         // ensure that we throw a StackOverflowError if optimization is removed.
251         final DetailAST root = TestUtil.getResultWithLimitedResources(
252                 () -> new JavaAstVisitor(tokenStream).visit(compilationUnit)
253         );
254 
255         assertWithMessage("File parsing and AST building should complete successfully.")
256                 .that(root)
257                 .isNotNull();
258     }

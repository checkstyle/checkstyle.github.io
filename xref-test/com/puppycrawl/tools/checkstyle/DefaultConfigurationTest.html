20  package com.puppycrawl.tools.checkstyle;
21  
22  import static org.junit.Assert.assertEquals;
23  import static org.junit.Assert.fail;
24  
25  import org.junit.Test;
26  
27  import com.puppycrawl.tools.checkstyle.api.CheckstyleException;
28  
29  public class DefaultConfigurationTest {
30  
31      @Test
32      public void testRemoveChild() {
33          final DefaultConfiguration config = new DefaultConfiguration("MyConfig");
34          final DefaultConfiguration configChild = new DefaultConfiguration("childConfig");
35          assertEquals("Invalid children count", 0, config.getChildren().length);
36          config.addChild(configChild);
37          assertEquals("Invalid children count", 1, config.getChildren().length);
38          config.removeChild(configChild);
39          assertEquals("Invalid children count", 0, config.getChildren().length);
40      }
41  
42      @Test
43      public void testExceptionForNonExistentAttribute() {
44          final String name = "MyConfig";
45          final DefaultConfiguration config = new DefaultConfiguration(name);
46          final String attributeName = "NonExistent#$%";
47          try {
48              config.getAttribute(attributeName);
49              fail("Exception is expected");
50          }
51          catch (CheckstyleException expected) {
52              assertEquals("Invalid exception message",
53                      "missing key '" + attributeName + "' in " + name,
54                      expected.getMessage());
55          }
56      }
57  
58      @Test
59      public void testDefaultMultiThreadConfiguration() {
60          final String name = "MyConfig";
61          final DefaultConfiguration config = new DefaultConfiguration(name);
62          final ThreadModeSettings singleThreadMode =
63                  ThreadModeSettings.SINGLE_THREAD_MODE_INSTANCE;
64          assertEquals("Invalid thread mode", singleThreadMode, config.getThreadModeSettings());
65      }
66  
67      @Test
68      public void testMultiThreadConfiguration() {
69          final String name = "MyConfig";
70          final ThreadModeSettings multiThreadMode =
71                  new ThreadModeSettings(4, 2);
72          final DefaultConfiguration config = new DefaultConfiguration(name, multiThreadMode);
73          assertEquals("Invalid thread mode", multiThreadMode, config.getThreadModeSettings());
74      }
75  
76  }

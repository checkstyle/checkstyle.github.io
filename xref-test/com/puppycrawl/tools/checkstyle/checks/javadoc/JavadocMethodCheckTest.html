427     @Test
428     public void testTokenToString() throws Exception {
429         final Class<?> tokenType = Class.forName("com.puppycrawl.tools.checkstyle.checks.javadoc."
430                 + "JavadocMethodCheck$Token");
431         final Constructor<?> tokenConstructor = tokenType.getDeclaredConstructor(String.class,
432                 int.class, int.class);
433         tokenConstructor.setAccessible(true);
434         final Object token = tokenConstructor.newInstance("tokenName", 1, 1);
435         final Method toString = token.getClass().getDeclaredMethod("toString");
436         final String result = (String) toString.invoke(token);
437         assertWithMessage("Invalid toString result")
438             .that(result)
439             .isEqualTo("Token[tokenName(1x1)]");
440     }

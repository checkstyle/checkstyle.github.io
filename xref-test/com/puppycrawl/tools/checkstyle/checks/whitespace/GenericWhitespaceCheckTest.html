159     @Test
160     public void testWrongTokenType() {
161         final GenericWhitespaceCheck genericWhitespaceCheckObj = new GenericWhitespaceCheck();
162         final DetailAST ast = new DetailAST();
163         ast.initialize(new CommonHiddenStreamToken(TokenTypes.INTERFACE_DEF, "interface"));
164         try {
165             genericWhitespaceCheckObj.visitToken(ast);
166             fail("exception expected");
167         }
168         catch (IllegalArgumentException ex) {
169             assertEquals("Invalid exception message",
170                 "Unknown type interface[0x-1]", ex.getMessage());
171         }

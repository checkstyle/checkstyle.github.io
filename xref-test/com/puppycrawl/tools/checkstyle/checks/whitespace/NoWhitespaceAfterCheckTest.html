219     @Test
220     public void testVisitTokenSwitchReflection() {
221         //unexpected parent for ARRAY_DECLARATOR token
222         final DetailAST astImport = mockAST(TokenTypes.IMPORT, "import", "mockfile");
223         final DetailAST astArrayDeclarator = mockAST(TokenTypes.ARRAY_DECLARATOR, "[", "mockfile");
224         final DetailAST astRightBracket = mockAST(TokenTypes.RBRACK, "[", "mockfile");
225         astImport.addChild(astArrayDeclarator);
226         astArrayDeclarator.addChild(astRightBracket);
227 
228         final NoWhitespaceAfterCheck check = new NoWhitespaceAfterCheck();
229         try {
230             check.visitToken(astArrayDeclarator);
231             fail("no intended exception thrown");
232         }
233         catch (IllegalStateException ex) {
234             assertEquals("Invalid exception message",
235                 "unexpected ast syntax import[0x-1]", ex.getMessage());
236         }
237     }

239     /**
240      * Creates MOCK lexical token and returns AST node for this token.
241      * @param tokenType type of token
242      * @param tokenText text of token
243      * @param tokenFileName file name of token
244      * @return AST node for the token
245      */
246     private static DetailAST mockAST(final int tokenType, final String tokenText,
247             final String tokenFileName) {
248         final CommonHiddenStreamToken tokenImportSemi = new CommonHiddenStreamToken();
249         tokenImportSemi.setType(tokenType);
250         tokenImportSemi.setText(tokenText);
251         tokenImportSemi.setFilename(tokenFileName);
252         final DetailAST astSemi = new DetailAST();
253         astSemi.initialize(tokenImportSemi);
254         return astSemi;
255     }

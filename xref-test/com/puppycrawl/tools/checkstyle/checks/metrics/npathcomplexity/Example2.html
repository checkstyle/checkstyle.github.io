19      if (a > b) { // non-nested if-else decision tree multiplies complexity by 2
20        bar();

22        baz();
23      }
24

28      print(t > 4 ? bar() : baz());
29      print(t > 5 ? bar() : baz());

32        baz();
33      } else {
34        bar();

36    }

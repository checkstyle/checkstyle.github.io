20  package com.puppycrawl.tools.checkstyle.checks.design;
21  
22  import static com.puppycrawl.tools.checkstyle.checks.design.InnerTypeLastCheck.MSG_KEY;
23  import static org.junit.Assert.assertArrayEquals;
24  
25  import org.junit.Test;
26  
27  import com.puppycrawl.tools.checkstyle.AbstractModuleTestSupport;
28  import com.puppycrawl.tools.checkstyle.DefaultConfiguration;
29  import com.puppycrawl.tools.checkstyle.api.TokenTypes;
30  import com.puppycrawl.tools.checkstyle.utils.CommonUtils;
31  
32  public class InnerTypeLastCheckTest extends AbstractModuleTestSupport {
33  
34      @Override
35      protected String getPackageLocation() {
36          return "com/puppycrawl/tools/checkstyle/checks/design/innertypelast";
37      }
38  
39      @Test
40      public void testGetRequiredTokens() {
41          final InnerTypeLastCheck checkObj = new InnerTypeLastCheck();
42          final int[] expected = {TokenTypes.CLASS_DEF, TokenTypes.INTERFACE_DEF};
43          assertArrayEquals("Default required tokens are invalid",
44              expected, checkObj.getRequiredTokens());
45      }
46  
47      @Test
48      public void testMembersBeforeInner() throws Exception {
49          final DefaultConfiguration checkConfig =
50              createModuleConfig(InnerTypeLastCheck.class);
51          final String[] expected = {
52              "44:9: " + getCheckMessage(MSG_KEY),
53              "65:9: " + getCheckMessage(MSG_KEY),
54              "69:9: " + getCheckMessage(MSG_KEY),
55              "78:5: " + getCheckMessage(MSG_KEY),
56              "95:9: " + getCheckMessage(MSG_KEY),
57          };
58          verify(checkConfig, getPath("InputInnerTypeLastClass.java"), expected);
59      }
60  
61      @Test
62      public void testIfRootClassChecked() throws Exception {
63          final DefaultConfiguration checkConfig =
64              createModuleConfig(InnerTypeLastCheck.class);
65          final String[] expected = CommonUtils.EMPTY_STRING_ARRAY;
66          verify(checkConfig, getPath("InputInnerTypeLastClassRootClass.java"), expected);
67      }
68  
69      @Test
70      public void testGetAcceptableTokens() {
71          final InnerTypeLastCheck obj = new InnerTypeLastCheck();
72          final int[] expected = {TokenTypes.CLASS_DEF, TokenTypes.INTERFACE_DEF};
73          assertArrayEquals("Default acceptable tokens are invalid",
74              expected, obj.getAcceptableTokens());
75      }
76  
77  }

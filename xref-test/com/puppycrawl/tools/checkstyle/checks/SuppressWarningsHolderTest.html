274     public void testGetAllAnnotationValuesWrongArg() throws ReflectiveOperationException {
275         final SuppressWarningsHolder holder = new SuppressWarningsHolder();
276         final Method getAllAnnotationValues = holder.getClass()
277                 .getDeclaredMethod("getAllAnnotationValues", DetailAST.class);
278         getAllAnnotationValues.setAccessible(true);
279 
280         final DetailAstImpl methodDef = new DetailAstImpl();
281         methodDef.setType(TokenTypes.METHOD_DEF);
282         methodDef.setText("Method Def");
283         methodDef.setLineNo(0);
284         methodDef.setColumnNo(0);
285 
286         final DetailAstImpl lparen = new DetailAstImpl();
287         lparen.setType(TokenTypes.LPAREN);
288 
289         final DetailAstImpl parent = new DetailAstImpl();
290         parent.addChild(lparen);
291         parent.addChild(methodDef);
292

295             assertWithMessage("Exception expected").fail();
296         }

298             assertWithMessage("Error type is unexpected")
299                     .that(ex)

302             assertWithMessage("Error message is unexpected")

409     public void testClearState() throws Exception {
410         final SuppressWarningsHolder check = new SuppressWarningsHolder();
411 
412         final Optional<DetailAST> annotationDef = TestUtil.findTokenInAstByPredicate(
413                 JavaParser.parseFile(

424                         entries -> ((ThreadLocal<List<Object>>) entries).get().isEmpty()))
425                 .isTrue();
426     }

428     private static void populateHolder(String checkName, int firstLine,
429                                                          int firstColumn, int lastLine,
430                                                          int lastColumn) throws Exception {
431         final Class<?> entry = Class
432                 .forName("com.puppycrawl.tools.checkstyle.checks.SuppressWarningsHolder$Entry");
433         final Constructor<?> entryConstr = entry.getDeclaredConstructor(String.class, int.class,
434                 int.class, int.class, int.class);
435         entryConstr.setAccessible(true);
436 
437         final Object entryInstance = entryConstr.newInstance(checkName, firstLine,
438                 firstColumn, lastLine, lastColumn);
439 
440         final ThreadLocal<List<Object>> entries = TestUtil
441                 .getInternalStaticState(SuppressWarningsHolder.class, "ENTRIES");
442         entries.get().add(entryInstance);
443     }

445     private static AuditEvent createAuditEvent(String moduleId, int line, int column) {
446         final Checker source = new Checker();
447         final Violation violation = new Violation(line, column, null, null, null,
448                 moduleId, MemberNameCheck.class, "violation");
449         return new AuditEvent(source, "filename", violation);
450     }

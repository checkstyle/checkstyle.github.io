326     private static SuppressWarningsHolder createHolder(String checkName, int firstLine,
327                                                        int firstColumn, int lastLine,
328                                                        int lastColumn) throws Exception {
329         final Class<?> entry = Class
330                 .forName("com.puppycrawl.tools.checkstyle.checks.SuppressWarningsHolder$Entry");
331         final Constructor<?> entryConstr = entry.getDeclaredConstructor(String.class, int.class,
332                 int.class, int.class, int.class);
333         entryConstr.setAccessible(true);
334 
335         final Object entryInstance = entryConstr.newInstance(checkName, firstLine,
336                 firstColumn, lastLine, lastColumn);
337 
338         final List<Object> entriesList = new ArrayList<>();
339         entriesList.add(entryInstance);
340 
341         final ThreadLocal<?> threadLocal = mock(ThreadLocal.class);
342         PowerMockito.doReturn(entriesList).when(threadLocal, "get");
343 
344         final SuppressWarningsHolder holder = new SuppressWarningsHolder();
345         final Field entries = holder.getClass().getDeclaredField("ENTRIES");
346         entries.setAccessible(true);
347         entries.set(holder, threadLocal);
348         return holder;
349     }

351     private static AuditEvent createAuditEvent(String moduleId, int line, int column) {
352         final Checker source = new Checker();
353         final LocalizedMessage message = new LocalizedMessage(line, column, null, null, null,
354                 moduleId, MemberNameCheck.class, "message");
355         return new AuditEvent(source, "filename", message);
356     }

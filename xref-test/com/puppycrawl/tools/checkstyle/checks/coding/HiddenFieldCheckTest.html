525         final DetailAST root = JavaParser.parseFile(new File(getPath("InputHiddenField8.java")),
526                 JavaParser.Options.WITHOUT_COMMENTS);

536                         "frame", new CheckIfStatefulFieldCleared()))
537                 .isTrue();
538     }

541

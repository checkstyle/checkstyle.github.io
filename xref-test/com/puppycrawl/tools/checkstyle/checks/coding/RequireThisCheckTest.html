403     @Test
404     public void testUnusedMethod() throws Exception {
405         final DetailAstImpl ident = new DetailAstImpl();
406         ident.setText("testName");
407 
408         final Class<?> cls = Class.forName(RequireThisCheck.class.getName() + "$CatchFrame");
409         final Constructor<?> constructor = cls.getDeclaredConstructors()[0];
410         constructor.setAccessible(true);
411         final Object o = constructor.newInstance(null, ident);
412 
413         Assert.assertEquals("expected ident token", ident,
414                 TestUtil.getClassDeclaredMethod(cls, "getFrameNameIdent").invoke(o));
415         Assert.assertEquals("expected catch frame type", "CATCH_FRAME",
416                 TestUtil.getClassDeclaredMethod(cls, "getType").invoke(o).toString());
417     }

419     /**
420      * We cannot reproduce situation when visitToken is called and leaveToken is not.
421      * So, we have to use reflection to be sure that even in such situation
422      * state of the field will be cleared.
423      *
424      * @throws Exception when code tested throws exception
425      */
426     @Test
427     public void testClearState() throws Exception {
428         final RequireThisCheck check = new RequireThisCheck();
429         final DetailAST root = JavaParser.parseFile(
430                 new File(getPath("InputRequireThisSimple.java")),
431                 JavaParser.Options.WITHOUT_COMMENTS);
432         final Optional<DetailAST> classDef = TestUtil.findTokenInAstByPredicate(root,
433             ast -> ast.getType() == TokenTypes.CLASS_DEF);
434 
435         Assert.assertTrue("Ast should contain CLASS_DEF", classDef.isPresent());
436         Assert.assertTrue("State is not cleared on beginTree",
437                 TestUtil.isStatefulFieldClearedDuringBeginTree(check, classDef.get(), "current",
438                     current -> ((Collection<?>) current).isEmpty()));
439     }

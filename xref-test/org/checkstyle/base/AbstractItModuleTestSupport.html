108     /**
109      * Returns {@link Configuration} instance for the given module name pulled
110      * from the {@code masterConfig}.
111      *
112      * @param masterConfig The master configuration to examine.
113      * @param moduleName module name.
114      * @param moduleId module id.
115      * @return {@link Configuration} instance for the given module name.
116      * @throws IllegalStateException if there is a problem retrieving the module
117      *         or config.
118      */
119     protected static Configuration getModuleConfig(Configuration masterConfig, String moduleName,
120             String moduleId) {
121         final Configuration result;
122         final List<Configuration> configs = getModuleConfigs(masterConfig, moduleName);
123         if (configs.size() == 1) {
124             result = configs.get(0);
125         }
126         else if (configs.isEmpty()) {
127             throw new IllegalStateException("no instances of the Module was found: " + moduleName);
128         }
129         else if (moduleId == null) {
130             throw new IllegalStateException("multiple instances of the same Module are detected");
131         }
132         else {
133             result = configs.stream().filter(conf -> isSameModuleId(conf, moduleId))
134             .findFirst()
135             .orElseThrow(() -> new IllegalStateException("problem with module config"));
136         }
137 
138         return result;
139     }

141     /**
142      * Verifies if the configuration's ID matches the expected {@code moduleId}.
143      *
144      * @param conf The config to examine.
145      * @param moduleId The module ID to match against.
146      * @return {@code true} if it matches.
147      * @throws IllegalStateException If there is an issue with finding the ID.
148      */
149     private static boolean isSameModuleId(Configuration conf, String moduleId) {
150         try {
151             return conf.getProperty("id").equals(moduleId);
152         }
153         catch (CheckstyleException ex) {
154             throw new IllegalStateException("problem to get ID attribute from " + conf, ex);
155         }
156     }

158     /**
159      * Returns a list of all {@link Configuration} instances for the given module IDs in the
160      * {@code masterConfig}.
161      *
162      * @param masterConfig The master configuration to pull results from.
163      * @param moduleIds module IDs.
164      * @return List of {@link Configuration} instances.
165      * @throws CheckstyleException if there is an error with the config.
166      */
167     protected static List<Configuration> getModuleConfigsByIds(Configuration masterConfig,
168             String... moduleIds) throws CheckstyleException {
169         final List<Configuration> result = new ArrayList<>();
170         for (Configuration currentConfig : masterConfig.getChildren()) {
171             if ("TreeWalker".equals(currentConfig.getName())) {
172                 for (Configuration moduleConfig : currentConfig.getChildren()) {
173                     final String id = getProperty(moduleConfig, "id");
174                     if (id != null && isIn(id, moduleIds)) {
175                         result.add(moduleConfig);
176                     }
177                 }
178             }
179             else {
180                 final String id = getProperty(currentConfig, "id");
181                 if (id != null && isIn(id, moduleIds)) {
182                     result.add(currentConfig);
183                 }
184             }
185         }
186         return result;
187     }

189     /**
190      * Finds the specific property {@code name} in the {@code config}.
191      *
192      * @param config The configuration to examine.
193      * @param name The property name to find.
194      * @return The property value or {@code null} if not found.
195      * @throws CheckstyleException if there is an error with the config.
196      */
197     private static String getProperty(Configuration config, String name)
198             throws CheckstyleException {
199         String result = null;
200 
201         if (isIn(name, config.getPropertyNames())) {
202             result = config.getProperty(name);
203         }
204 
205         return result;
206     }

208     /**
209      * Finds the specific ID in a list of IDs.
210      *
211      * @param find The ID to find.
212      * @param list The list of module IDs.
213      * @return {@code true} if the ID is in the list.
214      */
215     private static boolean isIn(String find, String... list) {
216         boolean found = false;
217 
218         for (String item : list) {
219             if (find.equals(item)) {
220                 found = true;
221                 break;
222             }
223         }
224 
225         return found;
226     }

229      * Returns a list of all {@link Configuration} instances for the given
230      * module name pulled from the {@code masterConfig}.
231      *
232      * @param masterConfig The master configuration to examine.
233      * @param moduleName module name.
234      * @return {@link Configuration} instance for the given module name.
235      */
236     private static List<Configuration> getModuleConfigs(Configuration masterConfig,
237             String moduleName) {
238         final List<Configuration> result = new ArrayList<>();
239         for (Configuration currentConfig : masterConfig.getChildren()) {
240             if ("TreeWalker".equals(currentConfig.getName())) {
241                 for (Configuration moduleConfig : currentConfig.getChildren()) {
242                     if (moduleName.equals(moduleConfig.getName())) {
243                         result.add(moduleConfig);
244                     }
245                 }
246             }
247             else if (moduleName.equals(currentConfig.getName())) {
248                 result.add(currentConfig);
249             }
250         }
251         return result;
252     }

268     /**
269      * Creates {@link Checker} instance based on the given {@link Configuration} instance.
270      *
271      * @param moduleConfig {@link Configuration} instance.
272      * @param moduleCreationOption {@code IN_TREEWALKER} if the {@code moduleConfig} should be added
273      *                                                  under {@link TreeWalker}.
274      * @return {@link Checker} instance based on the given {@link Configuration} instance.
275      * @throws Exception if an exception occurs during checker configuration.
276      */
277     protected final Checker createChecker(Configuration moduleConfig,
278                                  ModuleCreationOption moduleCreationOption)
279             throws Exception {
280         final Checker checker = new Checker();
281         checker.setModuleClassLoader(Thread.currentThread().getContextClassLoader());
282         // make sure the tests always run with English error messages
283         // so the tests don't fail in supported locales like German
284         final Locale locale = Locale.ENGLISH;
285         checker.setLocaleCountry(locale.getCountry());
286         checker.setLocaleLanguage(locale.getLanguage());
287 
288         if (moduleCreationOption == ModuleCreationOption.IN_TREEWALKER) {
289             final Configuration config = createTreeWalkerConfig(moduleConfig);
290             checker.configure(config);
291         }
292         else if (ROOT_MODULE_NAME.equals(moduleConfig.getName())) {
293             checker.configure(moduleConfig);
294         }
295         else {
296             final Configuration config = createRootConfig(moduleConfig);
297             checker.configure(config);
298         }
299         checker.addListener(getBriefUtLogger());
300         return checker;
301     }

303     /**
304      * Creates {@link DefaultConfiguration} for the {@link TreeWalker}
305      * based on the given {@link Configuration} instance.
306      *
307      * @param config {@link Configuration} instance.
308      * @return {@link DefaultConfiguration} for the {@link TreeWalker}
309      *     based on the given {@link Configuration} instance.
310      */
311     protected static DefaultConfiguration createTreeWalkerConfig(Configuration config) {
312         final DefaultConfiguration rootConfig =
313                 new DefaultConfiguration(ROOT_MODULE_NAME);
314         final DefaultConfiguration twConf = createModuleConfig(TreeWalker.class);
315         // make sure that the tests always run with this charset
316         rootConfig.addProperty("charset", StandardCharsets.UTF_8.name());
317         rootConfig.addChild(twConf);
318         twConf.addChild(config);
319         return rootConfig;
320     }

322     /**
323      * Creates {@link DefaultConfiguration} or the Checker.
324      * based on the the list of {@link Configuration}.
325      *
326      * @param configs list of {@link Configuration} instances.
327      * @return {@link DefaultConfiguration} for the Checker.
328      */
329     protected static DefaultConfiguration createTreeWalkerConfig(
330             List<Configuration> configs) {
331         DefaultConfiguration result = null;
332 
333         for (Configuration config : configs) {
334             if (result == null) {
335                 result = (DefaultConfiguration) createTreeWalkerConfig(config).getChildren()[0];
336             }
337             else {
338                 result.addChild(config);
339             }
340         }
341 
342         return result;
343     }

370     /**
371      * Performs verification of the file with the given file name. Uses specified configuration.
372      * Expected messages are represented by the array of strings, warning line numbers are
373      * represented by the array of integers.
374      * This implementation uses overloaded
375      * {@link AbstractItModuleTestSupport#verify(Checker, File[], String, String[], Integer...)}
376      * method inside.
377      *
378      * @param config configuration.
379      * @param fileName file name to verify.
380      * @param expected an array of expected messages.
381      * @param warnsExpected an array of expected warning numbers.
382      * @throws Exception if exception occurs during verification process.
383      */
384     protected final void verify(Configuration config, String fileName, String[] expected,
385             Integer... warnsExpected) throws Exception {
386         verify(createChecker(config),
387                 new File[] {new File(fileName)},
388                 fileName, expected, warnsExpected);
389     }

391     /**
392      * Performs verification of files.
393      * Uses provided {@link Checker} instance.
394      *
395      * @param checker {@link Checker} instance.
396      * @param processedFiles files to process.
397      * @param messageFileName message file name.
398      * @param expected an array of expected messages.
399      * @param warnsExpected an array of expected warning line numbers.
400      * @throws Exception if exception occurs during verification process.
401      */
402     protected final void verify(Checker checker,
403             File[] processedFiles,
404             String messageFileName,
405             String[] expected,
406             Integer... warnsExpected)
407             throws Exception {
408         stream.flush();
409         stream.reset();
410         final List<File> theFiles = new ArrayList<>();
411         Collections.addAll(theFiles, processedFiles);
412         final List<Integer> theWarnings = new ArrayList<>();
413         Collections.addAll(theWarnings, warnsExpected);
414         final int errs = checker.process(theFiles);
415 
416         // process each of the lines
417         try (ByteArrayInputStream inputStream =
418                 new ByteArrayInputStream(stream.toByteArray());
419             LineNumberReader lnr = new LineNumberReader(
420                 new InputStreamReader(inputStream, StandardCharsets.UTF_8))) {
421             int previousLineNumber = 0;
422             for (int index = 0; index < expected.length; index++) {
423                 final String expectedResult = messageFileName + ":" + expected[index];
424                 final String actual = lnr.readLine();
425                 assertWithMessage("Error message at position %s of 'expected' does "
426                         + "not match actual message", index)
427                     .that(actual)
428                     .isEqualTo(expectedResult);
429 
430                 String parseInt = removeDeviceFromPathOnWindows(actual);
431                 parseInt = parseInt.substring(parseInt.indexOf(':') + 1);
432                 parseInt = parseInt.substring(0, parseInt.indexOf(':'));
433                 final int lineNumber = Integer.parseInt(parseInt);
434                 assertWithMessage(
435                         "input file is expected to have a warning comment on line number %s",
436                         lineNumber)
437                     .that(previousLineNumber == lineNumber
438                             || theWarnings.remove((Integer) lineNumber))
439                     .isTrue();
440                 previousLineNumber = lineNumber;
441             }
442 
443             assertWithMessage("unexpected output: %s", lnr.readLine())
444                 .that(errs)
445                 .isEqualTo(expected.length);
446             assertWithMessage("unexpected warnings %s", theWarnings)
447                 .that(theWarnings)
448                 .isEmpty();
449         }
450 
451         checker.destroy();
452     }

454     /**
455      * Gets the check message 'as is' from appropriate 'messages.properties'
456      * file.
457      *
458      * @param aClass the package the message is located in.
459      * @param messageKey the key of message in 'messages.properties' file.
460      * @param arguments  the arguments of message in 'messages.properties' file.
461      * @return The message of the check with the arguments applied.
462      * @throws IOException if there is a problem loading the property file.
463      */
464     protected static String getCheckMessage(Class<? extends AbstractViolationReporter> aClass,
465             String messageKey, Object... arguments) throws IOException {
466         final Properties pr = new Properties();
467         pr.load(aClass.getResourceAsStream("messages.properties"));
468         final MessageFormat formatter = new MessageFormat(pr.getProperty(messageKey),
469                 Locale.ROOT);
470         return formatter.format(arguments);
471     }

473     /**
474      * Gets the check message 'as is' from appropriate 'messages.properties' file.
475      *
476      * @param messages the map of messages to scan.
477      * @param messageKey the key of message in 'messages.properties' file.
478      * @param arguments the arguments of message in 'messages.properties' file.
479      * @return The message of the check with the arguments applied.
480      */
481     protected static String getCheckMessage(Map<String, String> messages, String messageKey,
482             Object... arguments) {
483         String checkMessage = null;
484         for (Map.Entry<String, String> entry : messages.entrySet()) {
485             if (messageKey.equals(entry.getKey())) {
486                 final MessageFormat formatter = new MessageFormat(entry.getValue(), Locale.ROOT);
487                 checkMessage = formatter.format(arguments);
488                 break;
489             }
490         }
491         return checkMessage;
492     }

494     /**
495      * Remove device from path string for windows path.
496      *
497      * @param path path to correct.
498      * @return Path without device name.
499      */
500     private static String removeDeviceFromPathOnWindows(String path) {
501         String fixedPath = path;
502         final String os = System.getProperty("os.name", "Unix");
503         if (os.startsWith("Windows")) {
504             fixedPath = path.substring(path.indexOf(':') + 1);
505         }
506         return fixedPath;
507     }

509     /**
510      * Returns an array of integers which represents the warning line numbers in the file
511      * with the given file name.
512      *
513      * @param fileName file name.
514      * @return an array of integers which represents the warning line numbers.
515      * @throws IOException if I/O exception occurs while reading the file.
516      */
517     protected Integer[] getLinesWithWarn(String fileName) throws IOException {
518         final List<Integer> result = new ArrayList<>();
519         try (BufferedReader br = Files.newBufferedReader(
520                 Paths.get(fileName), StandardCharsets.UTF_8)) {
521             int lineNumber = 1;
522             while (true) {
523                 final String line = br.readLine();
524                 if (line == null) {
525                     break;
526                 }
527                 if (WARN_PATTERN.matcher(line).find()) {
528                     result.add(lineNumber);
529                 }
530                 lineNumber++;
531             }
532         }
533         return result.toArray(new Integer[0]);
534     }

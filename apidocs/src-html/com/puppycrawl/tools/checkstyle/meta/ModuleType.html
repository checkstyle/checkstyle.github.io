020package com.puppycrawl.tools.checkstyle.meta;
021
022/** Enum holding the types of module which can exist. */
023public enum ModuleType {
024
025    /** Module is a Checkstyle check. */
026    CHECK("check"),
027
028    /** Module is a Checkstyle filter. */
029    FILTER("filter"),
030
031    /** Module is a Checkstyle file filter. */
032    FILEFILTER("file-filter");
033
034    /** String representation of the module type. */
035    private final String label;
036
037    /**
038     * Creates a new instance.
039     *
040     * @param label label of module
041     */
042    ModuleType(String label) {
043        this.label = label;
044    }
045
046    /**
047     * Get label corresponding to a module type.
048     *
049     * @return module type label
050     */
051    public String getLabel() {
052        return label;
053    }
054}

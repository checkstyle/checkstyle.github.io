182    @Override
183    public void destroy() {
184        listeners.clear();
185        fileSetChecks.clear();
186        beforeExecutionFileFilters.clear();
187        filters.clear();
188        if (cacheFile != null) {
189            try {
190                cacheFile.persist();
191            }
192            catch (IOException ex) {
193                throw new IllegalStateException("Unable to persist cache file.", ex);
194            }
195        }
196    }

216    @Override
217    public int process(List<File> files) throws CheckstyleException {
218        if (cacheFile != null) {
219            cacheFile.putExternalResources(getExternalResourceLocations());
220        }
221
222        // Prepare to start
223        fireAuditStarted();
224        for (final FileSetCheck fsc : fileSetChecks) {
225            fsc.beginProcessing(charset);
226        }
227
228        final List<File> targetFiles = files.stream()
229                .filter(file -> CommonUtil.matchesFileExtension(file, fileExtensions))
230                .collect(Collectors.toUnmodifiableList());
231        processFiles(targetFiles);
232
233        // Finish up
234        // It may also log!!!
235        fileSetChecks.forEach(FileSetCheck::finishProcessing);
236
237        // It may also log!!!
238        fileSetChecks.forEach(FileSetCheck::destroy);
239
240        final int errorCount = counter.getCount();
241        fireAuditFinished();
242        return errorCount;
243    }

245    /**
246     * Returns a set of external configuration resource locations which are used by all file set
247     * checks and filters.
248     *
249     * @return a set of external configuration resource locations which are used by all file set
250     *         checks and filters.
251     */
252    private Set<String> getExternalResourceLocations() {
253        return Stream.concat(fileSetChecks.stream(), filters.getFilters().stream())
254            .filter(ExternalResourceHolder.class::isInstance)
255            .flatMap(resource -> {
256                return ((ExternalResourceHolder) resource)
257                        .getExternalResourceLocations().stream();
258            })
259            .collect(Collectors.toUnmodifiableSet());
260    }

278    /**
279     * Processes a list of files with all FileSetChecks.
280     *
281     * @param files a list of files to process.
282     * @throws CheckstyleException if error condition within Checkstyle occurs.
283     * @throws Error wraps any java.lang.Error happened during execution
284     * @noinspection ProhibitedExceptionThrown
285     * @noinspectionreason ProhibitedExceptionThrown - There is no other way to
286     *      deliver filename that was under processing.
287     */
288    // -@cs[CyclomaticComplexity] no easy way to split this logic of processing the file
289    private void processFiles(List<File> files) throws CheckstyleException {
290        for (final File file : files) {
291            String fileName = null;
292            final String filePath = file.getPath();
293            try {
294                fileName = file.getAbsolutePath();
295                final long timestamp = file.lastModified();
296                if (cacheFile != null && cacheFile.isInCache(fileName, timestamp)
297                        || !acceptFileStarted(fileName)) {
298                    continue;
299                }
300                if (cacheFile != null) {
301                    cacheFile.put(fileName, timestamp);
302                }
303                fireFileStarted(fileName);
304                final SortedSet<Violation> fileMessages = processFile(file);
305                fireErrors(fileName, fileMessages);
306                fireFileFinished(fileName);
307            }
308            // -@cs[IllegalCatch] There is no other way to deliver filename that was under
309            // processing. See https://github.com/checkstyle/checkstyle/issues/2285
310            catch (Exception ex) {
311                if (fileName != null && cacheFile != null) {
312                    cacheFile.remove(fileName);
313                }
314
315                // We need to catch all exceptions to put a reason failure (file name) in exception
316                throw new CheckstyleException("Exception was thrown while processing "
317                        + filePath, ex);
318            }
319            catch (Error error) {
320                if (fileName != null && cacheFile != null) {
321                    cacheFile.remove(fileName);
322                }
323
324                // We need to catch all errors to put a reason failure (file name) in error
325                throw new Error("Error was thrown while processing " + filePath, error);
326            }
327        }
328    }

330    /**
331     * Processes a file with all FileSetChecks.
332     *
333     * @param file a file to process.
334     * @return a sorted set of violations to be logged.
335     * @throws CheckstyleException if error condition within Checkstyle occurs.
336     * @noinspection ProhibitedExceptionThrown
337     * @noinspectionreason ProhibitedExceptionThrown - there is no other way to obey
338     *      haltOnException field
339     */
340    private SortedSet<Violation> processFile(File file) throws CheckstyleException {
341        final SortedSet<Violation> fileMessages = new TreeSet<>();
342        try {
343            final FileText theText = new FileText(file.getAbsoluteFile(), charset);
344            for (final FileSetCheck fsc : fileSetChecks) {
345                fileMessages.addAll(fsc.process(file, theText));
346            }
347        }
348        catch (final IOException ioe) {
349            log.debug("IOException occurred.", ioe);
350            fileMessages.add(new Violation(1,
351                    Definitions.CHECKSTYLE_BUNDLE, EXCEPTION_MSG,
352                    new String[] {ioe.getMessage()}, null, getClass(), null));
353        }
354        // -@cs[IllegalCatch] There is no other way to obey haltOnException field
355        catch (Exception ex) {
356            if (haltOnException) {
357                throw ex;
358            }
359
360            log.debug("Exception occurred.", ex);
361
362            final StringWriter sw = new StringWriter();
363            final PrintWriter pw = new PrintWriter(sw, true);
364
365            ex.printStackTrace(pw);
366
367            fileMessages.add(new Violation(1,
368                    Definitions.CHECKSTYLE_BUNDLE, EXCEPTION_MSG,
369                    new String[] {sw.getBuffer().toString()},
370                    null, getClass(), null));
371        }
372        return fileMessages;
373    }

375    /**
376     * Check if all before execution file filters accept starting the file.
377     *
378     * @param fileName
379     *            the file to be audited
380     * @return {@code true} if the file is accepted.
381     */
382    private boolean acceptFileStarted(String fileName) {
383        final String stripped = CommonUtil.relativizePath(basedir, fileName);
384        return beforeExecutionFileFilters.accept(stripped);
385    }

393    @Override

395        final String stripped = CommonUtil.relativizePath(basedir, fileName);

402    /**
403     * Notify all listeners about the errors in a file.
404     *
405     * @param fileName the audited file
406     * @param errors the audit errors from the file
407     */
408    @Override
409    public void fireErrors(String fileName, SortedSet<Violation> errors) {
410        final String stripped = CommonUtil.relativizePath(basedir, fileName);
411        boolean hasNonFilteredViolations = false;
412        for (final Violation element : errors) {
413            final AuditEvent event = new AuditEvent(this, stripped, element);
414            if (filters.accept(event)) {
415                hasNonFilteredViolations = true;
416                for (final AuditListener listener : listeners) {
417                    listener.addError(event);
418                }
419            }
420        }
421        if (hasNonFilteredViolations && cacheFile != null) {
422            cacheFile.remove(fileName);
423        }
424    }

441    @Override
442    protected void finishLocalSetup() throws CheckstyleException {
443        final Locale locale = new Locale(localeLanguage, localeCountry);
444        LocalizedMessage.setLocale(locale);
445
446        if (moduleFactory == null) {
447            if (moduleClassLoader == null) {
448                final String finishLocalSetupMessage = finishLocalSetupMsg.getMessage();
449                throw new CheckstyleException(finishLocalSetupMessage);
450            }
451
452            final Set<String> packageNames = PackageNamesLoader
453                    .getPackageNames(moduleClassLoader);
454            moduleFactory = new PackageObjectFactory(packageNames,
455                    moduleClassLoader);
456        }
457
458        final DefaultContext context = new DefaultContext();
459        context.add("charset", charset);
460        context.add("moduleFactory", moduleFactory);
461        context.add("severity", severity.getName());
462        context.add("basedir", basedir);
463        context.add("tabWidth", String.valueOf(tabWidth));
464        childContext = context;
465    }

467    /**
468     * {@inheritDoc} Creates child module.
469     *
470     * @noinspection ChainOfInstanceofChecks
471     * @noinspectionreason ChainOfInstanceofChecks - we treat checks and filters differently
472     */
473    @Override
474    protected void setupChild(Configuration childConf)
475            throws CheckstyleException {
476        final String name = childConf.getName();
477        final Object child;
478
479        try {
480            child = moduleFactory.createModule(name);
481
482            if (child instanceof AbstractAutomaticBean) {
483                final AbstractAutomaticBean bean = (AbstractAutomaticBean) child;
484                bean.contextualize(childContext);
485                bean.configure(childConf);
486            }
487        }
488        catch (final CheckstyleException ex) {
489            throw new CheckstyleException("cannot initialize module " + name
490                    + " - " + ex.getMessage(), ex);
491        }
492        if (child instanceof FileSetCheck) {
493            final FileSetCheck fsc = (FileSetCheck) child;
494            fsc.init();
495            addFileSetCheck(fsc);
496        }
497        else if (child instanceof BeforeExecutionFileFilter) {
498            final BeforeExecutionFileFilter filter = (BeforeExecutionFileFilter) child;
499            addBeforeExecutionFileFilter(filter);
500        }
501        else if (child instanceof Filter) {
502            final Filter filter = (Filter) child;
503            addFilter(filter);
504        }
505        else if (child instanceof AuditListener) {
506            final AuditListener listener = (AuditListener) child;
507            addListener(listener);
508        }
509        else {
510            throw new CheckstyleException(name
511                    + " is not allowed as a child in Checker");
512        }
513    }

549    /**
550     * Sets the file extensions that identify the files that pass the
551     * filter of this FileSetCheck.
552     *
553     * @param extensions the set of file extensions. A missing
554     *     initial '.' character of an extension is automatically added.
555     */
556    public final void setFileExtensions(String... extensions) {
557        if (extensions != null) {
558            fileExtensions = new String[extensions.length];
559            for (int i = 0; i < extensions.length; i++) {
560                final String extension = extensions[i];
561                if (extension.startsWith(EXTENSION_SEPARATOR)) {
562                    fileExtensions[i] = extension;
563                }
564                else {
565                    fileExtensions[i] = EXTENSION_SEPARATOR + extension;
566                }
567            }
568        }
569    }

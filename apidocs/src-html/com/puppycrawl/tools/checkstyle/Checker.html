174    @Override
175    public void destroy() {
176        listeners.clear();
177        fileSetChecks.clear();
178        beforeExecutionFileFilters.clear();
179        filters.clear();
180        if (cacheFile != null) {
181            try {
182                cacheFile.persist();
183            }
184            catch (IOException ex) {
185                throw new IllegalStateException("Unable to persist cache file.", ex);
186            }
187        }
188    }

208    @Override
209    public int process(List<File> files) throws CheckstyleException {
210        if (cacheFile != null) {
211            cacheFile.putExternalResources(getExternalResourceLocations());
212        }
213
214        // Prepare to start
215        fireAuditStarted();
216        for (final FileSetCheck fsc : fileSetChecks) {
217            fsc.beginProcessing(charset);
218        }
219
220        final List<File> targetFiles = files.stream()
221                .filter(file -> CommonUtil.matchesFileExtension(file, fileExtensions))
222                .collect(Collectors.toList());
223        processFiles(targetFiles);
224
225        // Finish up
226        // It may also log!!!
227        fileSetChecks.forEach(FileSetCheck::finishProcessing);
228
229        // It may also log!!!
230        fileSetChecks.forEach(FileSetCheck::destroy);
231
232        final int errorCount = counter.getCount();
233        fireAuditFinished();
234        return errorCount;
235    }

237    /**
238     * Returns a set of external configuration resource locations which are used by all file set
239     * checks and filters.
240     *
241     * @return a set of external configuration resource locations which are used by all file set
242     *         checks and filters.
243     */
244    private Set<String> getExternalResourceLocations() {
245        return Stream.concat(fileSetChecks.stream(), filters.getFilters().stream())
246            .filter(ExternalResourceHolder.class::isInstance)
247            .map(ExternalResourceHolder.class::cast)
248            .flatMap(resource -> resource.getExternalResourceLocations().stream())
249            .collect(Collectors.toSet());
250    }

268    /**
269     * Processes a list of files with all FileSetChecks.
270     *
271     * @param files a list of files to process.
272     * @throws CheckstyleException if error condition within Checkstyle occurs.
273     * @throws Error wraps any java.lang.Error happened during execution
274     * @noinspection ProhibitedExceptionThrown
275     * @noinspectionreason ProhibitedExceptionThrown - There is no other way to
276     *      deliver filename that was under processing.
277     */
278    // -@cs[CyclomaticComplexity] no easy way to split this logic of processing the file
279    private void processFiles(List<File> files) throws CheckstyleException {
280        for (final File file : files) {
281            String fileName = null;
282            try {
283                fileName = file.getAbsolutePath();
284                final long timestamp = file.lastModified();
285                if (cacheFile != null && cacheFile.isInCache(fileName, timestamp)
286                        || !acceptFileStarted(fileName)) {
287                    continue;
288                }
289                if (cacheFile != null) {
290                    cacheFile.put(fileName, timestamp);
291                }
292                fireFileStarted(fileName);
293                final SortedSet<Violation> fileMessages = processFile(file);
294                fireErrors(fileName, fileMessages);
295                fireFileFinished(fileName);
296            }
297            // -@cs[IllegalCatch] There is no other way to deliver filename that was under
298            // processing. See https://github.com/checkstyle/checkstyle/issues/2285
299            catch (Exception ex) {
300                if (fileName != null && cacheFile != null) {
301                    cacheFile.remove(fileName);
302                }
303
304                // We need to catch all exceptions to put a reason failure (file name) in exception
305                throw new CheckstyleException("Exception was thrown while processing "
306                        + file.getPath(), ex);
307            }
308            catch (Error error) {
309                if (fileName != null && cacheFile != null) {
310                    cacheFile.remove(fileName);
311                }
312
313                // We need to catch all errors to put a reason failure (file name) in error
314                throw new Error("Error was thrown while processing " + file.getPath(), error);
315            }
316        }
317    }

319    /**
320     * Processes a file with all FileSetChecks.
321     *
322     * @param file a file to process.
323     * @return a sorted set of violations to be logged.
324     * @throws CheckstyleException if error condition within Checkstyle occurs.
325     * @noinspection ProhibitedExceptionThrown
326     * @noinspectionreason ProhibitedExceptionThrown - there is no other way to obey
327     *      haltOnException field
328     */
329    private SortedSet<Violation> processFile(File file) throws CheckstyleException {
330        final SortedSet<Violation> fileMessages = new TreeSet<>();
331        try {
332            final FileText theText = new FileText(file.getAbsoluteFile(), charset);
333            for (final FileSetCheck fsc : fileSetChecks) {
334                fileMessages.addAll(fsc.process(file, theText));
335            }
336        }
337        catch (final IOException ioe) {
338            log.debug("IOException occurred.", ioe);
339            fileMessages.add(new Violation(1,
340                    Definitions.CHECKSTYLE_BUNDLE, EXCEPTION_MSG,
341                    new String[] {ioe.getMessage()}, null, getClass(), null));
342        }
343        // -@cs[IllegalCatch] There is no other way to obey haltOnException field
344        catch (Exception ex) {
345            if (haltOnException) {
346                throw ex;
347            }
348
349            log.debug("Exception occurred.", ex);
350
351            final StringWriter sw = new StringWriter();
352            final PrintWriter pw = new PrintWriter(sw, true);
353
354            ex.printStackTrace(pw);
355
356            fileMessages.add(new Violation(1,
357                    Definitions.CHECKSTYLE_BUNDLE, EXCEPTION_MSG,
358                    new String[] {sw.getBuffer().toString()},
359                    null, getClass(), null));
360        }
361        return fileMessages;
362    }

364    /**
365     * Check if all before execution file filters accept starting the file.
366     *
367     * @param fileName
368     *            the file to be audited
369     * @return {@code true} if the file is accepted.
370     */
371    private boolean acceptFileStarted(String fileName) {
372        final String stripped = CommonUtil.relativizeAndNormalizePath(basedir, fileName);
373        return beforeExecutionFileFilters.accept(stripped);
374    }

382    @Override

384        final String stripped = CommonUtil.relativizeAndNormalizePath(basedir, fileName);

391    /**
392     * Notify all listeners about the errors in a file.
393     *
394     * @param fileName the audited file
395     * @param errors the audit errors from the file
396     */
397    @Override
398    public void fireErrors(String fileName, SortedSet<Violation> errors) {
399        final String stripped = CommonUtil.relativizeAndNormalizePath(basedir, fileName);
400        boolean hasNonFilteredViolations = false;
401        for (final Violation element : errors) {
402            final AuditEvent event = new AuditEvent(this, stripped, element);
403            if (filters.accept(event)) {
404                hasNonFilteredViolations = true;
405                for (final AuditListener listener : listeners) {
406                    listener.addError(event);
407                }
408            }
409        }
410        if (hasNonFilteredViolations && cacheFile != null) {
411            cacheFile.remove(fileName);
412        }
413    }

430    @Override
431    protected void finishLocalSetup() throws CheckstyleException {
432        final Locale locale = new Locale(localeLanguage, localeCountry);
433        LocalizedMessage.setLocale(locale);
434
435        if (moduleFactory == null) {
436            if (moduleClassLoader == null) {
437                throw new CheckstyleException(
438                        "if no custom moduleFactory is set, "
439                                + "moduleClassLoader must be specified");
440            }
441
442            final Set<String> packageNames = PackageNamesLoader
443                    .getPackageNames(moduleClassLoader);
444            moduleFactory = new PackageObjectFactory(packageNames,
445                    moduleClassLoader);
446        }
447
448        final DefaultContext context = new DefaultContext();
449        context.add("charset", charset);
450        context.add("moduleFactory", moduleFactory);
451        context.add("severity", severity.getName());
452        context.add("basedir", basedir);
453        context.add("tabWidth", String.valueOf(tabWidth));
454        childContext = context;
455    }

457    /**
458     * {@inheritDoc} Creates child module.
459     *
460     * @noinspection ChainOfInstanceofChecks
461     * @noinspectionreason ChainOfInstanceofChecks - we treat checks and filters differently
462     */
463    @Override
464    protected void setupChild(Configuration childConf)
465            throws CheckstyleException {
466        final String name = childConf.getName();
467        final Object child;
468
469        try {
470            child = moduleFactory.createModule(name);
471
472            if (child instanceof AbstractAutomaticBean) {
473                final AbstractAutomaticBean bean = (AbstractAutomaticBean) child;
474                bean.contextualize(childContext);
475                bean.configure(childConf);
476            }
477        }
478        catch (final CheckstyleException ex) {
479            throw new CheckstyleException("cannot initialize module " + name
480                    + " - " + ex.getMessage(), ex);
481        }
482        if (child instanceof FileSetCheck) {
483            final FileSetCheck fsc = (FileSetCheck) child;
484            fsc.init();
485            addFileSetCheck(fsc);
486        }
487        else if (child instanceof BeforeExecutionFileFilter) {
488            final BeforeExecutionFileFilter filter = (BeforeExecutionFileFilter) child;
489            addBeforeExecutionFileFilter(filter);
490        }
491        else if (child instanceof Filter) {
492            final Filter filter = (Filter) child;
493            addFilter(filter);
494        }
495        else if (child instanceof AuditListener) {
496            final AuditListener listener = (AuditListener) child;
497            addListener(listener);
498        }
499        else {
500            throw new CheckstyleException(name
501                    + " is not allowed as a child in Checker");
502        }
503    }

539    /**
540     * Sets the file extensions that identify the files that pass the
541     * filter of this FileSetCheck.
542     *
543     * @param extensions the set of file extensions. A missing
544     *     initial '.' character of an extension is automatically added.
545     */
546    public final void setFileExtensions(String... extensions) {
547        if (extensions == null) {
548            fileExtensions = null;
549        }
550        else {
551            fileExtensions = new String[extensions.length];
552            for (int i = 0; i < extensions.length; i++) {
553                final String extension = extensions[i];
554                if (CommonUtil.startsWithChar(extension, '.')) {
555                    fileExtensions[i] = extension;
556                }
557                else {
558                    fileExtensions[i] = "." + extension;
559                }
560            }
561        }
562    }

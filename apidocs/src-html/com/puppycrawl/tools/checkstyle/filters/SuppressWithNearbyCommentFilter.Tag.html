040/**
041 * <p>
042 * Filter {@code SuppressWithNearbyCommentFilter} uses nearby comments to suppress audit events.
043 * </p>
044 * <p>
045 * Rationale: Same as {@code SuppressionCommentFilter}.
046 * Whereas the SuppressionCommentFilter uses matched pairs of filters to turn
047 * on/off comment matching, {@code SuppressWithNearbyCommentFilter} uses single comments.
048 * This requires fewer lines to mark a region, and may be aesthetically preferable in some contexts.
049 * </p>
050 * <p>
051 * Attention: This filter may only be specified within the TreeWalker module
052 * ({@code &lt;module name="TreeWalker"/&gt;}) and only applies to checks which are also
053 * defined within this module. To filter non-TreeWalker checks like {@code RegexpSingleline},
054 * a
055 * <a href="https://checkstyle.org/filters/suppresswithplaintextcommentfilter.html#SuppressWithPlainTextCommentFilter">
056 * SuppressWithPlainTextCommentFilter</a> or similar filter must be used.
057 * </p>
058 * <p>
059 * SuppressWithNearbyCommentFilter can suppress Checks that have
060 * Treewalker as parent module.
061 * </p>
062 * <ul>
063 * <li>
064 * Property {@code commentFormat} - Specify comment pattern to trigger filter to begin suppression.
065 * Type is {@code java.util.regex.Pattern}.
066 * Default value is {@code "SUPPRESS CHECKSTYLE (\w+)"}.
067 * </li>
068 * <li>
069 * Property {@code checkFormat} - Specify check pattern to suppress.
070 * Type is {@code java.util.regex.Pattern}.
071 * Default value is {@code ".*"}.
072 * </li>
073 * <li>
074 * Property {@code messageFormat} - Define message pattern to suppress.
075 * Type is {@code java.util.regex.Pattern}.
076 * Default value is {@code null}.
077 * </li>
078 * <li>
079 * Property {@code idFormat} - Specify check ID pattern to suppress.
080 * Type is {@code java.util.regex.Pattern}.
081 * Default value is {@code null}.
082 * </li>
083 * <li>
084 * Property {@code influenceFormat} - Specify negative/zero/positive value that
085 * defines the number of lines preceding/at/following the suppression comment.
086 * Type is {@code java.lang.String}.
087 * Default value is {@code "0"}.
088 * </li>
089 * <li>
090 * Property {@code checkCPP} - Control whether to check C++ style comments ({@code //}).
091 * Type is {@code boolean}.
092 * Default value is {@code true}.
093 * </li>
094 * <li>
095 * Property {@code checkC} - Control whether to check C style comments ({@code &#47;* ... *&#47;}).
096 * Type is {@code boolean}.
097 * Default value is {@code true}.
098 * </li>
099 * </ul>
100 * <p>
101 * To configure a filter to suppress audit events for <i>check</i> on any line
102 * with a comment {@code SUPPRESS CHECKSTYLE <i>check</i>}:
103 * </p>
104 * <pre>
105 * &lt;module name=&quot;SuppressWithNearbyCommentFilter&quot;/&gt;
106 * </pre>
107 * <pre>
108 * private int [] array; // SUPPRESS CHECKSTYLE
109 * </pre>
110 * <p>
111 * To configure a filter to suppress all audit events on any line containing
112 * the comment {@code CHECKSTYLE IGNORE THIS LINE}:
113 * </p>
114 * <pre>
115 * &lt;module name=&quot;SuppressWithNearbyCommentFilter&quot;&gt;
116 *   &lt;property name=&quot;commentFormat&quot; value=&quot;CHECKSTYLE IGNORE THIS LINE&quot;/&gt;
117 *   &lt;property name=&quot;checkFormat&quot; value=&quot;.*&quot;/&gt;
118 *   &lt;property name=&quot;influenceFormat&quot; value=&quot;0&quot;/&gt;
119 * &lt;/module&gt;
120 * </pre>
121 * <pre>
122 * public static final int lowerCaseConstant; // CHECKSTYLE IGNORE THIS LINE
123 * </pre>
124 * <p>
125 * To configure a filter so that {@code // OK to catch (Throwable|Exception|RuntimeException) here}
126 * permits the current and previous line to avoid generating an IllegalCatch audit event:
127 * </p>
128 * <pre>
129 * &lt;module name=&quot;SuppressWithNearbyCommentFilter&quot;&gt;
130 *   &lt;property name=&quot;commentFormat&quot; value=&quot;OK to catch (\w+) here&quot;/&gt;
131 *   &lt;property name=&quot;checkFormat&quot; value=&quot;IllegalCatchCheck&quot;/&gt;
132 *   &lt;property name=&quot;messageFormat&quot; value=&quot;$1&quot;/&gt;
133 *   &lt;property name=&quot;influenceFormat&quot; value=&quot;-1&quot;/&gt;
134 * &lt;/module&gt;
135 * </pre>
136 * <pre>
137 * . . .
138 * catch (RuntimeException re) {
139 * // OK to catch RuntimeException here
140 * }
141 * catch (Throwable th) { ... }
142 * . . .
143 * </pre>
144 * <p>
145 * To configure a filter so that {@code CHECKSTYLE IGNORE <i>check</i> FOR NEXT
146 * <i>var</i> LINES} avoids triggering any audits for the given check for
147 * the current line and the next <i>var</i> lines (for a total of <i>var</i>+1 lines):
148 * </p>
149 * <pre>
150 * &lt;module name=&quot;SuppressWithNearbyCommentFilter&quot;&gt;
151 *   &lt;property name=&quot;commentFormat&quot;
152 *       value=&quot;CHECKSTYLE IGNORE (\w+) FOR NEXT (\d+) LINES&quot;/&gt;
153 *   &lt;property name=&quot;checkFormat&quot; value=&quot;$1&quot;/&gt;
154 *   &lt;property name=&quot;influenceFormat&quot; value=&quot;$2&quot;/&gt;
155 * &lt;/module&gt;
156 * </pre>
157 * <pre>
158 * static final int lowerCaseConstant; // CHECKSTYLE IGNORE ConstantNameCheck FOR NEXT 3 LINES
159 * static final int lowerCaseConstant1;
160 * static final int lowerCaseConstant2;
161 * static final int lowerCaseConstant3;
162 * static final int lowerCaseConstant4; // will warn here
163 * </pre>
164 * <p>
165 * To configure a filter to avoid any audits on code like:
166 * </p>
167 * <pre>
168 * &lt;module name=&quot;SuppressWithNearbyCommentFilter&quot;&gt;
169 *   &lt;property name=&quot;commentFormat&quot;
170 *     value=&quot;ALLOW (\\w+) ON PREVIOUS LINE&quot;/&gt;
171 *   &lt;property name=&quot;checkFormat&quot; value=&quot;$1&quot;/&gt;
172 *   &lt;property name=&quot;influenceFormat&quot; value=&quot;-1&quot;/&gt;
173 * &lt;/module&gt;
174 * </pre>
175 * <pre>
176 * private int D2;
177 * // ALLOW MemberName ON PREVIOUS LINE
178 * . . .
179 * </pre>
180 * <p>
181 * To configure a filter to allow suppress one or more Checks (separated by "|")
182 * and demand comment no less than 14 symbols:
183 * </p>
184 * <pre>
185 * &lt;module name="SuppressWithNearbyCommentFilter"&gt;
186 *   &lt;property name="commentFormat"
187 *     value="@cs\.suppress \[(\w+(\|\w+)*)\] \w[-\.'`,:;\w ]{14,}"/&gt;
188 *   &lt;property name="checkFormat" value="$1"/&gt;
189 *   &lt;property name="influenceFormat" value="1"/&gt;
190 * &lt;/module&gt;
191 * </pre>
192 * <pre>
193 * public static final int [] array; // @cs.suppress [ConstantName|NoWhitespaceAfter] A comment here
194 * </pre>
195 * <p>
196 * It is possible to specify an ID of checks, so that it can be leveraged by
197 * the SuppressWithNearbyCommentFilter to skip validations. The following examples show how to skip
198 * validations near code that has comment like {@code // @cs-: &lt;ID/&gt; (reason)},
199 * where ID is the ID of checks you want to suppress.
200 * </p>
201 * <p>
202 * Examples of Checkstyle checks configuration:
203 * </p>
204 * <pre>
205 * &lt;module name="RegexpSinglelineJava"&gt;
206 *   &lt;property name="id" value="ignore"/&gt;
207 *   &lt;property name="format" value="^.*@Ignore\s*$"/&gt;
208 *   &lt;property name="message" value="@Ignore should have a reason."/&gt;
209 * &lt;/module&gt;
210 *
211 * &lt;module name="RegexpSinglelineJava"&gt;
212 *   &lt;property name="id" value="systemout"/&gt;
213 *   &lt;property name="format" value="^.*System\.(out|err).*$"/&gt;
214 *   &lt;property name="message" value="Don't use System.out/err, use SLF4J instead."/&gt;
215 * &lt;/module&gt;
216 * </pre>
217 * <p>
218 * Example of SuppressWithNearbyCommentFilter configuration (idFormat which is set to
219 * '$1' points that ID of the checks is in the first group of commentFormat regular expressions):
220 * </p>
221 * <pre>
222 * &lt;module name="SuppressWithNearbyCommentFilter"&gt;
223 *   &lt;property name="commentFormat" value="@cs-: (\w+) \(.*\)"/&gt;
224 *   &lt;property name="idFormat" value="$1"/&gt;
225 *   &lt;property name="influenceFormat" value="0"/&gt;
226 * &lt;/module&gt;
227 * </pre>
228 * <pre>
229 * &#64;Ignore // @cs-: ignore (test has not been implemented yet)
230 * &#64;Test
231 * public void testMethod() { }
232 *
233 * public static void foo() {
234 *   System.out.println("Debug info."); // @cs-: systemout (should not fail RegexpSinglelineJava)
235 * }
236 * </pre>
237 * <p>
238 * Example of how to configure the check to suppress more than one checks.
239 * The influence format is specified in the second regexp group.
240 * </p>
241 * <pre>
242 * &lt;module name="SuppressWithNearbyCommentFilter"&gt;
243 *   &lt;property name="commentFormat" value="@cs-\: ([\w\|]+) influence (\d+)"/&gt;
244 *   &lt;property name="checkFormat" value="$1"/&gt;
245 *   &lt;property name="influenceFormat" value="$2"/&gt;
246 * &lt;/module&gt;
247 * </pre>
248 * <pre>
249 * // @cs-: ClassDataAbstractionCoupling influence 2
250 * // @cs-: MagicNumber influence 4
251 * &#64;Service // no violations from ClassDataAbstractionCoupling here
252 * &#64;Transactional
253 * public class UserService {
254 *   private int value = 10022; // no violations from MagicNumber here
255 * }
256 * </pre>
257 * <p>
258 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
259 * </p>
260 *
261 * @since 5.0
262 */
263public class SuppressWithNearbyCommentFilter
264    extends AbstractAutomaticBean
265    implements TreeWalkerFilter {

267    /** Format to turn checkstyle reporting off. */

411    @Override

413        // No code by default

416    @Override
417    public boolean accept(TreeWalkerAuditEvent event) {
418        boolean accepted = true;
419
420        if (event.getViolation() != null) {
421            // Lazy update. If the first event for the current file, update file
422            // contents and tag suppressions
423            final FileContents currentContents = event.getFileContents();
424
425            if (getFileContents() != currentContents) {
426                setFileContents(currentContents);
427                tagSuppressions();
428            }
429            if (matchesTag(event)) {
430                accepted = false;
431            }
432        }
433        return accepted;
434    }
435
436    /**
437     * Whether current event matches any tag from {@link #tags}.
438     *
439     * @param event TreeWalkerAuditEvent to test match on {@link #tags}.
440     * @return true if event matches any tag from {@link #tags}, false otherwise.
441     */
442    private boolean matchesTag(TreeWalkerAuditEvent event) {
443        boolean result = false;
444        for (final Tag tag : tags) {
445            if (tag.isMatch(event)) {
446                result = true;
447                break;
448            }
449        }
450        return result;
451    }
452
453    /**
454     * Collects all the suppression tags for all comments into a list and
455     * sorts the list.
456     */
457    private void tagSuppressions() {
458        tags.clear();
459        final FileContents contents = getFileContents();
460        if (checkCPP) {
461            tagSuppressions(contents.getSingleLineComments().values());
462        }
463        if (checkC) {
464            final Collection<List<TextBlock>> cComments =
465                contents.getBlockComments().values();
466            cComments.forEach(this::tagSuppressions);
467        }
468    }
469
470    /**
471     * Appends the suppressions in a collection of comments to the full
472     * set of suppression tags.
473     *
474     * @param comments the set of comments.
475     */
476    private void tagSuppressions(Collection<TextBlock> comments) {
477        for (final TextBlock comment : comments) {
478            final int startLineNo = comment.getStartLineNo();
479            final String[] text = comment.getText();
480            tagCommentLine(text[0], startLineNo);
481            for (int i = 1; i < text.length; i++) {
482                tagCommentLine(text[i], startLineNo + i);
483            }
484        }
485    }
486
487    /**
488     * Tags a string if it matches the format for turning
489     * checkstyle reporting on or the format for turning reporting off.
490     *
491     * @param text the string to tag.
492     * @param line the line number of text.
493     */
494    private void tagCommentLine(String text, int line) {
495        final Matcher matcher = commentFormat.matcher(text);
496        if (matcher.find()) {
497            addTag(matcher.group(0), line);
498        }
499    }

501    /**
502     * Adds a comment suppression {@code Tag} to the list of all tags.
503     *
504     * @param text the text of the tag.
505     * @param line the line number of the tag.
506     */
507    private void addTag(String text, int line) {
508        final Tag tag = new Tag(text, line, this);
509        tags.add(tag);
510    }

512    /**
513     * A Tag holds a suppression comment and its location.
514     */
515    private static final class Tag {
516
517        /** The text of the tag. */
518        private final String text;
519
520        /** The first line where warnings may be suppressed. */
521        private final int firstLine;
522
523        /** The last line where warnings may be suppressed. */
524        private final int lastLine;
525
526        /** The parsed check regexp, expanded for the text of this tag. */
527        private final Pattern tagCheckRegexp;
528
529        /** The parsed message regexp, expanded for the text of this tag. */
530        private final Pattern tagMessageRegexp;
531
532        /** The parsed check ID regexp, expanded for the text of this tag. */
533        private final Pattern tagIdRegexp;
534
535        /**
536         * Constructs a tag.
537         *
538         * @param text the text of the suppression.
539         * @param line the line number.
540         * @param filter the {@code SuppressWithNearbyCommentFilter} with the context
541         * @throws IllegalArgumentException if unable to parse expanded text.
542         */
543        private Tag(String text, int line, SuppressWithNearbyCommentFilter filter) {
544            this.text = text;
545
546            // Expand regexp for check and message
547            // Does not intern Patterns with Utils.getPattern()
548            String format = "";
549            try {
550                format = CommonUtil.fillTemplateWithStringsByRegexp(
551                        filter.checkFormat, text, filter.commentFormat);
552                tagCheckRegexp = Pattern.compile(format);
553                if (filter.messageFormat == null) {
554                    tagMessageRegexp = null;
555                }
556                else {
557                    format = CommonUtil.fillTemplateWithStringsByRegexp(
558                            filter.messageFormat, text, filter.commentFormat);
559                    tagMessageRegexp = Pattern.compile(format);
560                }
561                if (filter.idFormat == null) {
562                    tagIdRegexp = null;
563                }
564                else {
565                    format = CommonUtil.fillTemplateWithStringsByRegexp(
566                            filter.idFormat, text, filter.commentFormat);
567                    tagIdRegexp = Pattern.compile(format);
568                }
569                format = CommonUtil.fillTemplateWithStringsByRegexp(
570                        filter.influenceFormat, text, filter.commentFormat);
571
572                final int influence = parseInfluence(format, filter.influenceFormat, text);
573
574                if (influence >= 1) {
575                    firstLine = line;
576                    lastLine = line + influence;
577                }
578                else {
579                    firstLine = line + influence;
580                    lastLine = line;
581                }
582            }
583            catch (final PatternSyntaxException ex) {
584                throw new IllegalArgumentException(
585                    "unable to parse expanded comment " + format, ex);
586            }
587        }
588
589        /**
590         * Gets influence from suppress filter influence format param.
591         *
592         * @param format          influence format to parse
593         * @param influenceFormat raw influence format
594         * @param text            text of the suppression
595         * @return parsed influence
596         * @throws IllegalArgumentException when unable to parse int in format
597         */
598        private static int parseInfluence(String format, String influenceFormat, String text) {
599            try {
600                return Integer.parseInt(format);
601            }
602            catch (final NumberFormatException ex) {
603                throw new IllegalArgumentException("unable to parse influence from '" + text
604                        + "' using " + influenceFormat, ex);
605            }
606        }
607
608        @Override
609        public boolean equals(Object other) {
610            if (this == other) {
611                return true;
612            }
613            if (other == null || getClass() != other.getClass()) {
614                return false;
615            }
616            final Tag tag = (Tag) other;
617            return Objects.equals(firstLine, tag.firstLine)
618                    && Objects.equals(lastLine, tag.lastLine)
619                    && Objects.equals(text, tag.text)
620                    && Objects.equals(tagCheckRegexp, tag.tagCheckRegexp)
621                    && Objects.equals(tagMessageRegexp, tag.tagMessageRegexp)
622                    && Objects.equals(tagIdRegexp, tag.tagIdRegexp);
623        }
624
625        @Override
626        public int hashCode() {
627            return Objects.hash(text, firstLine, lastLine, tagCheckRegexp, tagMessageRegexp,
628                    tagIdRegexp);
629        }
630
631        /**
632         * Determines whether the source of an audit event
633         * matches the text of this tag.
634         *
635         * @param event the {@code TreeWalkerAuditEvent} to check.
636         * @return true if the source of event matches the text of this tag.
637         */
638        public boolean isMatch(TreeWalkerAuditEvent event) {
639            return isInScopeOfSuppression(event)
640                    && isCheckMatch(event)
641                    && isIdMatch(event)
642                    && isMessageMatch(event);
643        }
644
645        /**
646         * Checks whether the {@link TreeWalkerAuditEvent} is in the scope of the suppression.
647         *
648         * @param event {@link TreeWalkerAuditEvent} instance.
649         * @return true if the {@link TreeWalkerAuditEvent} is in the scope of the suppression.
650         */
651        private boolean isInScopeOfSuppression(TreeWalkerAuditEvent event) {
652            final int line = event.getLine();
653            return line >= firstLine && line <= lastLine;
654        }
655
656        /**
657         * Checks whether {@link TreeWalkerAuditEvent} source name matches the check format.
658         *
659         * @param event {@link TreeWalkerAuditEvent} instance.
660         * @return true if the {@link TreeWalkerAuditEvent} source name matches the check format.
661         */
662        private boolean isCheckMatch(TreeWalkerAuditEvent event) {
663            final Matcher checkMatcher = tagCheckRegexp.matcher(event.getSourceName());
664            return checkMatcher.find();
665        }
666
667        /**
668         * Checks whether the {@link TreeWalkerAuditEvent} module ID matches the ID format.
669         *
670         * @param event {@link TreeWalkerAuditEvent} instance.
671         * @return true if the {@link TreeWalkerAuditEvent} module ID matches the ID format.
672         */
673        private boolean isIdMatch(TreeWalkerAuditEvent event) {
674            boolean match = true;
675            if (tagIdRegexp != null) {
676                if (event.getModuleId() == null) {
677                    match = false;
678                }
679                else {
680                    final Matcher idMatcher = tagIdRegexp.matcher(event.getModuleId());
681                    match = idMatcher.find();
682                }
683            }
684            return match;
685        }

041/**
042 * <p>
043 * Filter {@code SuppressWithNearbyTextFilter} uses plain text to suppress
044 * nearby audit events. The filter can suppress all checks which have Checker as a parent module.
045 * </p>
046 * <p>
047 * Setting {@code .*} value to {@code nearbyTextPattern} property will see <b>any</b>
048 * text as a suppression and will likely suppress all audit events in the file. It is
049 * best to set this to a key phrase not commonly used in the file to help denote it
050 * out of the rest of the file as a suppression. See the default value as an example.
051 * </p>
052 * <ul>
053 * <li>
054 * Property {@code nearbyTextPattern} - Specify nearby text
055 * pattern to trigger filter to begin suppression.
056 * Type is {@code java.util.regex.Pattern}.
057 * Default value is {@code "SUPPRESS CHECKSTYLE (\w+)"}.
058 * </li>
059 * <li>
060 * Property {@code checkPattern} - Specify check name pattern to suppress.
061 * Property can also be a RegExp group index at {@code nearbyTextPattern} in
062 * format of {@code $x} and be picked from line that matches {@code nearbyTextPattern}.
063 * Type is {@code java.util.regex.Pattern}.
064 * Default value is {@code ".*"}.
065 * </li>
066 * <li>
067 * Property {@code messagePattern} - Specify check violation message pattern to suppress.
068 * Type is {@code java.util.regex.Pattern}.
069 * Default value is {@code null}.
070 * </li>
071 * <li>
072 * Property {@code idPattern} - Specify check ID pattern to suppress.
073 * Type is {@code java.util.regex.Pattern}.
074 * Default value is {@code null}.
075 * </li>
076 * <li>
077 * Property {@code lineRange} - Specify negative/zero/positive value that
078 * defines the number of lines preceding/at/following the suppressing nearby text.
079 * Property can also be a RegExp group index at {@code nearbyTextPattern} in
080 * format of {@code $x} and be picked from line that matches {@code nearbyTextPattern}.
081 * Type is {@code java.lang.String}.
082 * Default value is {@code "0"}.
083 * </li>
084 * </ul>
085 * <p>
086 * To configure the filter to suppress audit events on the same line:
087 * </p>
088 * <pre>
089 * &lt;module name=&quot;Checker&quot;&gt;
090 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;/&gt;
091 *   &lt;module name=&quot;TreeWalker&quot;&gt;
092 *     &lt;module name=&quot;MagicNumber&quot;/&gt;
093 *   &lt;/module&gt;
094 * &lt;/module&gt;
095 * </pre>
096 * <p>
097 * Example:
098 * </p>
099 * <pre>
100 * int hoursInDay = 24; // SUPPRESS CHECKSTYLE because it is too obvious
101 * int daysInWeek = 7; // violation
102 * </pre>
103 * <p>
104 * To configure the filter to suppress audit events on any line that contains
105 * {@code DO NOT CHECK THIS LINE}:
106 * </p>
107 * <pre>
108 * &lt;module name=&quot;Checker&quot;&gt;
109 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
110 *     &lt;property name=&quot;nearbyTextPattern&quot; value=&quot;DO NOT CHECK THIS LINE&quot;/&gt;
111 *   &lt;/module&gt;
112 *   &lt;module name=&quot;TreeWalker&quot;&gt;
113 *     &lt;module name=&quot;MagicNumber&quot;/&gt;
114 *   &lt;/module&gt;
115 * &lt;/module&gt;
116 * </pre>
117 * <p>
118 * Example:
119 * </p>
120 * <pre>
121 * int a = 42; // DO NOT CHECK THIS LINE
122 * int b = 43; // violation
123 * </pre>
124 * <p>
125 * To configure the filter to suppress audit events whose check message contains
126 * the word {@code Line}. In this case, {@code LineLengthCheck}'s violation message
127 * contains it:
128 * </p>
129 * <pre>
130 * &lt;module name=&quot;Checker&quot;&gt;
131 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
132 *     &lt;property name=&quot;messagePattern&quot; value=&quot;.*Line.*&quot;/&gt;
133 *     &lt;property name=&quot;nearbyTextPattern&quot; value=&quot;.*&quot;/&gt;
134 *   &lt;/module&gt;
135 *   &lt;module name=&quot;LineLength&quot;&gt;
136 *     &lt;property name=&quot;max&quot; value=&quot;10&quot;/&gt;
137 *   &lt;/module&gt;
138 * &lt;/module&gt;
139 * </pre>
140 * <p>
141 * Example:
142 * </p>
143 * <pre>
144 * export FOO=BAR # ok, because violation message is matching suppress pattern
145 * </pre>
146 * <p>
147 * To configure the filter to suppress audit events only on a check whose id is
148 * {@code ignoreMe}:
149 * </p>
150 * <pre>
151 * &lt;module name=&quot;Checker&quot;&gt;
152 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
153 *     &lt;property name=&quot;idPattern&quot; value=&quot;ignoreMe&quot;/&gt;
154 *   &lt;/module&gt;
155 *   &lt;module name=&quot;LineLength&quot;&gt;
156 *     &lt;property name=&quot;max&quot; value=&quot;55&quot;/&gt;
157 *   &lt;/module&gt;
158 *   &lt;module name=&quot;TreeWalker&quot;&gt;
159 *     &lt;module name=&quot;MagicNumber&quot;&gt;
160 *       &lt;property name=&quot;id&quot; value=&quot;ignoreMe&quot;/&gt;
161 *     &lt;/module&gt;
162 *   &lt;/module&gt;
163 * &lt;/module&gt;
164 * </pre>
165 * <p>
166 * Example:
167 * </p>
168 * <pre>
169 * int a = 42; // SUPPRESS CHECKSTYLE because i want to
170 * static final int LONG_VAR_NAME_TO_TAKE_MORE_THAN_55_CHARS = 22; // LineLength violation
171 * </pre>
172 * <p>
173 * To configure the filter to suppress audit events for the current and next 2 lines:
174 * </p>
175 * <pre>
176 * &lt;module name=&quot;Checker&quot;&gt;
177 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
178 *     &lt;property name=&quot;lineRange&quot; value=&quot;2&quot;/&gt;
179 *   &lt;/module&gt;
180 *   &lt;module name=&quot;UniqueProperties&quot;/&gt;
181 * &lt;/module&gt;
182 * </pre>
183 * <p>
184 * Example:
185 * </p>
186 * <pre>
187 * key.one=41 # SUPPRESS CHECKSTYLE because I want to
188 * key.one=42 # ok
189 * key.one=43 # ok
190 * key.one=44 # violation
191 * </pre>
192 * <p>
193 * To configure the filter to suppress audit events for the current and previous line:
194 * </p>
195 * <pre>
196 * &lt;module name=&quot;Checker&quot;&gt;
197 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
198 *     &lt;property name=&quot;lineRange&quot; value=&quot;-1&quot;/&gt;
199 *   &lt;/module&gt;
200 *   &lt;module name=&quot;UniqueProperties&quot;/&gt;
201 * &lt;/module&gt;
202 * </pre>
203 * <p>
204 * Example:
205 * </p>
206 * <pre>
207 * key.one=41 # violation
208 * key.one=42 # ok
209 * key.one=43 # SUPPRESS CHECKSTYLE because I want to
210 * key.one=44 # violation
211 * </pre>
212 * <p>
213 * To configure the filter with a more compact {@code nearbyTextPattern}
214 * to accept variable {@code checkPattern}:
215 * </p>
216 * <pre>
217 * &lt;module name=&quot;Checker&quot;&gt;
218 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
219 *     &lt;property name=&quot;nearbyTextPattern&quot;
220 *       value=&quot;-@cs\[(\w+)\] (\w+)&quot;/&gt;
221 *     &lt;property name=&quot;checkPattern&quot; value=&quot;$1&quot;/&gt;
222 *   &lt;/module&gt;
223 *   &lt;module name=&quot;TreeWalker&quot;&gt;
224 *     &lt;module name=&quot;MagicNumber&quot;/&gt;
225 *   &lt;/module&gt;
226 * &lt;/module&gt;
227 * </pre>
228 * <p>
229 * Example:
230 * </p>
231 * <pre>
232 * int a = 42; // -@cs[MagicNumber] We do not consider this number as magic for some reason.
233 * int b = 43; // violation
234 * </pre>
235 * <p>
236 * To configure the filter to accept variable {@code checkPattern} and {@code lineRange}:
237 * </p>
238 * <pre>
239 * &lt;module name=&quot;Checker&quot;&gt;
240 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
241 *     &lt;property name=&quot;nearbyTextPattern&quot;
242 *       value=&quot;@cs-: (\w+) for ([+-]\d+) lines&quot;/&gt;
243 *     &lt;property name=&quot;checkPattern&quot; value=&quot;$1&quot;/&gt;
244 *     &lt;property name=&quot;lineRange&quot; value=&quot;$2&quot;/&gt;
245 *   &lt;/module&gt;
246 *   &lt;module name=&quot;TreeWalker&quot;&gt;
247 *     &lt;module name=&quot;MagicNumber&quot;/&gt;
248 *   &lt;/module&gt;
249 * &lt;/module&gt;
250 * </pre>
251 * <p>
252 * Example:
253 * </p>
254 * <pre>
255 * int a = 42; // @cs-: MagicNumber for +3 lines
256 * int b = 43;
257 * int c = 44;
258 * int d = 45;
259 * int e = 46; // violation
260 * </pre>
261 * <p>
262 * To configure the filter to suppress {@code LineLength} violations for lines containing a URL:
263 * </p>
264 * <pre>
265 * &lt;module name=&quot;Checker&quot;&gt;
266 *   &lt;module name=&quot;SuppressWithNearbyTextFilter&quot;&gt;
267 *     &lt;property name=&quot;checkPattern&quot; value=&quot;LineLength&quot;/&gt;
268 *     &lt;property name=&quot;nearbyTextPattern&quot;
269 *       value=&quot;&amp;lt;a href=&amp;quot;[^&amp;quot;]+&amp;quot;&amp;gt;&quot;/&gt;
270 *   &lt;/module&gt;
271 *   &lt;module name=&quot;LineLength&quot;&gt;
272 *     &lt;property name=&quot;max&quot; value=&quot;90&quot;/&gt;
273 *   &lt;/module&gt;
274 * &lt;/module&gt;
275 * </pre>
276 * <p>
277 * Example:
278 * </p>
279 * <pre>
280 * /&#42;&#42;
281 *  &#42; Flag description.
282 *  &#42;
283 *  &#42; Disabled until &lt;a href=&quot;www.github.com/owner/repo/issue/9#comment&quot;&gt; // ok
284 *  &#42;/
285 * public static final boolean SOME_FLAG = false;
286 * </pre>
287 * <p>
288 * Parent is {@code com.puppycrawl.tools.checkstyle.Checker}
289 * </p>
290 *
291 * @since 10.10.0
292 */
293public class SuppressWithNearbyTextFilter extends AbstractAutomaticBean implements Filter {
294
295    /** Default nearby text pattern to turn check reporting off. */
296    private static final String DEFAULT_NEARBY_TEXT_PATTERN = "SUPPRESS CHECKSTYLE (\\w+)";
297
298    /** Default regex for checks that should be suppressed. */
299    private static final String DEFAULT_CHECK_PATTERN = ".*";
300
301    /** Default number of lines that should be suppressed. */
302    private static final String DEFAULT_LINE_RANGE = "0";
303
304    /** Suppressions encountered in current file. */
305    private final List<Suppression> suppressions = new ArrayList<>();
306
307    /** Specify nearby text pattern to trigger filter to begin suppression. */
308    @XdocsPropertyType(PropertyType.PATTERN)
309    private Pattern nearbyTextPattern = Pattern.compile(DEFAULT_NEARBY_TEXT_PATTERN);
310
311    /**
312     * Specify check name pattern to suppress. Property can also be a RegExp group index
313     * at {@code nearbyTextPattern} in format of {@code $x} and be picked from line that
314     * matches {@code nearbyTextPattern}.
315     */
316    @XdocsPropertyType(PropertyType.PATTERN)
317    private String checkPattern = DEFAULT_CHECK_PATTERN;
318
319    /** Specify check violation message pattern to suppress. */
320    @XdocsPropertyType(PropertyType.PATTERN)
321    private String messagePattern;
322
323    /** Specify check ID pattern to suppress. */
324    @XdocsPropertyType(PropertyType.PATTERN)
325    private String idPattern;
326
327    /**
328     * Specify negative/zero/positive value that defines the number of lines
329     * preceding/at/following the suppressing nearby text. Property can also be a RegExp group
330     * index at {@code nearbyTextPattern} in format of {@code $x} and be picked
331     * from line that matches {@code nearbyTextPattern}.
332     */
333    private String lineRange = DEFAULT_LINE_RANGE;
334
335    /** The absolute path to the currently processed file. */
336    private String cachedFileAbsolutePath = "";
337
338    /**
339     * Setter to specify nearby text pattern to trigger filter to begin suppression.
340     *
341     * @param pattern a {@code Pattern} value.
342     */
343    public final void setNearbyTextPattern(Pattern pattern) {
344        nearbyTextPattern = pattern;
345    }
346
347    /**
348     * Setter to specify check name pattern to suppress. Property can also
349     * be a RegExp group index at {@code nearbyTextPattern} in
350     * format of {@code $x} and be picked from line that matches {@code nearbyTextPattern}.
351     *
352     * @param pattern a {@code String} value.
353     */
354    public final void setCheckPattern(String pattern) {
355        checkPattern = pattern;
356    }
357
358    /**
359     * Setter to specify check violation message pattern to suppress.
360     *
361     * @param pattern a {@code String} value.
362     */
363    public void setMessagePattern(String pattern) {
364        messagePattern = pattern;
365    }
366
367    /**
368     * Setter to specify check ID pattern to suppress.
369     *
370     * @param pattern a {@code String} value.
371     */
372    public void setIdPattern(String pattern) {
373        idPattern = pattern;
374    }
375
376    /**
377     * Setter to specify negative/zero/positive value that defines the number
378     * of lines preceding/at/following the suppressing nearby text. Property can also
379     * be a RegExp group index at {@code nearbyTextPattern} in
380     * format of {@code $x} and be picked from line that matches {@code nearbyTextPattern}.
381     *
382     * @param format a {@code String} value.
383     */
384    public final void setLineRange(String format) {
385        lineRange = format;
386    }
387
388    @Override
389    public boolean accept(AuditEvent event) {
390        boolean accepted = true;
391
392        if (event.getViolation() != null) {
393            final String eventFileTextAbsolutePath = event.getFileName();
394
395            if (!cachedFileAbsolutePath.equals(eventFileTextAbsolutePath)) {
396                final FileText currentFileText = getFileText(eventFileTextAbsolutePath);
397
398                if (currentFileText != null) {
399                    cachedFileAbsolutePath = currentFileText.getFile().getAbsolutePath();
400                    collectSuppressions(currentFileText);
401                }
402            }
403
404            final Optional<Suppression> nearestSuppression =
405                    getNearestSuppression(suppressions, event);
406            accepted = nearestSuppression.isEmpty();
407        }
408        return accepted;
409    }
410
411    @Override
412    protected void finishLocalSetup() {
413        // No code by default
414    }
415
416    /**
417     * Returns {@link FileText} instance created based on the given file name.
418     *
419     * @param fileName the name of the file.
420     * @return {@link FileText} instance.
421     * @throws IllegalStateException if the file could not be read.
422     */
423    private static FileText getFileText(String fileName) {
424        final File file = new File(fileName);
425        FileText result = null;
426
427        // some violations can be on a directory, instead of a file
428        if (!file.isDirectory()) {
429            try {
430                result = new FileText(file, StandardCharsets.UTF_8.name());
431            }
432            catch (IOException ex) {
433                throw new IllegalStateException("Cannot read source file: " + fileName, ex);
434            }
435        }
436
437        return result;
438    }
439
440    /**
441     * Collets all {@link Suppression} instances retrieved from the given {@link FileText}.
442     *
443     * @param fileText {@link FileText} instance.
444     */
445    private void collectSuppressions(FileText fileText) {
446        suppressions.clear();
447
448        for (int lineNo = 0; lineNo < fileText.size(); lineNo++) {
449            final Suppression suppression = getSuppression(fileText, lineNo);
450            if (suppression != null) {
451                suppressions.add(suppression);
452            }
453        }
454    }
455
456    /**
457     * Tries to extract the suppression from the given line.
458     *
459     * @param fileText {@link FileText} instance.
460     * @param lineNo line number.
461     * @return {@link Suppression} instance.
462     */
463    private Suppression getSuppression(FileText fileText, int lineNo) {
464        final String line = fileText.get(lineNo);
465        final Matcher nearbyTextMatcher = nearbyTextPattern.matcher(line);
466
467        Suppression suppression = null;
468        if (nearbyTextMatcher.find()) {
469            final String text = nearbyTextMatcher.group(0);
470            suppression = new Suppression(text, lineNo + 1, this);
471        }
472
473        return suppression;
474    }
475
476    /**
477     * Finds the nearest {@link Suppression} instance which can suppress
478     * the given {@link AuditEvent}. The nearest suppression is the suppression which scope
479     * is before the line and column of the event.
480     *
481     * @param suppressions collection of {@link Suppression} instances.
482     * @param event {@link AuditEvent} instance.
483     * @return {@link Suppression} instance.
484     */
485    private static Optional<Suppression> getNearestSuppression(Collection<Suppression> suppressions,
486                                                               AuditEvent event) {
487        return suppressions
488                .stream()
489                .filter(suppression -> suppression.isMatch(event))
490                .findFirst();
491    }
492
493    /** The class which represents the suppression. */
494    private static final class Suppression {
495
496        /** The first line where warnings may be suppressed. */
497        private final int firstLine;
498
499        /** The last line where warnings may be suppressed. */
500        private final int lastLine;
501
502        /** The regexp which is used to match the event source.*/
503        private final Pattern eventSourceRegexp;
504
505        /** The regexp which is used to match the event message.*/
506        private Pattern eventMessageRegexp;
507
508        /** The regexp which is used to match the event ID.*/
509        private Pattern eventIdRegexp;
510
511        /**
512         * Constructs new {@code Suppression} instance.
513         *
514         * @param text suppression text.
515         * @param lineNo suppression line number.
516         * @param filter the {@code SuppressWithNearbyTextFilter} with the context.
517         * @throws IllegalArgumentException if there is an error in the filter regex syntax.
518         */
519        private Suppression(
520                String text,
521                int lineNo,
522                SuppressWithNearbyTextFilter filter
523        ) {
524            final Pattern nearbyTextPattern = filter.nearbyTextPattern;
525            final String lineRange = filter.lineRange;
526            String format = "";
527            try {
528                format = CommonUtil.fillTemplateWithStringsByRegexp(
529                        filter.checkPattern, text, nearbyTextPattern);
530                eventSourceRegexp = Pattern.compile(format);
531                if (filter.messagePattern != null) {
532                    format = CommonUtil.fillTemplateWithStringsByRegexp(
533                            filter.messagePattern, text, nearbyTextPattern);
534                    eventMessageRegexp = Pattern.compile(format);
535                }
536                if (filter.idPattern != null) {
537                    format = CommonUtil.fillTemplateWithStringsByRegexp(
538                            filter.idPattern, text, nearbyTextPattern);
539                    eventIdRegexp = Pattern.compile(format);
540                }
541                format = CommonUtil.fillTemplateWithStringsByRegexp(lineRange,
542                                                                    text, nearbyTextPattern);
543
544                final int range = parseRange(format, lineRange, text);
545
546                firstLine = Math.min(lineNo, lineNo + range);
547                lastLine = Math.max(lineNo, lineNo + range);
548            }
549            catch (final PatternSyntaxException ex) {
550                throw new IllegalArgumentException(
551                    "unable to parse expanded comment " + format, ex);
552            }
553        }
554
555        /**
556         * Gets range from suppress filter range format param.
557         *
558         * @param format range format to parse
559         * @param lineRange raw line range
560         * @param text text of the suppression
561         * @return parsed range
562         * @throws IllegalArgumentException when unable to parse int in format
563         */
564        private static int parseRange(String format, String lineRange, String text) {
565            try {
566                return Integer.parseInt(format);
567            }
568            catch (final NumberFormatException ex) {
569                throw new IllegalArgumentException("unable to parse line range from '" + text
570                        + "' using " + lineRange, ex);
571            }
572        }
573
574        /**
575         * Determines whether the source of an audit event
576         * matches the text of this suppression.
577         *
578         * @param event the {@code AuditEvent} to check.
579         * @return true if the source of event matches the text of this suppression.
580         */
581        private boolean isMatch(AuditEvent event) {
582            return isInScopeOfSuppression(event)
583                    && isCheckMatch(event)
584                    && isIdMatch(event)
585                    && isMessageMatch(event);
586        }
587
588        /**
589         * Checks whether the {@link AuditEvent} is in the scope of the suppression.
590         *
591         * @param event {@link AuditEvent} instance.
592         * @return true if the {@link AuditEvent} is in the scope of the suppression.
593         */
594        private boolean isInScopeOfSuppression(AuditEvent event) {
595            final int eventLine = event.getLine();
596            return eventLine >= firstLine && eventLine <= lastLine;
597        }
598
599        /**
600         * Checks whether {@link AuditEvent} source name matches the check pattern.
601         *
602         * @param event {@link AuditEvent} instance.
603         * @return true if the {@link AuditEvent} source name matches the check pattern.
604         */
605        private boolean isCheckMatch(AuditEvent event) {
606            final Matcher checkMatcher = eventSourceRegexp.matcher(event.getSourceName());
607            return checkMatcher.find();
608        }
609
610        /**
611         * Checks whether the {@link AuditEvent} module ID matches the ID pattern.
612         *
613         * @param event {@link AuditEvent} instance.
614         * @return true if the {@link AuditEvent} module ID matches the ID pattern.
615         */
616        private boolean isIdMatch(AuditEvent event) {
617            boolean match = true;
618            if (eventIdRegexp != null) {
619                if (event.getModuleId() == null) {
620                    match = false;
621                }
622                else {
623                    final Matcher idMatcher = eventIdRegexp.matcher(event.getModuleId());
624                    match = idMatcher.find();
625                }
626            }
627            return match;
628        }
629
630        /**
631         * Checks whether the {@link AuditEvent} message matches the message pattern.
632         *
633         * @param event {@link AuditEvent} instance.
634         * @return true if the {@link AuditEvent} message matches the message pattern.
635         */
636        private boolean isMessageMatch(AuditEvent event) {
637            boolean match = true;
638            if (eventMessageRegexp != null) {
639                final Matcher messageMatcher = eventMessageRegexp.matcher(event.getMessage());
640                match = messageMatcher.find();
641            }
642            return match;
643        }
644    }

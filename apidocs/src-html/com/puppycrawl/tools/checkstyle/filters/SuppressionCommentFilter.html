041/**
042 * <p>
043 * Filter {@code SuppressionCommentFilter} uses pairs of comments to suppress audit events.
044 * </p>
045 * <p>
046 * Rationale:
047 * Sometimes there are legitimate reasons for violating a check. When
048 * this is a matter of the code in question and not personal
049 * preference, the best place to override the policy is in the code
050 * itself. Semi-structured comments can be associated with the check.
051 * This is sometimes superior to a separate suppressions file, which
052 * must be kept up-to-date as the source file is edited.
053 * </p>
054 * <p>
055 * Note that the suppression comment should be put before the violation.
056 * You can use more than one suppression comment each on separate line.
057 * </p>
058 * <p>
059 * Attention: This filter may only be specified within the TreeWalker module
060 * ({@code &lt;module name="TreeWalker"/&gt;}) and only applies to checks which are also
061 * defined within this module. To filter non-TreeWalker checks like {@code RegexpSingleline}, a
062 * <a href="https://checkstyle.org/filters/suppresswithplaintextcommentfilter.html#SuppressWithPlainTextCommentFilter">
063 * SuppressWithPlainTextCommentFilter</a> or similar filter must be used.
064 * </p>
065 * <p>
066 * {@code offCommentFormat} and {@code onCommentFormat} must have equal
067 * <a href="https://docs.oracle.com/en/java/javase/11/docs/api/java.base/java/util/regex/Matcher.html#groupCount()">
068 * paren counts</a>.
069 * </p>
070 * <p>
071 * SuppressionCommentFilter can suppress Checks that have Treewalker as parent module.
072 * </p>
073 * <ul>
074 * <li>
075 * Property {@code offCommentFormat} - Specify comment pattern to
076 * trigger filter to begin suppression.
077 * Type is {@code java.util.regex.Pattern}.
078 * Default value is {@code "CHECKSTYLE:OFF"}.
079 * </li>
080 * <li>
081 * Property {@code onCommentFormat} - Specify comment pattern to trigger filter to end suppression.
082 * Type is {@code java.util.regex.Pattern}.
083 * Default value is {@code "CHECKSTYLE:ON"}.
084 * </li>
085 * <li>
086 * Property {@code checkFormat} - Specify check pattern to suppress.
087 * Type is {@code java.util.regex.Pattern}.
088 * Default value is {@code ".*"}.
089 * </li>
090 * <li>
091 * Property {@code messageFormat} - Specify message pattern to suppress.
092 * Type is {@code java.util.regex.Pattern}.
093 * Default value is {@code null}.
094 * </li>
095 * <li>
096 * Property {@code idFormat} - Specify check ID pattern to suppress.
097 * Type is {@code java.util.regex.Pattern}.
098 * Default value is {@code null}.
099 * </li>
100 * <li>
101 * Property {@code checkCPP} - Control whether to check C++ style comments ({@code //}).
102 * Type is {@code boolean}.
103 * Default value is {@code true}.
104 * </li>
105 * <li>
106 * Property {@code checkC} - Control whether to check C style comments ({@code &#47;* ... *&#47;}).
107 * Type is {@code boolean}.
108 * Default value is {@code true}.
109 * </li>
110 * </ul>
111 * <p>
112 * To configure a filter to suppress audit events (MemberNameCheck,
113 * ConstantNameCheck and IllegalCatchCheck have been taken here for reference)
114 * between a comment containing {@code CHECKSTYLE:OFF} and a comment
115 * containing {@code CHECKSTYLE:ON}:
116 * </p>
117 * <pre>
118 *   &lt;module name="SuppressionCommentFilter"/&gt;
119 *   &lt;module name="MemberName"/&gt;
120 *   &lt;module name="ConstantName"/&gt;
121 *   &lt;module name="IllegalCatch"/&gt;
122 * </pre>
123 * <pre>
124 * class InputSuppressionCommentFilter
125 * {
126 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
127 *
128 *  //CHECKSTYLE:OFF
129 *  int VAR2; // suppressed violation
130 *  //CHECKSTYLE:ON
131 *
132 *  public static final int var3;
133 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
134 *
135 *  //CHECKSTYLE:OFF
136 *  public static final int var4; // suppressed violation
137 *  //CHECKSTYLE:ON
138 *
139 *  public void method1()
140 *  {
141 *    try {}
142 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
143 *
144 *    //CHECKSTYLE:OFF
145 *
146 *    try {}
147 *    catch(Exception ex) {} // suppressed violation
148 *    catch(Error err) {} // suppressed violation
149 *
150 *    //CHECKSTYLE:ON
151 *  }
152 * }
153 * </pre>
154 * <p>
155 * To configure a filter so that {@code // stop constant check} and
156 * {@code // resume constant check} marks legitimate constant names:
157 * </p>
158 * <pre>
159 * &lt;module name="SuppressionCommentFilter"&gt;
160 *   &lt;property name="offCommentFormat" value="stop constant check"/&gt;
161 *   &lt;property name="onCommentFormat" value="resume constant check"/&gt;
162 *   &lt;property name="checkFormat" value="ConstantNameCheck"/&gt;
163 * &lt;/module&gt;
164 * &lt;module name="MemberName"/&gt;
165 * &lt;module name="ConstantName"/&gt;
166 * &lt;module name="IllegalCatch"/&gt;
167 * </pre>
168 * <pre>
169 * class InputSuppressionCommentFilter
170 * {
171 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
172 *
173 *  //stop constant check
174 *  int VAR2; // violation , Name 'VAR2' must match pattern '^[a-z][a-zA-Z0-9]*$'
175 *  //resume constant check
176 *
177 *  public static final int var3;
178 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
179 *
180 *  //stop constant check
181 *  public static final int var4; // suppressed violation
182 *  //resume constant check
183 *
184 *  public void method1()
185 *  {
186 *    try {}
187 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
188 *
189 *    //stop constant check
190 *
191 *    try {}
192 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
193 *    catch(Error err) {} // violation , Catching 'Error' is not allowed
194 *
195 *    //resume constant check
196 *  }
197 * }
198 * </pre>
199 * <p>
200 * To configure a filter so that {@code UNUSED OFF: <i>var</i>} and
201 * {@code UNUSED ON: <i>var</i>} marks a variable or parameter known not to be
202 * used by the code by matching the variable name in the message through a
203 * specified message in messageFormat:
204 * </p>
205 * <pre>
206 * &lt;module name="SuppressionCommentFilter"&gt;
207 *   &lt;property name="offCommentFormat" value="ILLEGAL OFF\: (\w+)"/&gt;
208 *   &lt;property name="onCommentFormat" value="ILLEGAL ON\: (\w+)"/&gt;
209 *   &lt;property name="checkFormat" value="IllegalCatch"/&gt;
210 *   &lt;property name="messageFormat" value="^Catching '$1' is not allowed.$"/&gt;
211 * &lt;/module&gt;
212 * &lt;module name="MemberName"/&gt;
213 * &lt;module name="ConstantName"/&gt;
214 * &lt;module name="IllegalCatch"/&gt;
215 * </pre>
216 * <pre>
217 * class InputSuppressionCommentFilter
218 * {
219 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
220 *
221 *  //ILLEGAL OFF: Exception
222 *  int VAR2; // violation , Name 'VAR2' must match pattern '^[a-z][a-zA-Z0-9]*$'
223 *  //ILLEGAL ON: Exception
224 *
225 *  public static final int var3;
226 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
227 *
228 *  //ILLEGAL OFF: Exception
229 *  public static final int var4;
230 *  // violation above ,  Name 'var4' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
231 *  //ILLEGAL ON: Exception
232 *
233 *  public void method1()
234 *  {
235 *    try {}
236 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
237 *
238 *    //ILLEGAL OFF: Exception
239 *
240 *    try {}
241 *    catch(Exception ex) {} // suppressed violation
242 *    catch(Error err) {} // violation , Catching 'Error' is not allowed
243 *
244 *    //ILLEGAL ON: Exception
245 *  }
246 * }
247 * </pre>
248 * <p>
249 * To configure a filter so that name of suppressed check mentioned in comment
250 * {@code CSOFF: <i>regexp</i>} and {@code CSON: <i>regexp</i>} mark a matching check:
251 * </p>
252 * <pre>
253 * &lt;module name="SuppressionCommentFilter"&gt;
254 *   &lt;property name="offCommentFormat" value="CSOFF\: ([\w\|]+)"/&gt;
255 *   &lt;property name="onCommentFormat" value="CSON\: ([\w\|]+)"/&gt;
256 *   &lt;property name="checkFormat" value="$1"/&gt;
257 * &lt;/module&gt;
258 * &lt;module name="MemberName"/&gt;
259 * &lt;module name="ConstantName"/&gt;
260 * &lt;module name="IllegalCatch"/&gt;
261 * </pre>
262 * <pre>
263 * class InputSuppressionCommentFilter
264 * {
265 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
266 *
267 *  //CSOFF: MemberName
268 *  int VAR2; // suppressed violation
269 *  //CSON: MemberName
270 *
271 *  public static final int var3;
272 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
273 *
274 *  //CSOFF: ConstantName
275 *  public static final int var4; // suppressed violation
276 *  //CSON: ConstantName
277 *
278 *  public void method1()
279 *  {
280 *    try {}
281 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
282 *
283 *    //CSOFF: IllegalCatch
284 *
285 *    try {}
286 *    catch(Exception ex) {} // suppressed violation
287 *    catch(Error err) {} // suppressed violation
288 *
289 *    //CSON: IllegalCatch
290 *  }
291 * }
292 * </pre>
293 * <p>
294 * To configure a filter to suppress all audit events between a comment containing
295 * {@code CHECKSTYLE_OFF: ALMOST_ALL} and a comment containing
296 * {@code CHECKSTYLE_OFF: ALMOST_ALL} except for the <em>ConstantName</em> check:
297 * </p>
298 * <pre>
299 * &lt;module name="SuppressionCommentFilter"&gt;
300 *   &lt;property name="offCommentFormat" value="CHECKSTYLE_OFF: ALMOST_ALL"/&gt;
301 *   &lt;property name="onCommentFormat" value="CHECKSTYLE_ON: ALMOST_ALL"/&gt;
302 *   &lt;property name="checkFormat" value="^((?!(ConstantName)).)*$"/&gt;
303 * &lt;/module&gt;
304 * &lt;module name="MemberName"/&gt;
305 * &lt;module name="ConstantName"/&gt;
306 * &lt;module name="IllegalCatch"/&gt;
307 * </pre>
308 * <pre>
309 * class InputSuppressionCommentFilter
310 * {
311 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
312 *
313 *  //CHECKSTYLE_OFF: ALMOST_ALL
314 *  int VAR2; // suppressed violation
315 *  //CHECKSTYLE_ON: ALMOST_ALL
316 *
317 *  public static final int var3;
318 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
319 *
320 *  //CHECKSTYLE_OFF: ALMOST_ALL
321 *  public static final int var4;
322 *  // violation above , Name 'var4' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
323 *  //CHECKSTYLE_ON: ALMOST_ALL
324 *
325 *  public void method1()
326 *  {
327 *    try {}
328 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
329 *
330 *    //CHECKSTYLE_OFF: ALMOST_ALL
331 *
332 *    try {}
333 *    catch(Exception ex) {} // suppressed violation
334 *    catch(Error err) {} // suppressed violation
335 *
336 *    //CHECKSTYLE_ON: ALMOST_ALL
337 *  }
338 * }
339 * </pre>
340 * <p>
341 * It is possible to specify an ID of checks, so that it can be leveraged by the
342 * SuppressionCommentFilter to skip validations. The following examples show how
343 * to skip validations near code that is surrounded with {@code // CSOFF &lt;ID&gt;}
344 * and {@code // CSON &lt;ID&gt;}, where ID is the ID of checks you want to suppress.
345 * In the config of SuppressionCommentFilter, checkFormat is set to '$1' which points
346 * to the ID written in the offCommentFormat and onCommentFormat. Config for such a
347 * case is written below:
348 * </p>
349 * <pre>
350 * &lt;module name="SuppressionCommentFilter"&gt;
351 *   &lt;property name="offCommentFormat" value="CSOFF (\w+)"/&gt;
352 *   &lt;property name="onCommentFormat" value="CSON (\w+)"/&gt;
353 *   &lt;property name="idFormat" value="$1"/&gt;
354 * &lt;/module&gt;
355 * &lt;module name="MemberName"&gt;
356 *   &lt;property name="id" value="MemberID"/&gt;
357 * &lt;/module&gt;
358 * &lt;module name="ConstantName"&gt;
359 *   &lt;property name="id" value="ConstantID"/&gt;
360 * &lt;/module&gt;
361 * &lt;module name="IllegalCatch"&gt;
362 *   &lt;property name="id" value="IllegalID"/&gt;
363 * &lt;/module&gt;
364 * </pre>
365 * <pre>
366 * class InputSuppressionCommentFilter
367 * {
368 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
369 *
370 *  //CSOFF MemberID
371 *  int VAR2; // suppressed violation
372 *  //CSON: MemberID
373 *
374 *  public static final int var3;
375 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
376 *
377 *  //CSOFF ConstantID
378 *  public static final int var4; // suppressed violation
379 *  //CSON ConstantID
380 *
381 *  public void method1()
382 *  {
383 *    try {}
384 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
385 *
386 *    //CSOFF IllegalID
387 *
388 *    try {}
389 *    catch(Exception ex) {} // suppressed violation
390 *    catch(Error err) {} // suppressed violation
391 *
392 *    //CSON IllegalID
393 *  }
394 * }
395 * </pre>
396 * <p>
397 * Example of how to configure the check to suppress checks by name defined in comment.
398 * </p>
399 * <pre>
400 * &lt;module name="SuppressionCommentFilter"&gt;
401 *   &lt;property name="offCommentFormat" value="csoff (\w+)"/&gt;
402 *   &lt;property name="onCommentFormat" value="cson (\w+)"/&gt;
403 *   &lt;property name="checkFormat" value="$1"/&gt;
404 * &lt;/module&gt;
405 *   &lt;module name="MemberName"/&gt;
406 *   &lt;module name="ConstantName"/&gt;
407 *   &lt;module name="IllegalCatch"/&gt;
408 * </pre>
409 * <pre>
410 * class InputSuppressionCommentFilter
411 * {
412 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
413 *
414 *  //csoff MemberName
415 *  int VAR2; // suppressed violation
416 *  //cson MemberName
417 *
418 *  public static final int var3;
419 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
420 *
421 *  //csoff ConstantName
422 *  //csoff IllegalCatch
423 *
424 *  public static final int var4; // suppressed violation
425 *
426 *  public void method1()
427 *  {
428 *    try {}
429 *    catch(Exception ex) {} // suppressed violation
430 *
431 *    try {}
432 *    catch(Exception ex) {} // suppressed violation
433 *    catch(Error err) {} // suppressed violation
434 *  }
435 *
436 *    //cson ConstantName
437 *    //cson IllegalCatch
438 *
439 * }
440 * </pre>
441 * <p>
442 * Example depicting use of checkC and checkCPP style comments
443 * </p>
444 * <pre>
445 * &lt;module name="SuppressionCommentFilter"&gt;
446 *   &lt;property name="checkC" value="true"/&gt;
447 *   &lt;property name="checkCPP" value="false"/&gt;
448 * &lt;/module&gt;
449 * &lt;module name="MemberName"/&gt;
450 * &lt;module name="ConstantName"/&gt;
451 * &lt;module name="IllegalCatch"/&gt;
452 * </pre>
453 * <pre>
454 * class InputSuppressionCommentFilter
455 * {
456 *  int VAR1; // violation , Name 'VAR1' must match pattern '^[a-z][a-zA-Z0-9]*$'
457 *
458 *  //CHECKSTYLE:OFF
459 *  int VAR2; // violation , Name 'VAR2' must match pattern '^[a-z][a-zA-Z0-9]*$'
460 *  //CHECKSTYLE:ON
461 *
462 *  public static final int var3;
463 *  // violation above , Name 'var3' must match pattern '^[A-Z][A-Z0-9]*(_[A-Z0-9]+)*$'
464 *
465 *  /&#42;CHECKSTYLE:OFF&#42;/
466 *  public static final int var4; // suppressed violation
467 *  /&#42;CHECKSTYLE:ON&#42;/
468 *
469 *  public void method1()
470 *  {
471 *    try {}
472 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
473 *
474 *    //CHECKSTYLE:OFF
475 *
476 *    try {}
477 *    catch(Exception ex) {} // violation , Catching 'Exception' is not allowed
478 *    catch(Error err) {} // violation , Catching 'Error' is not allowed
479 *
480 *    //CHECKSTYLE:ON
481 *  }
482 * }
483 * </pre>

654    @Override
655    public boolean accept(TreeWalkerAuditEvent event) {
656        boolean accepted = true;
657
658        if (event.getViolation() != null) {
659            // Lazy update. If the first event for the current file, update file
660            // contents and tag suppressions
661            final FileContents currentContents = event.getFileContents();
662
663            if (getFileContents() != currentContents) {
664                setFileContents(currentContents);
665                tagSuppressions();
666            }
667            final Tag matchTag = findNearestMatch(event);
668            accepted = matchTag == null || matchTag.getTagType() == TagType.ON;
669        }
670        return accepted;
671    }
672
673    /**
674     * Finds the nearest comment text tag that matches an audit event.
675     * The nearest tag is before the line and column of the event.
676     *
677     * @param event the {@code TreeWalkerAuditEvent} to match.
678     * @return The {@code Tag} nearest event.
679     */
680    private Tag findNearestMatch(TreeWalkerAuditEvent event) {
681        Tag result = null;
682        for (Tag tag : tags) {
683            final int eventLine = event.getLine();
684            if (tag.getLine() > eventLine
685                || tag.getLine() == eventLine
686                    && tag.getColumn() > event.getColumn()) {
687                break;
688            }
689            if (tag.isMatch(event)) {
690                result = tag;
691            }
692        }
693        return result;
694    }

696    /**
697     * Collects all the suppression tags for all comments into a list and
698     * sorts the list.
699     */
700    private void tagSuppressions() {
701        tags.clear();
702        final FileContents contents = getFileContents();
703        if (checkCPP) {
704            tagSuppressions(contents.getSingleLineComments().values());
705        }
706        if (checkC) {
707            final Collection<List<TextBlock>> cComments = contents
708                    .getBlockComments().values();
709            cComments.forEach(this::tagSuppressions);
710        }
711        Collections.sort(tags);
712    }
713
714    /**
715     * Appends the suppressions in a collection of comments to the full
716     * set of suppression tags.
717     *
718     * @param comments the set of comments.
719     */
720    private void tagSuppressions(Collection<TextBlock> comments) {
721        for (TextBlock comment : comments) {
722            final int startLineNo = comment.getStartLineNo();
723            final String[] text = comment.getText();
724            tagCommentLine(text[0], startLineNo, comment.getStartColNo());
725            for (int i = 1; i < text.length; i++) {
726                tagCommentLine(text[i], startLineNo + i, 0);
727            }
728        }
729    }

731    /**
732     * Tags a string if it matches the format for turning
733     * checkstyle reporting on or the format for turning reporting off.
734     *
735     * @param text the string to tag.
736     * @param line the line number of text.
737     * @param column the column number of text.
738     */
739    private void tagCommentLine(String text, int line, int column) {
740        final Matcher offMatcher = offCommentFormat.matcher(text);
741        if (offMatcher.find()) {
742            addTag(offMatcher.group(0), line, column, TagType.OFF);
743        }
744        else {
745            final Matcher onMatcher = onCommentFormat.matcher(text);
746            if (onMatcher.find()) {
747                addTag(onMatcher.group(0), line, column, TagType.ON);
748            }
749        }
750    }

752    /**
753     * Adds a {@code Tag} to the list of all tags.
754     *
755     * @param text the text of the tag.
756     * @param line the line number of the tag.
757     * @param column the column number of the tag.
758     * @param reportingOn {@code true} if the tag turns checkstyle reporting on.
759     */
760    private void addTag(String text, int line, int column, TagType reportingOn) {
761        final Tag tag = new Tag(line, column, text, reportingOn, this);
762        tags.add(tag);
763    }

765    /**
766     * A Tag holds a suppression comment and its location, and determines
767     * whether the suppression turns checkstyle reporting on or off.
768     */
769    private static final class Tag
770        implements Comparable<Tag> {
771
772        /** The text of the tag. */
773        private final String text;
774
775        /** The line number of the tag. */
776        private final int line;
777
778        /** The column number of the tag. */
779        private final int column;
780
781        /** Determines whether the suppression turns checkstyle reporting on. */
782        private final TagType tagType;
783
784        /** The parsed check regexp, expanded for the text of this tag. */
785        private final Pattern tagCheckRegexp;
786
787        /** The parsed message regexp, expanded for the text of this tag. */
788        private final Pattern tagMessageRegexp;
789
790        /** The parsed check ID regexp, expanded for the text of this tag. */
791        private final Pattern tagIdRegexp;
792
793        /**
794         * Constructs a tag.
795         *
796         * @param line the line number.
797         * @param column the column number.
798         * @param text the text of the suppression.
799         * @param tagType {@code ON} if the tag turns checkstyle reporting.
800         * @param filter the {@code SuppressionCommentFilter} with the context
801         * @throws IllegalArgumentException if unable to parse expanded text.
802         */
803        private Tag(int line, int column, String text, TagType tagType,
804                   SuppressionCommentFilter filter) {
805            this.line = line;
806            this.column = column;
807            this.text = text;
808            this.tagType = tagType;
809
810            final Pattern commentFormat;
811            if (this.tagType == TagType.ON) {
812                commentFormat = filter.onCommentFormat;
813            }
814            else {
815                commentFormat = filter.offCommentFormat;
816            }
817
818            // Expand regexp for check and message
819            // Does not intern Patterns with Utils.getPattern()
820            String format = "";
821            try {
822                format = CommonUtil.fillTemplateWithStringsByRegexp(
823                        filter.checkFormat, text, commentFormat);
824                tagCheckRegexp = Pattern.compile(format);
825
826                if (filter.messageFormat == null) {
827                    tagMessageRegexp = null;
828                }
829                else {
830                    format = CommonUtil.fillTemplateWithStringsByRegexp(
831                            filter.messageFormat, text, commentFormat);
832                    tagMessageRegexp = Pattern.compile(format);
833                }
834
835                if (filter.idFormat == null) {
836                    tagIdRegexp = null;
837                }
838                else {
839                    format = CommonUtil.fillTemplateWithStringsByRegexp(
840                            filter.idFormat, text, commentFormat);
841                    tagIdRegexp = Pattern.compile(format);
842                }
843            }
844            catch (final PatternSyntaxException ex) {
845                throw new IllegalArgumentException(
846                    "unable to parse expanded comment " + format, ex);
847            }
848        }
849
850        /**
851         * Returns line number of the tag in the source file.
852         *
853         * @return the line number of the tag in the source file.
854         */
855        public int getLine() {
856            return line;
857        }
858
859        /**
860         * Determines the column number of the tag in the source file.
861         * Will be 0 for all lines of multiline comment, except the
862         * first line.
863         *
864         * @return the column number of the tag in the source file.
865         */
866        public int getColumn() {
867            return column;
868        }
869
870        /**
871         * Determines whether the suppression turns checkstyle reporting on or
872         * off.
873         *
874         * @return {@code ON} if the suppression turns reporting on.
875         */
876        public TagType getTagType() {
877            return tagType;
878        }
879
880        /**
881         * Compares the position of this tag in the file
882         * with the position of another tag.
883         *
884         * @param object the tag to compare with this one.
885         * @return a negative number if this tag is before the other tag,
886         *     0 if they are at the same position, and a positive number if this
887         *     tag is after the other tag.
888         */
889        @Override
890        public int compareTo(Tag object) {
891            final int result;
892            if (line == object.line) {
893                result = Integer.compare(column, object.column);
894            }
895            else {
896                result = Integer.compare(line, object.line);
897            }
898            return result;
899        }
900
901        /**
902         * Indicates whether some other object is "equal to" this one.
903         * Suppression on enumeration is needed so code stays consistent.
904         *
905         * @noinspection EqualsCalledOnEnumConstant
906         * @noinspectionreason EqualsCalledOnEnumConstant - enumeration is needed to keep
907         *      code consistent
908         */
909        @Override
910        public boolean equals(Object other) {
911            if (this == other) {
912                return true;
913            }
914            if (other == null || getClass() != other.getClass()) {
915                return false;
916            }
917            final Tag tag = (Tag) other;
918            return Objects.equals(line, tag.line)
919                    && Objects.equals(column, tag.column)
920                    && Objects.equals(tagType, tag.tagType)
921                    && Objects.equals(text, tag.text)
922                    && Objects.equals(tagCheckRegexp, tag.tagCheckRegexp)
923                    && Objects.equals(tagMessageRegexp, tag.tagMessageRegexp)
924                    && Objects.equals(tagIdRegexp, tag.tagIdRegexp);
925        }
926
927        @Override
928        public int hashCode() {
929            return Objects.hash(text, line, column, tagType, tagCheckRegexp, tagMessageRegexp,
930                    tagIdRegexp);
931        }
932
933        /**
934         * Determines whether the source of an audit event
935         * matches the text of this tag.
936         *
937         * @param event the {@code TreeWalkerAuditEvent} to check.
938         * @return true if the source of event matches the text of this tag.
939         */
940        public boolean isMatch(TreeWalkerAuditEvent event) {
941            return isCheckMatch(event) && isIdMatch(event) && isMessageMatch(event);
942        }
943
944        /**
945         * Checks whether {@link TreeWalkerAuditEvent} source name matches the check format.
946         *
947         * @param event {@link TreeWalkerAuditEvent} instance.
948         * @return true if the {@link TreeWalkerAuditEvent} source name matches the check format.
949         */
950        private boolean isCheckMatch(TreeWalkerAuditEvent event) {
951            final Matcher checkMatcher = tagCheckRegexp.matcher(event.getSourceName());
952            return checkMatcher.find();
953        }
954
955        /**
956         * Checks whether the {@link TreeWalkerAuditEvent} module ID matches the ID format.
957         *
958         * @param event {@link TreeWalkerAuditEvent} instance.
959         * @return true if the {@link TreeWalkerAuditEvent} module ID matches the ID format.
960         */
961        private boolean isIdMatch(TreeWalkerAuditEvent event) {
962            boolean match = true;
963            if (tagIdRegexp != null) {
964                if (event.getModuleId() == null) {
965                    match = false;
966                }
967                else {
968                    final Matcher idMatcher = tagIdRegexp.matcher(event.getModuleId());
969                    match = idMatcher.find();
970                }
971            }
972            return match;
973        }
974
975        /**
976         * Checks whether the {@link TreeWalkerAuditEvent} message matches the message format.
977         *
978         * @param event {@link TreeWalkerAuditEvent} instance.
979         * @return true if the {@link TreeWalkerAuditEvent} message matches the message format.
980         */
981        private boolean isMessageMatch(TreeWalkerAuditEvent event) {
982            boolean match = true;
983            if (tagMessageRegexp != null) {
984                final Matcher messageMatcher = tagMessageRegexp.matcher(event.getMessage());
985                match = messageMatcher.find();
986            }
987            return match;
988        }
989
990        @Override
991        public String toString() {
992            return "Tag[text='" + text + '\''
993                    + ", line=" + line
994                    + ", column=" + column
995                    + ", type=" + tagType
996                    + ", tagCheckRegexp=" + tagCheckRegexp
997                    + ", tagMessageRegexp=" + tagMessageRegexp
998                    + ", tagIdRegexp=" + tagIdRegexp + ']';
999        }
1000
1001    }

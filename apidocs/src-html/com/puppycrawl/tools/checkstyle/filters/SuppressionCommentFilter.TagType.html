039/**
040 * <p>
041 * Filter {@code SuppressionCommentFilter} uses pairs of comments to suppress audit events.
042 * </p>
043 * <p>
044 * Rationale:
045 * Sometimes there are legitimate reasons for violating a check. When
046 * this is a matter of the code in question and not personal
047 * preference, the best place to override the policy is in the code
048 * itself. Semi-structured comments can be associated with the check.
049 * This is sometimes superior to a separate suppressions file, which
050 * must be kept up-to-date as the source file is edited.
051 * </p>
052 * <p>
053 * Note that the suppression comment should be put before the violation.
054 * You can use more than one suppression comment each on separate line.
055 * </p>
056 * <p>
057 * Attention: This filter may only be specified within the TreeWalker module
058 * ({@code &lt;module name="TreeWalker"/&gt;}) and only applies to checks which are also
059 * defined within this module. To filter non-TreeWalker checks like {@code RegexpSingleline}, a
060 * <a href="https://checkstyle.org/config_filters.html#SuppressWithPlainTextCommentFilter">
061 * SuppressWithPlainTextCommentFilter</a> or similar filter must be used.
062 * </p>
063 * <p>
064 * {@code offCommentFormat} and {@code onCommentFormat} must have equal
065 * <a href="https://docs.oracle.com/en/java/javase/11/docs/api/java.base/java/util/regex/Matcher.html#groupCount()">
066 * paren counts</a>.
067 * </p>
068 * <ul>
069 * <li>
070 * Property {@code offCommentFormat} - Specify comment pattern to
071 * trigger filter to begin suppression.
072 * Default value is {@code "CHECKSTYLE:OFF"}.
073 * </li>
074 * <li>
075 * Property {@code onCommentFormat} - Specify comment pattern to trigger filter to end suppression.
076 * Default value is {@code "CHECKSTYLE:ON"}.
077 * </li>
078 * <li>
079 * Property {@code checkFormat} - Specify check pattern to suppress.
080 * Default value is {@code ".*"}.
081 * </li>
082 * <li>
083 * Property {@code messageFormat} - Specify message pattern to suppress.
084 * Default value is {@code null}.
085 * </li>
086 * <li>
087 * Property {@code checkCPP} - Control whether to check C++ style comments ({@code //}).
088 * Default value is {@code true}.
089 * </li>
090 * <li>
091 * Property {@code checkC} - Control whether to check C style comments ({@code &#47;* ... *&#47;}).
092 * Default value is {@code true}.
093 * </li>
094 * </ul>
095 * <p>
096 * To configure a filter to suppress audit events between a comment containing
097 * {@code CHECKSTYLE:OFF} and a comment containing {@code CHECKSTYLE:ON}:
098 * </p>
099 * <pre>
100 * &lt;module name="TreeWalker"&gt;
101 *               ...
102 *   &lt;module name="SuppressionCommentFilter"/&gt;
103 *               ...
104 * &lt;/module&gt;
105 * </pre>
106 * <p>
107 * To configure a filter to suppress audit events between a comment containing line
108 * {@code BEGIN GENERATED CODE} and a comment containing line {@code END GENERATED CODE}:
109 * </p>
110 * <pre>
111 * &lt;module name="SuppressionCommentFilter"&gt;
112 *   &lt;property name="offCommentFormat" value="BEGIN GENERATED CODE"/&gt;
113 *   &lt;property name="onCommentFormat" value="END GENERATED CODE"/&gt;
114 * &lt;/module&gt;
115 * </pre>
116 * <pre>
117 * //BEGIN GENERATED CODE
118 * &#64;Override
119 * public boolean equals(Object obj) { ... } // No violation events will be reported
120 *
121 * &#64;Override
122 * public int hashCode() { ... } // No violation events will be reported
123 * //END GENERATED CODE
124 * . . .
125 * </pre>
126 * <p>
127 * To configure a filter so that {@code // stop constant check} and
128 * {@code // resume constant check} marks legitimate constant names:
129 * </p>
130 * <pre>
131 * &lt;module name="SuppressionCommentFilter"&gt;
132 *   &lt;property name="offCommentFormat" value="stop constant check"/&gt;
133 *   &lt;property name="onCommentFormat" value="resume constant check"/&gt;
134 *   &lt;property name="checkFormat" value="ConstantNameCheck"/&gt;
135 * &lt;/module&gt;
136 * </pre>
137 * <pre>
138 * //stop constant check
139 * public static final int someConstant; // won't warn here
140 * //resume constant check
141 * public static final int someConstant; // will warn here as constant's name doesn't match the
142 * // pattern "^[A-Z][A-Z0-9]*$"
143 * </pre>
144 * <p>
145 * To configure a filter so that {@code UNUSED OFF: <i>var</i>} and
146 * {@code UNUSED ON: <i>var</i>} marks a variable or parameter known not to be
147 * used by the code by matching the variable name in the message:
148 * </p>
149 * <pre>
150 * &lt;module name="SuppressionCommentFilter"&gt;
151 *   &lt;property name="offCommentFormat" value="UNUSED OFF\: (\w+)"/&gt;
152 *   &lt;property name="onCommentFormat" value="UNUSED ON\: (\w+)"/&gt;
153 *   &lt;property name="checkFormat" value="Unused"/&gt;
154 *   &lt;property name="messageFormat" value="^Unused \w+ '$1'.$"/&gt;
155 * &lt;/module&gt;
156 * </pre>
157 * <pre>
158 * private static void foo(int a, int b) // UNUSED OFF: b
159 * {
160 * System.out.println(a);
161 * }
162 *
163 * private static void foo1(int a, int b) // UNUSED ON: b
164 * {
165 * System.out.println(a);
166 * }
167 * </pre>
168 * <p>
169 * To configure a filter so that name of suppressed check mentioned in comment
170 * {@code CSOFF: <i>regexp</i>} and {@code CSON: <i>regexp</i>} mark a matching check:
171 * </p>
172 * <pre>
173 * &lt;module name="SuppressionCommentFilter"&gt;
174 *   &lt;property name="offCommentFormat" value="CSOFF\: ([\w\|]+)"/&gt;
175 *   &lt;property name="onCommentFormat" value="CSON\: ([\w\|]+)"/&gt;
176 *   &lt;property name="checkFormat" value="$1"/&gt;
177 * &lt;/module&gt;
178 * </pre>
179 * <pre>
180 * public static final int lowerCaseConstant; // CSOFF: ConstantNameCheck
181 * public static final int lowerCaseConstant1; // CSON: ConstantNameCheck
182 * </pre>
183 * <p>
184 * To configure a filter to suppress all audit events between a comment containing
185 * {@code CHECKSTYLE_OFF: ALMOST_ALL} and a comment containing
186 * {@code CHECKSTYLE_OFF: ALMOST_ALL} except for the <em>EqualsHashCode</em> check:
187 * </p>
188 * <pre>
189 * &lt;module name="SuppressionCommentFilter"&gt;
190 *   &lt;property name="offCommentFormat" value="CHECKSTYLE_OFF: ALMOST_ALL"/&gt;
191 *   &lt;property name="onCommentFormat" value="CHECKSTYLE_ON: ALMOST_ALL"/&gt;
192 *   &lt;property name="checkFormat" value="^((?!(EqualsHashCode)).)*$"/&gt;
193 * &lt;/module&gt;
194 * </pre>
195 * <pre>
196 * public static final int array []; // CHECKSTYLE_OFF: ALMOST_ALL
197 * private String [] strArray;
198 * private int array1 []; // CHECKSTYLE_ON: ALMOST_ALL
199 * </pre>
200 * <p>
201 * To configure a filter to suppress Check's violation message
202 * <b>which matches specified message in messageFormat</b>
203 * (so suppression will be not only by Check's name, but by message text
204 * additionally, as the same Check could report different by message format violations)
205 * between a comment containing {@code stop} and comment containing {@code resume}:
206 * </p>
207 * <pre>
208 * &lt;module name="SuppressionCommentFilter"&gt;
209 *   &lt;property name="offCommentFormat" value="stop"/&gt;
210 *   &lt;property name="onCommentFormat" value="resume"/&gt;
211 *   &lt;property name="checkFormat" value="IllegalTypeCheck"/&gt;
212 *   &lt;property name="messageFormat"
213 *       value="^Declaring variables, return values or parameters of type 'GregorianCalendar'
214 *         is not allowed.$"/&gt;
215 * &lt;/module&gt;
216 * </pre>
217 * <p>
218 * Code before filter above is applied with Check's audit events:
219 * </p>
220 * <pre>
221 * ...
222 * // Warning below: Declaring variables, return values or parameters of type 'GregorianCalendar'
223 * // is not allowed.
224 * GregorianCalendar calendar;
225 * // Warning below here: Declaring variables, return values or parameters of type 'HashSet'
226 * // is not allowed.
227 * HashSet hashSet;
228 * ...
229 * </pre>
230 * <p>
231 * Code after filter is applied:
232 * </p>
233 * <pre>
234 * ...
235 * //stop
236 * GregorianCalendar calendar; // No warning here as it is suppressed by filter.
237 * HashSet hashSet;
238 * // Warning above here: Declaring variables, return values or parameters of type 'HashSet'
239 * //is not allowed.
240 *
241 * //resume
242 * ...
243 * </pre>
244 * <p>
245 * It is possible to specify an ID of checks, so that it can be leveraged by the
246 * SuppressionCommentFilter to skip validations. The following examples show how
247 * to skip validations near code that is surrounded with {@code // CSOFF &lt;ID&gt; (reason)}
248 * and {@code // CSON &lt;ID&gt;}, where ID is the ID of checks you want to suppress.
249 * </p>
250 * <p>
251 * Examples of Checkstyle checks configuration:
252 * </p>
253 * <pre>
254 * &lt;module name="RegexpSinglelineJava"&gt;
255 *   &lt;property name="id" value="ignore"/&gt;
256 *   &lt;property name="format" value="^.*@Ignore\s*$"/&gt;
257 *   &lt;property name="message" value="@Ignore should have a reason."/&gt;
258 * &lt;/module&gt;
259 *
260 * &lt;module name="RegexpSinglelineJava"&gt;
261 *   &lt;property name="id" value="systemout"/&gt;
262 *   &lt;property name="format" value="^.*System\.(out|err).*$"/&gt;
263 *   &lt;property name="message" value="Don't use System.out/err, use SLF4J instead."/&gt;
264 * &lt;/module&gt;
265 * </pre>
266 * <p>
267 * Example of SuppressionCommentFilter configuration (checkFormat which is set
268 * to '$1' points that ID of the checks is in the first group of offCommentFormat
269 * and onCommentFormat regular expressions):
270 * </p>
271 * <pre>
272 * &lt;module name="SuppressionCommentFilter"&gt;
273 *   &lt;property name="offCommentFormat" value="CSOFF (\w+) \(\w+\)"/&gt;
274 *   &lt;property name="onCommentFormat" value="CSON (\w+)"/&gt;
275 *   &lt;property name="checkFormat" value="$1"/&gt;
276 * &lt;/module&gt;
277 * </pre>
278 * <pre>
279 * // CSOFF ignore (test has not been implemented yet)
280 * &#64;Ignore // should NOT fail RegexpSinglelineJava
281 * &#64;Test
282 * public void testMethod() { }
283 * // CSON ignore
284 *
285 * // CSOFF systemout (debug)
286 * public static void foo() {
287 *   System.out.println("Debug info."); // should NOT fail RegexpSinglelineJava
288 * }
289 * // CSON systemout
290 * </pre>
291 * <p>
292 * Example of how to configure the check to suppress more than one checks.
293 * </p>
294 * <pre>
295 * &lt;module name="SuppressionCommentFilter"&gt;
296 *   &lt;property name="offCommentFormat" value="@cs-\: ([\w\|]+)"/&gt;
297 *   &lt;property name="checkFormat" value="$1"/&gt;
298 * &lt;/module&gt;
299 * </pre>
300 * <pre>
301 * // @cs-: ClassDataAbstractionCoupling
302 * // @cs-: MagicNumber
303 * &#64;Service // no violations from ClassDataAbstractionCoupling here
304 * &#64;Transactional
305 * public class UserService {
306 *   private int value = 10022; // no violations from MagicNumber here
307 * }
308 * </pre>
309 *
310 * @since 3.5
311 */

440    @Override
441    protected void finishLocalSetup() {
442        // No code by default

445    @Override
446    public boolean accept(TreeWalkerAuditEvent event) {
447        boolean accepted = true;
448
449        if (event.getLocalizedMessage() != null) {
450            // Lazy update. If the first event for the current file, update file
451            // contents and tag suppressions
452            final FileContents currentContents = event.getFileContents();
453
454            if (getFileContents() != currentContents) {
455                setFileContents(currentContents);
456                tagSuppressions();
457            }
458            final Tag matchTag = findNearestMatch(event);
459            accepted = matchTag == null || matchTag.getTagType() == TagType.ON;
460        }
461        return accepted;
462    }
463
464    /**
465     * Finds the nearest comment text tag that matches an audit event.
466     * The nearest tag is before the line and column of the event.
467     * @param event the {@code TreeWalkerAuditEvent} to match.
468     * @return The {@code Tag} nearest event.
469     */
470    private Tag findNearestMatch(TreeWalkerAuditEvent event) {
471        Tag result = null;
472        for (Tag tag : tags) {
473            if (tag.getLine() > event.getLine()
474                || tag.getLine() == event.getLine()
475                    && tag.getColumn() > event.getColumn()) {
476                break;
477            }
478            if (tag.isMatch(event)) {
479                result = tag;
480            }
481        }
482        return result;
483    }
484
485    /**
486     * Collects all the suppression tags for all comments into a list and
487     * sorts the list.
488     */
489    private void tagSuppressions() {
490        tags.clear();
491        final FileContents contents = getFileContents();
492        if (checkCPP) {
493            tagSuppressions(contents.getSingleLineComments().values());
494        }
495        if (checkC) {
496            final Collection<List<TextBlock>> cComments = contents
497                    .getBlockComments().values();
498            cComments.forEach(this::tagSuppressions);
499        }
500        Collections.sort(tags);
501    }
502
503    /**
504     * Appends the suppressions in a collection of comments to the full
505     * set of suppression tags.
506     * @param comments the set of comments.
507     */
508    private void tagSuppressions(Collection<TextBlock> comments) {
509        for (TextBlock comment : comments) {
510            final int startLineNo = comment.getStartLineNo();
511            final String[] text = comment.getText();
512            tagCommentLine(text[0], startLineNo, comment.getStartColNo());
513            for (int i = 1; i < text.length; i++) {
514                tagCommentLine(text[i], startLineNo + i, 0);
515            }
516        }
517    }
518
519    /**
520     * Tags a string if it matches the format for turning
521     * checkstyle reporting on or the format for turning reporting off.
522     * @param text the string to tag.
523     * @param line the line number of text.
524     * @param column the column number of text.
525     */
526    private void tagCommentLine(String text, int line, int column) {
527        final Matcher offMatcher = offCommentFormat.matcher(text);
528        if (offMatcher.find()) {
529            addTag(offMatcher.group(0), line, column, TagType.OFF);
530        }
531        else {
532            final Matcher onMatcher = onCommentFormat.matcher(text);
533            if (onMatcher.find()) {
534                addTag(onMatcher.group(0), line, column, TagType.ON);
535            }
536        }
537    }

539    /**
540     * Adds a {@code Tag} to the list of all tags.
541     * @param text the text of the tag.
542     * @param line the line number of the tag.
543     * @param column the column number of the tag.
544     * @param reportingOn {@code true} if the tag turns checkstyle reporting on.
545     */
546    private void addTag(String text, int line, int column, TagType reportingOn) {
547        final Tag tag = new Tag(line, column, text, reportingOn, this);
548        tags.add(tag);
549    }

551    /**
552     * A Tag holds a suppression comment and its location, and determines
553     * whether the suppression turns checkstyle reporting on or off.
554     */
555    private static final class Tag
556        implements Comparable<Tag> {
557
558        /** The text of the tag. */
559        private final String text;
560
561        /** The line number of the tag. */
562        private final int line;
563
564        /** The column number of the tag. */
565        private final int column;
566
567        /** Determines whether the suppression turns checkstyle reporting on. */
568        private final TagType tagType;
569
570        /** The parsed check regexp, expanded for the text of this tag. */
571        private final Pattern tagCheckRegexp;
572
573        /** The parsed message regexp, expanded for the text of this tag. */
574        private final Pattern tagMessageRegexp;
575
576        /**
577         * Constructs a tag.
578         * @param line the line number.
579         * @param column the column number.
580         * @param text the text of the suppression.
581         * @param tagType {@code ON} if the tag turns checkstyle reporting.
582         * @param filter the {@code SuppressionCommentFilter} with the context
583         * @throws IllegalArgumentException if unable to parse expanded text.
584         */
585        /* package */ Tag(int line, int column, String text, TagType tagType,
586                   SuppressionCommentFilter filter) {
587            this.line = line;
588            this.column = column;
589            this.text = text;
590            this.tagType = tagType;
591
592            //Expand regexp for check and message
593            //Does not intern Patterns with Utils.getPattern()
594            String format = "";
595            try {
596                if (this.tagType == TagType.ON) {
597                    format = CommonUtil.fillTemplateWithStringsByRegexp(
598                            filter.checkFormat, text, filter.onCommentFormat);
599                    tagCheckRegexp = Pattern.compile(format);
600                    if (filter.messageFormat == null) {
601                        tagMessageRegexp = null;
602                    }
603                    else {
604                        format = CommonUtil.fillTemplateWithStringsByRegexp(
605                                filter.messageFormat, text, filter.onCommentFormat);
606                        tagMessageRegexp = Pattern.compile(format);
607                    }
608                }
609                else {
610                    format = CommonUtil.fillTemplateWithStringsByRegexp(
611                            filter.checkFormat, text, filter.offCommentFormat);
612                    tagCheckRegexp = Pattern.compile(format);
613                    if (filter.messageFormat == null) {
614                        tagMessageRegexp = null;
615                    }
616                    else {
617                        format = CommonUtil.fillTemplateWithStringsByRegexp(
618                                filter.messageFormat, text, filter.offCommentFormat);
619                        tagMessageRegexp = Pattern.compile(format);
620                    }
621                }
622            }
623            catch (final PatternSyntaxException ex) {
624                throw new IllegalArgumentException(
625                    "unable to parse expanded comment " + format, ex);
626            }
627        }
628
629        /**
630         * Returns line number of the tag in the source file.
631         * @return the line number of the tag in the source file.
632         */
633        public int getLine() {
634            return line;
635        }
636
637        /**
638         * Determines the column number of the tag in the source file.
639         * Will be 0 for all lines of multiline comment, except the
640         * first line.
641         * @return the column number of the tag in the source file.
642         */
643        public int getColumn() {
644            return column;
645        }
646
647        /**
648         * Determines whether the suppression turns checkstyle reporting on or
649         * off.
650         * @return {@code ON} if the suppression turns reporting on.
651         */
652        public TagType getTagType() {
653            return tagType;
654        }
655
656        /**
657         * Compares the position of this tag in the file
658         * with the position of another tag.
659         * @param object the tag to compare with this one.
660         * @return a negative number if this tag is before the other tag,
661         *     0 if they are at the same position, and a positive number if this
662         *     tag is after the other tag.
663         */
664        @Override
665        public int compareTo(Tag object) {
666            final int result;
667            if (line == object.line) {
668                result = Integer.compare(column, object.column);
669            }
670            else {
671                result = Integer.compare(line, object.line);
672            }
673            return result;
674        }
675
676        /**
677         * Indicates whether some other object is "equal to" this one.
678         * Suppression on enumeration is needed so code stays consistent.
679         * @noinspection EqualsCalledOnEnumConstant
680         */
681        @Override
682        public boolean equals(Object other) {
683            if (this == other) {
684                return true;
685            }
686            if (other == null || getClass() != other.getClass()) {
687                return false;
688            }
689            final Tag tag = (Tag) other;
690            return Objects.equals(line, tag.line)
691                    && Objects.equals(column, tag.column)
692                    && Objects.equals(tagType, tag.tagType)
693                    && Objects.equals(text, tag.text)
694                    && Objects.equals(tagCheckRegexp, tag.tagCheckRegexp)
695                    && Objects.equals(tagMessageRegexp, tag.tagMessageRegexp);
696        }
697
698        @Override
699        public int hashCode() {
700            return Objects.hash(text, line, column, tagType, tagCheckRegexp, tagMessageRegexp);
701        }
702
703        /**
704         * Determines whether the source of an audit event
705         * matches the text of this tag.
706         * @param event the {@code TreeWalkerAuditEvent} to check.
707         * @return true if the source of event matches the text of this tag.
708         */
709        public boolean isMatch(TreeWalkerAuditEvent event) {
710            return (isCheckMatch(event) || isIdMatch(event)) && isMessageMatch(event);
711        }
712
713        /**
714         * Checks whether {@link TreeWalkerAuditEvent} source name matches the check format.
715         * @param event {@link TreeWalkerAuditEvent} instance.
716         * @return true if the {@link TreeWalkerAuditEvent} source name matches the check format.
717         */
718        private boolean isCheckMatch(TreeWalkerAuditEvent event) {
719            final Matcher checkMatcher = tagCheckRegexp.matcher(event.getSourceName());
720            return checkMatcher.find();
721        }
722
723        /**
724         * Checks whether the {@link TreeWalkerAuditEvent} module ID matches the ID format.
725         * @param event {@link TreeWalkerAuditEvent} instance.
726         * @return true if the {@link TreeWalkerAuditEvent} module ID matches the ID format.
727         */
728        private boolean isIdMatch(TreeWalkerAuditEvent event) {
729            boolean match = false;
730            if (event.getModuleId() != null) {
731                final Matcher idMatcher = tagCheckRegexp.matcher(event.getModuleId());
732                match = idMatcher.find();
733            }
734            return match;
735        }
736
737        /**
738         * Checks whether the {@link TreeWalkerAuditEvent} message matches the message format.
739         * @param event {@link TreeWalkerAuditEvent} instance.
740         * @return true if the {@link TreeWalkerAuditEvent} message matches the message format.
741         */
742        private boolean isMessageMatch(TreeWalkerAuditEvent event) {
743            boolean match = true;
744            if (tagMessageRegexp != null) {
745                final Matcher messageMatcher = tagMessageRegexp.matcher(event.getMessage());
746                match = messageMatcher.find();
747            }
748            return match;
749        }
750
751        @Override
752        public String toString() {
753            return "Tag[text='" + text + '\''
754                    + ", line=" + line
755                    + ", column=" + column
756                    + ", type=" + tagType
757                    + ", tagCheckRegexp=" + tagCheckRegexp
758                    + ", tagMessageRegexp=" + tagMessageRegexp + ']';
759        }

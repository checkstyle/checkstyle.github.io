020package com.puppycrawl.tools.checkstyle;
021
022import java.util.Collection;
023import java.util.HashMap;
024import java.util.HashSet;
025import java.util.Map;
026
027import com.puppycrawl.tools.checkstyle.api.Context;
028
029/**
030 * A default implementation of the Context interface.
031 */
032public final class DefaultContext implements Context {
033
034    /** Stores the context entries. */
035    private final Map<String, Object> entries = new HashMap<>();
036
037    @Override
038    public Object get(String key) {
039        return entries.get(key);
040    }
041
042    @Override
043    public Collection<String> getAttributeNames() {
044        return new HashSet<>(entries.keySet());
045    }
046
047    /**
048     * Adds a context entry.
049     *
050     * @param key the context key
051     * @param value the value for key
052     */
053    public void add(String key, Object value) {
054        entries.put(key, value);
055    }
056
057}

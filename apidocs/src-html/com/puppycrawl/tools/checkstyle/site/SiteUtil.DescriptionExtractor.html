296    /**
297     * Get string values of the message keys from the given check class.
298     *
299     * @param module class to examine.
300     * @return a set of checkstyle's module message keys.
301     * @throws MacroExecutionException if extraction of message keys fails.
302     */
303    public static Set<String> getMessageKeys(Class<?> module)
304            throws MacroExecutionException {
305        final Set<Field> messageKeyFields = getCheckMessageKeys(module);
306        // We use a TreeSet to sort the message keys alphabetically
307        final Set<String> messageKeys = new TreeSet<>();
308        for (Field field : messageKeyFields) {
309            messageKeys.add(getFieldValue(field, module).toString());
310        }
311        return messageKeys;
312    }

314    /**
315     * Gets the check's messages keys.
316     *
317     * @param module class to examine.
318     * @return a set of checkstyle's module message fields.
319     * @throws MacroExecutionException if the attempt to read a protected class fails.
320     * @noinspection ChainOfInstanceofChecks
321     * @noinspectionreason ChainOfInstanceofChecks - We will deal with this at
322     *                     <a href="https://github.com/checkstyle/checkstyle/issues/13500">13500</a>
323     *
324     */
325    private static Set<Field> getCheckMessageKeys(Class<?> module)
326            throws MacroExecutionException {
327        try {
328            final Set<Field> checkstyleMessages = new HashSet<>();
329
330            // get all fields from current class
331            final Field[] fields = module.getDeclaredFields();
332
333            for (Field field : fields) {
334                if (field.getName().startsWith("MSG_")) {
335                    checkstyleMessages.add(field);
336                }
337            }
338
339            // deep scan class through hierarchy
340            final Class<?> superModule = module.getSuperclass();
341
342            if (superModule != null) {
343                checkstyleMessages.addAll(getCheckMessageKeys(superModule));
344            }
345
346            // special cases that require additional classes
347            if (module == RegexpMultilineCheck.class) {
348                checkstyleMessages.addAll(getCheckMessageKeys(Class
349                    .forName("com.puppycrawl.tools.checkstyle.checks.regexp.MultilineDetector")));
350            }
351            else if (module == RegexpSinglelineCheck.class
352                    || module == RegexpSinglelineJavaCheck.class) {
353                checkstyleMessages.addAll(getCheckMessageKeys(Class
354                    .forName("com.puppycrawl.tools.checkstyle.checks.regexp.SinglelineDetector")));
355            }
356
357            return checkstyleMessages;
358        }
359        catch (ClassNotFoundException ex) {
360            final String message = String.format(Locale.ROOT, "Couldn't find class: %s",
361                    module.getName());
362            throw new MacroExecutionException(message, ex);
363        }
364    }

474    /**
475     * Returns the parent module name for the given module class. Returns either
476     * "TreeWalker" or "Checker". Returns null if the module class is null.
477     *
478     * @param moduleClass the module class.
479     * @return the parent module name as a string.
480     * @throws MacroExecutionException if the parent module cannot be found.
481     */
482    public static String getParentModule(Class<?> moduleClass)
483                throws MacroExecutionException {
484        String parentModuleName = "";
485        Class<?> parentClass = moduleClass.getSuperclass();
486
487        while (parentClass != null) {
488            parentModuleName = CLASS_TO_PARENT_MODULE.get(parentClass);
489            if (parentModuleName != null) {
490                break;
491            }
492            parentClass = parentClass.getSuperclass();
493        }
494
495        // If parent class is not found, check interfaces
496        if (parentModuleName == null || parentModuleName.isEmpty()) {
497            final Class<?>[] interfaces = moduleClass.getInterfaces();
498            for (Class<?> interfaceClass : interfaces) {
499                parentModuleName = CLASS_TO_PARENT_MODULE.get(interfaceClass);
500                if (parentModuleName != null) {
501                    break;
502                }
503            }
504        }
505
506        if (parentModuleName == null || parentModuleName.isEmpty()) {
507            final String message = String.format(Locale.ROOT,
508                    "Failed to find parent module for %s", moduleClass.getSimpleName());
509            throw new MacroExecutionException(message);
510        }
511
512        return parentModuleName;
513    }

518     * @param clss the class to get the properties for.
519     * @param instance the instance of the module.
520     * @return a set of properties for the given class.

522    public static Set<String> getPropertiesForDocumentation(Class<?> clss, Object instance) {

525                    .filter(prop -> {

529        properties.addAll(getNonExplicitProperties(instance, clss));
530        return new TreeSet<>(properties);

533    /**
534     * Get the javadocs of the properties of the module. If the property is not present in the
535     * module, then the javadoc of the property from the superclass(es) is used.
536     *
537     * @param properties the properties of the module.
538     * @param moduleName the name of the module.
539     * @param moduleFile the module file.
540     * @return the javadocs of the properties of the module.
541     * @throws MacroExecutionException if an error occurs during processing.
542     */
543    public static Map<String, DetailNode> getPropertiesJavadocs(Set<String> properties,
544                                                                String moduleName, File moduleFile)
545            throws MacroExecutionException {
546        // lazy initialization
547        if (SUPER_CLASS_PROPERTIES_JAVADOCS.isEmpty()) {
548            processSuperclasses();
549        }
550
551        processModule(moduleName, moduleFile);
552
553        final Map<String, DetailNode> unmodifiableJavadocs =
554                ClassAndPropertiesSettersJavadocScraper.getJavadocsForModuleOrProperty();
555        final Map<String, DetailNode> javadocs = new LinkedHashMap<>(unmodifiableJavadocs);
556
557        properties.forEach(property -> {
558            final DetailNode superClassPropertyJavadoc =
559                    SUPER_CLASS_PROPERTIES_JAVADOCS.get(property);
560            if (superClassPropertyJavadoc != null) {
561                javadocs.putIfAbsent(property, superClassPropertyJavadoc);
562            }
563        });
564
565        assertAllPropertySetterJavadocsAreFound(properties, moduleName, javadocs);
566
567        return javadocs;
568    }

570    /**
571     * Assert that each property has a corresponding setter javadoc that is not null.
572     * 'tokens' and 'javadocTokens' are excluded from this check, because their
573     * description is different from the description of the setter.
574     *
575     * @param properties the properties of the module.
576     * @param moduleName the name of the module.
577     * @param javadocs the javadocs of the properties of the module.
578     * @throws MacroExecutionException if an error occurs during processing.
579     */
580    private static void assertAllPropertySetterJavadocsAreFound(
581            Set<String> properties, String moduleName, Map<String, DetailNode> javadocs)
582            throws MacroExecutionException {
583        for (String property : properties) {
584            final boolean isPropertySetterJavadocFound = javadocs.containsKey(property)
585                       || TOKENS.equals(property) || JAVADOC_TOKENS.equals(property);
586            if (!isPropertySetterJavadocFound) {
587                final String message = String.format(Locale.ROOT,
588                        "%s: Failed to find setter javadoc for property '%s'",
589                        moduleName, property);
590                throw new MacroExecutionException(message);
591            }
592        }
593    }

612     * Scrape the Javadocs of the class and its properties setters with
613     * ClassAndPropertiesSettersJavadocScraper.
614     *
615     * @param moduleName the name of the module.
616     * @param moduleFile the module file.
617     * @throws MacroExecutionException if an error occurs during processing.
618     */
619    private static void processModule(String moduleName, File moduleFile)
620            throws MacroExecutionException {
621        if (!moduleFile.isFile()) {
622            final String message = String.format(Locale.ROOT,
623                    "File %s is not a file. Please check the 'modulePath' property.", moduleFile);
624            throw new MacroExecutionException(message);
625        }
626        ClassAndPropertiesSettersJavadocScraper.initialize(moduleName);
627        final Checker checker = new Checker();
628        checker.setModuleClassLoader(Checker.class.getClassLoader());
629        final DefaultConfiguration scraperCheckConfig =
630                        new DefaultConfiguration(
631                                ClassAndPropertiesSettersJavadocScraper.class.getName());
632        final DefaultConfiguration defaultConfiguration =
633                new DefaultConfiguration("configuration");
634        final DefaultConfiguration treeWalkerConfig =
635                new DefaultConfiguration(TreeWalker.class.getName());
636        defaultConfiguration.addProperty(CHARSET, StandardCharsets.UTF_8.name());
637        defaultConfiguration.addChild(treeWalkerConfig);
638        treeWalkerConfig.addChild(scraperCheckConfig);
639        try {
640            checker.configure(defaultConfiguration);
641            final List<File> filesToProcess = List.of(moduleFile);
642            checker.process(filesToProcess);
643            checker.destroy();
644        }
645        catch (CheckstyleException checkstyleException) {
646            final String message = String.format(Locale.ROOT, "Failed processing %s", moduleName);
647            throw new MacroExecutionException(message, checkstyleException);
648        }
649    }

699     * Gets properties that are not explicitly captured but should be documented if
700     * certain conditions are met.
701     *
702     * @param instance the instance of the module.
703     * @param clss the class of the module.
704     * @return the non explicit properties.
705     */
706    private static Set<String> getNonExplicitProperties(
707            Object instance, Class<?> clss) {
708        final Set<String> result = new TreeSet<>();
709        if (AbstractCheck.class.isAssignableFrom(clss)) {
710            final AbstractCheck check = (AbstractCheck) instance;
711
712            final int[] acceptableTokens = check.getAcceptableTokens();
713            Arrays.sort(acceptableTokens);
714            final int[] defaultTokens = check.getDefaultTokens();
715            Arrays.sort(defaultTokens);
716            final int[] requiredTokens = check.getRequiredTokens();
717            Arrays.sort(requiredTokens);
718
719            if (!Arrays.equals(acceptableTokens, defaultTokens)
720                    || !Arrays.equals(acceptableTokens, requiredTokens)) {
721                result.add(TOKENS);
722            }
723        }
724
725        if (AbstractJavadocCheck.class.isAssignableFrom(clss)) {
726            final AbstractJavadocCheck check = (AbstractJavadocCheck) instance;
727            result.add("violateExecutionOnNonTightHtml");
728
729            final int[] acceptableJavadocTokens = check.getAcceptableJavadocTokens();
730            Arrays.sort(acceptableJavadocTokens);
731            final int[] defaultJavadocTokens = check.getDefaultJavadocTokens();
732            Arrays.sort(defaultJavadocTokens);
733            final int[] requiredJavadocTokens = check.getRequiredJavadocTokens();
734            Arrays.sort(requiredJavadocTokens);
735
736            if (!Arrays.equals(acceptableJavadocTokens, defaultJavadocTokens)
737                    || !Arrays.equals(acceptableJavadocTokens, requiredJavadocTokens)) {
738                result.add(JAVADOC_TOKENS);
739            }
740        }
741
742        if (AbstractFileSetCheck.class.isAssignableFrom(clss)) {
743            result.add(FILE_EXTENSIONS);
744        }
745        return result;
746    }

839    private static DetailNode getSinceJavadocTag(DetailNode javadoc) {

879     * @param field the field to get the default value of.
880     * @param classInstance the instance of the class to get the default value of.
881     * @param moduleName the name of the module.
882     * @return the default value of the property.
883     * @throws MacroExecutionException if an error occurs during getting the default value.
884     * @noinspection IfStatementWithTooManyBranches
885     * @noinspectionreason IfStatementWithTooManyBranches - complex nature of getting properties
886     *      from XML files requires giant if/else statement
887     */
888    // -@cs[CyclomaticComplexity] Splitting would not make the code more readable
889    public static String getDefaultValue(String propertyName, Field field,
890                                         Object classInstance, String moduleName)
891            throws MacroExecutionException {
892        final Object value = getFieldValue(field, classInstance);
893        final Class<?> fieldClass = getFieldClass(field, propertyName, moduleName, classInstance);
894        String result = null;
895        if (CHARSET.equals(propertyName)) {
896            result = "the charset property of the parent"
897                    + " <a href=\"https://checkstyle.org/config.html#Checker\">Checker</a> module";
898        }
899        else if (classInstance instanceof PropertyCacheFile) {
900            result = "null (no cache file)";
901        }
902        else if (fieldClass == boolean.class) {
903            result = value.toString();
904        }
905        else if (fieldClass == int.class) {
906            result = value.toString();
907        }
908        else if (fieldClass == int[].class) {
909            result = getIntArrayPropertyValue(value);
910        }
911        else if (fieldClass == double[].class) {
912            result = removeSquareBrackets(Arrays.toString((double[]) value).replace(".0", ""));
913            if (result.isEmpty()) {
914                result = CURLY_BRACKETS;
915            }
916        }
917        else if (fieldClass == String[].class) {
918            result = getStringArrayPropertyValue(propertyName, value);
919        }
920        else if (fieldClass == URI.class || fieldClass == String.class) {
921            if (value != null) {
922                result = '"' + value.toString() + '"';
923            }
924        }
925        else if (fieldClass == Pattern.class) {
926            if (value != null) {
927                result = '"' + value.toString().replace("\n", "\\n").replace("\t", "\\t")
928                        .replace("\r", "\\r").replace("\f", "\\f") + '"';
929            }
930        }
931        else if (fieldClass == Pattern[].class) {
932            result = getPatternArrayPropertyValue(value);
933        }
934        else if (fieldClass.isEnum()) {
935            if (value != null) {
936                result = value.toString().toLowerCase(Locale.ENGLISH);
937            }
938        }
939        else if (fieldClass == AccessModifierOption[].class) {
940            result = removeSquareBrackets(Arrays.toString((Object[]) value));
941        }
942        else {
943            final String message = String.format(Locale.ROOT,
944                    "Unknown property type: %s", fieldClass.getSimpleName());
945            throw new MacroExecutionException(message);
946        }
947
948        if (result == null) {
949            result = "null";
950        }
951
952        return result;
953    }

1099    // -@cs[CyclomaticComplexity] Splitting would not make the code more readable
1100    private static Class<?> getFieldClass(Field field, String propertyName,
1101                                          String moduleName, Object instance)
1102            throws MacroExecutionException {
1103        Class<?> result = null;
1104
1105        if (field != null) {
1106            result = field.getType();
1107        }
1108        if (result == null) {
1109            if (!PROPERTIES_ALLOWED_GET_TYPES_FROM_METHOD
1110                    .contains(moduleName + DOT + propertyName)) {
1111                throw new MacroExecutionException(
1112                        "Could not find field " + propertyName + " in class " + moduleName);
1113            }
1114
1115            result = getPropertyClass(propertyName, instance);
1116        }
1117        if (field != null && (result == List.class || result == Set.class)) {
1118            final ParameterizedType type = (ParameterizedType) field.getGenericType();
1119            final Class<?> parameterClass = (Class<?>) type.getActualTypeArguments()[0];
1120
1121            if (parameterClass == Integer.class) {
1122                result = int[].class;
1123            }
1124            else if (parameterClass == String.class) {
1125                result = String[].class;
1126            }
1127            else if (parameterClass == Pattern.class) {
1128                result = Pattern[].class;
1129            }
1130            else {
1131                final String message = "Unknown parameterized type: "
1132                        + parameterClass.getSimpleName();
1133                throw new MacroExecutionException(message);
1134            }
1135        }
1136        else if (result == BitSet.class) {
1137            result = int[].class;
1138        }
1139
1140        return result;
1141    }

1194        while (!Object.class.equals(currentClass)) {

1197                result.trySetAccessible();

1199            }

1202            }
1203        }

1254            boolean isInHtmlElement = false;
1255            boolean isInHrefAttribute = false;
1256            final StringBuilder description = new StringBuilder(128);
1257            final Deque<DetailNode> queue = new ArrayDeque<>();
1258            final List<DetailNode> descriptionNodes = getDescriptionNodes(javadoc);
1259            Lists.reverse(descriptionNodes).forEach(queue::push);
1260
1261            // Perform DFS traversal on description nodes
1262            while (!queue.isEmpty()) {
1263                final DetailNode node = queue.pop();
1264                Lists.reverse(Arrays.asList(node.getChildren())).forEach(queue::push);
1265
1266                if (node.getType() == JavadocTokenTypes.HTML_TAG_NAME
1267                        && "href".equals(node.getText())) {
1268                    isInHrefAttribute = true;
1269                }
1270                if (isInHrefAttribute && node.getType() == JavadocTokenTypes.ATTR_VALUE) {
1271                    final String href = node.getText();
1272                    if (href.contains(CHECKSTYLE_ORG_URL)) {
1273                        handleInternalLink(description, moduleName, href);
1274                    }
1275                    else {
1276                        description.append(href);
1277                    }
1278
1279                    isInHrefAttribute = false;
1280                    continue;
1281                }
1282                if (node.getType() == JavadocTokenTypes.HTML_ELEMENT) {
1283                    isInHtmlElement = true;
1284                }
1285                if (node.getType() == JavadocTokenTypes.END
1286                        && node.getParent().getType() == JavadocTokenTypes.HTML_ELEMENT_END) {
1287                    description.append(node.getText());
1288                    isInHtmlElement = false;
1289                }
1290                if (node.getType() == JavadocTokenTypes.TEXT
1291                        // If a node has children, its text is not part of the description
1292                        || isInHtmlElement && node.getChildren().length == 0
1293                            // Some HTML elements span multiple lines, so we avoid the asterisk
1294                            && node.getType() != JavadocTokenTypes.LEADING_ASTERISK) {
1295                    description.append(node.getText());
1296                }
1297                if (node.getType() == JavadocTokenTypes.CODE_LITERAL) {
1298                    isInCodeLiteral = true;
1299                    description.append("<code>");
1300                }
1301                if (isInCodeLiteral
1302                        && node.getType() == JavadocTokenTypes.JAVADOC_INLINE_TAG_END) {
1303                    isInCodeLiteral = false;
1304                    description.append("</code>");
1305                }
1306            }
1307            return description.toString().trim();
1308        }

1310        /**
1311         * Converts the href value to a relative link to the document and appends it to the
1312         * description.
1313         *
1314         * @param description the description to append the relative link to.
1315         * @param moduleName the name of the module.
1316         * @param value the href value.
1317         * @throws MacroExecutionException if the relative link could not be created.
1318         */
1319        private static void handleInternalLink(StringBuilder description,
1320                                               String moduleName, String value)
1321                throws MacroExecutionException {
1322            String href = value;
1323            href = href.replace(CHECKSTYLE_ORG_URL, "");
1324            // Remove first and last characters, they are always double quotes
1325            href = href.substring(1, href.length() - 1);
1326
1327            final String relativeHref = getLinkToDocument(moduleName, href);
1328            final char doubleQuote = '\"';
1329            description.append(doubleQuote).append(relativeHref).append(doubleQuote);
1330        }

1332        /**
1333         * Extracts description nodes from javadoc.
1334         *
1335         * @param javadoc the Javadoc to extract the description from.
1336         * @return the description nodes of the setter.
1337         */
1338        private static List<DetailNode> getDescriptionNodes(DetailNode javadoc) {
1339            final DetailNode[] children = javadoc.getChildren();
1340            final List<DetailNode> descriptionNodes = new ArrayList<>();
1341            for (final DetailNode child : children) {
1342                if (isEndOfDescription(child)) {
1343                    break;
1344                }
1345                descriptionNodes.add(child);
1346            }
1347            return descriptionNodes;
1348        }
1349
1350        /**
1351         * Determines if the given child index is the end of the description. The end of the
1352         * description is defined as 4 consecutive nodes of type NEWLINE, LEADING_ASTERISK, NEWLINE,
1353         * LEADING_ASTERISK. This is an asterisk that is alone on a line. Just like the one below
1354         * this line.
1355         *
1356         * @param child the child to check.
1357         * @return true if the given child index is the end of the description.
1358         */
1359        private static boolean isEndOfDescription(DetailNode child) {
1360            final DetailNode nextSibling = JavadocUtil.getNextSibling(child);
1361            final DetailNode secondNextSibling = JavadocUtil.getNextSibling(nextSibling);
1362            final DetailNode thirdNextSibling = JavadocUtil.getNextSibling(secondNextSibling);
1363
1364            return child.getType() == JavadocTokenTypes.NEWLINE
1365                        && nextSibling.getType() == JavadocTokenTypes.LEADING_ASTERISK
1366                        && secondNextSibling.getType() == JavadocTokenTypes.NEWLINE
1367                        && thirdNextSibling.getType() == JavadocTokenTypes.LEADING_ASTERISK;
1368        }

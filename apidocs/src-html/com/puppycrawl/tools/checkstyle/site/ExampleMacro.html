020package com.puppycrawl.tools.checkstyle.site;
021
022import java.io.IOException;
023import java.nio.file.Files;
024import java.nio.file.Path;
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.List;
028import java.util.Locale;
029import java.util.stream.Collectors;
030
031import org.apache.maven.doxia.macro.AbstractMacro;
032import org.apache.maven.doxia.macro.Macro;
033import org.apache.maven.doxia.macro.MacroExecutionException;
034import org.apache.maven.doxia.macro.MacroRequest;
035import org.apache.maven.doxia.sink.Sink;
036import org.apache.maven.doxia.sink.impl.SinkEventAttributeSet;
037import org.codehaus.plexus.component.annotations.Component;
038
039/**
040 * A macro that inserts a snippet of code or configuration from a file.
041 */
042@Component(role = Macro.class, hint = "example")
043public class ExampleMacro extends AbstractMacro {
044
045    /** Starting delimiter for config snippets. */
046    private static final String XML_CONFIG_START = "/*xml";
047
048    /** Ending delimiter for config snippets. */
049    private static final String XML_CONFIG_END = "*/";
050
051    /** Starting delimiter for code snippets. */
052    private static final String CODE_SNIPPET_START = "// xdoc section -- start";
053
054    /** Ending delimiter for code snippets. */
055    private static final String CODE_SNIPPET_END = "// xdoc section -- end";
056
057    /** Newline character. */
058    private static final String NEWLINE = System.lineSeparator();
059
060    /** Eight whitespace characters. All example source tags are indented 8 spaces. */
061    private static final String INDENTATION = "        ";
062
063    /** The path of the last file. */
064    private String lastPath = "";
065
066    /** The line contents of the last file. */
067    private List<String> lastLines = new ArrayList<>();
068
069    @Override
070    public void execute(Sink sink, MacroRequest request) throws MacroExecutionException {
071        final String path = (String) request.getParameter("path");
072        final String type = (String) request.getParameter("type");
073
074        List<String> lines = lastLines;
075        if (!path.equals(lastPath)) {
076            lines = readFile("src/xdocs-examples/" + path);
077            lastPath = path;
078            lastLines = lines;
079        }
080
081        if ("config".equals(type)) {
082            final String config = getConfigSnippet(lines);
083            writeSnippet(sink, config);
084        }
085        else if ("code".equals(type)) {
086            String code = getCodeSnippet(lines);
087            // Replace tabs with spaces for FileTabCharacterCheck examples
088            if (path.contains("filetabcharacter")) {
089                code = code.replace("\t", "  ");
090            }
091            writeSnippet(sink, code);
092        }
093        else {
094            final String message = String.format(Locale.ROOT, "Unknown example type: %s", type);
095            throw new MacroExecutionException(message);
096        }
097    }
098
099    /**
100     * Read the file at the given path and returns its contents as a list of lines.
101     *
102     * @param path the path to the file to read.
103     * @return the contents of the file as a list of lines.
104     * @throws MacroExecutionException if the file could not be read.
105     */
106    private static List<String> readFile(String path) throws MacroExecutionException {
107        try {
108            final Path exampleFilePath = Path.of(path);
109            return Files.readAllLines(exampleFilePath);
110        }
111        catch (IOException ioException) {
112            final String message = String.format(Locale.ROOT, "Failed to read %s", path);
113            throw new MacroExecutionException(message, ioException);
114        }
115    }
116
117    /**
118     * Extract a configuration snippet from the given lines. Config delimiters use the whole
119     * line for themselves and have no indentation. We use equals() instead of contains()
120     * to be more strict because some examples contain those delimiters.
121     *
122     * @param lines the lines to extract the snippet from.
123     * @return the configuration snippet.
124     */
125    private static String getConfigSnippet(Collection<String> lines) {
126        return lines.stream()
127                .dropWhile(line -> !XML_CONFIG_START.equals(line))
128                .skip(1)
129                .takeWhile(line -> !XML_CONFIG_END.equals(line))
130                .collect(Collectors.joining(NEWLINE));
131    }
132
133    /**
134     * Extract a code snippet from the given lines. Code delimiters can be indented, so
135     * we use contains() instead of equals().
136     *
137     * @param lines the lines to extract the snippet from.
138     * @return the code snippet.
139     */
140    private static String getCodeSnippet(Collection<String> lines) {
141        return lines.stream()
142                .dropWhile(line -> !line.contains(CODE_SNIPPET_START))
143                .skip(1)
144                .takeWhile(line -> !line.contains(CODE_SNIPPET_END))
145                .collect(Collectors.joining(NEWLINE));
146    }
147
148    /**
149     * Write the given snippet to the file inside a source block.
150     *
151     * @param sink the sink to write to.
152     * @param snippet the snippet to write.
153     */
154    private static void writeSnippet(Sink sink, String snippet) {
155        sink.verbatim(SinkEventAttributeSet.BOXED);
156        final String text = NEWLINE
157                + String.join(NEWLINE, snippet.stripTrailing(), INDENTATION);
158        sink.text(text);
159        sink.verbatim_();
160    }
161}

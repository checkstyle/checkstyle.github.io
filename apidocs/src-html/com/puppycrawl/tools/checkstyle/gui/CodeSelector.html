020package com.puppycrawl.tools.checkstyle.gui;
021
022import java.awt.Color;
023import java.util.List;
024
025import javax.swing.JTextArea;
026
027import com.google.common.collect.ImmutableList;
028import com.puppycrawl.tools.checkstyle.api.DetailAST;
029import com.puppycrawl.tools.checkstyle.api.DetailNode;
030
031/**
032 * Helper class to select a code.
033 * @author unknown
034 */
035public class CodeSelector {
036
037    /** Editor. */
038    private final JTextArea editor;
039    /** Presentation model. */
040    private final CodeSelectorPresentation pModel;
041
042    /**
043     * Constructor.
044     * @param node ast node.
045     * @param editor text area editor.
046     * @param lines2position list to map lines.
047     */
048    public CodeSelector(final Object node, final JTextArea editor,
049                        final List<Integer> lines2position) {
050        this.editor = editor;
051        if (node instanceof DetailAST) {
052            pModel = new CodeSelectorPresentation((DetailAST) node,
053                    ImmutableList.copyOf(lines2position));
054        }
055        else {
056            pModel = new CodeSelectorPresentation((DetailNode) node,
057                    ImmutableList.copyOf(lines2position));
058        }
059    }
060
061    /**
062     * Set selection.
063     */
064    public void select() {
065        pModel.findSelectionPositions();
066        editor.setSelectedTextColor(Color.blue);
067        editor.requestFocusInWindow();
068        editor.setCaretPosition(pModel.getSelectionStart());
069        editor.moveCaretPosition(pModel.getSelectionEnd());
070    }
071
072}

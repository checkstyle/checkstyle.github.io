020package com.puppycrawl.tools.checkstyle.gui;
021
022import java.io.File;
023
024import javax.swing.SwingUtilities;
025import javax.swing.WindowConstants;
026
027/**
028 * Entry point for starting the checkstyle GUI.
029 *
030 * @author unknown
031 */
032public final class Main {
033
034    /** Hidden constructor of the current utility class. */
035    private Main() {
036        // no code
037    }
038
039    /**
040     * Entry point.
041     * @param args the command line arguments.
042     */
043    public static void main(final String... args) {
044        SwingUtilities.invokeLater(() -> {
045            final MainFrame mainFrame = new MainFrame();
046
047            if (args.length > 0) {
048                final File sourceFile = new File(args[0]);
049                mainFrame.openFile(sourceFile);
050            }
051            mainFrame.setTitle("Checkstyle GUI");
052            mainFrame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
053            mainFrame.setVisible(true);
054        });
055    }
056
057}

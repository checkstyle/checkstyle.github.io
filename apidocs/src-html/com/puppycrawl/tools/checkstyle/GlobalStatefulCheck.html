020package com.puppycrawl.tools.checkstyle;
021
022import java.lang.annotation.ElementType;
023import java.lang.annotation.Inherited;
024import java.lang.annotation.Retention;
025import java.lang.annotation.RetentionPolicy;
026import java.lang.annotation.Target;
027
028/**
029 * This annotation means that the check contains global context,
030 * which will be updated while Checkstyle processes files. This also means,
031 * that all files will be processed by the same check instance.
032 * This annotation should be used, if a check accumulates some information during the audit,
033 * and processed only once at the end of the audit (however, the check still
034 * can produce some messages, while collecting information).
035 * The check methods and fields should be thread safe, because they may be accessed from others
036 * threads at the same time.
037 * Checker guarantees that there will be exactly one check instance
038 * This is similar to multi-file validation, which checkstyle does not support fully yet.
039 * Please refer to https://github.com/checkstyle/checkstyle/issues/3540 for details.
040 * @author Andrew Kuchev
041 * @noinspection AnnotationClass, ClassIndependentOfModule, unused
042 */
043@Retention(RetentionPolicy.RUNTIME)
044@Target(ElementType.TYPE)
045@Inherited
046public @interface GlobalStatefulCheck {
047
048    // this annotation does not have properties
049
050}

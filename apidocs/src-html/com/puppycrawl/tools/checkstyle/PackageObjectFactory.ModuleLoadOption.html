038/**
039 * A factory for creating objects from package names and names.
040 * Consider the below example for better understanding.
041 * <ul>
042 *     <li>module name - name of java class that represents module;</li>
043 *     <li>module full name - fully qualifies name of java class that represents module;</li>
044 *     <li>check module short name - name of Check without 'Check' suffix;</li>
045 *     <li>check module name - name of java class that represents Check (with 'Check' suffix);</li>
046 *     <li>
047 *         check module full name - fully qualifies name of java class
048 *         that represents Check (with 'Check' suffix).
049 *     </li>
050 * </ul>
051 */
052public class PackageObjectFactory implements ModuleFactory {
053
054    /**
055     * Enum class to define loading options.
056     */
057    public enum ModuleLoadOption {
058
059        /**
060         * Searching from registered checkstyle modules and from packages given in constructor.
061         **/
062        SEARCH_REGISTERED_PACKAGES,
063        /**
064         * As SEARCH_REGISTERED_PACKAGES and also try to load class from all of packages given in
065         * constructor.
066         * Required for eclipse-cs plugin.
067         **/
068        TRY_IN_ALL_REGISTERED_PACKAGES,
069
070    }

118     * Creates a new {@code PackageObjectFactory} instance.
119     * @param packageNames the list of package names to use
120     * @param moduleClassLoader class loader used to load Checkstyle
121     *          core and custom modules

127    /**
128     * Creates a new {@code PackageObjectFactory} instance.
129     * @param packageNames the list of package names to use
130     * @param moduleClassLoader class loader used to load Checkstyle
131     *          core and custom modules
132     * @param moduleLoadOption loading option
133     * @throws IllegalArgumentException if moduleClassLoader is null or packageNames contains null
134     */
135    public PackageObjectFactory(Set<String> packageNames, ClassLoader moduleClassLoader,
136            ModuleLoadOption moduleLoadOption) {
137        if (moduleClassLoader == null) {
138            throw new IllegalArgumentException(NULL_LOADER_MESSAGE);
139        }
140        if (packageNames.contains(null)) {
141            throw new IllegalArgumentException(NULL_PACKAGE_MESSAGE);
142        }
143
144        //create a copy of the given set, but retain ordering
145        packages = new LinkedHashSet<>(packageNames);
146        this.moduleClassLoader = moduleClassLoader;
147        this.moduleLoadOption = moduleLoadOption;
148    }

150    /**
151     * Creates a new {@code PackageObjectFactory} instance.
152     * @param packageName The package name to use
153     * @param moduleClassLoader class loader used to load Checkstyle
154     *          core and custom modules
155     * @throws IllegalArgumentException if moduleClassLoader is null or packageNames is null
156     */
157    public PackageObjectFactory(String packageName, ClassLoader moduleClassLoader) {
158        if (moduleClassLoader == null) {
159            throw new IllegalArgumentException(NULL_LOADER_MESSAGE);
160        }
161        if (packageName == null) {
162            throw new IllegalArgumentException(NULL_PACKAGE_MESSAGE);
163        }
164
165        packages = Collections.singleton(packageName);
166        this.moduleClassLoader = moduleClassLoader;
167    }

169    /**
170     * Creates a new instance of a class from a given name, or that name
171     * concatenated with &quot;Check&quot;. If the name is
172     * a class name, creates an instance of the named class. Otherwise, creates
173     * an instance of a class name obtained by concatenating the given name
174     * to a package name from a given list of package names.
175     * @param name the name of a class.
176     * @return the {@code Object} created by loader.
177     * @throws CheckstyleException if an error occurs.
178     */
179    @Override
180    public Object createModule(String name) throws CheckstyleException {
181        Object instance = null;
182        // if the name is a simple class name, try to find it in maps at first
183        if (!name.contains(PACKAGE_SEPARATOR)) {
184            instance = createFromStandardCheckSet(name);
185            // find the name in third party map
186            if (instance == null) {
187                if (thirdPartyNameToFullModuleNames == null) {
188                    thirdPartyNameToFullModuleNames =
189                            generateThirdPartyNameToFullModuleName(moduleClassLoader);
190                }
191                instance = createObjectFromMap(name, thirdPartyNameToFullModuleNames);
192            }
193        }
194        if (instance == null) {
195            instance = createObject(name);
196        }
197        if (instance == null
198                && moduleLoadOption == ModuleLoadOption.TRY_IN_ALL_REGISTERED_PACKAGES) {
199            instance = createModuleByTryInEachPackage(name);
200        }
201        if (instance == null) {
202            String attemptedNames = null;
203            if (!name.contains(PACKAGE_SEPARATOR)) {
204                final String nameCheck = name + CHECK_SUFFIX;
205                attemptedNames = joinPackageNamesWithClassName(name, packages)
206                        + STRING_SEPARATOR + nameCheck + STRING_SEPARATOR
207                        + joinPackageNamesWithClassName(nameCheck, packages);
208            }
209            final LocalizedMessage exceptionMessage = new LocalizedMessage(1,
210                Definitions.CHECKSTYLE_BUNDLE, UNABLE_TO_INSTANTIATE_EXCEPTION_MESSAGE,
211                new String[] {name, attemptedNames}, null, getClass(), null);
212            throw new CheckstyleException(exceptionMessage.getMessage());
213        }
214        return instance;
215    }

217    /**
218     * Create object from one of Checkstyle module names.
219     * @param name name of module.
220     * @return instance of module.
221     * @throws CheckstyleException if the class fails to instantiate or there are ambiguous classes.
222     */
223    private Object createFromStandardCheckSet(String name) throws CheckstyleException {
224        final String fullModuleName = NAME_TO_FULL_MODULE_NAME.get(name);
225        Object instance = null;
226        if (fullModuleName == null) {
227            final String fullCheckModuleName =
228                    NAME_TO_FULL_MODULE_NAME.get(name + CHECK_SUFFIX);
229            if (fullCheckModuleName != null) {
230                instance = createObject(fullCheckModuleName);
231            }
232        }
233        else {
234            instance = createObject(fullModuleName);
235        }
236        return instance;
237    }
238
239    /**
240     * Create object with the help of the supplied map.
241     * @param name name of module.
242     * @param map the supplied map.
243     * @return instance of module if it is found in modules map and no ambiguous classes exist.
244     * @throws CheckstyleException if the class fails to instantiate or there are ambiguous classes.
245     */
246    private Object createObjectFromMap(String name, Map<String, Set<String>> map)
247            throws CheckstyleException {
248        final Set<String> fullModuleNames = map.get(name);
249        Object instance = null;
250        if (fullModuleNames == null) {
251            final Set<String> fullCheckModuleNames = map.get(name + CHECK_SUFFIX);
252            if (fullCheckModuleNames != null) {
253                instance = createObjectFromFullModuleNames(name, fullCheckModuleNames);
254            }
255        }
256        else {
257            instance = createObjectFromFullModuleNames(name, fullModuleNames);
258        }
259        return instance;
260    }
261
262    /**
263     * Create Object from optional full module names.
264     * In most case, there should be only one element in {@code fullModuleName}, otherwise
265     * an exception would be thrown.
266     * @param name name of module
267     * @param fullModuleNames the supplied full module names set
268     * @return instance of module if there is only one element in {@code fullModuleName}
269     * @throws CheckstyleException if the class fails to instantiate or there are more than one
270     *      element in {@code fullModuleName}
271     */
272    private Object createObjectFromFullModuleNames(String name, Set<String> fullModuleNames)
273            throws CheckstyleException {
274        final Object returnValue;
275        if (fullModuleNames.size() == 1) {
276            returnValue = createObject(fullModuleNames.iterator().next());
277        }
278        else {
279            final String optionalNames = fullModuleNames.stream()
280                    .sorted()
281                    .collect(Collectors.joining(STRING_SEPARATOR));
282            final LocalizedMessage exceptionMessage = new LocalizedMessage(1,
283                    Definitions.CHECKSTYLE_BUNDLE, AMBIGUOUS_MODULE_NAME_EXCEPTION_MESSAGE,
284                    new String[] {name, optionalNames}, null, getClass(), null);
285            throw new CheckstyleException(exceptionMessage.getMessage());
286        }
287        return returnValue;
288    }

290    /**
291     * Generate the map of third party Checkstyle module names to the set of their fully qualified
292     * names.
293     * @param loader the class loader used to load Checkstyle package names
294     * @return the map of third party Checkstyle module names to the set of their fully qualified
295     *      names
296     */
297    private Map<String, Set<String>> generateThirdPartyNameToFullModuleName(ClassLoader loader) {
298        Map<String, Set<String>> returnValue;
299        try {
300            returnValue = ModuleReflectionUtil.getCheckstyleModules(packages, loader).stream()
301                .collect(Collectors.groupingBy(Class::getSimpleName,
302                    Collectors.mapping(Class::getCanonicalName, Collectors.toSet())));
303        }
304        catch (IOException ignore) {
305            returnValue = Collections.emptyMap();
306        }
307        return returnValue;
308    }

310    /**
311     * Returns simple check name from full modules names map.
312     * @param fullName name of the class for joining.
313     * @return simple check name.
314     */
315    public static String getShortFromFullModuleNames(String fullName) {
316        return NAME_TO_FULL_MODULE_NAME
317                .entrySet()
318                .stream()
319                .filter(entry -> entry.getValue().equals(fullName))
320                .map(Entry::getKey)
321                .findFirst()
322                .orElse(fullName);
323    }

325    /**
326     * Creates a string by joining package names with a class name.
327     * @param className name of the class for joining.
328     * @param packages packages names.
329     * @return a string which is obtained by joining package names with a class name.
330     */
331    private static String joinPackageNamesWithClassName(String className, Set<String> packages) {
332        return packages.stream().collect(
333            Collectors.joining(PACKAGE_SEPARATOR + className + STRING_SEPARATOR, "",
334                    PACKAGE_SEPARATOR + className));
335    }

337    /**
338     * Creates a new instance of a named class.
339     * @param className the name of the class to instantiate.
340     * @return the {@code Object} created by loader or null.
341     * @throws CheckstyleException if the class fails to instantiate.
342     */
343    private Object createObject(String className) throws CheckstyleException {
344        Class<?> clazz = null;
345
346        try {
347            clazz = Class.forName(className, true, moduleClassLoader);
348        }
349        catch (final ReflectiveOperationException | NoClassDefFoundError ignored) {
350            // ignore the exception
351        }
352
353        Object instance = null;
354
355        if (clazz != null) {
356            try {
357                final Constructor<?> declaredConstructor = clazz.getDeclaredConstructor();
358                declaredConstructor.setAccessible(true);
359                instance = declaredConstructor.newInstance();
360            }
361            catch (final ReflectiveOperationException ex) {
362                throw new CheckstyleException("Unable to instantiate " + className, ex);
363            }
364        }
365
366        return instance;
367    }

369    /**
370     * Searching to class with given name (or name concatenated with &quot;Check&quot;) in existing
371     * packages. Returns instance if class found or, otherwise, null.
372     * @param name the name of a class.
373     * @return the {@code Object} created by loader.
374     * @throws CheckstyleException if an error occurs.
375     */
376    private Object createModuleByTryInEachPackage(String name) throws CheckstyleException {
377        final List<String> possibleNames = packages.stream()
378            .map(packageName -> packageName + PACKAGE_SEPARATOR + name)
379            .flatMap(className -> Stream.of(className, className + CHECK_SUFFIX))
380            .collect(Collectors.toList());
381        Object instance = null;
382        for (String possibleName : possibleNames) {
383            instance = createObject(possibleName);
384            if (instance != null) {
385                break;
386            }
387        }
388        return instance;
389    }

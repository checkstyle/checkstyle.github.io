082    /**
083     * Tests whether a method definition AST defines an equals covariant.
084     *
085     * @param ast the method definition AST to test.
086     *     Precondition: ast is a TokenTypes.METHOD_DEF node.
087     * @return true if ast defines an equals covariant.
088     */
089    public static boolean isEqualsMethod(DetailAST ast) {
090        boolean equalsMethod = false;
091
092        if (ast.getType() == TokenTypes.METHOD_DEF) {
093            final DetailAST modifiers = ast.findFirstToken(TokenTypes.MODIFIERS);
094            final boolean staticOrAbstract =
095                    modifiers.findFirstToken(TokenTypes.LITERAL_STATIC) != null
096                    || modifiers.findFirstToken(TokenTypes.ABSTRACT) != null;
097
098            if (!staticOrAbstract) {
099                final DetailAST nameNode = ast.findFirstToken(TokenTypes.IDENT);
100                final String name = nameNode.getText();
101
102                if ("equals".equals(name)) {
103                    // one parameter?
104                    final DetailAST paramsNode = ast.findFirstToken(TokenTypes.PARAMETERS);
105                    equalsMethod = paramsNode.getChildCount() == 1;
106                }
107            }
108        }
109        return equalsMethod;
110    }

112    /**
113     * Returns the value represented by the specified string of the specified
114     * type. Returns 0 for types other than float, double, int, and long.
115     *
116     * @param text the string to be parsed.
117     * @param type the token type of the text. Should be a constant of
118     *     {@link TokenTypes}.
119     * @return the double value represented by the string argument.
120     */
121    public static double parseDouble(String text, int type) {
122        String txt = UNDERSCORE_PATTERN.matcher(text).replaceAll("");
123        final double result;
124        switch (type) {
125            case TokenTypes.NUM_FLOAT:
126            case TokenTypes.NUM_DOUBLE:
127                result = Double.parseDouble(txt);
128                break;
129            case TokenTypes.NUM_INT:
130            case TokenTypes.NUM_LONG:
131                int radix = BASE_10;
132                if (txt.startsWith("0x") || txt.startsWith("0X")) {
133                    radix = BASE_16;
134                    txt = txt.substring(2);
135                }
136                else if (txt.startsWith("0b") || txt.startsWith("0B")) {
137                    radix = BASE_2;
138                    txt = txt.substring(2);
139                }
140                else if (txt.startsWith("0")) {
141                    radix = BASE_8;
142                }
143                result = parseNumber(txt, radix, type);
144                break;
145            default:
146                result = Double.NaN;
147                break;
148        }
149        return result;
150    }

152    /**
153     * Parses the string argument as an integer or a long in the radix specified by
154     * the second argument. The characters in the string must all be digits of
155     * the specified radix.
156     *
157     * @param text the String containing the integer representation to be
158     *     parsed. Precondition: text contains a parsable int.
159     * @param radix the radix to be used while parsing text.
160     * @param type the token type of the text. Should be a constant of
161     *     {@link TokenTypes}.
162     * @return the number represented by the string argument in the specified radix.
163     */
164    private static double parseNumber(final String text, final int radix, final int type) {
165        String txt = text;
166        if (txt.endsWith("L") || txt.endsWith("l")) {
167            txt = txt.substring(0, txt.length() - 1);
168        }
169        final double result;
170
171        final boolean negative = txt.charAt(0) == '-';
172        if (type == TokenTypes.NUM_INT) {
173            if (negative) {
174                result = Integer.parseInt(txt, radix);
175            }
176            else {
177                result = Integer.parseUnsignedInt(txt, radix);
178            }
179        }
180        else {
181            if (negative) {
182                result = Long.parseLong(txt, radix);
183            }
184            else {
185                result = Long.parseUnsignedLong(txt, radix);
186            }
187        }
188
189        return result;
190    }

192    /**
193     * Finds sub-node for given node minimal (line, column) pair.
194     *
195     * @param node the root of tree for search.
196     * @return sub-node with minimal (line, column) pair.
197     */
198    public static DetailAST getFirstNode(final DetailAST node) {
199        DetailAST currentNode = node;
200        DetailAST child = node.getFirstChild();
201        while (child != null) {
202            final DetailAST newNode = getFirstNode(child);
203            if (isBeforeInSource(newNode, currentNode)) {
204                currentNode = newNode;
205            }
206            child = child.getNextSibling();
207        }
208
209        return currentNode;
210    }

213     * Retrieves whether ast1 is located before ast2.
214     *
215     * @param ast1 the first node.
216     * @param ast2 the second node.
217     * @return true, if ast1 is located before ast2.
218     */
219    public static boolean isBeforeInSource(DetailAST ast1, DetailAST ast2) {
220        return ast1.getLineNo() < ast2.getLineNo()
221            || TokenUtil.areOnSameLine(ast1, ast2)
222                && ast1.getColumnNo() < ast2.getColumnNo();
223    }

225    /**
226     * Retrieves the names of the type parameters to the node.
227     *
228     * @param node the parameterized AST node
229     * @return a list of type parameter names
230     */
231    public static List<String> getTypeParameterNames(final DetailAST node) {
232        final DetailAST typeParameters =
233            node.findFirstToken(TokenTypes.TYPE_PARAMETERS);
234
235        final List<String> typeParameterNames = new ArrayList<>();
236        if (typeParameters != null) {
237            final DetailAST typeParam =
238                typeParameters.findFirstToken(TokenTypes.TYPE_PARAMETER);
239            typeParameterNames.add(
240                    typeParam.findFirstToken(TokenTypes.IDENT).getText());
241
242            DetailAST sibling = typeParam.getNextSibling();
243            while (sibling != null) {
244                if (sibling.getType() == TokenTypes.TYPE_PARAMETER) {
245                    typeParameterNames.add(
246                            sibling.findFirstToken(TokenTypes.IDENT).getText());
247                }
248                sibling = sibling.getNextSibling();
249            }
250        }
251
252        return typeParameterNames;
253    }

257     *
258     * @param node the parameterized AST node
259     * @return a list of type parameter names

261    public static List<DetailAST> getTypeParameters(final DetailAST node) {
262        final DetailAST typeParameters =
263            node.findFirstToken(TokenTypes.TYPE_PARAMETERS);
264
265        final List<DetailAST> typeParams = new ArrayList<>();
266        if (typeParameters != null) {
267            final DetailAST typeParam =
268                typeParameters.findFirstToken(TokenTypes.TYPE_PARAMETER);
269            typeParams.add(typeParam);
270
271            DetailAST sibling = typeParam.getNextSibling();
272            while (sibling != null) {
273                if (sibling.getType() == TokenTypes.TYPE_PARAMETER) {
274                    typeParams.add(sibling);
275                }
276                sibling = sibling.getNextSibling();
277            }
278        }
279
280        return typeParams;
281    }

283    /**
284     * Returns whether an AST represents a setter method.
285     *
286     * @param ast the AST to check with
287     * @return whether the AST represents a setter method
288     */
289    public static boolean isSetterMethod(final DetailAST ast) {
290        boolean setterMethod = false;
291
292        // Check have a method with exactly 7 children which are all that
293        // is allowed in a proper setter method which does not throw any
294        // exceptions.
295        if (ast.getType() == TokenTypes.METHOD_DEF
296                && ast.getChildCount() == SETTER_GETTER_MAX_CHILDREN) {
297            final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
298            final String name = type.getNextSibling().getText();
299            final boolean matchesSetterFormat = SETTER_PATTERN.matcher(name).matches();
300
301            final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
302            final boolean singleParam = params.getChildCount(TokenTypes.PARAMETER_DEF) == 1;
303
304            if (matchesSetterFormat && singleParam) {
305                // Now verify that the body consists of:
306                // SLIST -> EXPR -> ASSIGN
307                // SEMI
308                // RCURLY
309                final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
310
311                if (slist != null && slist.getChildCount() == SETTER_BODY_SIZE) {
312                    final DetailAST expr = slist.getFirstChild();
313                    setterMethod = expr.getFirstChild().getType() == TokenTypes.ASSIGN;
314                }
315            }
316        }
317        return setterMethod;
318    }

320    /**
321     * Returns whether an AST represents a getter method.
322     *
323     * @param ast the AST to check with
324     * @return whether the AST represents a getter method
325     */
326    public static boolean isGetterMethod(final DetailAST ast) {
327        boolean getterMethod = false;
328
329        // Check have a method with exactly 7 children which are all that
330        // is allowed in a proper getter method which does not throw any
331        // exceptions.
332        if (ast.getType() == TokenTypes.METHOD_DEF
333                && ast.getChildCount() == SETTER_GETTER_MAX_CHILDREN) {
334            final DetailAST type = ast.findFirstToken(TokenTypes.TYPE);
335            final String name = type.getNextSibling().getText();
336            final boolean matchesGetterFormat = GETTER_PATTERN.matcher(name).matches();
337
338            final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
339            final boolean noParams = params.getChildCount(TokenTypes.PARAMETER_DEF) == 0;
340
341            if (matchesGetterFormat && noParams) {
342                // Now verify that the body consists of:
343                // SLIST -> RETURN
344                // RCURLY
345                final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
346
347                if (slist != null) {
348                    final DetailAST expr = slist.getFirstChild();
349                    getterMethod = expr.getType() == TokenTypes.LITERAL_RETURN;
350                }
351            }
352        }
353        return getterMethod;
354    }

357     * Checks whether a method is a not void one.
358     *
359     * @param methodDefAst the method node.
360     * @return true if method is a not void one.
361     */
362    public static boolean isNonVoidMethod(DetailAST methodDefAst) {
363        boolean returnValue = false;
364        if (methodDefAst.getType() == TokenTypes.METHOD_DEF) {
365            final DetailAST typeAST = methodDefAst.findFirstToken(TokenTypes.TYPE);
366            if (typeAST.findFirstToken(TokenTypes.LITERAL_VOID) == null) {
367                returnValue = true;
368            }
369        }
370        return returnValue;
371    }

374     * Checks whether a parameter is a receiver.
375     *
376     * <p>A receiver parameter is a special parameter that
377     * represents the object for which the method is invoked.
378     * It is denoted by the reserved keyword {@code this}
379     * in the method declaration. Check
380     * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#PARAMETER_DEF">
381     * PARAMETER_DEF</a>
382     * </p>
383     *
384     * @param parameterDefAst the parameter node.
385     * @return true if the parameter is a receiver.
386     * @see <a href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-8.html#jls-8.4.1">
387     *     ReceiverParameter</a>
388     */
389    public static boolean isReceiverParameter(DetailAST parameterDefAst) {
390        return parameterDefAst.findFirstToken(TokenTypes.IDENT) == null;
391    }

393    /**
394     * Returns the access modifier of the method/constructor at the specified AST. If
395     * the method is in an interface or annotation block, the access modifier is assumed
396     * to be public.
397     *
398     * @param ast the token of the method/constructor.
399     * @return the access modifier of the method/constructor.
400     */
401    public static AccessModifierOption getAccessModifierFromModifiersToken(DetailAST ast) {
402        final DetailAST modsToken = ast.findFirstToken(TokenTypes.MODIFIERS);
403        AccessModifierOption accessModifier =
404                getAccessModifierFromModifiersTokenDirectly(modsToken);
405
406        if (accessModifier == AccessModifierOption.PACKAGE) {
407            if (ScopeUtil.isInEnumBlock(ast) && ast.getType() == TokenTypes.CTOR_DEF) {
408                accessModifier = AccessModifierOption.PRIVATE;
409            }
410            else if (ScopeUtil.isInInterfaceOrAnnotationBlock(ast)) {
411                accessModifier = AccessModifierOption.PUBLIC;
412            }
413        }
414
415        return accessModifier;
416    }

418    /**
419     * Returns {@link AccessModifierOption} based on the information about access modifier
420     * taken from the given token of type {@link TokenTypes#MODIFIERS}.
421     *
422     * @param modifiersToken token of type {@link TokenTypes#MODIFIERS}.
423     * @return {@link AccessModifierOption}.
424     * @throws IllegalArgumentException when expected non-null modifiersToken with type 'MODIFIERS'
425     */
426    private static AccessModifierOption getAccessModifierFromModifiersTokenDirectly(
427            DetailAST modifiersToken) {
428        if (modifiersToken == null) {
429            throw new IllegalArgumentException("expected non-null AST-token with type 'MODIFIERS'");
430        }
431
432        AccessModifierOption accessModifier = AccessModifierOption.PACKAGE;
433        for (DetailAST token = modifiersToken.getFirstChild(); token != null;
434             token = token.getNextSibling()) {
435            final int tokenType = token.getType();
436            if (tokenType == TokenTypes.LITERAL_PUBLIC) {
437                accessModifier = AccessModifierOption.PUBLIC;
438            }
439            else if (tokenType == TokenTypes.LITERAL_PROTECTED) {
440                accessModifier = AccessModifierOption.PROTECTED;
441            }
442            else if (tokenType == TokenTypes.LITERAL_PRIVATE) {
443                accessModifier = AccessModifierOption.PRIVATE;
444            }
445        }
446        return accessModifier;
447    }

449    /**
450     * Returns the access modifier of the surrounding "block".
451     *
452     * @param node the node to return the access modifier for
453     * @return the access modifier of the surrounding block
454     */
455    public static AccessModifierOption getSurroundingAccessModifier(DetailAST node) {
456        AccessModifierOption returnValue = null;
457        for (DetailAST token = node;
458             returnValue == null && !TokenUtil.isRootNode(token);
459             token = token.getParent()) {
460            final int type = token.getType();
461            if (type == TokenTypes.CLASS_DEF
462                || type == TokenTypes.INTERFACE_DEF
463                || type == TokenTypes.ANNOTATION_DEF
464                || type == TokenTypes.ENUM_DEF) {
465                returnValue = getAccessModifierFromModifiersToken(token);
466            }
467            else if (type == TokenTypes.LITERAL_NEW) {
468                break;
469            }
470        }
471
472        return returnValue;
473    }

476     * Create set of class names and short class names.
477     *
478     * @param classNames array of class names.
479     * @return set of class names and short class names.
480     */
481    public static Set<String> parseClassNames(String... classNames) {
482        return Arrays.stream(classNames)
483                .flatMap(className -> Stream.of(className, CommonUtil.baseClassName(className)))
484                .filter(Predicate.not(String::isEmpty))
485                .collect(Collectors.toUnmodifiableSet());
486    }

488    /**
489     * Strip initial newline and preceding whitespace on each line from text block content.
490     * In order to be consistent with how javac handles this task, we have modeled this
491     * implementation after the code from:
492     * github.com/openjdk/jdk14u/blob/master/src/java.base/share/classes/java/lang/String.java
493     *
494     * @param textBlockContent the actual content of the text block.
495     * @return string consistent with javac representation.
496     */
497    public static String stripIndentAndInitialNewLineFromTextBlock(String textBlockContent) {
498        final String contentWithInitialNewLineRemoved =
499            ALL_NEW_LINES.matcher(textBlockContent).replaceFirst("");
500        final List<String> lines =
501            Arrays.asList(ALL_NEW_LINES.split(contentWithInitialNewLineRemoved));
502        final int indent = getSmallestIndent(lines);
503        final String suffix = "";
504
505        return lines.stream()
506                .map(line -> stripIndentAndTrailingWhitespaceFromLine(line, indent))
507                .collect(Collectors.joining(System.lineSeparator(), suffix, suffix));
508    }

510    /**
511     * Helper method for stripIndentAndInitialNewLineFromTextBlock, strips correct indent
512     * from string, and trailing whitespace, or returns empty string if no text.
513     *
514     * @param line the string to strip indent and trailing whitespace from
515     * @param indent the amount of indent to remove
516     * @return modified string with removed indent and trailing whitespace, or empty string.
517     */
518    private static String stripIndentAndTrailingWhitespaceFromLine(String line, int indent) {
519        final int lastNonWhitespace = lastIndexOfNonWhitespace(line);
520        String returnString = "";
521        if (lastNonWhitespace > 0) {
522            returnString = line.substring(indent, lastNonWhitespace);
523        }
524        return returnString;
525    }

527    /**
528     * Helper method for stripIndentAndInitialNewLineFromTextBlock, to determine the smallest
529     * indent in a text block string literal.
530     *
531     * @param lines collection of actual text block content, split by line.
532     * @return number of spaces representing the smallest indent in this text block.
533     */
534    private static int getSmallestIndent(Collection<String> lines) {
535        return lines.stream()
536            .mapToInt(CommonUtil::indexOfNonWhitespace)
537            .min()
538            .orElse(0);
539    }

541    /**
542     * Helper method to find the index of the last non-whitespace character in a string.
543     *
544     * @param line the string to find the last index of a non-whitespace character for.
545     * @return the index of the last non-whitespace character.
546     */
547    private static int lastIndexOfNonWhitespace(String line) {
548        int length;
549        for (length = line.length(); length > 0; length--) {
550            if (!Character.isWhitespace(line.charAt(length - 1))) {
551                break;
552            }
553        }
554        return length;
555    }

557    /**
558     * Calculates and returns the type declaration name matching count.
559     *
560     * <p>
561     * Suppose our pattern class is {@code foo.a.b} and class to be matched is
562     * {@code foo.a.ball} then type declaration name matching count would be calculated by
563     * comparing every character, and updating main counter when we hit "." to prevent matching
564     * "a.b" with "a.ball". In this case type declaration name matching count
565     * would be equal to 6 and not 7 (b of ball is not counted).
566     * </p>
567     *
568     * @param patternClass class against which the given class has to be matched
569     * @param classToBeMatched class to be matched
570     * @return class name matching count
571     */
572    public static int typeDeclarationNameMatchingCount(String patternClass,
573                                                       String classToBeMatched) {
574        final int length = Math.min(classToBeMatched.length(), patternClass.length());
575        int result = 0;
576        for (int i = 0; i < length && patternClass.charAt(i) == classToBeMatched.charAt(i); ++i) {
577            if (patternClass.charAt(i) == PACKAGE_SEPARATOR) {
578                result = i;
579            }
580        }
581        return result;
582    }

584    /**
585     * Get the qualified name of type declaration by combining {@code packageName},
586     * {@code outerClassQualifiedName} and {@code className}.
587     *
588     * @param packageName packageName
589     * @param outerClassQualifiedName outerClassQualifiedName
590     * @param className className
591     * @return the qualified name of type declaration by combining {@code packageName},
592     *         {@code outerClassQualifiedName} and {@code className}
593     */
594    public static String getQualifiedTypeDeclarationName(String packageName,
595                                                         String outerClassQualifiedName,
596                                                         String className) {
597        final String qualifiedClassName;
598
599        if (outerClassQualifiedName == null) {
600            if (packageName == null) {
601                qualifiedClassName = className;
602            }
603            else {
604                qualifiedClassName = packageName + PACKAGE_SEPARATOR + className;
605            }
606        }
607        else {
608            qualifiedClassName = outerClassQualifiedName + PACKAGE_SEPARATOR + className;
609        }
610        return qualifiedClassName;
611    }

624    /**
625     * Get the short name of super class of anonymous inner class.
626     * Example:
627     * <pre>
628     * TestClass.NestedClass obj = new Test().new NestedClass() {};
629     * // Short name will be Test.NestedClass
630     * </pre>
631     *
632     * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
633     * @return short name of base class of anonymous inner class
634     */
635    public static String getShortNameOfAnonInnerClass(DetailAST literalNewAst) {
636        DetailAST parentAst = literalNewAst;
637        while (TokenUtil.isOfType(parentAst, TokenTypes.LITERAL_NEW, TokenTypes.DOT)) {
638            parentAst = parentAst.getParent();
639        }
640        final DetailAST firstChild = parentAst.getFirstChild();
641        return extractQualifiedName(firstChild);
642    }

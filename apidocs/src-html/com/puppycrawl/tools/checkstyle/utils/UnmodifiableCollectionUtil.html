020package com.puppycrawl.tools.checkstyle.utils;
021
022import java.util.Collections;
023import java.util.Set;
024
025/**
026 * <p>Note: it simply wraps the existing JDK methods to provide a workaround
027 * for Pitest survival on mutation for removal of immutable wrapping,
028 * see #13127 for more details.
029 * </p>
030 *
031 */
032public final class UnmodifiableCollectionUtil {
033
034    /**
035     * Private constructor for UnmodifiableCollectionUtil.
036     *
037     */
038    private UnmodifiableCollectionUtil() {
039    }
040
041    /**
042     * Creates an unmodifiable set based on the provided collection.
043     *
044     * @param collection the collection to create an unmodifiable set from
045     * @param <T> the type of elements in the set
046     * @return an unmodifiable set containing the elements from the provided collection
047     */
048    public static <T> Set<T> unmodifiableSet(Set<T> collection) {
049        return Collections.unmodifiableSet(collection);
050    }
051}

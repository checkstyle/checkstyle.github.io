020package com.puppycrawl.tools.checkstyle.utils;
021
022import java.io.IOException;
023import java.io.InputStream;
024
025import com.puppycrawl.tools.checkstyle.api.CheckstyleException;
026
027/**
028 * Utility methods for suppression filters.
029 *
030 */
031public final class FilterUtil {
032
033    /** Stop instances being created. **/
034    private FilterUtil() {
035    }
036
037    /**
038     * Checks if suppression source with given fileName exists.
039     *
040     * @param fileName name of the suppressions file.
041     * @return true if suppression file exists, otherwise false
042     */
043    public static boolean isFileExists(String fileName) {
044        boolean suppressionSourceExists;
045        try (InputStream stream = CommonUtil.getUriByFilename(fileName).toURL().openStream()) {
046            suppressionSourceExists = true;
047        }
048        catch (CheckstyleException | IOException ignored) {
049            suppressionSourceExists = false;
050        }
051        return suppressionSourceExists;
052    }
053
054}

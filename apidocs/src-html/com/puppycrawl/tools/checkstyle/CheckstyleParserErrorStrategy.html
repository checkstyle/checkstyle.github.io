020package com.puppycrawl.tools.checkstyle;
021
022import org.antlr.v4.runtime.BailErrorStrategy;
023import org.antlr.v4.runtime.InputMismatchException;
024import org.antlr.v4.runtime.Parser;
025import org.antlr.v4.runtime.Token;
026
027/**
028 * Extending BailErrorStrategy allows us to report errors while
029 * cancelling the parsing operation. If input is not syntactically
030 * correct, we do not need to use antlr's recovery strategy, which
031 * can add significant overhead.
032 *
033 */
034public class CheckstyleParserErrorStrategy extends BailErrorStrategy {
035
036    @Override
037    public Token recoverInline(Parser recognizer) {
038        reportError(recognizer, new InputMismatchException(recognizer));
039        return super.recoverInline(recognizer);
040    }
041}

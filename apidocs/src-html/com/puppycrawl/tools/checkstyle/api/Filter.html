022/**
023 * An interface for filtering AuditEvents.
024 */
025@FunctionalInterface
026public interface Filter {
027
028    /**
029     * Determines whether or not a filtered AuditEvent is accepted.
030     *
031     * @param event the AuditEvent to filter.
032     * @return true if the event is accepted.
033     */
034    boolean accept(AuditEvent event);
035
036}

020package com.puppycrawl.tools.checkstyle.api;
021
022import java.util.Collection;
023
024/**
025 * A context to be used in subcomponents. The general idea of
026 * Context/Contextualizable was taken from <a target="_top"
027 * href="https://avalon.apache.org/closed.html">Jakarta's Avalon framework</a>.
028 *
029 * @see Contextualizable
030 */
031public interface Context {
032
033    /**
034     * Searches for the value with the specified attribute key in this context.
035     *
036     * @param key the attribute key.
037     * @return the value in this context with the specified attribute key value.
038     */
039    Object get(String key);
040
041    /**
042     * Returns the names of all attributes of this context.
043     *
044     * @return the names of all attributes of this context.
045     */
046    Collection<String> getAttributeNames();
047
048}

039/**
040 * <p>
041 * Checks visibility of class members. Only static final, immutable or annotated
042 * by specified annotation members may be public;
043 * other class members must be private unless the property {@code protectedAllowed}
044 * or {@code packageAllowed} is set.
045 * </p>
046 * <p>
047 * Public members are not flagged if the name matches the public
048 * member regular expression (contains {@code "^serialVersionUID$"} by
049 * default).
050 * </p>
051 * <p>
052 * Note that Checkstyle 2 used to include {@code "^f[A-Z][a-zA-Z0-9]*$"} in the default pattern
053 * to allow names used in container-managed persistence for Enterprise JavaBeans (EJB) 1.1 with
054 * the default settings. With EJB 2.0 it is no longer necessary to have public access for
055 * persistent fields, so the default has been changed.
056 * </p>
057 * <p>
058 * Rationale: Enforce encapsulation.
059 * </p>
060 * <p>
061 * Check also has options making it less strict:
062 * </p>
063 * <p>
064 * <b>ignoreAnnotationCanonicalNames</b>- the list of annotations which ignore
065 * variables in consideration. If user want to provide short annotation name that
066 * type will match to any named the same type without consideration of package.
067 * </p>
068 * <p>
069 * <b>allowPublicFinalFields</b>- which allows public final fields.
070 * </p>
071 * <p>
072 * <b>allowPublicImmutableFields</b>- which allows immutable fields to be
073 * declared as public if defined in final class.
074 * </p>
075 * <p>
076 * Field is known to be immutable if:
077 * </p>
078 * <ul>
079 * <li>It's declared as final</li>
080 * <li>Has either a primitive type or instance of class user defined to be immutable
081 * (such as String, ImmutableCollection from Guava, etc.)</li>
082 * </ul>
083 * <p>
084 * Classes known to be immutable are listed in <b>immutableClassCanonicalNames</b>
085 * by their canonical names.
086 * </p>
087 * <p>
088 * Property Rationale: Forcing all fields of class to have private modifier by default is
089 * good in most cases, but in some cases it drawbacks in too much boilerplate get/set code.
090 * One of such cases are immutable classes.
091 * </p>
092 * <p>
093 * Restriction: Check doesn't check if class is immutable, there's no checking
094 * if accessory methods are missing and all fields are immutable, we only check
095 * if current field is immutable or final.
096 * Under the flag <b>allowPublicImmutableFields</b>, the enclosing class must
097 * also be final, to encourage immutability.
098 * Under the flag <b>allowPublicFinalFields</b>, the final modifier
099 * on the enclosing class is optional.
100 * </p>
101 * <p>
102 * Star imports are out of scope of this Check. So if one of type imported via
103 * star import collides with user specified one by its short name - there
104 * won't be Check's violation.
105 * </p>
106 * <ul>
107 * <li>
108 * Property {@code packageAllowed} - Control whether package visible members are allowed.
109 * Type is {@code boolean}.
110 * Default value is {@code false}.
111 * </li>
112 * <li>
113 * Property {@code protectedAllowed} - Control whether protected members are allowed.
114 * Type is {@code boolean}.
115 * Default value is {@code false}.
116 * </li>
117 * <li>
118 * Property {@code publicMemberPattern} - Specify pattern for public members that should be ignored.
119 * Type is {@code java.util.regex.Pattern}.
120 * Default value is {@code "^serialVersionUID$"}.
121 * </li>
122 * <li>
123 * Property {@code allowPublicFinalFields} - Allow final fields to be declared as public.
124 * Type is {@code boolean}.
125 * Default value is {@code false}.
126 * </li>
127 * <li>
128 * Property {@code allowPublicImmutableFields} - Allow immutable fields to be
129 * declared as public if defined in final class.
130 * Type is {@code boolean}.
131 * Default value is {@code false}.
132 * </li>
133 * <li>
134 * Property {@code immutableClassCanonicalNames} - Specify immutable classes canonical names.
135 * Type is {@code java.lang.String[]}.
136 * Default value is {@code java.io.File, java.lang.Boolean, java.lang.Byte,
137 * java.lang.Character, java.lang.Double, java.lang.Float, java.lang.Integer,
138 * java.lang.Long, java.lang.Short, java.lang.StackTraceElement, java.lang.String,
139 * java.math.BigDecimal, java.math.BigInteger, java.net.Inet4Address, java.net.Inet6Address,
140 * java.net.InetSocketAddress, java.net.URI, java.net.URL, java.util.Locale, java.util.UUID}.
141 * </li>
142 * <li>
143 * Property {@code ignoreAnnotationCanonicalNames} - Specify annotations canonical
144 * names which ignore variables in consideration.
145 * Type is {@code java.lang.String[]}.
146 * Default value is {@code com.google.common.annotations.VisibleForTesting,
147 * org.junit.ClassRule, org.junit.Rule}.
148 * </li>
149 * </ul>
150 * <p>

915    /**
916     * Checks access modifier of given variable.
917     * If it is not proper according to Check - puts violation on it.
918     *
919     * @param variableDef variable to check.
920     */
921    private void visitVariableDef(DetailAST variableDef) {
922        final boolean inInterfaceOrAnnotationBlock =
923                ScopeUtil.isInInterfaceOrAnnotationBlock(variableDef);
924
925        if (!inInterfaceOrAnnotationBlock && !hasIgnoreAnnotation(variableDef)) {
926            final DetailAST varNameAST = variableDef.findFirstToken(TokenTypes.TYPE)
927                .getNextSibling();
928            final String varName = varNameAST.getText();
929            if (!hasProperAccessModifier(variableDef, varName)) {
930                log(varNameAST, MSG_KEY, varName);
931            }
932        }
933    }

947    /**
948     * Checks imported type. If type's canonical name was not specified in
949     * <b>immutableClassCanonicalNames</b>, but its short name collides with one from
950     * <b>immutableClassShortNames</b> - removes it from the last one.
951     *
952     * @param importAst {@link TokenTypes#IMPORT Import}
953     */
954    private void visitImport(DetailAST importAst) {
955        if (!isStarImport(importAst)) {
956            final String canonicalName = getCanonicalName(importAst);
957            final String shortName = getClassShortName(canonicalName);
958
959            // If imported canonical class name is not specified as allowed immutable class,
960            // but its short name collides with one of specified class - removes the short name
961            // from list to avoid names collision
962            if (!immutableClassCanonicalNames.contains(canonicalName)) {
963                immutableClassShortNames.remove(shortName);
964            }
965            if (!ignoreAnnotationCanonicalNames.contains(canonicalName)) {
966                ignoreAnnotationShortNames.remove(shortName);
967            }
968        }
969    }

972     * Checks if current import is star import. E.g.:
973     * <p>
974     * {@code
975     * import java.util.*;
976     * }
977     * </p>
978     *
979     * @param importAst {@link TokenTypes#IMPORT Import}
980     * @return true if it is star import
981     */
982    private static boolean isStarImport(DetailAST importAst) {
983        boolean result = false;
984        DetailAST toVisit = importAst;
985        while (toVisit != null) {
986            toVisit = getNextSubTreeNode(toVisit, importAst);
987            if (toVisit != null && toVisit.getType() == TokenTypes.STAR) {
988                result = true;
989                break;
990            }
991        }
992        return result;
993    }

995    /**
996     * Checks if current variable has proper access modifier according to Check's options.
997     *
998     * @param variableDef Variable definition node.
999     * @param variableName Variable's name.
1000     * @return true if variable has proper access modifier.
1001     */
1002    private boolean hasProperAccessModifier(DetailAST variableDef, String variableName) {
1003        boolean result = true;
1004
1005        final String variableScope = getVisibilityScope(variableDef);
1006
1007        if (!PRIVATE_ACCESS_MODIFIER.equals(variableScope)) {
1008            result =
1009                isStaticFinalVariable(variableDef)
1010                || packageAllowed && PACKAGE_ACCESS_MODIFIER.equals(variableScope)
1011                || protectedAllowed && PROTECTED_ACCESS_MODIFIER.equals(variableScope)
1012                || isIgnoredPublicMember(variableName, variableScope)
1013                || isAllowedPublicField(variableDef);
1014        }
1015
1016        return result;
1017    }

1068     * Returns the set of modifier Strings for a VARIABLE_DEF or CLASS_DEF AST.
1069     *
1070     * @param defAST AST for a variable or class definition.
1071     * @return the set of modifier Strings for defAST.
1072     */
1073    private static Set<String> getModifiers(DetailAST defAST) {
1074        final DetailAST modifiersAST = defAST.findFirstToken(TokenTypes.MODIFIERS);
1075        final Set<String> modifiersSet = new HashSet<>();
1076        if (modifiersAST != null) {
1077            DetailAST modifier = modifiersAST.getFirstChild();
1078            while (modifier != null) {
1079                modifiersSet.add(modifier.getText());
1080                modifier = modifier.getNextSibling();
1081            }
1082        }
1083        return modifiersSet;
1084    }

1087     * Returns the visibility scope for the variable.
1088     *
1089     * @param variableDef Variable definition node.
1090     * @return one of "public", "private", "protected", "package"
1091     */
1092    private static String getVisibilityScope(DetailAST variableDef) {
1093        final Set<String> modifiers = getModifiers(variableDef);
1094        String accessModifier = PACKAGE_ACCESS_MODIFIER;
1095        for (final String modifier : EXPLICIT_MODS) {
1096            if (modifiers.contains(modifier)) {
1097                accessModifier = modifier;
1098                break;

1104    /**
1105     * Checks if current field is immutable:
1106     * has final modifier and either a primitive type or instance of class
1107     * known to be immutable (such as String, ImmutableCollection from Guava, etc.).
1108     * Classes known to be immutable are listed in
1109     * {@link VisibilityModifierCheck#immutableClassCanonicalNames}
1110     *
1111     * @param variableDef Field in consideration.
1112     * @return true if field is immutable.
1113     */
1114    private boolean isImmutableField(DetailAST variableDef) {
1115        boolean result = false;
1116        if (isFinalField(variableDef)) {
1117            final DetailAST type = variableDef.findFirstToken(TokenTypes.TYPE);
1118            final boolean isCanonicalName = isCanonicalName(type);
1119            final String typeName = getCanonicalName(type);
1120            if (immutableClassShortNames.contains(typeName)
1121                    || isCanonicalName && immutableClassCanonicalNames.contains(typeName)) {
1122                final DetailAST typeArgs = getGenericTypeArgs(type, isCanonicalName);
1123
1124                if (typeArgs == null) {
1125                    result = true;
1126                }
1127                else {
1128                    final List<String> argsClassNames = getTypeArgsClassNames(typeArgs);
1129                    result = areImmutableTypeArguments(argsClassNames);
1130                }
1131            }
1132            else {
1133                result = !isCanonicalName && isPrimitive(type);
1134            }
1135        }
1136        return result;
1137    }

1169     * Returns a list of type parameters class names.
1170     *
1171     * @param typeArgs type arguments token.
1172     * @return a list of type parameters class names.
1173     */
1174    private static List<String> getTypeArgsClassNames(DetailAST typeArgs) {
1175        final List<String> typeClassNames = new ArrayList<>();
1176        DetailAST type = typeArgs.findFirstToken(TokenTypes.TYPE_ARGUMENT);
1177        DetailAST sibling;
1178        do {
1179            final String typeName = getCanonicalName(type);
1180            typeClassNames.add(typeName);
1181            sibling = type.getNextSibling();
1182            type = sibling.getNextSibling();

1184        return typeClassNames;

1228    /**
1229     * Gets canonical type's name from given {@link TokenTypes#TYPE TYPE} node.
1230     *
1231     * @param type DetailAST {@link TokenTypes#TYPE TYPE} node.
1232     * @return canonical type's name
1233     */
1234    private static String getCanonicalName(DetailAST type) {
1235        final StringBuilder canonicalNameBuilder = new StringBuilder(256);
1236        DetailAST toVisit = type;
1237        while (toVisit != null) {
1238            toVisit = getNextSubTreeNode(toVisit, type);
1239            if (toVisit != null && toVisit.getType() == TokenTypes.IDENT) {
1240                if (canonicalNameBuilder.length() > 0) {
1241                    canonicalNameBuilder.append('.');
1242                }
1243                canonicalNameBuilder.append(toVisit.getText());
1244                final DetailAST nextSubTreeNode = getNextSubTreeNode(toVisit, type);
1245                if (nextSubTreeNode != null
1246                        && nextSubTreeNode.getType() == TokenTypes.TYPE_ARGUMENTS) {
1247                    break;
1248                }
1249            }
1250        }
1251        return canonicalNameBuilder.toString();
1252    }

1254    /**
1255     * Gets the next node of a syntactical tree (child of a current node or
1256     * sibling of a current node, or sibling of a parent of a current node).
1257     *
1258     * @param currentNodeAst Current node in considering
1259     * @param subTreeRootAst SubTree root
1260     * @return Current node after bypassing, if current node reached the root of a subtree
1261     *        method returns null
1262     */
1263    private static DetailAST
1264        getNextSubTreeNode(DetailAST currentNodeAst, DetailAST subTreeRootAst) {
1265        DetailAST currentNode = currentNodeAst;
1266        DetailAST toVisitAst = currentNode.getFirstChild();
1267        while (toVisitAst == null) {
1268            toVisitAst = currentNode.getNextSibling();
1269            if (currentNode.getParent().getColumnNo() == subTreeRootAst.getColumnNo()) {
1270                break;
1271            }
1272            currentNode = currentNode.getParent();
1273        }
1274        return toVisitAst;
1275    }

1300    /**
1301     * Checks whether the AST is annotated with
1302     * an annotation containing the passed in regular
1303     * expression and return the AST representing that
1304     * annotation.
1305     *
1306     * <p>
1307     * This method will not look for imports or package
1308     * statements to detect the passed in annotation.
1309     * </p>
1310     *
1311     * <p>
1312     * To check if an AST contains a passed in annotation
1313     * taking into account fully-qualified names
1314     * (ex: java.lang.Override, Override)
1315     * this method will need to be called twice. Once for each
1316     * name given.
1317     * </p>
1318     *
1319     * @param variableDef {@link TokenTypes#VARIABLE_DEF variable def node}.
1320     * @return the AST representing the first such annotation or null if
1321     *         no such annotation was found
1322     */
1323    private DetailAST findMatchingAnnotation(DetailAST variableDef) {
1324        DetailAST matchingAnnotation = null;
1325
1326        final DetailAST holder = AnnotationUtil.getAnnotationHolder(variableDef);
1327
1328        for (DetailAST child = holder.getFirstChild();
1329            child != null; child = child.getNextSibling()) {
1330            if (child.getType() == TokenTypes.ANNOTATION) {
1331                final DetailAST ast = child.getFirstChild();
1332                final String name =
1333                    FullIdent.createFullIdent(ast.getNextSibling()).getText();
1334                if (ignoreAnnotationCanonicalNames.contains(name)
1335                         || ignoreAnnotationShortNames.contains(name)) {
1336                    matchingAnnotation = child;
1337                    break;
1338                }
1339            }
1340        }
1341
1342        return matchingAnnotation;
1343    }

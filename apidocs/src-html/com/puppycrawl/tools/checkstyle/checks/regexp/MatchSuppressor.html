020package com.puppycrawl.tools.checkstyle.checks.regexp;
021
022/**
023 * Represents a suppressor for matches.
024 */
025@FunctionalInterface
026public interface MatchSuppressor {
027
028    /**
029     * Checks if the specified selection should be suppressed.
030     *
031     * @param startLineNo the starting line number
032     * @param startColNo the starting column number
033     * @param endLineNo the ending line number
034     * @param endColNo the ending column number
035     * @return true if the positions intersects with a comment.
036     **/
037    boolean shouldSuppress(int startLineNo, int startColNo, int endLineNo,
038            int endColNo);
039
040}

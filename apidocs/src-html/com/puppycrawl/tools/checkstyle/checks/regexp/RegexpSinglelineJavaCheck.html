020package com.puppycrawl.tools.checkstyle.checks.regexp;
021
022import com.puppycrawl.tools.checkstyle.PropertyType;
023import com.puppycrawl.tools.checkstyle.StatelessCheck;
024import com.puppycrawl.tools.checkstyle.XdocsPropertyType;
025import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
026import com.puppycrawl.tools.checkstyle.api.DetailAST;
027import com.puppycrawl.tools.checkstyle.utils.CommonUtil;
028
029/**
030 * <p>
031 * Checks that a specified pattern matches a single-line in Java files.
032 * </p>
033 * <p>
034 * This class is variation on
035 * <a href="https://checkstyle.org/checks/regexp/regexpsingleline.html#RegexpSingleline">
036 * RegexpSingleline</a>
037 * for detecting single-lines that match a supplied regular expression in Java files.
038 * It supports suppressing matches in Java comments.
039 * </p>
040 * <ul>
041 * <li>
042 * Property {@code format} - Specify the format of the regular expression to match.
043 * Type is {@code java.util.regex.Pattern}.
044 * Default value is {@code "$."}.
045 * </li>
046 * <li>
047 * Property {@code message} - Specify the message which is used to notify about
048 * violations, if empty then default (hard-coded) message is used.
049 * Type is {@code java.lang.String}.
050 * Default value is {@code null}.
051 * </li>
052 * <li>
053 * Property {@code ignoreCase} - Control whether to ignore case when searching.
054 * Type is {@code boolean}.
055 * Default value is {@code false}.
056 * </li>
057 * <li>
058 * Property {@code minimum} - Specify the minimum number of matches required in each file.
059 * Type is {@code int}.
060 * Default value is {@code 0}.
061 * </li>
062 * <li>
063 * Property {@code maximum} - Specify the maximum number of matches required in each file.
064 * Type is {@code int}.
065 * Default value is {@code 0}.
066 * </li>
067 * <li>
068 * Property {@code ignoreComments} - Control whether to ignore text in comments when searching.
069 * Type is {@code boolean}.
070 * Default value is {@code false}.
071 * </li>
072 * </ul>
073 * <p>
074 *   To configure the default check:
075 * </p>
076 * <pre>
077 * &lt;module name=&quot;RegexpSinglelineJava&quot;/&gt;
078 * </pre>
079 * <p>
080 *   This configuration does not match to anything,
081 *   so we do not provide any code example for it
082 *   as no violation will ever be reported.
083 * </p>
084 * <p>
085 * To configure the check for calls to {@code System.out.println}, except in comments:
086 * </p>
087 * <pre>
088 * &lt;module name="RegexpSinglelineJava"&gt;
089 *   &lt;!-- . matches any character, so we need to
090 *        escape it and use \. to match dots. --&gt;
091 *   &lt;property name="format" value="System\.out\.println"/&gt;
092 *   &lt;property name="ignoreComments" value="true"/&gt;
093 * &lt;/module&gt;
094 * </pre>
095 * <p>Example:</p>
096 * <pre>
097 * System.out.println(""); // violation, instruction matches illegal pattern
098 * System.out.
099 *   println(""); // OK
100 * &#47;* System.out.println *&#47; // OK, comments are ignored
101 * </pre>
102 * <p>
103 * To configure the check to find case-insensitive occurrences of "debug":
104 * </p>
105 * <pre>
106 * &lt;module name="RegexpSinglelineJava"&gt;
107 *   &lt;property name="format" value="debug"/&gt;
108 *   &lt;property name="ignoreCase" value="true"/&gt;
109 * &lt;/module&gt;
110 * </pre>
111 * <p>Example:</p>
112 * <pre>
113 * int debug = 0; // violation, variable name matches illegal pattern
114 * public class Debug { // violation, class name matches illegal pattern
115 * &#47;* this is for de
116 *   bug only; *&#47; // OK
117 * </pre>
118 * <p>
119 * To configure the check to find occurrences of
120 * &quot;\.read(.*)|\.write(.*)&quot;
121 * and display &quot;IO found&quot; for each violation.
122 * </p>
123 * <pre>
124 * &lt;module name=&quot;RegexpSinglelineJava&quot;&gt;
125 *   &lt;property name=&quot;format&quot; value=&quot;\.read(.*)|\.write(.*)&quot;/&gt;
126 *   &lt;property name=&quot;message&quot; value=&quot;IO found&quot;/&gt;
127 * &lt;/module&gt;
128 * </pre>
129 * <p>Example:</p>
130 * <pre>
131 * FileReader in = new FileReader("path/to/input");
132 * int ch = in.read(); // violation
133 * while(ch != -1) {
134 *   System.out.print((char)ch);
135 *   ch = in.read(); // violation
136 * }
137 *
138 * FileWriter out = new FileWriter("path/to/output");
139 * out.write("something"); // violation
140 * </pre>
141 * <p>
142 * To configure the check to find occurrences of
143 * &quot;\.log(.*)&quot;. We want to allow a maximum of 2 occurrences.
144 * </p>
145 * <pre>
146 * &lt;module name=&quot;RegexpSinglelineJava&quot;&gt;
147 *   &lt;property name=&quot;format&quot; value=&quot;\.log(.*)&quot;/&gt;
148 *   &lt;property name=&quot;maximum&quot; value=&quot;2&quot;/&gt;
149 * &lt;/module&gt;
150 * </pre>
151 * <p>Example:</p>
152 * <pre>
153 * public class Foo{
154 *   public void bar(){
155 *     Logger.log("first"); // OK, first occurrence is allowed
156 *     Logger.log("second"); // OK, second occurrence is allowed
157 *     Logger.log("third"); // violation
158 *     System.out.println("fourth");
159 *     Logger.log("fifth"); // violation
160 *   }
161 * }
162 * </pre>
163 * <p>
164 * To configure the check to find all occurrences of
165 * &quot;public&quot;. We want to ignore comments,
166 * display &quot;public member found&quot; for each violation
167 * and say if less than 2 occurrences.
168 * </p>
169 * <pre>
170 * &lt;module name=&quot;RegexpSinglelineJava&quot;&gt;
171 *   &lt;property name=&quot;format&quot; value=&quot;public&quot;/&gt;
172 *   &lt;property name=&quot;minimum&quot; value=&quot;2&quot;/&gt;
173 *   &lt;property name=&quot;message&quot; value=&quot;public member found&quot;/&gt;
174 *   &lt;property name=&quot;ignoreComments&quot; value=&quot;true&quot;/&gt;
175 * &lt;/module&gt;
176 * </pre>
177 * <p>Example:</p>
178 * <pre>
179 * class Foo{ // violation, file contains less than 2 occurrences of "public"
180 *   private int a;
181 *   &#47;* public comment *&#47; // OK, comment is ignored
182 *   private void bar1() {}
183 *   public void bar2() {} // violation
184 * }
185 * </pre>
186 * <p>Example:</p>
187 * <pre>
188 * class Foo{
189 *   private int a;
190 *  &#47;* public comment *&#47; // OK, comment is ignored
191 *   public void bar1() {} // violation
192 *   public void bar2() {} // violation
193 * }
194 * </pre>
195 * <p>
196 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
197 * </p>
198 * <p>
199 * Violation Message Keys:
200 * </p>
201 * <ul>
202 * <li>
203 * {@code regexp.exceeded}
204 * </li>
205 * <li>
206 * {@code regexp.minimum}
207 * </li>
208 * </ul>
209 *
210 * @since 6.0
211 */
212@StatelessCheck
213public class RegexpSinglelineJavaCheck extends AbstractCheck {
214
215    /** Specify the format of the regular expression to match. */
216    @XdocsPropertyType(PropertyType.PATTERN)
217    private String format = "$.";
218    /**
219     * Specify the message which is used to notify about violations,
220     * if empty then default (hard-coded) message is used.
221     */
222    private String message;
223    /** Specify the minimum number of matches required in each file. */
224    private int minimum;
225    /** Specify the maximum number of matches required in each file. */
226    private int maximum;
227    /** Control whether to ignore case when searching. */
228    private boolean ignoreCase;
229    /** Control whether to ignore text in comments when searching. */
230    private boolean ignoreComments;
231
232    @Override
233    public int[] getDefaultTokens() {
234        return getRequiredTokens();
235    }
236
237    @Override
238    public int[] getAcceptableTokens() {
239        return getRequiredTokens();
240    }
241
242    @Override
243    public int[] getRequiredTokens() {
244        return CommonUtil.EMPTY_INT_ARRAY;
245    }
246
247    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
248    @SuppressWarnings("deprecation")
249    @Override
250    public void beginTree(DetailAST rootAST) {
251        MatchSuppressor suppressor = null;
252        if (ignoreComments) {
253            suppressor = new CommentSuppressor(getFileContents());
254        }
255
256        final DetectorOptions options = DetectorOptions.newBuilder()
257            .reporter(this)
258            .suppressor(suppressor)
259            .format(format)
260            .message(message)
261            .minimum(minimum)
262            .maximum(maximum)
263            .ignoreCase(ignoreCase)
264            .build();
265        final SinglelineDetector detector = new SinglelineDetector(options);
266        detector.processLines(getFileContents().getText());
267    }
268
269    /**
270     * Setter to specify the format of the regular expression to match.
271     *
272     * @param format the format of the regular expression to match.
273     * @since 5.0
274     */
275    public void setFormat(String format) {
276        this.format = format;
277    }
278
279    /**
280     * Setter to specify the message which is used to notify about violations,
281     * if empty then default (hard-coded) message is used.
282     *
283     * @param message the message to report for a match.
284     * @since 6.0
285     */
286    public void setMessage(String message) {
287        this.message = message;
288    }
289
290    /**
291     * Setter to specify the minimum number of matches required in each file.
292     *
293     * @param minimum the minimum number of matches required in each file.
294     * @since 5.0
295     */
296    public void setMinimum(int minimum) {
297        this.minimum = minimum;
298    }
299
300    /**
301     * Setter to specify the maximum number of matches required in each file.
302     *
303     * @param maximum the maximum number of matches required in each file.
304     * @since 5.0
305     */
306    public void setMaximum(int maximum) {
307        this.maximum = maximum;
308    }
309
310    /**
311     * Setter to control whether to ignore case when searching.
312     *
313     * @param ignoreCase whether to ignore case when searching.
314     * @since 5.0
315     */
316    public void setIgnoreCase(boolean ignoreCase) {
317        this.ignoreCase = ignoreCase;
318    }
319
320    /**
321     * Setter to control whether to ignore text in comments when searching.
322     *
323     * @param ignore whether to ignore text in comments when searching.
324     * @since 5.0
325     */
326    public void setIgnoreComments(boolean ignore) {
327        ignoreComments = ignore;
328    }
329
330}

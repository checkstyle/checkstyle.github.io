020package com.puppycrawl.tools.checkstyle.checks.naming;
021
022import java.util.Arrays;
023import java.util.Optional;
024
025import com.puppycrawl.tools.checkstyle.api.DetailAST;
026import com.puppycrawl.tools.checkstyle.api.TokenTypes;
027import com.puppycrawl.tools.checkstyle.utils.CheckUtil;
028
029/**
030 * <p>
031 * Checks that method parameter names conform to a specified pattern.
032 * By using {@code accessModifiers} property it is possible
033 * to specify different formats for methods at different visibility levels.
034 * </p>
035 * <p>
036 * To validate {@code catch} parameters please use
037 * <a href="https://checkstyle.org/checks/naming/catchparametername.html#CatchParameterName">
038 * CatchParameterName</a>.
039 * </p>
040 * <p>
041 * To validate lambda parameters please use
042 * <a href="https://checkstyle.org/checks/naming/lambdaparametername.html#LambdaParameterName">
043 * LambdaParameterName</a>.
044 * </p>
045 * <ul>
046 * <li>
047 * Property {@code format} - Specifies valid identifiers.
048 * Type is {@code java.util.regex.Pattern}.
049 * Default value is {@code "^[a-z][a-zA-Z0-9]*$"}.
050 * </li>
051 * <li>
052 * Property {@code ignoreOverridden} - Allows to skip methods with Override annotation from
053 * validation.
054 * Type is {@code boolean}.
055 * Default value is {@code false}.
056 * </li>
057 * <li>
058 * Property {@code accessModifiers} - Access modifiers of methods where parameters are
059 * checked.
060 * Type is {@code com.puppycrawl.tools.checkstyle.checks.naming.AccessModifierOption[]}.
061 * Default value is {@code public, protected, package, private}.
062 * </li>
063 * </ul>
064 * <p>
065 * To configure the check:
066 * </p>
067 * <pre>
068 * &lt;module name=&quot;ParameterName&quot;/&gt;
069 * </pre>
070 * <p>Code Example:</p>
071 * <pre>
072 * class MyClass {
073 *   void method1(int v1) {} // OK
074 *   void method2(int V2) {} // violation, name 'V2' must match pattern '^[a-z][a-zA-Z0-9]*$'
075 * }
076 * </pre>
077 * <p>
078 * An example of how to configure the check for names that begin with
079 * a lower case letter, followed by letters, digits, and underscores:
080 * </p>
081 * <pre>
082 * &lt;module name=&quot;ParameterName&quot;&gt;
083 *   &lt;property name=&quot;format&quot; value=&quot;^[a-z][_a-zA-Z0-9]+$&quot;/&gt;
084 * &lt;/module&gt;
085 * </pre>
086 * <p>Code Example:</p>
087 * <pre>
088 * class MyClass {
089 *   void method1(int v1) {} // OK
090 *   void method2(int v_2) {} // OK
091 *   void method3(int V3) {} // violation, name 'V3' must match pattern '^[a-z][_a-zA-Z0-9]+$'
092 * }
093 * </pre>
094 * <p>
095 * An example of how to configure the check to skip methods with Override annotation from
096 * validation:
097 * </p>
098 * <pre>
099 * &lt;module name=&quot;ParameterName&quot;&gt;
100 *   &lt;property name=&quot;ignoreOverridden&quot; value=&quot;true&quot;/&gt;
101 * &lt;/module&gt;
102 * </pre>
103 * <p>Code Example:</p>
104 * <pre>
105 * class MyClass {
106 *   void method1(int v1) {} // OK
107 *   void method2(int V2) {} // violation, name 'V2' must match pattern '^[a-z][a-zA-Z0-9]*$'
108 *   &#064;Override
109 *   public boolean equals(Object V3) { // OK
110 *       return true;
111 *   }
112 * }
113 * </pre>
114 * <p>
115 * An example of how to configure the check for names that begin with a lower case letter, followed
116 * by letters and digits is:
117 * </p>
118 * <pre>
119 * &lt;module name=&quot;ParameterName&quot;&gt;
120 *   &lt;property name=&quot;format&quot; value=&quot;^[a-z][a-zA-Z0-9]+$&quot;/&gt;
121 * &lt;/module&gt;
122 * </pre>
123 * <p>Code Example:</p>
124 * <pre>
125 * class MyClass {
126 *   void method1(int v1) {} // OK
127 *   void method2(int v_2) {} // violation, name 'v_2' must match pattern '^[a-z][a-zA-Z0-9]+$'
128 *   void method3(int V3) {} // violation, name 'V3' must match pattern '^[a-z][a-zA-Z0-9]+$'
129 * }
130 * </pre>
131 * <p>
132 * The following configuration checks that the parameters always start with two lowercase
133 * characters and, in addition, that public method parameters cannot be one character long:
134 * </p>
135 * <pre>
136 * &lt;module name=&quot;ParameterName&quot;&gt;
137 *   &lt;property name=&quot;format&quot; value=&quot;^[a-z]([a-z0-9][a-zA-Z0-9]*)?$&quot;/&gt;
138 *   &lt;property name=&quot;accessModifiers&quot;
139 *     value=&quot;protected, package, private&quot;/&gt;
140 *   &lt;message key=&quot;name.invalidPattern&quot;
141 *     value=&quot;Parameter name ''{0}'' must match pattern ''{1}''&quot;/&gt;
142 * &lt;/module&gt;
143 * &lt;module name=&quot;ParameterName&quot;&gt;
144 *   &lt;property name=&quot;format&quot; value=&quot;^[a-z][a-z0-9][a-zA-Z0-9]*$&quot;/&gt;
145 *   &lt;property name=&quot;accessModifiers&quot; value=&quot;public&quot;/&gt;
146 *   &lt;message key=&quot;name.invalidPattern&quot;
147 *     value=&quot;Parameter name ''{0}'' must match pattern ''{1}''&quot;/&gt;
148 * &lt;/module&gt;
149 * </pre>
150 * <p>Code Example:</p>
151 * <pre>
152 * class MyClass {
153 *   void method1(int v1) {} // OK
154 *   protected method2(int V2) {} // violation, Parameter name 'V2'
155 *                                // must match pattern '^[a-z]([a-z0-9][a-zA-Z0-9]*)?$'
156 *   private method3(int a) {} // OK
157 *   public method4(int b) {} // violation, Parameter name 'b'
158 *                            // must match pattern '^[a-z][a-z0-9][a-zA-Z0-9]*$'
159 * }
160 * </pre>
161 * <p>
162 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
163 * </p>
164 * <p>
165 * Violation Message Keys:
166 * </p>
167 * <ul>
168 * <li>
169 * {@code name.invalidPattern}
170 * </li>
171 * </ul>
172 *
173 * @since 3.0
174 */
175public class ParameterNameCheck extends AbstractNameCheck {
176
177    /**
178     * Allows to skip methods with Override annotation from validation.
179     */
180    private boolean ignoreOverridden;
181
182    /** Access modifiers of methods where parameters are checked. */
183    private AccessModifierOption[] accessModifiers = {
184        AccessModifierOption.PUBLIC,
185        AccessModifierOption.PROTECTED,
186        AccessModifierOption.PACKAGE,
187        AccessModifierOption.PRIVATE,
188    };
189
190    /**
191     * Creates a new {@code ParameterNameCheck} instance.
192     */
193    public ParameterNameCheck() {
194        super("^[a-z][a-zA-Z0-9]*$");
195    }
196
197    /**
198     * Setter to allows to skip methods with Override annotation from validation.
199     *
200     * @param ignoreOverridden Flag for skipping methods with Override annotation.
201     */
202    public void setIgnoreOverridden(boolean ignoreOverridden) {
203        this.ignoreOverridden = ignoreOverridden;
204    }
205
206    /**
207     * Setter to access modifiers of methods where parameters are checked.
208     *
209     * @param accessModifiers access modifiers of methods which should be checked.
210     */
211    public void setAccessModifiers(AccessModifierOption... accessModifiers) {
212        this.accessModifiers =
213            Arrays.copyOf(accessModifiers, accessModifiers.length);
214    }
215
216    @Override
217    public int[] getDefaultTokens() {
218        return getRequiredTokens();
219    }
220
221    @Override
222    public int[] getAcceptableTokens() {
223        return getRequiredTokens();
224    }
225
226    @Override
227    public int[] getRequiredTokens() {
228        return new int[] {TokenTypes.PARAMETER_DEF};
229    }
230
231    @Override
232    protected boolean mustCheckName(DetailAST ast) {
233        boolean checkName = true;
234        final DetailAST parent = ast.getParent();
235        if (ignoreOverridden && isOverriddenMethod(ast)
236                || parent.getType() == TokenTypes.LITERAL_CATCH
237                || parent.getParent().getType() == TokenTypes.LAMBDA
238                || CheckUtil.isReceiverParameter(ast)
239                || !matchAccessModifiers(
240                        CheckUtil.getAccessModifierFromModifiersToken(parent.getParent()))) {
241            checkName = false;
242        }
243        return checkName;
244    }
245
246    /**
247     * Checks whether a method has the correct access modifier to be checked.
248     *
249     * @param accessModifier the access modifier of the method.
250     * @return whether the method matches the expected access modifier.
251     */
252    private boolean matchAccessModifiers(final AccessModifierOption accessModifier) {
253        return Arrays.stream(accessModifiers)
254                .anyMatch(modifier -> modifier == accessModifier);
255    }
256
257    /**
258     * Checks whether a method is annotated with Override annotation.
259     *
260     * @param ast method parameter definition token.
261     * @return true if a method is annotated with Override annotation.
262     */
263    private static boolean isOverriddenMethod(DetailAST ast) {
264        boolean overridden = false;
265
266        final DetailAST parent = ast.getParent().getParent();
267        final Optional<DetailAST> annotation =
268            Optional.ofNullable(parent.getFirstChild().getFirstChild());
269
270        if (annotation.isPresent()) {
271            final Optional<DetailAST> overrideToken =
272                Optional.ofNullable(annotation.get().findFirstToken(TokenTypes.IDENT));
273            if (overrideToken.isPresent() && "Override".equals(overrideToken.get().getText())) {
274                overridden = true;
275            }
276        }
277        return overridden;
278    }
279
280}

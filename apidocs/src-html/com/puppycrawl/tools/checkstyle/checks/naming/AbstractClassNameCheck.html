020package com.puppycrawl.tools.checkstyle.checks.naming;
021
022import java.util.regex.Pattern;
023
024import com.puppycrawl.tools.checkstyle.StatelessCheck;
025import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
026import com.puppycrawl.tools.checkstyle.api.DetailAST;
027import com.puppycrawl.tools.checkstyle.api.TokenTypes;
028
029/**
030 * <p>
031 * Ensures that the names of abstract classes conforming to some pattern
032 * and check that {@code abstract} modifier exists.
033 * </p>
034 * <p>
035 * Rationale: Abstract classes are convenience base class implementations of
036 * interfaces. For this reason, it should be made obvious that a given class
037 * is abstract by prefacing the class name with 'Abstract'.
038 * </p>
039 * <ul>
040 * <li>
041 * Property {@code format} - Specify valid identifiers.
042 * Type is {@code java.util.regex.Pattern}.
043 * Default value is {@code "^Abstract.+$"}.</li>
044 * <li>
045 * Property {@code ignoreModifier} - Control whether to ignore checking for the
046 * {@code abstract} modifier on classes that match the name.
047 * Type is {@code boolean}.
048 * Default value is {@code false}.</li>
049 * <li>
050 * Property {@code ignoreName} - Control whether to ignore checking the name.
051 * Realistically only useful if using the check to identify that match name and
052 * do not have the {@code abstract} modifier.
053 * Type is {@code boolean}.
054 * Default value is {@code false}.
055 * </li>
056 * </ul>
057 * <p>
058 * To configure the check:
059 * </p>
060 * <pre>
061 * &lt;module name="AbstractClassName"/&gt;
062 * </pre>
063 * <p>Example:</p>
064 * <pre>
065 * class Example1 {
066 *   abstract class AbstractFirst {}
067 *   abstract class Second {} // violation 'must match pattern '\^Abstract\.\+\$''
068 *   class AbstractThird {} // violation 'must be declared as 'abstract''
069 *   class Fourth {}
070 *   abstract class GeneratorFifth {} // violation 'must match pattern '\^Abstract\.\+\$''
071 *   class GeneratorSixth {}
072 * }
073 * </pre>
074 * <p>
075 * To configure the check so that it check name
076 * but ignore {@code abstract} modifier:
077 * </p>
078 * <pre>
079 * &lt;module name="AbstractClassName"&gt;
080 *   &lt;property name="ignoreModifier" value="true"/&gt;
081 * &lt;/module&gt;
082 * </pre>
083 * <p>Example:</p>
084 * <pre>
085 * class Example2 {
086 *   abstract class AbstractFirst {}
087 *   abstract class Second {} // violation 'must match pattern '\^Abstract\.\+\$''
088 *   class AbstractThird {}
089 *   class Fourth {}
090 *   abstract class GeneratorFifth {} // violation 'must match pattern '\^Abstract\.\+\$''
091 *   class GeneratorSixth {}
092 * }
093 * </pre>
094 * <p>
095 * To configure the check to ignore name
096 * validation when class declared as 'abstract'
097 * </p>
098 * <pre>
099 * &lt;module name="AbstractClassName"&gt;
100 *   &lt;property name="ignoreName" value="true"/&gt;
101 * &lt;/module&gt;
102 * </pre>
103 * <p>Example:</p>
104 * <pre>
105 * class Example3 {
106 *   abstract class AbstractFirst {}
107 *   abstract class Second {}
108 *   class AbstractThird {} // violation 'must be declared as 'abstract''
109 *   class Fourth {}
110 *   abstract class GeneratorFifth {}
111 *   class GeneratorSixth {}
112 * }
113 * </pre>
114 * <p>
115 * To configure the check
116 * with {@code format}:
117 * </p>
118 * <pre>
119 * &lt;module name="AbstractClassName"&gt;
120 *   &lt;property name="format" value="^Generator.+$"/&gt;
121 * &lt;/module&gt;
122 * </pre>
123 * <p>Example:</p>
124 * <pre>
125 * class Example4 {
126 *   abstract class AbstractFirst {} // violation 'must match pattern '\^Generator\.\+\$''
127 *   abstract class Second {} // violation 'must match pattern '\^Generator\.\+\$''
128 *   class AbstractThird {}
129 *   class Fourth {}
130 *   abstract class GeneratorFifth {}
131 *   class GeneratorSixth {} // violation 'must be declared as 'abstract''
132 * }
133 * </pre>
134 * <p>
135 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
136 * </p>
137 * <p>
138 * Violation Message Keys:
139 * </p>
140 * <ul>
141 * <li>
142 * {@code illegal.abstract.class.name}
143 * </li>
144 * <li>
145 * {@code no.abstract.class.modifier}
146 * </li>
147 * </ul>
148 *
149 * @since 3.2
150 */
151@StatelessCheck
152public final class AbstractClassNameCheck extends AbstractCheck {
153
154    /**
155     * A key is pointing to the warning message text in "messages.properties"
156     * file.
157     */
158    public static final String MSG_ILLEGAL_ABSTRACT_CLASS_NAME = "illegal.abstract.class.name";
159
160    /**
161     * A key is pointing to the warning message text in "messages.properties"
162     * file.
163     */
164    public static final String MSG_NO_ABSTRACT_CLASS_MODIFIER = "no.abstract.class.modifier";
165
166    /**
167     * Control whether to ignore checking for the {@code abstract} modifier on
168     * classes that match the name.
169     */
170    private boolean ignoreModifier;
171
172    /**
173     * Control whether to ignore checking the name. Realistically only useful
174     * if using the check to identify that match name and do not have the
175     * {@code abstract} modifier.
176     */
177    private boolean ignoreName;
178
179    /** Specify valid identifiers. */
180    private Pattern format = Pattern.compile("^Abstract.+$");
181
182    /**
183     * Setter to control whether to ignore checking for the {@code abstract} modifier on
184     * classes that match the name.
185     *
186     * @param value new value
187     */
188    public void setIgnoreModifier(boolean value) {
189        ignoreModifier = value;
190    }
191
192    /**
193     * Setter to control whether to ignore checking the name. Realistically only useful if
194     * using the check to identify that match name and do not have the {@code abstract} modifier.
195     *
196     * @param value new value.
197     */
198    public void setIgnoreName(boolean value) {
199        ignoreName = value;
200    }
201
202    /**
203     * Setter to specify valid identifiers.
204     *
205     * @param pattern the new pattern
206     */
207    public void setFormat(Pattern pattern) {
208        format = pattern;
209    }
210
211    @Override
212    public int[] getDefaultTokens() {
213        return getRequiredTokens();
214    }
215
216    @Override
217    public int[] getRequiredTokens() {
218        return new int[] {TokenTypes.CLASS_DEF};
219    }
220
221    @Override
222    public int[] getAcceptableTokens() {
223        return getRequiredTokens();
224    }
225
226    @Override
227    public void visitToken(DetailAST ast) {
228        visitClassDef(ast);
229    }
230
231    /**
232     * Checks class definition.
233     *
234     * @param ast class definition for check.
235     */
236    private void visitClassDef(DetailAST ast) {
237        final String className =
238            ast.findFirstToken(TokenTypes.IDENT).getText();
239        if (isAbstract(ast)) {
240            // if class has abstract modifier
241            if (!ignoreName && !isMatchingClassName(className)) {
242                log(ast, MSG_ILLEGAL_ABSTRACT_CLASS_NAME, className, format.pattern());
243            }
244        }
245        else if (!ignoreModifier && isMatchingClassName(className)) {
246            log(ast, MSG_NO_ABSTRACT_CLASS_MODIFIER, className);
247        }
248    }
249
250    /**
251     * Checks if declared class is abstract or not.
252     *
253     * @param ast class definition for check.
254     * @return true if a given class declared as abstract.
255     */
256    private static boolean isAbstract(DetailAST ast) {
257        final DetailAST abstractAST = ast.findFirstToken(TokenTypes.MODIFIERS)
258            .findFirstToken(TokenTypes.ABSTRACT);
259
260        return abstractAST != null;
261    }
262
263    /**
264     * Returns true if class name matches format of abstract class names.
265     *
266     * @param className class name for check.
267     * @return true if class name matches format of abstract class names.
268     */
269    private boolean isMatchingClassName(String className) {
270        return format.matcher(className).find();
271    }
272
273}

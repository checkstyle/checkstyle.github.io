020package com.puppycrawl.tools.checkstyle.checks.naming;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * <p>
027 * Checks that interface type parameter names conform to a specified pattern.
028 * </p>
029 * <ul>
030 * <li>
031 * Property {@code format} - Specifies valid identifiers.
032 * Type is {@code java.util.regex.Pattern}.
033 * Default value is {@code "^[A-Z]$"}.
034 * </li>
035 * </ul>
036 * <p>
037 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
038 * </p>
039 * <p>
040 * Violation Message Keys:
041 * </p>
042 * <ul>
043 * <li>
044 * {@code name.invalidPattern}
045 * </li>
046 * </ul>
047 *
048 * @since 5.8
049 */
050public class InterfaceTypeParameterNameCheck
051    extends AbstractNameCheck {
052
053    /** Creates a new {@code InterfaceTypeParameterNameCheck} instance. */
054    public InterfaceTypeParameterNameCheck() {
055        super("^[A-Z]$");
056    }
057
058    @Override
059    public int[] getDefaultTokens() {
060        return getRequiredTokens();
061    }
062
063    @Override
064    public int[] getAcceptableTokens() {
065        return getRequiredTokens();
066    }
067
068    @Override
069    public int[] getRequiredTokens() {
070        return new int[] {
071            TokenTypes.TYPE_PARAMETER,
072        };
073    }
074
075    @Override
076    protected final boolean mustCheckName(DetailAST ast) {
077        final DetailAST location =
078            ast.getParent().getParent();
079        return location.getType() == TokenTypes.INTERFACE_DEF;
080    }
081
082}

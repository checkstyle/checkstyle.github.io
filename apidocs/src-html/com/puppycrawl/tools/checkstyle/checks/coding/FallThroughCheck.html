022import java.util.regex.Matcher;
023import java.util.regex.Pattern;
024
025import com.puppycrawl.tools.checkstyle.StatelessCheck;
026import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
027import com.puppycrawl.tools.checkstyle.api.DetailAST;
028import com.puppycrawl.tools.checkstyle.api.TokenTypes;
029import com.puppycrawl.tools.checkstyle.utils.CodePointUtil;

031/**
032 * <p>
033 * Checks for fall-through in {@code switch} statements.
034 * Finds locations where a {@code case} <b>contains</b> Java code but lacks a
035 * {@code break}, {@code return}, {@code yield}, {@code throw} or {@code continue} statement.
036 * </p>
037 * <p>
038 * The check honors special comments to suppress the warning.
039 * By default, the texts
040 * "fallthru", "fall thru", "fall-thru",
041 * "fallthrough", "fall through", "fall-through"
042 * "fallsthrough", "falls through", "falls-through" (case-sensitive).
043 * The comment containing these words must be all on one line,
044 * and must be on the last non-empty line before the {@code case} triggering
045 * the warning or on the same line before the {@code case}(ugly, but possible).
046 * </p>
047 * <p>
048 * Note: The check assumes that there is no unreachable code in the {@code case}.
049 * </p>
050 * <ul>
051 * <li>
052 * Property {@code checkLastCaseGroup} - Control whether the last case group must be checked.
053 * Type is {@code boolean}.
054 * Default value is {@code false}.
055 * </li>
056 * <li>
057 * Property {@code reliefPattern} - Define the RegExp to match the relief comment that suppresses
058 * the warning about a fall through.
059 * Type is {@code java.util.regex.Pattern}.
060 * Default value is {@code "falls?[ -]?thr(u|ough)"}.
061 * </li>
062 * </ul>
063 * <p>
064 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
065 * </p>
066 * <p>
067 * Violation Message Keys:
068 * </p>
069 * <ul>
070 * <li>
071 * {@code fall.through}
072 * </li>
073 * <li>
074 * {@code fall.through.last}
075 * </li>
076 * </ul>
077 *
078 * @since 3.4
079 */
080@StatelessCheck
081public class FallThroughCheck extends AbstractCheck {
082
083    /**
084     * A key is pointing to the warning message text in "messages.properties"
085     * file.
086     */
087    public static final String MSG_FALL_THROUGH = "fall.through";
088
089    /**
090     * A key is pointing to the warning message text in "messages.properties"
091     * file.
092     */
093    public static final String MSG_FALL_THROUGH_LAST = "fall.through.last";
094
095    /** Control whether the last case group must be checked. */
096    private boolean checkLastCaseGroup;
097
098    /**
099     * Define the RegExp to match the relief comment that suppresses
100     * the warning about a fall through.
101     */
102    private Pattern reliefPattern = Pattern.compile("falls?[ -]?thr(u|ough)");
103
104    @Override
105    public int[] getDefaultTokens() {
106        return getRequiredTokens();
107    }
108
109    @Override
110    public int[] getRequiredTokens() {
111        return new int[] {TokenTypes.CASE_GROUP};
112    }
113
114    @Override
115    public int[] getAcceptableTokens() {
116        return getRequiredTokens();
117    }
118
119    /**
120     * Setter to define the RegExp to match the relief comment that suppresses
121     * the warning about a fall through.
122     *
123     * @param pattern
124     *            The regular expression pattern.
125     * @since 4.0
126     */
127    public void setReliefPattern(Pattern pattern) {
128        reliefPattern = pattern;
129    }
130
131    /**
132     * Setter to control whether the last case group must be checked.
133     *
134     * @param value new value of the property.
135     * @since 4.0
136     */
137    public void setCheckLastCaseGroup(boolean value) {
138        checkLastCaseGroup = value;
139    }
140
141    @Override
142    public void visitToken(DetailAST ast) {
143        final DetailAST nextGroup = ast.getNextSibling();
144        final boolean isLastGroup = nextGroup.getType() != TokenTypes.CASE_GROUP;
145        if (!isLastGroup || checkLastCaseGroup) {
146            final DetailAST slist = ast.findFirstToken(TokenTypes.SLIST);
147
148            if (slist != null && !isTerminated(slist, true, true)
149                && !hasFallThroughComment(ast, nextGroup)) {
150                if (isLastGroup) {
151                    log(ast, MSG_FALL_THROUGH_LAST);
152                }
153                else {
154                    log(nextGroup, MSG_FALL_THROUGH);
155                }
156            }
157        }
158    }
159
160    /**
161     * Checks if a given subtree terminated by return, throw or,
162     * if allowed break, continue.
163     *
164     * @param ast root of given subtree
165     * @param useBreak should we consider break as terminator
166     * @param useContinue should we consider continue as terminator
167     * @return true if the subtree is terminated.
168     */
169    private boolean isTerminated(final DetailAST ast, boolean useBreak,
170                                 boolean useContinue) {
171        final boolean terminated;
172
173        switch (ast.getType()) {
174            case TokenTypes.LITERAL_RETURN:
175            case TokenTypes.LITERAL_YIELD:
176            case TokenTypes.LITERAL_THROW:
177                terminated = true;
178                break;
179            case TokenTypes.LITERAL_BREAK:
180                terminated = useBreak;
181                break;
182            case TokenTypes.LITERAL_CONTINUE:
183                terminated = useContinue;
184                break;
185            case TokenTypes.SLIST:
186                terminated = checkSlist(ast, useBreak, useContinue);
187                break;
188            case TokenTypes.LITERAL_IF:
189                terminated = checkIf(ast, useBreak, useContinue);
190                break;
191            case TokenTypes.LITERAL_FOR:
192            case TokenTypes.LITERAL_WHILE:
193            case TokenTypes.LITERAL_DO:
194                terminated = checkLoop(ast);
195                break;
196            case TokenTypes.LITERAL_TRY:
197                terminated = checkTry(ast, useBreak, useContinue);
198                break;
199            case TokenTypes.LITERAL_SWITCH:
200                terminated = checkSwitch(ast, useContinue);
201                break;
202            case TokenTypes.LITERAL_SYNCHRONIZED:
203                terminated = checkSynchronized(ast, useBreak, useContinue);
204                break;
205            default:
206                terminated = false;
207        }
208        return terminated;
209    }
210
211    /**
212     * Checks if a given SLIST terminated by return, throw or,
213     * if allowed break, continue.
214     *
215     * @param slistAst SLIST to check
216     * @param useBreak should we consider break as terminator
217     * @param useContinue should we consider continue as terminator
218     * @return true if SLIST is terminated.
219     */
220    private boolean checkSlist(final DetailAST slistAst, boolean useBreak,
221                               boolean useContinue) {
222        DetailAST lastStmt = slistAst.getLastChild();
223
224        if (lastStmt.getType() == TokenTypes.RCURLY) {
225            lastStmt = lastStmt.getPreviousSibling();
226        }
227
228        return lastStmt != null
229            && isTerminated(lastStmt, useBreak, useContinue);
230    }
231
232    /**
233     * Checks if a given IF terminated by return, throw or,
234     * if allowed break, continue.
235     *
236     * @param ast IF to check
237     * @param useBreak should we consider break as terminator
238     * @param useContinue should we consider continue as terminator
239     * @return true if IF is terminated.
240     */
241    private boolean checkIf(final DetailAST ast, boolean useBreak,
242                            boolean useContinue) {
243        final DetailAST thenStmt = ast.findFirstToken(TokenTypes.RPAREN)
244                .getNextSibling();
245        final DetailAST elseStmt = thenStmt.getNextSibling();
246
247        return elseStmt != null
248                && isTerminated(thenStmt, useBreak, useContinue)
249                && isTerminated(elseStmt.getFirstChild(), useBreak, useContinue);
250    }
251
252    /**
253     * Checks if a given loop terminated by return, throw or,
254     * if allowed break, continue.
255     *
256     * @param ast loop to check
257     * @return true if loop is terminated.
258     */
259    private boolean checkLoop(final DetailAST ast) {
260        final DetailAST loopBody;
261        if (ast.getType() == TokenTypes.LITERAL_DO) {
262            final DetailAST lparen = ast.findFirstToken(TokenTypes.DO_WHILE);
263            loopBody = lparen.getPreviousSibling();
264        }
265        else {
266            final DetailAST rparen = ast.findFirstToken(TokenTypes.RPAREN);
267            loopBody = rparen.getNextSibling();
268        }
269        return isTerminated(loopBody, false, false);
270    }
271
272    /**
273     * Checks if a given try/catch/finally block terminated by return, throw or,
274     * if allowed break, continue.
275     *
276     * @param ast loop to check
277     * @param useBreak should we consider break as terminator
278     * @param useContinue should we consider continue as terminator
279     * @return true if try/catch/finally block is terminated
280     */
281    private boolean checkTry(final DetailAST ast, boolean useBreak,
282                             boolean useContinue) {
283        final DetailAST finalStmt = ast.getLastChild();
284        boolean isTerminated = finalStmt.getType() == TokenTypes.LITERAL_FINALLY
285                && isTerminated(finalStmt.findFirstToken(TokenTypes.SLIST), useBreak, useContinue);
286
287        if (!isTerminated) {
288            DetailAST firstChild = ast.getFirstChild();
289
290            if (firstChild.getType() == TokenTypes.RESOURCE_SPECIFICATION) {
291                firstChild = firstChild.getNextSibling();
292            }
293
294            isTerminated = isTerminated(firstChild,
295                    useBreak, useContinue);
296
297            DetailAST catchStmt = ast.findFirstToken(TokenTypes.LITERAL_CATCH);
298            while (catchStmt != null
299                    && isTerminated
300                    && catchStmt.getType() == TokenTypes.LITERAL_CATCH) {
301                final DetailAST catchBody =
302                        catchStmt.findFirstToken(TokenTypes.SLIST);
303                isTerminated = isTerminated(catchBody, useBreak, useContinue);
304                catchStmt = catchStmt.getNextSibling();
305            }
306        }
307        return isTerminated;
308    }
309
310    /**
311     * Checks if a given switch terminated by return, throw or,
312     * if allowed break, continue.
313     *
314     * @param literalSwitchAst loop to check
315     * @param useContinue should we consider continue as terminator
316     * @return true if switch is terminated
317     */
318    private boolean checkSwitch(final DetailAST literalSwitchAst, boolean useContinue) {
319        DetailAST caseGroup = literalSwitchAst.findFirstToken(TokenTypes.CASE_GROUP);
320        boolean isTerminated = caseGroup != null;
321        while (isTerminated && caseGroup.getType() != TokenTypes.RCURLY) {
322            final DetailAST caseBody =
323                caseGroup.findFirstToken(TokenTypes.SLIST);
324            isTerminated = caseBody != null && isTerminated(caseBody, false, useContinue);
325            caseGroup = caseGroup.getNextSibling();
326        }
327        return isTerminated;
328    }
329
330    /**
331     * Checks if a given synchronized block terminated by return, throw or,
332     * if allowed break, continue.
333     *
334     * @param synchronizedAst synchronized block to check.
335     * @param useBreak should we consider break as terminator
336     * @param useContinue should we consider continue as terminator
337     * @return true if synchronized block is terminated
338     */
339    private boolean checkSynchronized(final DetailAST synchronizedAst, boolean useBreak,
340                                      boolean useContinue) {
341        return isTerminated(
342            synchronizedAst.findFirstToken(TokenTypes.SLIST), useBreak, useContinue);
343    }
344
345    /**
346     * Determines if the fall through case between {@code currentCase} and
347     * {@code nextCase} is relieved by an appropriate comment.
348     *
349     * <p>Handles</p>
350     * <pre>
351     * case 1:
352     * /&#42; FALLTHRU &#42;/ case 2:
353     *
354     * switch(i) {
355     * default:
356     * /&#42; FALLTHRU &#42;/}
357     *
358     * case 1:
359     * // FALLTHRU
360     * case 2:
361     *
362     * switch(i) {
363     * default:
364     * // FALLTHRU
365     * </pre>
366     *
367     * @param currentCase AST of the case that falls through to the next case.
368     * @param nextCase AST of the next case.
369     * @return True if a relief comment was found
370     */
371    private boolean hasFallThroughComment(DetailAST currentCase, DetailAST nextCase) {
372        boolean allThroughComment = false;
373        final int endLineNo = nextCase.getLineNo();
374
375        if (matchesComment(reliefPattern, endLineNo)) {
376            allThroughComment = true;
377        }
378        else {
379            final int startLineNo = currentCase.getLineNo();
380            for (int i = endLineNo - 2; i > startLineNo - 1; i--) {
381                final int[] line = getLineCodePoints(i);
382                if (!CodePointUtil.isBlank(line)) {
383                    allThroughComment = matchesComment(reliefPattern, i + 1);
384                    break;
385                }
386            }
387        }
388        return allThroughComment;
389    }
390
391    /**
392     * Does a regular expression match on the given line and checks that a
393     * possible match is within a comment.
394     *
395     * @param pattern The regular expression pattern to use.
396     * @param lineNo The line number in the file.
397     * @return True if a match was found inside a comment.
398     */
399    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
400    @SuppressWarnings("deprecation")
401    private boolean matchesComment(Pattern pattern, int lineNo) {
402        final String line = getLine(lineNo - 1);
403
404        final Matcher matcher = pattern.matcher(line);
405        return matcher.find()
406                && getFileContents().hasIntersectionWithComment(
407                        lineNo, matcher.start(), lineNo, matcher.end());
408    }

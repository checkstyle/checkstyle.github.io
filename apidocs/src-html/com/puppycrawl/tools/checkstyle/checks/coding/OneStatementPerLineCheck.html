020package com.puppycrawl.tools.checkstyle.checks.coding;
021
022import java.util.ArrayDeque;
023import java.util.Deque;
024
025import com.puppycrawl.tools.checkstyle.FileStatefulCheck;
026import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
027import com.puppycrawl.tools.checkstyle.api.DetailAST;
028import com.puppycrawl.tools.checkstyle.api.TokenTypes;
029
030/**
031 * <p>
032 * Checks that there is only one statement per line.
033 * </p>
034 * <p>
035 * Rationale: It's very difficult to read multiple statements on one line.
036 * </p>
037 * <p>
038 * In the Java programming language, statements are the fundamental unit of
039 * execution. All statements except blocks are terminated by a semicolon.
040 * Blocks are denoted by open and close curly braces.
041 * </p>
042 * <p>
043 * OneStatementPerLineCheck checks the following types of statements:
044 * variable declaration statements, empty statements, import statements,
045 * assignment statements, expression statements, increment statements,
046 * object creation statements, 'for loop' statements, 'break' statements,
047 * 'continue' statements, 'return' statements.
048 * </p>
049 * <p>
050 * The following examples will be flagged as a violation:
051 * </p>
052 * <pre>
053 * //Each line causes violation:
054 * int var1; int var2;
055 * var1 = 1; var2 = 2;
056 * int var1 = 1; int var2 = 2;
057 * var1++; var2++;
058 * Object obj1 = new Object(); Object obj2 = new Object();
059 * import java.io.EOFException; import java.io.BufferedReader;
060 * ;; //two empty statements on the same line.
061 *
062 * //Multi-line statements:
063 * int var1 = 1
064 * ; var2 = 2; //violation here
065 * int o = 1, p = 2,
066 * r = 5; int t; //violation here
067 * </pre>
068 * <p>
069 * An example of how to configure this Check:
070 * </p>
071 * <pre>
072 * &lt;module name=&quot;OneStatementPerLine&quot;/&gt;
073 * </pre>
074 *
075 * @since 5.3
076 */
077@FileStatefulCheck
078public final class OneStatementPerLineCheck extends AbstractCheck {
079
080    /**
081     * A key is pointing to the warning message text in "messages.properties"
082     * file.
083     */
084    public static final String MSG_KEY = "multiple.statements.line";
085
086    /**
087     * Counts number of semicolons in nested lambdas.
088     */
089    private final Deque<Integer> countOfSemiInLambda = new ArrayDeque<>();
090
091    /**
092     * Hold the line-number where the last statement ended.
093     */
094    private int lastStatementEnd = -1;
095
096    /**
097     * Hold the line-number where the last 'for-loop' statement ended.
098     */
099    private int forStatementEnd = -1;
100
101    /**
102     * The for-header usually has 3 statements on one line, but THIS IS OK.
103     */
104    private boolean inForHeader;
105
106    /**
107     * Holds if current token is inside lambda.
108     */
109    private boolean isInLambda;
110
111    /**
112     * Hold the line-number where the last lambda statement ended.
113     */
114    private int lambdaStatementEnd = -1;
115
116    @Override
117    public int[] getDefaultTokens() {
118        return getRequiredTokens();
119    }
120
121    @Override
122    public int[] getAcceptableTokens() {
123        return getRequiredTokens();
124    }
125
126    @Override
127    public int[] getRequiredTokens() {
128        return new int[] {
129            TokenTypes.SEMI,
130            TokenTypes.FOR_INIT,
131            TokenTypes.FOR_ITERATOR,
132            TokenTypes.LAMBDA,
133        };
134    }
135
136    @Override
137    public void beginTree(DetailAST rootAST) {
138        inForHeader = false;
139        lastStatementEnd = -1;
140        forStatementEnd = -1;
141        isInLambda = false;
142    }
143
144    @Override
145    public void visitToken(DetailAST ast) {
146        switch (ast.getType()) {
147            case TokenTypes.SEMI:
148                checkIfSemicolonIsInDifferentLineThanPrevious(ast);
149                break;
150            case TokenTypes.FOR_ITERATOR:
151                forStatementEnd = ast.getLineNo();
152                break;
153            case TokenTypes.LAMBDA:
154                isInLambda = true;
155                countOfSemiInLambda.push(0);
156                break;
157            default:
158                inForHeader = true;
159                break;
160        }
161    }
162
163    @Override
164    public void leaveToken(DetailAST ast) {
165        switch (ast.getType()) {
166            case TokenTypes.SEMI:
167                lastStatementEnd = ast.getLineNo();
168                forStatementEnd = -1;
169                lambdaStatementEnd = -1;
170                break;
171            case TokenTypes.FOR_ITERATOR:
172                inForHeader = false;
173                break;
174            case TokenTypes.LAMBDA:
175                countOfSemiInLambda.pop();
176                if (countOfSemiInLambda.isEmpty()) {
177                    isInLambda = false;
178                }
179                lambdaStatementEnd = ast.getLineNo();
180                break;
181            default:
182                break;
183        }
184    }
185
186    /**
187     * Checks if given semicolon is in different line than previous.
188     * @param ast semicolon to check
189     */
190    private void checkIfSemicolonIsInDifferentLineThanPrevious(DetailAST ast) {
191        DetailAST currentStatement = ast;
192        final boolean hasResourcesPrevSibling =
193                currentStatement.getPreviousSibling() != null
194                        && currentStatement.getPreviousSibling().getType() == TokenTypes.RESOURCES;
195        if (!hasResourcesPrevSibling && isMultilineStatement(currentStatement)) {
196            currentStatement = ast.getPreviousSibling();
197        }
198        if (isInLambda) {
199            int countOfSemiInCurrentLambda = countOfSemiInLambda.pop();
200            countOfSemiInCurrentLambda++;
201            countOfSemiInLambda.push(countOfSemiInCurrentLambda);
202            if (!inForHeader && countOfSemiInCurrentLambda > 1
203                    && isOnTheSameLine(currentStatement,
204                    lastStatementEnd, forStatementEnd,
205                    lambdaStatementEnd)) {
206                log(ast, MSG_KEY);
207            }
208        }
209        else if (!inForHeader && isOnTheSameLine(currentStatement, lastStatementEnd,
210                forStatementEnd, lambdaStatementEnd)) {
211            log(ast, MSG_KEY);
212        }
213    }
214
215    /**
216     * Checks whether two statements are on the same line.
217     * @param ast token for the current statement.
218     * @param lastStatementEnd the line-number where the last statement ended.
219     * @param forStatementEnd the line-number where the last 'for-loop'
220     *                        statement ended.
221     * @param lambdaStatementEnd the line-number where the last lambda
222     *                        statement ended.
223     * @return true if two statements are on the same line.
224     */
225    private static boolean isOnTheSameLine(DetailAST ast, int lastStatementEnd,
226                                           int forStatementEnd, int lambdaStatementEnd) {
227        return lastStatementEnd == ast.getLineNo() && forStatementEnd != ast.getLineNo()
228                && lambdaStatementEnd != ast.getLineNo();
229    }
230
231    /**
232     * Checks whether statement is multiline.
233     * @param ast token for the current statement.
234     * @return true if one statement is distributed over two or more lines.
235     */
236    private static boolean isMultilineStatement(DetailAST ast) {
237        final boolean multiline;
238        if (ast.getPreviousSibling() == null) {
239            multiline = false;
240        }
241        else {
242            final DetailAST prevSibling = ast.getPreviousSibling();
243            multiline = prevSibling.getLineNo() != ast.getLineNo()
244                    && ast.getParent() != null;
245        }
246        return multiline;
247    }
248
249}

020package com.puppycrawl.tools.checkstyle.checks.coding;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import com.puppycrawl.tools.checkstyle.FileStatefulCheck;
026import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
027import com.puppycrawl.tools.checkstyle.api.DetailAST;
028import com.puppycrawl.tools.checkstyle.api.FullIdent;
029import com.puppycrawl.tools.checkstyle.api.TokenTypes;
030import com.puppycrawl.tools.checkstyle.utils.CheckUtil;
031
032/**
033 * <p>
034 * Checks that classes that either override {@code equals()} or {@code hashCode()} also
035 * overrides the other.
036 * This check only verifies that the method declarations match {@code Object.equals(Object)} and
037 * {@code Object.hashCode()} exactly to be considered an override. This check does not verify
038 * invalid method names, parameters other than {@code Object}, or anything else.
039 * </p>
040 * <p>
041 * Rationale: The contract of {@code equals()} and {@code hashCode()} requires that
042 * equal objects have the same hashCode. Therefore, whenever you override
043 * {@code equals()} you must override {@code hashCode()} to ensure that your class can
044 * be used in hash-based collections.
045 * </p>
046 * <p>
047 * To configure the check:
048 * </p>
049 * <pre>
050 * &lt;module name=&quot;EqualsHashCode&quot;/&gt;
051 * </pre>
052 * <p>Example:</p>
053 * <pre>
054 * public static class Example1 {
055 *     public int hashCode() {
056 *         // code
057 *     }
058 *     public boolean equals(String o) { // violation, overloaded implementation of 'equals'
059 *         // code
060 *     }
061 * }
062 * public static class Example2 {
063 *     public boolean equals(Object o) { // violation, no 'hashCode'
064 *         // code
065 *     }
066 *     public boolean equals(String o) {
067 *         // code
068 *     }
069 * }
070 * public static class Example3 {
071 *     public int hashCode() {
072 *         // code
073 *     }
074 *     public boolean equals(Object o) { // OK
075 *         // code
076 *     }
077 *     public boolean equals(String o) {
078 *         // code
079 *     }
080 * }
081 * public static class Example4 {
082 *     public int hashCode() {
083 *         // code
084 *     }
085 *     public boolean equals(java.lang.Object o) { // OK
086 *         // code
087 *    }
088 * }
089 * public static class Example5 {
090 *     public static int hashCode(int i) {
091 *         // code
092 *     }
093 *     public boolean equals(Object o) { // violation, overloaded implementation of 'hashCode'
094 *         // code
095 *     }
096 * }
097 * public static class Example6 {
098 *     public int hashCode() { // violation, overloaded implementation of 'equals'
099 *         // code
100 *     }
101 *     public static boolean equals(Object o, Object o2) {
102 *         // code
103 *     }
104 * }
105 * </pre>
106 * @since 3.0
107 */
108@FileStatefulCheck
109public class EqualsHashCodeCheck
110        extends AbstractCheck {
111
112    // implementation note: we have to use the following members to
113    // keep track of definitions in different inner classes
114
115    /**
116     * A key is pointing to the warning message text in "messages.properties"
117     * file.
118     */
119    public static final String MSG_KEY_HASHCODE = "equals.noHashCode";
120
121    /**
122     * A key is pointing to the warning message text in "messages.properties"
123     * file.
124     */
125    public static final String MSG_KEY_EQUALS = "equals.noEquals";
126
127    /** Maps OBJ_BLOCK to the method definition of equals(). */
128    private final Map<DetailAST, DetailAST> objBlockWithEquals = new HashMap<>();
129
130    /** Maps OBJ_BLOCKs to the method definition of hashCode(). */
131    private final Map<DetailAST, DetailAST> objBlockWithHashCode = new HashMap<>();
132
133    @Override
134    public int[] getDefaultTokens() {
135        return getRequiredTokens();
136    }
137
138    @Override
139    public int[] getAcceptableTokens() {
140        return getRequiredTokens();
141    }
142
143    @Override
144    public int[] getRequiredTokens() {
145        return new int[] {TokenTypes.METHOD_DEF};
146    }
147
148    @Override
149    public void beginTree(DetailAST rootAST) {
150        objBlockWithEquals.clear();
151        objBlockWithHashCode.clear();
152    }
153
154    @Override
155    public void visitToken(DetailAST ast) {
156        if (isEqualsMethod(ast)) {
157            objBlockWithEquals.put(ast.getParent(), ast);
158        }
159        else if (isHashCodeMethod(ast)) {
160            objBlockWithHashCode.put(ast.getParent(), ast);
161        }
162    }
163
164    /**
165     * Determines if an AST is a valid Equals method implementation.
166     *
167     * @param ast the AST to check
168     * @return true if the {code ast} is a Equals method.
169     */
170    private static boolean isEqualsMethod(DetailAST ast) {
171        final DetailAST modifiers = ast.getFirstChild();
172        final DetailAST parameters = ast.findFirstToken(TokenTypes.PARAMETERS);
173
174        return CheckUtil.isEqualsMethod(ast)
175                && isObjectParam(parameters.getFirstChild())
176                && (ast.findFirstToken(TokenTypes.SLIST) != null
177                        || modifiers.findFirstToken(TokenTypes.LITERAL_NATIVE) != null);
178    }
179
180    /**
181     * Determines if an AST is a valid HashCode method implementation.
182     *
183     * @param ast the AST to check
184     * @return true if the {code ast} is a HashCode method.
185     */
186    private static boolean isHashCodeMethod(DetailAST ast) {
187        final DetailAST modifiers = ast.getFirstChild();
188        final DetailAST methodName = ast.findFirstToken(TokenTypes.IDENT);
189        final DetailAST parameters = ast.findFirstToken(TokenTypes.PARAMETERS);
190
191        return "hashCode".equals(methodName.getText())
192                && parameters.getFirstChild() == null
193                && (ast.findFirstToken(TokenTypes.SLIST) != null
194                        || modifiers.findFirstToken(TokenTypes.LITERAL_NATIVE) != null);
195    }
196
197    /**
198     * Determines if an AST is a formal param of type Object.
199     * @param paramNode the AST to check
200     * @return true if firstChild is a parameter of an Object type.
201     */
202    private static boolean isObjectParam(DetailAST paramNode) {
203        final DetailAST typeNode = paramNode.findFirstToken(TokenTypes.TYPE);
204        final FullIdent fullIdent = FullIdent.createFullIdentBelow(typeNode);
205        final String name = fullIdent.getText();
206        return "Object".equals(name) || "java.lang.Object".equals(name);
207    }
208
209    @Override
210    public void finishTree(DetailAST rootAST) {
211        objBlockWithEquals
212            .entrySet().stream().filter(detailASTDetailASTEntry -> {
213                return objBlockWithHashCode.remove(detailASTDetailASTEntry.getKey()) == null;
214            }).forEach(detailASTDetailASTEntry -> {
215                final DetailAST equalsAST = detailASTDetailASTEntry.getValue();
216                log(equalsAST, MSG_KEY_HASHCODE);
217            });
218        objBlockWithHashCode.forEach((key, equalsAST) -> log(equalsAST, MSG_KEY_EQUALS));
219    }
220
221}

037/**
038 * <p>
039 * Checks the distance between declaration of variable and its first usage.
040 * Note : Variable declaration/initialization statements are not counted while calculating length.
041 * </p>
042 * <ul>
043 * <li>
044 * Property {@code allowedDistance} - Specify distance between declaration
045 * of variable and its first usage. Values should be greater than 0.
046 * Type is {@code int}.
047 * Default value is {@code 3}.
048 * </li>
049 * <li>
050 * Property {@code ignoreVariablePattern} - Define RegExp to ignore distance calculation
051 * for variables listed in this pattern.
052 * Type is {@code java.util.regex.Pattern}.
053 * Default value is {@code ""}.
054 * </li>
055 * <li>
056 * Property {@code validateBetweenScopes} - Allow to calculate the distance between
057 * declaration of variable and its first usage in the different scopes.
058 * Type is {@code boolean}.
059 * Default value is {@code false}.
060 * </li>
061 * <li>
062 * Property {@code ignoreFinal} - Allow to ignore variables with a 'final' modifier.
063 * Type is {@code boolean}.
064 * Default value is {@code true}.
065 * </li>
066 * </ul>
067 * <p>
068 * To configure the check with default config:
069 * </p>
070 * <pre>
071 * &lt;module name=&quot;VariableDeclarationUsageDistance&quot;/&gt;
072 * </pre>
073 * <p>Example:</p>
074 * <pre>
075 * public class Test {
076 *
077 *   public void foo1() {
078 *     int num;        // violation, distance = 4
079 *     final double PI;   // OK, final variables not checked
080 *     System.out.println("Statement 1");
081 *     System.out.println("Statement 2");
082 *     System.out.println("Statement 3");
083 *     num = 1;
084 *     PI = 3.14;
085 *   }
086 *
087 *   public void foo2() {
088 *     int a;          // OK, used in different scope
089 *     int b;          // OK, used in different scope
090 *     int count = 0;  // OK, used in different scope
091 *
092 *     {
093 *       System.out.println("Inside inner scope");
094 *       a = 1;
095 *       b = 2;
096 *       count++;
097 *     }
098 *   }
099 * }
100 * </pre>
101 * <p>
102 * Check can detect a block of initialization methods. If a variable is used in
103 * such a block and there are no other statements after variable declaration, then distance = 1.
104 * </p>
105 * <p>Case #1:</p>
106 * <pre>
107 * int minutes = 5;
108 * Calendar cal = Calendar.getInstance();
109 * cal.setTimeInMillis(timeNow);
110 * cal.set(Calendar.SECOND, 0);
111 * cal.set(Calendar.MILLISECOND, 0);
112 * cal.set(Calendar.HOUR_OF_DAY, hh);
113 * cal.set(Calendar.MINUTE, minutes);
114 * </pre>
115 * <p>
116 * The distance for the variable "minutes" is 1 even
117 * though this variable is used in the fifth method's call.
118 * </p>
119 * <p>Case #2:</p>
120 * <pre>
121 * int minutes = 5;
122 * Calendar cal = Calendar.getInstance();
123 * cal.setTimeInMillis(timeNow);
124 * cal.set(Calendar.SECOND, 0);
125 * cal.set(Calendar.MILLISECOND, 0);
126 * <i>System.out.println(cal);</i>
127 * cal.set(Calendar.HOUR_OF_DAY, hh);
128 * cal.set(Calendar.MINUTE, minutes);
129 * </pre>
130 * <p>
131 * The distance for the variable "minutes" is 6 because there is one more expression
132 * (except the initialization block) between the declaration of this variable and its usage.
133 * </p>
134 * <p>
135 * To configure the check to set allowed distance:
136 * </p>
137 * <pre>
138 * &lt;module name=&quot;VariableDeclarationUsageDistance&quot;&gt;
139 *   &lt;property name=&quot;allowedDistance&quot; value=&quot;4&quot;/&gt;
140 * &lt;/module&gt;
141 * </pre>
142 * <p>Example:</p>
143 * <pre>
144 * public class Test {
145 *
146 *   public void foo1() {
147 *     int num;        // OK, distance = 4
148 *     final double PI;   // OK, final variables not checked
149 *     System.out.println("Statement 1");
150 *     System.out.println("Statement 2");
151 *     System.out.println("Statement 3");
152 *     num = 1;
153 *     PI = 3.14;
154 *   }
155 *
156 *   public void foo2() {
157 *     int a;          // OK, used in different scope
158 *     int b;          // OK, used in different scope
159 *     int count = 0;  // OK, used in different scope
160 *
161 *     {
162 *       System.out.println("Inside inner scope");
163 *       a = 1;
164 *       b = 2;
165 *       count++;
166 *     }
167 *   }
168 * }
169 * </pre>
170 * <p>
171 * To configure the check to ignore certain variables:
172 * </p>
173 * <pre>
174 * &lt;module name=&quot;VariableDeclarationUsageDistance&quot;&gt;
175 *   &lt;property name=&quot;ignoreVariablePattern&quot; value=&quot;^num$&quot;/&gt;
176 * &lt;/module&gt;
177 * </pre>
178 * <p>
179 * This configuration ignores variables named "num".
180 * </p>
181 * <p>Example:</p>
182 * <pre>
183 * public class Test {
184 *
185 *   public void foo1() {
186 *     int num;        // OK, variable ignored
187 *     final double PI;   // OK, final variables not checked
188 *     System.out.println("Statement 1");
189 *     System.out.println("Statement 2");
190 *     System.out.println("Statement 3");
191 *     num = 1;
192 *     PI = 3.14;
193 *   }
194 *
195 *   public void foo2() {
196 *     int a;          // OK, used in different scope
197 *     int b;          // OK, used in different scope
198 *     int count = 0;  // OK, used in different scope
199 *
200 *     {
201 *       System.out.println("Inside inner scope");
202 *       a = 1;
203 *       b = 2;
204 *       count++;
205 *     }
206 *   }
207 * }
208 * </pre>
209 * <p>
210 * To configure the check to force validation between scopes:
211 * </p>
212 * <pre>
213 * &lt;module name=&quot;VariableDeclarationUsageDistance&quot;&gt;
214 *   &lt;property name=&quot;validateBetweenScopes&quot; value=&quot;true&quot;/&gt;
215 * &lt;/module&gt;
216 * </pre>
217 * <p>Example:</p>
218 * <pre>
219 * public class Test {
220 *
221 *   public void foo1() {
222 *     int num;        // violation, distance = 4
223 *     final double PI;   // OK, final variables not checked
224 *     System.out.println("Statement 1");
225 *     System.out.println("Statement 2");
226 *     System.out.println("Statement 3");
227 *     num = 1;
228 *     PI = 3.14;
229 *   }
230 *
231 *   public void foo2() {
232 *     int a;          // OK, distance = 2
233 *     int b;          // OK, distance = 3
234 *     int count = 0;  // violation, distance = 4
235 *
236 *     {
237 *       System.out.println("Inside inner scope");
238 *       a = 1;
239 *       b = 2;
240 *       count++;
241 *     }
242 *   }
243 * }
244 * </pre>
245 * <p>
246 * To configure the check to check final variables:
247 * </p>
248 * <pre>
249 * &lt;module name=&quot;VariableDeclarationUsageDistance&quot;&gt;
250 *   &lt;property name=&quot;ignoreFinal&quot; value=&quot;false&quot;/&gt;
251 * &lt;/module&gt;
252 * </pre>
253 * <p>Example:</p>
254 * <pre>
255 * public class Test {
256 *
257 *   public void foo1() {
258 *     int num;        // violation, distance = 4
259 *     final double PI;   // violation, distance = 5
260 *     System.out.println("Statement 1");
261 *     System.out.println("Statement 2");
262 *     System.out.println("Statement 3");
263 *     num = 1;
264 *     PI = 3.14;
265 *   }
266 *
267 *   public void foo2() {
268 *     int a;          // OK, used in different scope
269 *     int b;          // OK, used in different scope
270 *     int count = 0;  // OK, used in different scope
271 *
272 *     {
273 *       System.out.println("Inside inner scope");
274 *       a = 1;
275 *       b = 2;
276 *       count++;
277 *     }
278 *   }
279 * }
280 * </pre>
281 * <p>
282 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
283 * </p>
284 * <p>
285 * Violation Message Keys:
286 * </p>
287 * <ul>
288 * <li>
289 * {@code variable.declaration.usage.distance}
290 * </li>
291 * <li>
292 * {@code variable.declaration.usage.distance.extend}
293 * </li>
294 * </ul>
295 *
296 * @since 5.8
297 */

400    @Override
401    public void visitToken(DetailAST ast) {
402        final int parentType = ast.getParent().getType();
403        final DetailAST modifiers = ast.getFirstChild();
404
405        if (parentType != TokenTypes.OBJBLOCK
406                && (!ignoreFinal || modifiers.findFirstToken(TokenTypes.FINAL) == null)) {
407            final DetailAST variable = ast.findFirstToken(TokenTypes.IDENT);
408
409            if (!isVariableMatchesIgnorePattern(variable.getText())) {
410                final DetailAST semicolonAst = ast.getNextSibling();
411                final Entry<DetailAST, Integer> entry;
412                if (validateBetweenScopes) {
413                    entry = calculateDistanceBetweenScopes(semicolonAst, variable);
414                }
415                else {
416                    entry = calculateDistanceInSingleScope(semicolonAst, variable);
417                }
418                final DetailAST variableUsageAst = entry.getKey();
419                final int dist = entry.getValue();
420                if (dist > allowedDistance
421                        && !isInitializationSequence(variableUsageAst, variable.getText())) {
422                    if (ignoreFinal) {
423                        log(ast, MSG_KEY_EXT, variable.getText(), dist, allowedDistance);
424                    }
425                    else {
426                        log(ast, MSG_KEY, variable.getText(), dist, allowedDistance);
427                    }
428                }
429            }
430        }
431    }

433    /**
434     * Get name of instance whose method is called.
435     *
436     * @param methodCallAst
437     *        DetailAST of METHOD_CALL.
438     * @return name of instance.
439     */
440    private static String getInstanceName(DetailAST methodCallAst) {
441        final String methodCallName =
442                FullIdent.createFullIdentBelow(methodCallAst).getText();
443        final int lastDotIndex = methodCallName.lastIndexOf('.');
444        String instanceName = "";
445        if (lastDotIndex != -1) {
446            instanceName = methodCallName.substring(0, lastDotIndex);
447        }
448        return instanceName;
449    }

451    /**
452     * Processes statements until usage of variable to detect sequence of
453     * initialization methods.
454     *
455     * @param variableUsageAst
456     *        DetailAST of expression that uses variable named variableName.
457     * @param variableName
458     *        name of considered variable.
459     * @return true if statements between declaration and usage of variable are
460     *         initialization methods.
461     */
462    private static boolean isInitializationSequence(
463            DetailAST variableUsageAst, String variableName) {
464        boolean result = true;
465        boolean isUsedVariableDeclarationFound = false;
466        DetailAST currentSiblingAst = variableUsageAst;
467        String initInstanceName = "";
468
469        while (result && !isUsedVariableDeclarationFound && currentSiblingAst != null) {
470            if (currentSiblingAst.getType() == TokenTypes.EXPR
471                    && currentSiblingAst.getFirstChild().getType() == TokenTypes.METHOD_CALL) {
472                final DetailAST methodCallAst = currentSiblingAst.getFirstChild();
473                final String instanceName = getInstanceName(methodCallAst);
474                if (instanceName.isEmpty()) {
475                    result = false;
476                }
477                else if (!instanceName.equals(initInstanceName)) {
478                    if (initInstanceName.isEmpty()) {
479                        initInstanceName = instanceName;
480                    }
481                    else {
482                        result = false;
483                    }
484                }
485
486            }
487            else if (currentSiblingAst.getType() == TokenTypes.VARIABLE_DEF) {
488                final String currentVariableName =
489                        currentSiblingAst.findFirstToken(TokenTypes.IDENT).getText();
490                isUsedVariableDeclarationFound = variableName.equals(currentVariableName);
491            }
492            else {
493                result = currentSiblingAst.getType() == TokenTypes.SEMI;
494            }
495            currentSiblingAst = currentSiblingAst.getPreviousSibling();
496        }
497        return result;
498    }

500    /**
501     * Calculates distance between declaration of variable and its first usage
502     * in single scope.
503     *
504     * @param semicolonAst
505     *        Regular node of Ast which is checked for content of checking
506     *        variable.
507     * @param variableIdentAst
508     *        Variable which distance is calculated for.
509     * @return entry which contains expression with variable usage and distance.
510     *         If variable usage is not found, then the expression node is null,
511     *         although the distance can be greater than zero.
512     */
513    private static Entry<DetailAST, Integer> calculateDistanceInSingleScope(
514            DetailAST semicolonAst, DetailAST variableIdentAst) {
515        int dist = 0;
516        boolean firstUsageFound = false;
517        DetailAST currentAst = semicolonAst;
518        DetailAST variableUsageAst = null;
519
520        while (!firstUsageFound && currentAst != null) {
521            if (currentAst.getFirstChild() != null) {
522                if (isChild(currentAst, variableIdentAst)) {
523                    dist = getDistToVariableUsageInChildNode(currentAst, variableIdentAst, dist);
524                    variableUsageAst = currentAst;
525                    firstUsageFound = true;
526                }
527                else if (currentAst.getType() != TokenTypes.VARIABLE_DEF) {
528                    dist++;
529                }
530            }
531            currentAst = currentAst.getNextSibling();
532        }
533
534        return new SimpleEntry<>(variableUsageAst, dist);
535    }

537    /**
538     * Returns the distance to variable usage for in the child node.
539     *
540     * @param childNode child node.
541     * @param varIdent variable variable identifier.
542     * @param currentDistToVarUsage current distance to the variable usage.
543     * @return the distance to variable usage for in the child node.
544     */
545    private static int getDistToVariableUsageInChildNode(DetailAST childNode, DetailAST varIdent,
546                                                         int currentDistToVarUsage) {
547        DetailAST examineNode = childNode;
548        if (examineNode.getType() == TokenTypes.LABELED_STAT) {
549            examineNode = examineNode.getFirstChild().getNextSibling();
550        }
551
552        int resultDist = currentDistToVarUsage;
553        switch (examineNode.getType()) {
554            case TokenTypes.SLIST:
555                resultDist = 0;
556                break;
557            case TokenTypes.LITERAL_FOR:
558            case TokenTypes.LITERAL_WHILE:
559            case TokenTypes.LITERAL_DO:
560            case TokenTypes.LITERAL_IF:
561            case TokenTypes.LITERAL_SWITCH:
562                if (isVariableInOperatorExpr(examineNode, varIdent)) {
563                    resultDist++;
564                }
565                else {
566                    // variable usage is in inner scope
567                    // reset counters, because we can't determine distance
568                    resultDist = 0;
569                }
570                break;
571            default:
572                if (examineNode.findFirstToken(TokenTypes.SLIST) == null) {
573                    resultDist++;
574                }
575                else {
576                    resultDist = 0;
577                }
578        }
579        return resultDist;
580    }

582    /**
583     * Calculates distance between declaration of variable and its first usage
584     * in multiple scopes.
585     *
586     * @param ast
587     *        Regular node of Ast which is checked for content of checking
588     *        variable.
589     * @param variable
590     *        Variable which distance is calculated for.
591     * @return entry which contains expression with variable usage and distance.
592     */
593    private static Entry<DetailAST, Integer> calculateDistanceBetweenScopes(
594            DetailAST ast, DetailAST variable) {
595        int dist = 0;
596        DetailAST currentScopeAst = ast;
597        DetailAST variableUsageAst = null;
598        while (currentScopeAst != null) {
599            final Entry<List<DetailAST>, Integer> searchResult =
600                    searchVariableUsageExpressions(variable, currentScopeAst);
601
602            currentScopeAst = null;
603
604            final List<DetailAST> variableUsageExpressions = searchResult.getKey();
605            dist += searchResult.getValue();
606
607            // If variable usage exists in a single scope, then look into
608            // this scope and count distance until variable usage.
609            if (variableUsageExpressions.size() == 1) {
610                final DetailAST blockWithVariableUsage = variableUsageExpressions
611                        .get(0);
612                DetailAST exprWithVariableUsage = null;
613                switch (blockWithVariableUsage.getType()) {
614                    case TokenTypes.VARIABLE_DEF:
615                    case TokenTypes.EXPR:
616                        dist++;
617                        break;
618                    case TokenTypes.LITERAL_FOR:
619                    case TokenTypes.LITERAL_WHILE:
620                    case TokenTypes.LITERAL_DO:
621                        exprWithVariableUsage = getFirstNodeInsideForWhileDoWhileBlocks(
622                            blockWithVariableUsage, variable);
623                        break;
624                    case TokenTypes.LITERAL_IF:
625                        exprWithVariableUsage = getFirstNodeInsideIfBlock(
626                            blockWithVariableUsage, variable);
627                        break;
628                    case TokenTypes.LITERAL_SWITCH:
629                        exprWithVariableUsage = getFirstNodeInsideSwitchBlock(
630                            blockWithVariableUsage, variable);
631                        break;
632                    case TokenTypes.LITERAL_TRY:
633                        exprWithVariableUsage =
634                            getFirstNodeInsideTryCatchFinallyBlocks(blockWithVariableUsage,
635                                variable);
636                        break;
637                    default:
638                        exprWithVariableUsage = blockWithVariableUsage.getFirstChild();
639                }
640                currentScopeAst = exprWithVariableUsage;
641                variableUsageAst = blockWithVariableUsage;
642            }
643
644            // If there's no any variable usage, then distance = 0.
645            else if (variableUsageExpressions.isEmpty()) {
646                variableUsageAst = null;
647            }
648            // If variable usage exists in different scopes, then distance =
649            // distance until variable first usage.
650            else {
651                dist++;
652                variableUsageAst = variableUsageExpressions.get(0);
653            }
654        }
655        return new SimpleEntry<>(variableUsageAst, dist);
656    }

658    /**
659     * Searches variable usages starting from specified statement.
660     *
661     * @param variableAst Variable that is used.
662     * @param statementAst DetailAST to start searching from.
663     * @return entry which contains list with found expressions that use the variable
664     *     and distance from specified statement to first found expression.
665     */
666    private static Entry<List<DetailAST>, Integer>
667        searchVariableUsageExpressions(final DetailAST variableAst, final DetailAST statementAst) {
668        final List<DetailAST> variableUsageExpressions = new ArrayList<>();
669        int distance = 0;
670        DetailAST currentStatementAst = statementAst;
671        while (currentStatementAst != null) {
672            if (currentStatementAst.getFirstChild() != null) {
673                if (isChild(currentStatementAst, variableAst)) {
674                    variableUsageExpressions.add(currentStatementAst);
675                }
676                // If expression hasn't been met yet, then distance + 1.
677                else if (variableUsageExpressions.isEmpty()
678                        && !isZeroDistanceToken(currentStatementAst.getType())) {
679                    distance++;
680                }
681            }
682            currentStatementAst = currentStatementAst.getNextSibling();
683        }
684        return new SimpleEntry<>(variableUsageExpressions, distance);
685    }

687    /**
688     * Gets first Ast node inside FOR, WHILE or DO-WHILE blocks if variable
689     * usage is met only inside the block (not in its declaration!).
690     *
691     * @param block
692     *        Ast node represents FOR, WHILE or DO-WHILE block.
693     * @param variable
694     *        Variable which is checked for content in block.
695     * @return If variable usage is met only inside the block
696     *         (not in its declaration!) then return the first Ast node
697     *         of this block, otherwise - null.
698     */
699    private static DetailAST getFirstNodeInsideForWhileDoWhileBlocks(
700            DetailAST block, DetailAST variable) {
701        DetailAST firstNodeInsideBlock = null;
702
703        if (!isVariableInOperatorExpr(block, variable)) {
704            final DetailAST currentNode;
705
706            // Find currentNode for DO-WHILE block.
707            if (block.getType() == TokenTypes.LITERAL_DO) {
708                currentNode = block.getFirstChild();
709            }
710            // Find currentNode for FOR or WHILE block.
711            else {
712                // Looking for RPAREN ( ')' ) token to mark the end of operator
713                // expression.
714                currentNode = block.findFirstToken(TokenTypes.RPAREN).getNextSibling();
715            }
716
717            final int currentNodeType = currentNode.getType();
718
719            if (currentNodeType != TokenTypes.EXPR) {
720                firstNodeInsideBlock = currentNode;
721            }
722        }
723
724        return firstNodeInsideBlock;
725    }

727    /**
728     * Gets first Ast node inside IF block if variable usage is met
729     * only inside the block (not in its declaration!).
730     *
731     * @param block
732     *        Ast node represents IF block.
733     * @param variable
734     *        Variable which is checked for content in block.
735     * @return If variable usage is met only inside the block
736     *         (not in its declaration!) then return the first Ast node
737     *         of this block, otherwise - null.
738     */
739    private static DetailAST getFirstNodeInsideIfBlock(
740            DetailAST block, DetailAST variable) {
741        DetailAST firstNodeInsideBlock = null;
742
743        if (!isVariableInOperatorExpr(block, variable)) {
744            final Optional<DetailAST> slistToken = TokenUtil
745                .findFirstTokenByPredicate(block, token -> token.getType() == TokenTypes.SLIST);
746            final DetailAST lastNode = block.getLastChild();
747            DetailAST previousNode = lastNode.getPreviousSibling();
748
749            if (slistToken.isEmpty()
750                && lastNode.getType() == TokenTypes.LITERAL_ELSE) {
751
752                // Is if statement without '{}' and has a following else branch,
753                // then change previousNode to the if statement body.
754                previousNode = previousNode.getPreviousSibling();
755            }
756
757            final List<DetailAST> variableUsageExpressions = new ArrayList<>();
758            if (isChild(previousNode, variable)) {
759                variableUsageExpressions.add(previousNode);
760            }
761
762            if (isChild(lastNode, variable)) {
763                variableUsageExpressions.add(lastNode);
764            }
765
766            // If variable usage exists in several related blocks, then
767            // firstNodeInsideBlock = null, otherwise if variable usage exists
768            // only inside one block, then get node from
769            // variableUsageExpressions.
770            if (variableUsageExpressions.size() == 1) {
771                firstNodeInsideBlock = variableUsageExpressions.get(0);
772            }
773        }
774
775        return firstNodeInsideBlock;
776    }

779     * Gets first Ast node inside SWITCH block if variable usage is met
780     * only inside the block (not in its declaration!).
781     *
782     * @param block
783     *        Ast node represents SWITCH block.
784     * @param variable
785     *        Variable which is checked for content in block.
786     * @return If variable usage is met only inside the block
787     *         (not in its declaration!) then return the first Ast node
788     *         of this block, otherwise - null.
789     */
790    private static DetailAST getFirstNodeInsideSwitchBlock(
791            DetailAST block, DetailAST variable) {
792        final List<DetailAST> variableUsageExpressions =
793                getVariableUsageExpressionsInsideSwitchBlock(block, variable);
794
795        // If variable usage exists in several related blocks, then
796        // firstNodeInsideBlock = null, otherwise if variable usage exists
797        // only inside one block, then get node from
798        // variableUsageExpressions.
799        DetailAST firstNodeInsideBlock = null;
800        if (variableUsageExpressions.size() == 1) {
801            firstNodeInsideBlock = variableUsageExpressions.get(0);
802        }
803
804        return firstNodeInsideBlock;

807    /**
808     * Helper method for getFirstNodeInsideSwitchBlock to return all variable
809     * usage expressions inside a given switch block.
810     *
811     * @param block the switch block to check.
812     * @param variable variable which is checked for in switch block.
813     * @return List of usages or empty list if none are found.
814     */
815    private static List<DetailAST> getVariableUsageExpressionsInsideSwitchBlock(DetailAST block,
816                                                                            DetailAST variable) {
817        final Optional<DetailAST> firstToken = TokenUtil.findFirstTokenByPredicate(block, child -> {
818            return child.getType() == TokenTypes.SWITCH_RULE
819                    || child.getType() == TokenTypes.CASE_GROUP;
820        });
821
822        final List<DetailAST> variableUsageExpressions = new ArrayList<>();
823
824        firstToken.ifPresent(token -> {
825            TokenUtil.forEachChild(block, token.getType(), child -> {
826                final DetailAST lastNodeInCaseGroup = child.getLastChild();
827                if (isChild(lastNodeInCaseGroup, variable)) {
828                    variableUsageExpressions.add(lastNodeInCaseGroup);
829                }
830            });
831        });
832
833        return variableUsageExpressions;
834    }

836    /**
837     * Gets first Ast node inside TRY-CATCH-FINALLY blocks if variable usage is
838     * met only inside the block (not in its declaration!).
839     *
840     * @param block
841     *        Ast node represents TRY-CATCH-FINALLY block.
842     * @param variable
843     *        Variable which is checked for content in block.
844     * @return If variable usage is met only inside the block
845     *         (not in its declaration!) then return the first Ast node
846     *         of this block, otherwise - null.
847     */
848    private static DetailAST getFirstNodeInsideTryCatchFinallyBlocks(
849            DetailAST block, DetailAST variable) {
850        DetailAST currentNode = block.getFirstChild();
851        final List<DetailAST> variableUsageExpressions =
852                new ArrayList<>();
853
854        // Checking variable usage inside TRY block.
855        if (isChild(currentNode, variable)) {
856            variableUsageExpressions.add(currentNode);
857        }
858
859        // Switch on CATCH block.
860        currentNode = currentNode.getNextSibling();
861
862        // Checking variable usage inside all CATCH blocks.
863        while (currentNode != null
864                && currentNode.getType() == TokenTypes.LITERAL_CATCH) {
865            final DetailAST catchBlock = currentNode.getLastChild();
866
867            if (isChild(catchBlock, variable)) {
868                variableUsageExpressions.add(catchBlock);
869            }
870            currentNode = currentNode.getNextSibling();
871        }
872
873        // Checking variable usage inside FINALLY block.
874        if (currentNode != null) {
875            final DetailAST finalBlock = currentNode.getLastChild();
876
877            if (isChild(finalBlock, variable)) {
878                variableUsageExpressions.add(finalBlock);
879            }
880        }
881
882        DetailAST variableUsageNode = null;
883
884        // If variable usage exists in several related blocks, then
885        // firstNodeInsideBlock = null, otherwise if variable usage exists
886        // only inside one block, then get node from
887        // variableUsageExpressions.
888        if (variableUsageExpressions.size() == 1) {
889            variableUsageNode = variableUsageExpressions.get(0).getFirstChild();
890        }
891
892        return variableUsageNode;
893    }

895    /**
896     * Checks if variable is in operator declaration. For instance:
897     * <pre>
898     * boolean b = true;
899     * if (b) {...}
900     * </pre>
901     * Variable 'b' is in declaration of operator IF.
902     *
903     * @param operator
904     *        Ast node which represents operator.
905     * @param variable
906     *        Variable which is checked for content in operator.
907     * @return true if operator contains variable in its declaration, otherwise
908     *         - false.
909     */
910    private static boolean isVariableInOperatorExpr(
911            DetailAST operator, DetailAST variable) {
912        boolean isVarInOperatorDeclaration = false;
913
914        DetailAST ast = operator.findFirstToken(TokenTypes.LPAREN);
915
916        // Look if variable is in operator expression
917        while (ast.getType() != TokenTypes.RPAREN) {
918            if (isChild(ast, variable)) {
919                isVarInOperatorDeclaration = true;
920                break;
921            }
922            ast = ast.getNextSibling();
923        }
924
925        // Variable may be met in ELSE declaration
926        // So, check variable usage in these declarations.
927        if (!isVarInOperatorDeclaration) {
928            final DetailAST elseBlock = operator.getLastChild();
929
930            if (elseBlock.getType() == TokenTypes.LITERAL_ELSE) {
931                // Get IF followed by ELSE
932                final DetailAST firstNodeInsideElseBlock = elseBlock.getFirstChild();
933
934                if (firstNodeInsideElseBlock.getType() == TokenTypes.LITERAL_IF) {
935                    isVarInOperatorDeclaration =
936                        isVariableInOperatorExpr(firstNodeInsideElseBlock, variable);
937                }
938            }
939        }
940
941        return isVarInOperatorDeclaration;
942    }

944    /**
945     * Checks if Ast node contains given element.
946     *
947     * @param parent
948     *        Node of AST.
949     * @param ast
950     *        Ast element which is checked for content in Ast node.
951     * @return true if Ast element was found in Ast node, otherwise - false.
952     */
953    private static boolean isChild(DetailAST parent, DetailAST ast) {
954        boolean isChild = false;
955        DetailAST curNode = parent.getFirstChild();
956
957        while (curNode != null) {
958            if (curNode.getType() == ast.getType() && curNode.getText().equals(ast.getText())) {
959                isChild = true;
960                break;
961            }
962
963            DetailAST toVisit = curNode.getFirstChild();
964            while (toVisit == null) {
965                toVisit = curNode.getNextSibling();
966                curNode = curNode.getParent();
967
968                if (curNode == parent) {
969                    break;
970                }
971            }
972
973            curNode = toVisit;
974        }
975
976        return isChild;
977    }

991    /**
992     * Check if the token should be ignored for distance counting.
993     * For example,
994     * <pre>
995     *     try (final AutoCloseable t = new java.io.StringReader(a);) {
996     *     }
997     * </pre>
998     * final is a zero-distance token and should be ignored for distance counting.
999     * <pre>
1000     *     class Table implements Comparator&lt;Integer&gt;{
1001     *     }
1002     * </pre>
1003     * An inner class may be defined. Both tokens implements and extends
1004     * are zero-distance tokens.
1005     * <pre>
1006     *     public int method(Object b){
1007     *     }
1008     * </pre>
1009     * public is a modifier and zero-distance token. int is a type and
1010     * zero-distance token.
1011     *
1012     * @param type
1013     *        Token type of the ast node.
1014     * @return true if it should be ignored for distance counting, otherwise false.
1015     */
1016    private static boolean isZeroDistanceToken(int type) {
1017        return type == TokenTypes.VARIABLE_DEF
1018                || type == TokenTypes.TYPE
1019                || type == TokenTypes.MODIFIERS
1020                || type == TokenTypes.RESOURCE
1021                || type == TokenTypes.EXTENDS_CLAUSE
1022                || type == TokenTypes.IMPLEMENTS_CLAUSE;
1023    }

042/**
043 * <p>
044 * Checks that a local variable is declared and/or assigned, but not used.
045 * Doesn't support
046 * <a href="https://docs.oracle.com/javase/specs/jls/se17/html/jls-14.html#jls-14.30">
047 * pattern variables yet</a>.
048 * Doesn't check
049 * <a href="https://docs.oracle.com/javase/specs/jls/se17/html/jls-4.html#jls-4.12.3">
050 * array components</a> as array
051 * components are classified as different kind of variables by
052 * <a href="https://docs.oracle.com/javase/specs/jls/se17/html/index.html">JLS</a>.
053 * </p>
054 * <p>
055 * To configure the check:
056 * </p>
057 * <pre>
058 * &lt;module name=&quot;UnusedLocalVariable&quot;/&gt;
059 * </pre>
060 * <p>
061 * Example:
062 * </p>
063 * <pre>
064 * class Test {
065 *
066 *     int a;
067 *
068 *     {
069 *         int k = 12; // violation, assigned and updated but never used
070 *         k++;
071 *     }
072 *
073 *     Test(int a) {   // ok as 'a' is a constructor parameter not a local variable
074 *         this.a = 12;
075 *     }
076 *
077 *     void method(int b) {
078 *         int a = 10;             // violation
079 *         int[] arr = {1, 2, 3};  // violation
080 *         int[] anotherArr = {1}; // ok
081 *         anotherArr[0] = 4;
082 *     }
083 *
084 *     String convertValue(String newValue) {
085 *         String s = newValue.toLowerCase(); // violation
086 *         return newValue.toLowerCase();
087 *     }
088 *
089 *     void read() throws IOException {
090 *         BufferedReader reader = new BufferedReader(new InputStreamReader(System.in));
091 *         String s; // violation
092 *         while ((s = reader.readLine()) != null) {
093 *         }
094 *         try (BufferedReader reader1 // ok as 'reader1' is a resource and resources are closed
095 *                                     // at the end of the statement
096 *             = new BufferedReader(new FileReader("abc.txt"))) {
097 *         }
098 *         try {
099 *         } catch (Exception e) {     // ok as e is an exception parameter
100 *         }
101 *     }
102 *
103 *     void loops() {
104 *         int j = 12;
105 *         for (int i = 0; j &lt; 11; i++) { // violation, unused local variable 'i'.
106 *         }
107 *         for (int p = 0; j &lt; 11; p++)   // ok
108 *             p /= 2;
109 *     }
110 *
111 *     void lambdas() {
112 *         Predicate&lt;String&gt; obj = (String str) -&gt; { // ok as 'str' is a lambda parameter
113 *             return true;
114 *         };
115 *         obj.test("test");
116 *     }
117 * }
118 * </pre>
119 * <p>
120 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
121 * </p>
122 * <p>
123 * Violation Message Keys:
124 * </p>
125 * <ul>
126 * <li>
127 * {@code unused.local.var}
128 * </li>
129 * </ul>
130 *
131 * @since 9.3
132 */
133@FileStatefulCheck
134public class UnusedLocalVariableCheck extends AbstractCheck {

274    @Override
275    public void beginTree(DetailAST root) {
276        variables.clear();
277        typeDeclarations.clear();
278        typeDeclAstToTypeDeclDesc.clear();
279        anonInnerAstToTypeDeclDesc.clear();
280        anonInnerClassHolders.clear();
281        packageName = null;
282        depth = 0;
283    }
284
285    @Override
286    public void visitToken(DetailAST ast) {
287        final int type = ast.getType();
288        if (type == TokenTypes.DOT) {
289            visitDotToken(ast, variables);
290        }
291        else if (type == TokenTypes.VARIABLE_DEF) {
292            visitVariableDefToken(ast);
293        }
294        else if (type == TokenTypes.IDENT) {
295            visitIdentToken(ast, variables);
296        }
297        else if (type == TokenTypes.LITERAL_NEW
298                && isInsideLocalAnonInnerClass(ast)) {
299            visitLocalAnonInnerClass(ast);
300        }
301        else if (TokenUtil.isTypeDeclaration(type)) {
302            visitTypeDeclarationToken(ast);
303        }
304        else if (type == TokenTypes.PACKAGE_DEF) {
305            packageName = CheckUtil.extractQualifiedName(ast.getFirstChild().getNextSibling());
306        }
307    }
308
309    @Override
310    public void leaveToken(DetailAST ast) {
311        if (TokenUtil.isOfType(ast, SCOPES)) {
312            logViolations(ast, variables);
313        }
314        else if (ast.getType() == TokenTypes.COMPILATION_UNIT) {
315            leaveCompilationUnit();
316        }
317        else if (isNonLocalTypeDeclaration(ast)) {
318            depth--;
319            typeDeclarations.pop();
320        }
321    }

323    /**
324     * Visit ast of type {@link TokenTypes#DOT}.
325     *
326     * @param dotAst dotAst
327     * @param variablesStack stack of all the relevant variables in the scope
328     */
329    private static void visitDotToken(DetailAST dotAst, Deque<VariableDesc> variablesStack) {
330        if (dotAst.getParent().getType() != TokenTypes.LITERAL_NEW
331                && shouldCheckIdentTokenNestedUnderDot(dotAst)) {
332            checkIdentifierAst(dotAst.findFirstToken(TokenTypes.IDENT), variablesStack);
333        }
334    }
335
336    /**
337     * Visit ast of type {@link TokenTypes#VARIABLE_DEF}.
338     *
339     * @param varDefAst varDefAst
340     */
341    private void visitVariableDefToken(DetailAST varDefAst) {
342        addLocalVariables(varDefAst, variables);
343        addInstanceOrClassVar(varDefAst);
344    }
345
346    /**
347     * Visit ast of type {@link TokenTypes#IDENT}.
348     *
349     * @param identAst identAst
350     * @param variablesStack stack of all the relevant variables in the scope
351     */
352    private static void visitIdentToken(DetailAST identAst, Deque<VariableDesc> variablesStack) {
353        final DetailAST parent = identAst.getParent();
354        final boolean isMethodReferenceMethodName = parent.getType() == TokenTypes.METHOD_REF
355                && parent.getFirstChild() != identAst;
356        final boolean isConstructorReference = parent.getType() == TokenTypes.METHOD_REF
357                && parent.getLastChild().getType() == TokenTypes.LITERAL_NEW;
358        final boolean isNestedClassInitialization =
359                TokenUtil.isOfType(identAst.getNextSibling(), TokenTypes.LITERAL_NEW)
360                && parent.getType() == TokenTypes.DOT;
361
362        if (isNestedClassInitialization || !isMethodReferenceMethodName
363                && !isConstructorReference
364                && !TokenUtil.isOfType(parent, UNACCEPTABLE_PARENT_OF_IDENT)) {
365            checkIdentifierAst(identAst, variablesStack);
366        }
367    }
368
369    /**
370     * Visit the type declaration token.
371     *
372     * @param typeDeclAst type declaration ast
373     */
374    private void visitTypeDeclarationToken(DetailAST typeDeclAst) {
375        if (isNonLocalTypeDeclaration(typeDeclAst)) {
376            final String qualifiedName = getQualifiedTypeDeclarationName(typeDeclAst);
377            final TypeDeclDesc currTypeDecl = new TypeDeclDesc(qualifiedName, depth, typeDeclAst);
378            depth++;
379            typeDeclarations.push(currTypeDecl);
380            typeDeclAstToTypeDeclDesc.put(typeDeclAst, currTypeDecl);
381        }
382    }
383
384    /**
385     * Visit the local anon inner class.
386     *
387     * @param literalNewAst literalNewAst
388     */
389    private void visitLocalAnonInnerClass(DetailAST literalNewAst) {
390        anonInnerAstToTypeDeclDesc.put(literalNewAst, typeDeclarations.peek());
391        anonInnerClassHolders.add(getBlockContainingLocalAnonInnerClass(literalNewAst));
392    }
393
394    /**
395     * Whether ast node of type {@link TokenTypes#LITERAL_NEW} is a part of a local
396     * anonymous inner class.
397     *
398     * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
399     * @return true if variableDefAst is an instance variable in local anonymous inner class
400     */
401    private static boolean isInsideLocalAnonInnerClass(DetailAST literalNewAst) {
402        boolean result = false;
403        final DetailAST lastChild = literalNewAst.getLastChild();
404        if (lastChild != null && lastChild.getType() == TokenTypes.OBJBLOCK) {
405            DetailAST currentAst = literalNewAst;
406            while (currentAst.getType() != TokenTypes.SLIST) {
407                if (TokenUtil.isTypeDeclaration(currentAst.getParent().getType())) {
408                    break;
409                }
410                currentAst = currentAst.getParent();
411            }
412            result = currentAst.getType() == TokenTypes.SLIST;
413        }
414        return result;
415    }
416
417    /**
418     * Traverse {@code variablesStack} stack and log the violations.
419     *
420     * @param scopeAst ast node of type {@link UnusedLocalVariableCheck#SCOPES}
421     * @param variablesStack stack of all the relevant variables in the scope
422     */
423    private void logViolations(DetailAST scopeAst, Deque<VariableDesc> variablesStack) {
424        while (!variablesStack.isEmpty() && variablesStack.peek().getScope() == scopeAst) {
425            final VariableDesc variableDesc = variablesStack.pop();
426            if (!variableDesc.isUsed()
427                    && !variableDesc.isInstVarOrClassVar()) {
428                final DetailAST typeAst = variableDesc.getTypeAst();
429                log(typeAst, MSG_UNUSED_LOCAL_VARIABLE, variableDesc.getName());
430            }
431        }
432    }
433
434    /**
435     * We process all the blocks containing local anonymous inner classes
436     * separately after processing all the other nodes. This is being done
437     * due to the fact the instance variables of local anon inner classes can
438     * cast a shadow on local variables.
439     */
440    private void leaveCompilationUnit() {
441        anonInnerClassHolders.forEach(holder -> {
442            iterateOverBlockContainingLocalAnonInnerClass(holder, new ArrayDeque<>());
443        });
444    }
445
446    /**
447     * Whether a type declaration is non-local. Annotated interfaces are always non-local.
448     *
449     * @param typeDeclAst type declaration ast
450     * @return true if type declaration is non-local
451     */
452    private static boolean isNonLocalTypeDeclaration(DetailAST typeDeclAst) {
453        return TokenUtil.isTypeDeclaration(typeDeclAst.getType())
454                && typeDeclAst.getParent().getType() != TokenTypes.SLIST;
455    }
456
457    /**
458     * Get the block containing local anon inner class.
459     *
460     * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
461     * @return the block containing local anon inner class
462     */
463    private static DetailAST getBlockContainingLocalAnonInnerClass(DetailAST literalNewAst) {
464        DetailAST currentAst = literalNewAst;
465        DetailAST result = null;
466        while (!TokenUtil.isOfType(currentAst, CONTAINERS_FOR_ANON_INNERS)) {
467            if (currentAst.getType() == TokenTypes.LAMBDA
468                    && currentAst.getParent()
469                    .getParent().getParent().getType() == TokenTypes.OBJBLOCK) {
470                result = currentAst;
471                break;
472            }
473            currentAst = currentAst.getParent();
474            result = currentAst;
475        }
476        return result;
477    }

479    /**
480     * Add local variables to the {@code variablesStack} stack.
481     * Also adds the instance variables defined in a local anonymous inner class.
482     *
483     * @param varDefAst ast node of type {@link TokenTypes#VARIABLE_DEF}
484     * @param variablesStack stack of all the relevant variables in the scope
485     */
486    private static void addLocalVariables(DetailAST varDefAst, Deque<VariableDesc> variablesStack) {
487        final DetailAST parentAst = varDefAst.getParent();
488        final DetailAST grandParent = parentAst.getParent();
489        final boolean isInstanceVarInAnonymousInnerClass =
490                grandParent.getType() == TokenTypes.LITERAL_NEW;
491        if (isInstanceVarInAnonymousInnerClass
492                || parentAst.getType() != TokenTypes.OBJBLOCK) {
493            final DetailAST ident = varDefAst.findFirstToken(TokenTypes.IDENT);
494            final VariableDesc desc = new VariableDesc(ident.getText(),
495                    varDefAst.findFirstToken(TokenTypes.TYPE), findScopeOfVariable(varDefAst));
496            if (isInstanceVarInAnonymousInnerClass) {
497                desc.registerAsInstOrClassVar();
498            }
499            variablesStack.push(desc);
500        }
501    }
502
503    /**
504     * Add instance variables and class variables to the
505     * {@link TypeDeclDesc#instanceAndClassVarStack}.
506     *
507     * @param varDefAst ast node of type {@link TokenTypes#VARIABLE_DEF}
508     */
509    private void addInstanceOrClassVar(DetailAST varDefAst) {
510        final DetailAST parentAst = varDefAst.getParent();
511        if (isNonLocalTypeDeclaration(parentAst.getParent())
512                && !isPrivateInstanceVariable(varDefAst)) {
513            final DetailAST ident = varDefAst.findFirstToken(TokenTypes.IDENT);
514            final VariableDesc desc = new VariableDesc(ident.getText());
515            typeDeclAstToTypeDeclDesc.get(parentAst.getParent()).addInstOrClassVar(desc);
516        }
517    }
518
519    /**
520     * Whether instance variable or class variable have private access modifier.
521     *
522     * @param varDefAst ast node of type {@link TokenTypes#VARIABLE_DEF}
523     * @return true if instance variable or class variable have private access modifier
524     */
525    private static boolean isPrivateInstanceVariable(DetailAST varDefAst) {
526        final AccessModifierOption varAccessModifier =
527                CheckUtil.getAccessModifierFromModifiersToken(varDefAst);
528        return varAccessModifier == AccessModifierOption.PRIVATE;
529    }

531    /**
532     * Get the {@link TypeDeclDesc} of the super class of anonymous inner class.
533     *
534     * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
535     * @return {@link TypeDeclDesc} of the super class of anonymous inner class
536     */
537    private TypeDeclDesc getSuperClassOfAnonInnerClass(DetailAST literalNewAst) {
538        TypeDeclDesc obtainedClass = null;
539        final String shortNameOfClass = CheckUtil.getShortNameOfAnonInnerClass(literalNewAst);
540        if (packageName != null && shortNameOfClass.startsWith(packageName)) {
541            final Optional<TypeDeclDesc> classWithCompletePackageName =
542                    typeDeclAstToTypeDeclDesc.values()
543                    .stream()
544                    .filter(typeDeclDesc -> {
545                        return typeDeclDesc.getQualifiedName().equals(shortNameOfClass);
546                    })
547                    .findFirst();
548            if (classWithCompletePackageName.isPresent()) {
549                obtainedClass = classWithCompletePackageName.get();
550            }
551        }
552        else {
553            final List<TypeDeclDesc> typeDeclWithSameName = typeDeclWithSameName(shortNameOfClass);
554            if (!typeDeclWithSameName.isEmpty()) {
555                obtainedClass = getTheNearestClass(
556                        anonInnerAstToTypeDeclDesc.get(literalNewAst).getQualifiedName(),
557                        typeDeclWithSameName);
558            }
559        }
560        return obtainedClass;
561    }

563    /**
564     * Add non-private instance and class variables of the super class of the anonymous class
565     * to the variables stack.
566     *
567     * @param obtainedClass super class of the anon inner class
568     * @param variablesStack stack of all the relevant variables in the scope
569     * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
570     */
571    private void modifyVariablesStack(TypeDeclDesc obtainedClass,
572            Deque<VariableDesc> variablesStack,
573            DetailAST literalNewAst) {
574        if (obtainedClass != null) {
575            final Deque<VariableDesc> instAndClassVarDeque = typeDeclAstToTypeDeclDesc
576                    .get(obtainedClass.getTypeDeclAst())
577                    .getUpdatedCopyOfVarStack(literalNewAst);
578            instAndClassVarDeque.forEach(variablesStack::push);
579        }
580    }

582    /**
583     * Checks if there is a type declaration with same name as the super class.
584     *
585     * @param superClassName name of the super class
586     * @return true if there is another type declaration with same name.
587     */
588    private List<TypeDeclDesc> typeDeclWithSameName(String superClassName) {
589        return typeDeclAstToTypeDeclDesc.values().stream()
590                .filter(typeDeclDesc -> {
591                    return hasSameNameAsSuperClass(superClassName, typeDeclDesc);
592                })
593                .collect(Collectors.toList());
594    }
595
596    /**
597     * Whether the qualified name of {@code typeDeclDesc} matches the super class name.
598     *
599     * @param superClassName name of the super class
600     * @param typeDeclDesc type declaration description
601     * @return {@code true} if the qualified name of {@code typeDeclDesc}
602     *         matches the super class name
603     */
604    private boolean hasSameNameAsSuperClass(String superClassName, TypeDeclDesc typeDeclDesc) {
605        final boolean result;
606        if (packageName == null && typeDeclDesc.getDepth() == 0) {
607            result = typeDeclDesc.getQualifiedName().equals(superClassName);
608        }
609        else {
610            result = typeDeclDesc.getQualifiedName()
611                    .endsWith(PACKAGE_SEPARATOR + superClassName);
612        }
613        return result;
614    }

616    /**
617     * For all type declarations with the same name as the superclass, gets the nearest type
618     * declaration.
619     *
620     * @param outerTypeDeclName outer type declaration of anonymous inner class
621     * @param typeDeclWithSameName typeDeclarations which have the same name as the super class
622     * @return the nearest class
623     */
624    private static TypeDeclDesc getTheNearestClass(String outerTypeDeclName,
625            List<TypeDeclDesc> typeDeclWithSameName) {
626        return Collections.min(typeDeclWithSameName, (first, second) -> {
627            return getTypeDeclarationNameMatchingCountDiff(outerTypeDeclName, first, second);
628        });
629    }
630
631    /**
632     * Get the difference between type declaration name matching count. If the
633     * difference between them is zero, then their depth is compared to obtain the result.
634     *
635     * @param outerTypeDeclName outer type declaration of anonymous inner class
636     * @param firstTypeDecl first input type declaration
637     * @param secondTypeDecl second input type declaration
638     * @return difference between type declaration name matching count
639     */
640    private static int getTypeDeclarationNameMatchingCountDiff(String outerTypeDeclName,
641                                                               TypeDeclDesc firstTypeDecl,
642                                                               TypeDeclDesc secondTypeDecl) {
643        int diff = Integer.compare(
644            CheckUtil.typeDeclarationNameMatchingCount(
645                outerTypeDeclName, secondTypeDecl.getQualifiedName()),
646            CheckUtil.typeDeclarationNameMatchingCount(
647                outerTypeDeclName, firstTypeDecl.getQualifiedName()));
648        if (diff == 0) {
649            diff = Integer.compare(firstTypeDecl.getDepth(), secondTypeDecl.getDepth());
650        }
651        return diff;
652    }

654    /**
655     * Get qualified type declaration name from type ast.
656     *
657     * @param typeDeclAst type declaration ast
658     * @return qualified name of type declaration
659     */
660    private String getQualifiedTypeDeclarationName(DetailAST typeDeclAst) {
661        final String className = typeDeclAst.findFirstToken(TokenTypes.IDENT).getText();
662        String outerClassQualifiedName = null;
663        if (!typeDeclarations.isEmpty()) {
664            outerClassQualifiedName = typeDeclarations.peek().getQualifiedName();
665        }
666        return CheckUtil
667            .getQualifiedTypeDeclarationName(packageName, outerClassQualifiedName, className);
668    }

670    /**
671     * Iterate over all the ast nodes present under {@code ast}.
672     *
673     * @param ast ast
674     * @param variablesStack stack of all the relevant variables in the scope
675     */
676    private void iterateOverBlockContainingLocalAnonInnerClass(
677            DetailAST ast, Deque<VariableDesc> variablesStack) {
678        DetailAST currNode = ast;
679        while (currNode != null) {
680            customVisitToken(currNode, variablesStack);
681            DetailAST toVisit = currNode.getFirstChild();
682            while (currNode != ast && toVisit == null) {
683                customLeaveToken(currNode, variablesStack);
684                toVisit = currNode.getNextSibling();
685                currNode = currNode.getParent();
686            }
687            currNode = toVisit;
688        }
689    }
690
691    /**
692     * Visit all ast nodes under {@link UnusedLocalVariableCheck#anonInnerClassHolders} once
693     * again.
694     *
695     * @param ast ast
696     * @param variablesStack stack of all the relevant variables in the scope
697     */
698    private void customVisitToken(DetailAST ast, Deque<VariableDesc> variablesStack) {
699        final int type = ast.getType();
700        if (type == TokenTypes.DOT) {
701            visitDotToken(ast, variablesStack);
702        }
703        else if (type == TokenTypes.VARIABLE_DEF) {
704            addLocalVariables(ast, variablesStack);
705        }
706        else if (type == TokenTypes.IDENT) {
707            visitIdentToken(ast, variablesStack);
708        }
709        else if (isInsideLocalAnonInnerClass(ast)) {
710            final TypeDeclDesc obtainedClass = getSuperClassOfAnonInnerClass(ast);
711            modifyVariablesStack(obtainedClass, variablesStack, ast);
712        }
713    }

715    /**
716     * Leave all ast nodes under {@link UnusedLocalVariableCheck#anonInnerClassHolders} once
717     * again.
718     *
719     * @param ast ast
720     * @param variablesStack stack of all the relevant variables in the scope
721     */
722    private void customLeaveToken(DetailAST ast, Deque<VariableDesc> variablesStack) {
723        logViolations(ast, variablesStack);
724    }

726    /**
727     * Whether to check identifier token nested under dotAst.
728     *
729     * @param dotAst dotAst
730     * @return true if ident nested under dotAst should be checked
731     */
732    private static boolean shouldCheckIdentTokenNestedUnderDot(DetailAST dotAst) {
733
734        return TokenUtil.findFirstTokenByPredicate(dotAst,
735                        childAst -> {
736                            return TokenUtil.isOfType(childAst,
737                                    UNACCEPTABLE_CHILD_OF_DOT);
738                        })
739                .isEmpty();
740    }

742    /**
743     * Checks the identifier ast.
744     *
745     * @param identAst ast of type {@link TokenTypes#IDENT}
746     * @param variablesStack stack of all the relevant variables in the scope
747     */
748    private static void checkIdentifierAst(DetailAST identAst, Deque<VariableDesc> variablesStack) {
749        for (VariableDesc variableDesc : variablesStack) {
750            if (identAst.getText().equals(variableDesc.getName())
751                    && !isLeftHandSideValue(identAst)) {
752                variableDesc.registerAsUsed();
753                break;
754            }
755        }
756    }

758    /**
759     * Find the scope of variable.
760     *
761     * @param variableDef ast of type {@link TokenTypes#VARIABLE_DEF}
762     * @return scope of variableDef
763     */
764    private static DetailAST findScopeOfVariable(DetailAST variableDef) {
765        final DetailAST result;
766        final DetailAST parentAst = variableDef.getParent();
767        if (TokenUtil.isOfType(parentAst, TokenTypes.SLIST, TokenTypes.OBJBLOCK)) {
768            result = parentAst;
769        }
770        else {
771            result = parentAst.getParent();
772        }
773        return result;
774    }

776    /**
777     * Checks whether the ast of type {@link TokenTypes#IDENT} is
778     * used as left-hand side value. An identifier is being used as a left-hand side
779     * value if it is used as the left operand of an assignment or as an
780     * operand of a stand-alone increment or decrement.
781     *
782     * @param identAst ast of type {@link TokenTypes#IDENT}
783     * @return true if identAst is used as a left-hand side value
784     */
785    private static boolean isLeftHandSideValue(DetailAST identAst) {
786        final DetailAST parent = identAst.getParent();
787        return isStandAloneIncrementOrDecrement(identAst)
788                || parent.getType() == TokenTypes.ASSIGN
789                && identAst != parent.getLastChild();
790    }
791
792    /**
793     * Checks whether the ast of type {@link TokenTypes#IDENT} is used as
794     * an operand of a stand-alone increment or decrement.
795     *
796     * @param identAst ast of type {@link TokenTypes#IDENT}
797     * @return true if identAst is used as an operand of stand-alone
798     *         increment or decrement
799     */
800    private static boolean isStandAloneIncrementOrDecrement(DetailAST identAst) {
801        final DetailAST parent = identAst.getParent();
802        final DetailAST grandParent = parent.getParent();
803        return TokenUtil.isOfType(parent, INCREMENT_AND_DECREMENT_TOKENS)
804                && TokenUtil.isOfType(grandParent, TokenTypes.EXPR)
805                && !isIncrementOrDecrementVariableUsed(grandParent);
806    }

808    /**
809     * A variable with increment or decrement operator is considered used if it
810     * is used as an argument or as an array index or for assigning value
811     * to a variable.
812     *
813     * @param exprAst ast of type {@link TokenTypes#EXPR}
814     * @return true if variable nested in exprAst is used
815     */
816    private static boolean isIncrementOrDecrementVariableUsed(DetailAST exprAst) {
817        return TokenUtil.isOfType(exprAst.getParent(),
818                TokenTypes.ELIST, TokenTypes.INDEX_OP, TokenTypes.ASSIGN)
819                && exprAst.getParent().getParent().getType() != TokenTypes.FOR_ITERATOR;
820    }

822    /**
823     * Maintains information about the variable.
824     */
825    private static final class VariableDesc {
826
827        /**
828         * The name of the variable.
829         */

831
832        /**
833         * Ast of type {@link TokenTypes#TYPE}.
834         */
835        private final DetailAST typeAst;
836
837        /**
838         * The scope of variable is determined by the ast of type
839         * {@link TokenTypes#SLIST} or {@link TokenTypes#LITERAL_FOR}

843
844        /**
845         * Is an instance variable or a class variable.

849        /**

854        /**

876        }

896        /**

941    /**
942     * Maintains information about the type declaration.
943     * Any ast node of type {@link TokenTypes#CLASS_DEF} or {@link TokenTypes#INTERFACE_DEF}
944     * or {@link TokenTypes#ENUM_DEF} or {@link TokenTypes#ANNOTATION_DEF}
945     * or {@link TokenTypes#RECORD_DEF} is considered as a type declaration.
946     */
947    private static final class TypeDeclDesc {
948
949        /**
950         * Complete type declaration name with package name and outer type declaration name.
951         */
952        private final String qualifiedName;
953
954        /**
955         * Depth of nesting of type declaration.
956         */
957        private final int depth;
958
959        /**
960         * Type declaration ast node.
961         */
962        private final DetailAST typeDeclAst;
963
964        /**
965         * A stack of type declaration's instance and static variables.
966         */
967        private final Deque<VariableDesc> instanceAndClassVarStack;
968
969        /**
970         * Create a new TypeDeclDesc instance.
971         *
972         * @param qualifiedName qualified name
973         * @param depth depth of nesting
974         * @param typeDeclAst type declaration ast node
975         */
976        private TypeDeclDesc(String qualifiedName, int depth,
977                DetailAST typeDeclAst) {
978            this.qualifiedName = qualifiedName;
979            this.depth = depth;
980            this.typeDeclAst = typeDeclAst;
981            instanceAndClassVarStack = new ArrayDeque<>();
982        }
983
984        /**
985         * Get the complete type declaration name i.e. type declaration name with package name
986         * and outer type declaration name.
987         *
988         * @return qualified class name
989         */
990        public String getQualifiedName() {
991            return qualifiedName;
992        }
993
994        /**
995         * Get the depth of type declaration.
996         *
997         * @return the depth of nesting of type declaration
998         */
999        public int getDepth() {
1000            return depth;
1001        }
1002
1003        /**
1004         * Get the type declaration ast node.
1005         *
1006         * @return ast node of the type declaration
1007         */
1008        public DetailAST getTypeDeclAst() {
1009            return typeDeclAst;
1010        }
1011
1012        /**
1013         * Get the copy of variables in instanceAndClassVar stack with updated scope.
1014         *
1015         * @param literalNewAst ast node of type {@link TokenTypes#LITERAL_NEW}
1016         * @return copy of variables in instanceAndClassVar stack with updated scope.
1017         */
1018        public Deque<VariableDesc> getUpdatedCopyOfVarStack(DetailAST literalNewAst) {
1019            final DetailAST updatedScope = literalNewAst.getLastChild();
1020            final Deque<VariableDesc> instAndClassVarDeque = new ArrayDeque<>();
1021            instanceAndClassVarStack.forEach(instVar -> {
1022                final VariableDesc variableDesc = new VariableDesc(instVar.getName(),
1023                        instVar.getTypeAst(), updatedScope);
1024                variableDesc.registerAsInstOrClassVar();
1025                instAndClassVarDeque.push(variableDesc);
1026            });
1027            return instAndClassVarDeque;
1028        }
1029
1030        /**
1031         * Add an instance variable or class variable to the stack.
1032         *
1033         * @param variableDesc variable to be added
1034         */
1035        public void addInstOrClassVar(VariableDesc variableDesc) {
1036            instanceAndClassVarStack.push(variableDesc);
1037        }
1038    }

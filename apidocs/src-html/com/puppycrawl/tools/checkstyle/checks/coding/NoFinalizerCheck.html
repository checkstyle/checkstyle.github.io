020package com.puppycrawl.tools.checkstyle.checks.coding;
021
022import com.puppycrawl.tools.checkstyle.StatelessCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.TokenTypes;
026
027/**
028 * Checks that no method having zero parameters is defined
029 * using the name <em>finalize</em>.
030 *
031 * @author fqian@google.com (Feng Qian)
032 * @author smckay@google.com (Steve McKay)
033 * @author lkuehne
034 */
035@StatelessCheck
036public class NoFinalizerCheck extends AbstractCheck {
037
038    /**
039     * A key is pointing to the warning message text in "messages.properties"
040     * file.
041     */
042    public static final String MSG_KEY = "avoid.finalizer.method";
043
044    @Override
045    public int[] getDefaultTokens() {
046        return getRequiredTokens();
047    }
048
049    @Override
050    public int[] getAcceptableTokens() {
051        return getRequiredTokens();
052    }
053
054    @Override
055    public int[] getRequiredTokens() {
056        return new int[] {TokenTypes.METHOD_DEF};
057    }
058
059    @Override
060    public void visitToken(DetailAST aAST) {
061        final DetailAST mid = aAST.findFirstToken(TokenTypes.IDENT);
062        final String name = mid.getText();
063
064        if ("finalize".equals(name)) {
065            final DetailAST params = aAST.findFirstToken(TokenTypes.PARAMETERS);
066            final boolean hasEmptyParamList =
067                params.findFirstToken(TokenTypes.PARAMETER_DEF) == null;
068
069            if (hasEmptyParamList) {
070                log(aAST.getLineNo(), MSG_KEY);
071            }
072        }
073    }
074
075}

037/**
038 * <p>
039 * Checks for empty line separators before package, all import declarations,
040 * fields, constructors, methods, nested classes,
041 * static initializers and instance initializers.
042 * </p>
043 * <p>
044 * Checks for empty line separators before not only statements but
045 * implementation and documentation comments and blocks as well.
046 * </p>
047 * <p>
048 * ATTENTION: empty line separator is required between token siblings,
049 * not after line where token is found.
050 * If token does not have a sibling of the same type, then empty line
051 * is required at its end (for example for CLASS_DEF it is after '}').
052 * Also, trailing comments are skipped.
053 * </p>
054 * <p>
055 * ATTENTION: violations from multiple empty lines cannot be suppressed via XPath:
056 * <a href="https://github.com/checkstyle/checkstyle/issues/8179">#8179</a>.
057 * </p>
058 * <ul>
059 * <li>
060 * Property {@code allowNoEmptyLineBetweenFields} - Allow no empty line between fields.
061 * Type is {@code boolean}.
062 * Default value is {@code false}.
063 * </li>
064 * <li>
065 * Property {@code allowMultipleEmptyLines} - Allow multiple empty lines between class members.
066 * Type is {@code boolean}.
067 * Default value is {@code true}.
068 * </li>
069 * <li>
070 * Property {@code allowMultipleEmptyLinesInsideClassMembers} - Allow multiple
071 * empty lines inside class members.
072 * Type is {@code boolean}.
073 * Default value is {@code true}.
074 * </li>
075 * <li>
076 * Property {@code tokens} - tokens to check
077 * Type is {@code java.lang.String[]}.
078 * Validation type is {@code tokenSet}.
079 * Default value is:
080 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#PACKAGE_DEF">
081 * PACKAGE_DEF</a>,
082 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#IMPORT">
083 * IMPORT</a>,
084 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#STATIC_IMPORT">
085 * STATIC_IMPORT</a>,
086 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
087 * CLASS_DEF</a>,
088 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
089 * INTERFACE_DEF</a>,
090 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
091 * ENUM_DEF</a>,
092 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
093 * STATIC_INIT</a>,
094 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INSTANCE_INIT">
095 * INSTANCE_INIT</a>,
096 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
097 * METHOD_DEF</a>,
098 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
099 * CTOR_DEF</a>,
100 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#VARIABLE_DEF">
101 * VARIABLE_DEF</a>,
102 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#RECORD_DEF">
103 * RECORD_DEF</a>,
104 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#COMPACT_CTOR_DEF">
105 * COMPACT_CTOR_DEF</a>.
106 * </li>
107 * </ul>
108 * <p>
109 * To configure the default check:
110 * </p>
111 * <pre>
112 * &lt;module name=&quot;EmptyLineSeparator&quot;/&gt;
113 * </pre>
114 * <p>
115 * Example of declarations without empty line separator:
116 * </p>
117 *
118 * <pre>
119 * ///////////////////////////////////////////////////
120 * //HEADER
121 * ///////////////////////////////////////////////////
122 * package com.whitespace; // violation , 'package' should be separated from previous line
123 * import java.io.Serializable; // violation , 'import' should be separated from previous line
124 *
125 * class FirstClass {
126 *
127 *   int var1 = 1;
128 *   int var2 = 2; // violation , 'VARIABLE_DEF' should be separated from previous line
129 *
130 *
131 *   int var3 = 3;
132 *
133 *
134 *   void method1() {}
135 *   void method2() { // violation , 'METHOD_DEF' should be separated from previous line
136 *      int var4 = 4;
137 *
138 *
139 *      int var5 = 5;
140 *   }
141 * }
142 * </pre>
143 *
144 * <p>
145 * To check empty line before
146 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#VARIABLE_DEF">
147 * VARIABLE_DEF</a> and
148 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
149 * METHOD_DEF</a>:
150 * </p>
151 *
152 * <pre>
153 *   &lt;module name=&quot;EmptyLineSeparator&quot;&gt;
154 *   &lt;property name=&quot;tokens&quot; value=&quot;VARIABLE_DEF, METHOD_DEF&quot;/&gt;
155 *   &lt;/module&gt;
156 * </pre>
157 *
158 * <pre>
159 * ///////////////////////////////////////////////////
160 * //HEADER
161 * ///////////////////////////////////////////////////
162 * package com.whitespace;
163 * import java.io.Serializable;
164 *
165 * class FirstClass {
166 *
167 *   int var1 = 1;
168 *   int var2 = 2; // violation , 'VARIABLE_DEF' should be separated from previous line
169 *
170 *
171 *   int var3 = 3;
172 *
173 *
174 *   void method1() {}
175 *    void method2() { // violation , 'METHOD_DEF' should be separated from previous line
176 *      int var4 = 4;
177 *
178 *
179 *      int var5 = 5;
180 *   }
181 * }
182 * </pre>
183 *
184 *
185 * <p>
186 * To allow no empty line between fields:
187 * </p>
188 * <pre>
189 * &lt;module name="EmptyLineSeparator"&gt;
190 *   &lt;property name="allowNoEmptyLineBetweenFields" value="true"/&gt;
191 * &lt;/module&gt;
192 * </pre>
193 *
194 * <p>
195 * Example:
196 * </p>
197 *
198 * <pre>
199 * ///////////////////////////////////////////////////
200 * //HEADER
201 * ///////////////////////////////////////////////////
202 * package com.whitespace; // violation , 'package' should be separated from previous line
203 * import java.io.Serializable; // violation , 'import' should be separated from previous line
204 *
205 * class FirstClass {
206 *
207 *   int var1 = 1;
208 *   int var2 = 2;
209 *
210 *
211 *   int var3 = 3;
212 *
213 *
214 *   void method1() {}
215 *    void method2() { // violation , 'METHOD_DEF' should be separated from previous line
216 *      int var4 = 4;
217 *
218 *
219 *      int var5 = 5;
220 *   }
221 * }
222 * </pre>
223 *
224 * <p>
225 * To disallow multiple empty lines between class members:
226 * </p>
227 * <pre>
228 * &lt;module name=&quot;EmptyLineSeparator&quot;&gt;
229 *   &lt;property name=&quot;allowMultipleEmptyLines&quot; value=&quot;false&quot;/&gt;
230 * &lt;/module&gt;
231 * </pre>
232 * <pre>
233 * ///////////////////////////////////////////////////
234 * //HEADER
235 * ///////////////////////////////////////////////////
236 * package com.whitespace; // violation , 'package' should be separated from previous line
237 * import java.io.Serializable; // violation , 'import' should be separated from previous line
238 *
239 * class FirstClass {
240 *
241 *   int var1 = 1;
242 *   int var2 = 2; // violation , 'VARIABLE_DEF' should be separated from previous line
243 *
244 *
245 *   int var3 = 3; // violation , 'VARIABLE_DEF' has more than 1 empty lines before
246 *
247 *
248 *   void method1() {} // violation , 'METHOD_DEF' has more than 1 empty lines before
249 *    void method2() { // violation , 'METHOD_DEF' should be separated from previous line
250 *      int var4 = 4;
251 *
252 *
253 *      int var5 = 5;
254 *   }
255 * }
256 * </pre>
257 *
258 * <p>
259 * To disallow multiple empty lines inside constructor, initialization block and method:
260 * </p>
261 * <pre>
262 * &lt;module name="EmptyLineSeparator"&gt;
263 *   &lt;property name="allowMultipleEmptyLinesInsideClassMembers" value="false"/&gt;
264 * &lt;/module&gt;
265 * </pre>
266 *
267 * <p>
268 * The check is valid only for statements that have body:
269 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
270 * CLASS_DEF</a>,
271 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
272 * INTERFACE_DEF</a>,
273 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
274 * ENUM_DEF</a>,
275 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
276 * STATIC_INIT</a>,
277 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INSTANCE_INIT">
278 * INSTANCE_INIT</a>,
279 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
280 * METHOD_DEF</a>,
281 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
282 * CTOR_DEF</a>.
283 * </p>
284 * <p>
285 * Example of declarations with multiple empty lines inside method:
286 * </p>
287 *
288 * <pre>
289 * ///////////////////////////////////////////////////
290 * //HEADER
291 * ///////////////////////////////////////////////////
292 * package com.whitespace; // violation , 'package' should be separated from previous line
293 * import java.io.Serializable; // violation , 'import' should be separated from previous line
294 *
295 * class FirstClass {
296 *
297 *   int var1 = 1;
298 *   int var2 = 2; // violation , 'VARIABLE_DEF' should be separated from previous line
299 *
300 *
301 *   int var3 = 3;
302 *
303 *
304 *   void method1() {}
305 *   void method2() { // violation , 'METHOD_DEF' should be separated from previous line
306 *      int var4 = 4; // violation , There is more than 1 empty line after this line

308 *
309 *      int var5 = 5;
310 *   }
311 * }
312 * </pre>
313 *
314 * <p>
315 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
316 * </p>
317 * <p>

441    @Override
442    public void visitToken(DetailAST ast) {
443        checkComments(ast);
444        if (hasMultipleLinesBefore(ast)) {
445            log(ast, MSG_MULTIPLE_LINES, ast.getText());
446        }
447        if (!allowMultipleEmptyLinesInsideClassMembers) {
448            processMultipleLinesInside(ast);
449        }
450        if (ast.getType() == TokenTypes.PACKAGE_DEF) {
451            checkCommentInModifiers(ast);
452        }
453        DetailAST nextToken = ast.getNextSibling();
454        while (nextToken != null && TokenUtil.isCommentType(nextToken.getType())) {
455            nextToken = nextToken.getNextSibling();
456        }
457        if (nextToken != null) {
458            checkToken(ast, nextToken);
459        }
460    }

462    /**
463     * Checks that token and next token are separated.
464     *
465     * @param ast token to validate
466     * @param nextToken next sibling of the token
467     */
468    private void checkToken(DetailAST ast, DetailAST nextToken) {
469        final int astType = ast.getType();
470        switch (astType) {
471            case TokenTypes.VARIABLE_DEF:
472                processVariableDef(ast, nextToken);
473                break;
474            case TokenTypes.IMPORT:
475            case TokenTypes.STATIC_IMPORT:
476                processImport(ast, nextToken);
477                break;
478            case TokenTypes.PACKAGE_DEF:
479                processPackage(ast, nextToken);
480                break;
481            default:
482                if (nextToken.getType() == TokenTypes.RCURLY) {
483                    if (hasNotAllowedTwoEmptyLinesBefore(nextToken)) {
484                        final DetailAST result = getLastElementBeforeEmptyLines(ast,
485                                nextToken.getLineNo());
486                        log(result, MSG_MULTIPLE_LINES_AFTER, result.getText());
487                    }
488                }
489                else if (!hasEmptyLineAfter(ast)) {
490                    log(nextToken, MSG_SHOULD_BE_SEPARATED,
491                        nextToken.getText());
492                }
493        }
494    }
495
496    /**
497     * Checks that packageDef token is separated from comment in modifiers.
498     *
499     * @param packageDef package def token
500     */
501    private void checkCommentInModifiers(DetailAST packageDef) {
502        final Optional<DetailAST> comment = findCommentUnder(packageDef);
503        if (comment.isPresent()) {
504            log(comment.get(), MSG_SHOULD_BE_SEPARATED, comment.get().getText());
505        }
506    }

508    /**
509     * Log violation in case there are multiple empty lines inside constructor,
510     * initialization block or method.
511     *
512     * @param ast the ast to check.
513     */
514    private void processMultipleLinesInside(DetailAST ast) {
515        final int astType = ast.getType();
516        if (isClassMemberBlock(astType)) {
517            final List<Integer> emptyLines = getEmptyLines(ast);
518            final List<Integer> emptyLinesToLog = getEmptyLinesToLog(emptyLines);
519            for (Integer lineNo : emptyLinesToLog) {
520                log(getLastElementBeforeEmptyLines(ast, lineNo), MSG_MULTIPLE_LINES_INSIDE);
521            }
522        }
523    }
524
525    /**
526     * Returns the element after which empty lines exist.
527     *
528     * @param ast the ast to check.
529     * @param line the empty line which gives violation.
530     * @return The DetailAST after which empty lines are present.
531     */
532    private static DetailAST getLastElementBeforeEmptyLines(DetailAST ast, int line) {
533        DetailAST result = ast;
534        if (ast.getFirstChild().getLineNo() <= line) {
535            result = ast.getFirstChild();
536            while (result.getNextSibling() != null
537                    && result.getNextSibling().getLineNo() <= line) {
538                result = result.getNextSibling();
539            }
540            if (result.hasChildren()) {
541                result = getLastElementBeforeEmptyLines(result, line);
542            }
543        }
544
545        if (result.getNextSibling() != null) {
546            final Optional<DetailAST> postFixNode = getPostFixNode(result.getNextSibling());
547            if (postFixNode.isPresent()) {
548                // A post fix AST will always have a sibling METHOD CALL
549                // METHOD CALL will at least have two children
550                // The first child is DOT in case of POSTFIX which have at least 2 children
551                // First child of DOT again puts us back to normal AST tree which will
552                // recurse down below from here
553                final DetailAST firstChildAfterPostFix = postFixNode.get();
554                result = getLastElementBeforeEmptyLines(firstChildAfterPostFix, line);
555            }
556        }
557        return result;
558    }
559
560    /**
561     * Gets postfix Node from AST if present.
562     *
563     * @param ast the AST used to get postfix Node.
564     * @return Optional postfix node.
565     */
566    private static Optional<DetailAST> getPostFixNode(DetailAST ast) {
567        Optional<DetailAST> result = Optional.empty();
568        if (ast.getType() == TokenTypes.EXPR
569            // EXPR always has at least one child
570            && ast.getFirstChild().getType() == TokenTypes.METHOD_CALL) {
571            // METHOD CALL always has at two least child
572            final DetailAST node = ast.getFirstChild().getFirstChild();
573            if (node.getType() == TokenTypes.DOT) {
574                result = Optional.of(node);
575            }
576        }
577        return result;
578    }

580    /**
581     * Whether the AST is a class member block.
582     *
583     * @param astType the AST to check.
584     * @return true if the AST is a class member block.
585     */
586    private static boolean isClassMemberBlock(int astType) {
587        return TokenUtil.isOfType(astType,
588            TokenTypes.STATIC_INIT, TokenTypes.INSTANCE_INIT, TokenTypes.METHOD_DEF,
589            TokenTypes.CTOR_DEF, TokenTypes.COMPACT_CTOR_DEF);

592    /**
593     * Get list of empty lines.
594     *
595     * @param ast the ast to check.
596     * @return list of line numbers for empty lines.
597     */
598    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
599    @SuppressWarnings("deprecation")
600    private List<Integer> getEmptyLines(DetailAST ast) {
601        final DetailAST lastToken = ast.getLastChild().getLastChild();
602        int lastTokenLineNo = 0;
603        if (lastToken != null) {
604            // -1 as count starts from 0
605            // -2 as last token line cannot be empty, because it is a RCURLY
606            lastTokenLineNo = lastToken.getLineNo() - 2;
607        }
608        final List<Integer> emptyLines = new ArrayList<>();
609        final FileContents fileContents = getFileContents();
610
611        for (int lineNo = ast.getLineNo(); lineNo <= lastTokenLineNo; lineNo++) {
612            if (fileContents.lineIsBlank(lineNo)) {
613                emptyLines.add(lineNo);
614            }
615        }
616        return emptyLines;
617    }

619    /**
620     * Get list of empty lines to log.
621     *
622     * @param emptyLines list of empty lines.
623     * @return list of empty lines to log.
624     */
625    private static List<Integer> getEmptyLinesToLog(Iterable<Integer> emptyLines) {
626        final List<Integer> emptyLinesToLog = new ArrayList<>();
627        int previousEmptyLineNo = -1;
628        for (int emptyLineNo : emptyLines) {
629            if (previousEmptyLineNo + 1 == emptyLineNo) {
630                emptyLinesToLog.add(previousEmptyLineNo);
631            }
632            previousEmptyLineNo = emptyLineNo;
633        }
634        return emptyLinesToLog;
635    }

637    /**
638     * Whether the token has not allowed multiple empty lines before.
639     *
640     * @param ast the ast to check.
641     * @return true if the token has not allowed multiple empty lines before.
642     */
643    private boolean hasMultipleLinesBefore(DetailAST ast) {
644        return (ast.getType() != TokenTypes.VARIABLE_DEF || isTypeField(ast))
645                && hasNotAllowedTwoEmptyLinesBefore(ast);
646    }

648    /**
649     * Process Package.
650     *
651     * @param ast token
652     * @param nextToken next token
653     */
654    private void processPackage(DetailAST ast, DetailAST nextToken) {
655        if (ast.getLineNo() > 1 && !hasEmptyLineBefore(ast)) {
656            if (CheckUtil.isPackageInfo(getFilePath())) {
657                if (!ast.getFirstChild().hasChildren() && !isPrecededByJavadoc(ast)) {
658                    log(ast, MSG_SHOULD_BE_SEPARATED, ast.getText());
659                }
660            }
661            else {
662                log(ast, MSG_SHOULD_BE_SEPARATED, ast.getText());
663            }
664        }
665        if (isLineEmptyAfterPackage(ast)) {
666            final DetailAST elementAst = getViolationAstForPackage(ast);
667            log(elementAst, MSG_SHOULD_BE_SEPARATED, elementAst.getText());
668        }
669        else if (!hasEmptyLineAfter(ast)) {
670            log(nextToken, MSG_SHOULD_BE_SEPARATED, nextToken.getText());
671        }
672    }

674    /**
675     * Checks if there is another element at next line of package declaration.
676     *
677     * @param ast Package ast.
678     * @return true, if there is an element.
679     */
680    private static boolean isLineEmptyAfterPackage(DetailAST ast) {
681        DetailAST nextElement = ast;
682        final int lastChildLineNo = ast.getLastChild().getLineNo();
683        while (nextElement.getLineNo() < lastChildLineNo + 1
684                && nextElement.getNextSibling() != null) {
685            nextElement = nextElement.getNextSibling();
686        }
687        return nextElement.getLineNo() == lastChildLineNo + 1;
688    }

691     * Gets the Ast on which violation is to be given for package declaration.
692     *
693     * @param ast Package ast.
694     * @return Violation ast.
695     */
696    private static DetailAST getViolationAstForPackage(DetailAST ast) {
697        DetailAST nextElement = ast;
698        final int lastChildLineNo = ast.getLastChild().getLineNo();
699        while (nextElement.getLineNo() < lastChildLineNo + 1) {
700            nextElement = nextElement.getNextSibling();
701        }
702        return nextElement;

706     * Process Import.
707     *
708     * @param ast token
709     * @param nextToken next token

711    private void processImport(DetailAST ast, DetailAST nextToken) {
712        if (!TokenUtil.isOfType(nextToken, TokenTypes.IMPORT, TokenTypes.STATIC_IMPORT)
713            && !hasEmptyLineAfter(ast)) {
714            log(nextToken, MSG_SHOULD_BE_SEPARATED, nextToken.getText());

719     * Process Variable.
720     *
721     * @param ast token
722     * @param nextToken next Token
723     */
724    private void processVariableDef(DetailAST ast, DetailAST nextToken) {
725        if (isTypeField(ast) && !hasEmptyLineAfter(ast)
726                && isViolatingEmptyLineBetweenFieldsPolicy(nextToken)) {
727            log(nextToken, MSG_SHOULD_BE_SEPARATED,
728                    nextToken.getText());

732    /**
733     * Checks whether token placement violates policy of empty line between fields.
734     *
735     * @param detailAST token to be analyzed
736     * @return true if policy is violated and warning should be raised; false otherwise
737     */
738    private boolean isViolatingEmptyLineBetweenFieldsPolicy(DetailAST detailAST) {
739        return detailAST.getType() != TokenTypes.RCURLY
740                && (!allowNoEmptyLineBetweenFields
741                    || !TokenUtil.isOfType(detailAST, TokenTypes.COMMA, TokenTypes.VARIABLE_DEF));
742    }

744    /**
745     * Checks if a token has empty two previous lines and multiple empty lines is not allowed.
746     *
747     * @param token DetailAST token
748     * @return true, if token has empty two lines before and allowMultipleEmptyLines is false
749     */
750    private boolean hasNotAllowedTwoEmptyLinesBefore(DetailAST token) {
751        return !allowMultipleEmptyLines && hasEmptyLineBefore(token)
752                && isPrePreviousLineEmpty(token);
753    }

755    /**
756     * Check if group of comments located right before token has more than one previous empty line.
757     *
758     * @param token DetailAST token
759     */
760    private void checkComments(DetailAST token) {
761        if (!allowMultipleEmptyLines) {
762            if (TokenUtil.isOfType(token,
763                TokenTypes.PACKAGE_DEF, TokenTypes.IMPORT,
764                TokenTypes.STATIC_IMPORT, TokenTypes.STATIC_INIT)) {
765                DetailAST previousNode = token.getPreviousSibling();
766                while (isCommentInBeginningOfLine(previousNode)) {
767                    if (hasEmptyLineBefore(previousNode) && isPrePreviousLineEmpty(previousNode)) {
768                        log(previousNode, MSG_MULTIPLE_LINES, previousNode.getText());
769                    }
770                    previousNode = previousNode.getPreviousSibling();
771                }
772            }
773            else {
774                checkCommentsInsideToken(token);
775            }
776        }
777    }

779    /**
780     * Check if group of comments located at the start of token has more than one previous empty
781     * line.
782     *
783     * @param token DetailAST token
784     */
785    private void checkCommentsInsideToken(DetailAST token) {
786        final List<DetailAST> childNodes = new LinkedList<>();
787        DetailAST childNode = token.getLastChild();
788        while (childNode != null) {
789            if (childNode.getType() == TokenTypes.MODIFIERS) {
790                for (DetailAST node = token.getFirstChild().getLastChild();
791                         node != null;
792                         node = node.getPreviousSibling()) {
793                    if (isCommentInBeginningOfLine(node)) {
794                        childNodes.add(node);
795                    }
796                }
797            }
798            else if (isCommentInBeginningOfLine(childNode)) {
799                childNodes.add(childNode);
800            }
801            childNode = childNode.getPreviousSibling();
802        }
803        for (DetailAST node : childNodes) {
804            if (hasEmptyLineBefore(node) && isPrePreviousLineEmpty(node)) {
805                log(node, MSG_MULTIPLE_LINES, node.getText());
806            }
807        }
808    }

810    /**
811     * Checks if a token has empty pre-previous line.
812     *
813     * @param token DetailAST token.
814     * @return true, if token has empty lines before.
815     */
816    private boolean isPrePreviousLineEmpty(DetailAST token) {
817        boolean result = false;
818        final int lineNo = token.getLineNo();
819        // 3 is the number of the pre-previous line because the numbering starts from zero.
820        final int number = 3;
821        if (lineNo >= number) {
822            final String prePreviousLine = getLine(lineNo - number);
823            result = CommonUtil.isBlank(prePreviousLine);
824        }
825        return result;
826    }

828    /**
829     * Checks if token have empty line after.
830     *
831     * @param token token.
832     * @return true if token have empty line after.
833     */
834    private boolean hasEmptyLineAfter(DetailAST token) {
835        DetailAST lastToken = token.getLastChild().getLastChild();
836        if (lastToken == null) {
837            lastToken = token.getLastChild();
838        }
839        DetailAST nextToken = token.getNextSibling();
840        if (TokenUtil.isCommentType(nextToken.getType())) {
841            nextToken = nextToken.getNextSibling();
842        }
843        // Start of the next token
844        final int nextBegin = nextToken.getLineNo();
845        // End of current token.
846        final int currentEnd = lastToken.getLineNo();
847        return hasEmptyLine(currentEnd + 1, nextBegin - 1);
848    }

850    /**
851     * Finds comment in next sibling of given packageDef.
852     *
853     * @param packageDef token to check
854     * @return comment under the token
855     */
856    private static Optional<DetailAST> findCommentUnder(DetailAST packageDef) {
857        return Optional.ofNullable(packageDef.getNextSibling())
858            .map(sibling -> sibling.findFirstToken(TokenTypes.MODIFIERS))
859            .map(DetailAST::getFirstChild)
860            .filter(token -> TokenUtil.isCommentType(token.getType()))
861            .filter(comment -> comment.getLineNo() == packageDef.getLineNo() + 1);
862    }

864    /**
865     * Checks, whether there are empty lines within the specified line range. Line numbering is
866     * started from 1 for parameter values
867     *
868     * @param startLine number of the first line in the range
869     * @param endLine number of the second line in the range
870     * @return {@code true} if found any blank line within the range, {@code false}
871     *         otherwise
872     */
873    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
874    @SuppressWarnings("deprecation")
875    private boolean hasEmptyLine(int startLine, int endLine) {
876        // Initial value is false - blank line not found
877        boolean result = false;
878        final FileContents fileContents = getFileContents();
879        for (int line = startLine; line <= endLine; line++) {
880            // Check, if the line is blank. Lines are numbered from 0, so subtract 1
881            if (fileContents.lineIsBlank(line - 1)) {
882                result = true;
883                break;
884            }
885        }
886        return result;
887    }

889    /**
890     * Checks if a token has an empty line before.
891     *
892     * @param token token.
893     * @return true, if token have empty line before.
894     */
895    private boolean hasEmptyLineBefore(DetailAST token) {
896        boolean result = false;
897        final int lineNo = token.getLineNo();
898        if (lineNo != 1) {
899            // [lineNo - 2] is the number of the previous line as the numbering starts from zero.
900            final String lineBefore = getLine(lineNo - 2);
901            result = CommonUtil.isBlank(lineBefore);
902        }
903        return result;
904    }

906    /**
907     * Check if token is comment, which starting in beginning of line.
908     *
909     * @param comment comment token for check.
910     * @return true, if token is comment, which starting in beginning of line.
911     */
912    private boolean isCommentInBeginningOfLine(DetailAST comment) {
913        // comment.getLineNo() - 1 is the number of the previous line as the numbering starts
914        // from zero.
915        boolean result = false;
916        if (comment != null) {
917            final String lineWithComment = getLine(comment.getLineNo() - 1).trim();
918            result = lineWithComment.startsWith("//") || lineWithComment.startsWith("/*");
919        }
920        return result;
921    }

030/**
031 * <p>
032 * Checks that there is no whitespace after a token.
033 * More specifically, it checks that it is not followed by whitespace,
034 * or (if linebreaks are allowed) all characters on the line after are
035 * whitespace. To forbid linebreaks after a token, set property
036 * {@code allowLineBreaks} to {@code false}.
037 * </p>
038 * <p>
039 * The check processes
040 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ARRAY_DECLARATOR">
041 * ARRAY_DECLARATOR</a> and
042 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INDEX_OP">
043 * INDEX_OP</a> tokens specially from other tokens. Actually it is checked that
044 * there is no whitespace before these tokens, not after them. Space after the
045 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATIONS">
046 * ANNOTATIONS</a> before
047 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ARRAY_DECLARATOR">
048 * ARRAY_DECLARATOR</a> and
049 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INDEX_OP">
050 * INDEX_OP</a> will be ignored.
051 * </p>
052 * <p>
053 * If the annotation is between the type and the array, like {@code char @NotNull [] param},
054 * the check will skip validation for spaces.
055 * </p>
056 * <p>
057 * Note: This check processes the
058 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_SYNCHRONIZED">
059 * LITERAL_SYNCHRONIZED</a> token only when it appears as a part of a
060 * <a href="https://docs.oracle.com/javase/specs/jls/se19/html/jls-14.html#jls-14.19">
061 * synchronized statement</a>, i.e. {@code synchronized(this) {}}.
062 * </p>
063 * <ul>
064 * <li>
065 * Property {@code allowLineBreaks} - Control whether whitespace is allowed
066 * if the token is at a linebreak.
067 * Type is {@code boolean}.
068 * Default value is {@code true}.
069 * </li>
070 * <li>
071 * Property {@code tokens} - tokens to check
072 * Type is {@code java.lang.String[]}.
073 * Validation type is {@code tokenSet}.
074 * Default value is:
075 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ARRAY_INIT">
076 * ARRAY_INIT</a>,
077 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#AT">
078 * AT</a>,
079 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INC">
080 * INC</a>,
081 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#DEC">
082 * DEC</a>,
083 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#UNARY_MINUS">
084 * UNARY_MINUS</a>,
085 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#UNARY_PLUS">
086 * UNARY_PLUS</a>,
087 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#BNOT">
088 * BNOT</a>,
089 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LNOT">
090 * LNOT</a>,
091 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#DOT">
092 * DOT</a>,
093 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ARRAY_DECLARATOR">
094 * ARRAY_DECLARATOR</a>,
095 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INDEX_OP">
096 * INDEX_OP</a>.
097 * </li>
098 * </ul>
099 * <p>
100 * To configure the check:
101 * </p>
102 * <pre>
103 * &lt;module name=&quot;NoWhitespaceAfter&quot;/&gt;
104 * </pre>
105 * <p>Example:</p>
106 * <pre>
107 * class Test {
108 *
109 *   public void lineBreak(String x) {
110 *     Integer.
111 *         parseInt(x); // Ok
112 *     Integer.parseInt(x); // Ok
113 *   }
114 *
115 *   public void dotOperator(String s) {
116 *     Integer.parseInt(s); // Ok
117 *     Integer. parseInt(s); // violation, '.' is followed by whitespace
118 *   }
119 *
120 *   public void arrayDec() {
121 *     int[] arr = arr; // Ok
122 *     int [] arr = arr; // violation, int is followed by whitespace
123 *   }
124 *
125 *   public void bitwiseNot(int a) {
126 *     a = ~ a; // violation '~' is followed by whitespace
127 *     a = ~a; // Ok
128 *   }
129 * }
130 * </pre>
131 * <p>To configure the check to forbid linebreaks after a DOT token:
132 * </p>
133 * <pre>
134 * &lt;module name=&quot;NoWhitespaceAfter&quot;&gt;
135 *   &lt;property name=&quot;tokens&quot; value=&quot;DOT&quot;/&gt;
136 *   &lt;property name=&quot;allowLineBreaks&quot; value=&quot;false&quot;/&gt;
137 * &lt;/module&gt;
138 * </pre>
139 * <p>Example:</p>
140 * <pre>
141 * class Test {
142 *
143 *   public void lineBreak(String x) {
144 *     Integer.
145 *         parseInt(x); // violation, '.' is followed by whitespace
146 *     Integer.parseInt(x); // Ok
147 *   }
148 *
149 *   public void dotOperator(String s) {
150 *     Integer.parseInt(s); // Ok
151 *     Integer. parseInt(s); // violation, '.' is followed by whitespace
152 *   }
153 *
154 *   public void arrayDec() {
155 *     int[] arr = arr; // Ok
156 *     int [] arr = arr; // Ok
157 *   }
158 *
159 *   public void bitwiseNot(int a) {
160 *     a = ~ a; // Ok
161 *     a = ~a; // Ok
162 *   }
163 * }
164 * </pre>
165 * <p>
166 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
167 * </p>
168 * <p>
169 * Violation Message Keys:
170 * </p>
171 * <ul>
172 * <li>
173 * {@code ws.followed}
174 * </li>
175 * </ul>
176 *
177 * @since 3.0
178 */
179@StatelessCheck
180public class NoWhitespaceAfterCheck extends AbstractCheck {

233    /**
234     * Setter to control whether whitespace is allowed if the token is at a linebreak.
235     *
236     * @param allowLineBreaks whether whitespace should be
237     *     flagged at linebreaks.
238     * @since 3.0
239     */
240    public void setAllowLineBreaks(boolean allowLineBreaks) {
241        this.allowLineBreaks = allowLineBreaks;
242    }
243
244    @Override
245    public void visitToken(DetailAST ast) {
246        if (shouldCheckWhitespaceAfter(ast)) {
247            final DetailAST whitespaceFollowedAst = getWhitespaceFollowedNode(ast);
248            final int whitespaceColumnNo = getPositionAfter(whitespaceFollowedAst);
249            final int whitespaceLineNo = whitespaceFollowedAst.getLineNo();
250
251            if (hasTrailingWhitespace(ast, whitespaceColumnNo, whitespaceLineNo)) {
252                log(ast, MSG_KEY, whitespaceFollowedAst.getText());
253            }
254        }
255    }

257    /**
258     * For a visited ast node returns node that should be checked
259     * for not being followed by whitespace.
260     *
261     * @param ast
262     *        , visited node.
263     * @return node before ast.
264     */
265    private static DetailAST getWhitespaceFollowedNode(DetailAST ast) {
266        final DetailAST whitespaceFollowedAst;
267        switch (ast.getType()) {
268            case TokenTypes.TYPECAST:
269                whitespaceFollowedAst = ast.findFirstToken(TokenTypes.RPAREN);
270                break;
271            case TokenTypes.ARRAY_DECLARATOR:
272                whitespaceFollowedAst = getArrayDeclaratorPreviousElement(ast);
273                break;
274            case TokenTypes.INDEX_OP:
275                whitespaceFollowedAst = getIndexOpPreviousElement(ast);
276                break;
277            default:
278                whitespaceFollowedAst = ast;
279        }
280        return whitespaceFollowedAst;
281    }
282
283    /**
284     * Returns whether whitespace after a visited node should be checked. For example, whitespace
285     * is not allowed between a type and an array declarator (returns true), except when there is
286     * an annotation in between the type and array declarator (returns false).
287     *
288     * @param ast the visited node
289     * @return true if whitespace after ast should be checked
290     */
291    private static boolean shouldCheckWhitespaceAfter(DetailAST ast) {
292        final DetailAST previousSibling = ast.getPreviousSibling();
293        final boolean isSynchronizedMethod = ast.getType() == TokenTypes.LITERAL_SYNCHRONIZED
294                        && ast.getFirstChild() == null;
295        return !isSynchronizedMethod
296                && (previousSibling == null || previousSibling.getType() != TokenTypes.ANNOTATIONS);
297    }

299    /**
300     * Gets position after token (place of possible redundant whitespace).
301     *
302     * @param ast Node representing token.
303     * @return position after token.
304     */
305    private static int getPositionAfter(DetailAST ast) {
306        final int after;
307        // If target of possible redundant whitespace is in method definition.
308        if (ast.getType() == TokenTypes.IDENT
309                && ast.getNextSibling() != null
310                && ast.getNextSibling().getType() == TokenTypes.LPAREN) {
311            final DetailAST methodDef = ast.getParent();
312            final DetailAST endOfParams = methodDef.findFirstToken(TokenTypes.RPAREN);
313            after = endOfParams.getColumnNo() + 1;
314        }
315        else {
316            after = ast.getColumnNo() + ast.getText().length();
317        }
318        return after;
319    }
320
321    /**
322     * Checks if there is unwanted whitespace after the visited node.
323     *
324     * @param ast
325     *        , visited node.
326     * @param whitespaceColumnNo
327     *        , column number of a possible whitespace.
328     * @param whitespaceLineNo
329     *        , line number of a possible whitespace.
330     * @return true if whitespace found.
331     */
332    private boolean hasTrailingWhitespace(DetailAST ast,
333        int whitespaceColumnNo, int whitespaceLineNo) {
334        final boolean result;
335        final int astLineNo = ast.getLineNo();
336        final int[] line = getLineCodePoints(astLineNo - 1);
337        if (astLineNo == whitespaceLineNo && whitespaceColumnNo < line.length) {
338            result = CommonUtil.isCodePointWhitespace(line, whitespaceColumnNo);
339        }
340        else {
341            result = !allowLineBreaks;
342        }
343        return result;
344    }

346    /**
347     * Returns proper argument for getPositionAfter method, it is a token after
348     * {@link TokenTypes#ARRAY_DECLARATOR ARRAY_DECLARATOR}, in can be {@link TokenTypes#RBRACK
349     * RBRACK}, {@link TokenTypes#IDENT IDENT} or an array type definition (literal).
350     *
351     * @param ast
352     *        , {@link TokenTypes#ARRAY_DECLARATOR ARRAY_DECLARATOR} node.
353     * @return previous node by text order.
354     * @throws IllegalStateException if an unexpected token type is encountered.
355     */
356    private static DetailAST getArrayDeclaratorPreviousElement(DetailAST ast) {
357        final DetailAST previousElement;
358
359        if (ast.getPreviousSibling() != null
360                && ast.getPreviousSibling().getType() == TokenTypes.ARRAY_DECLARATOR) {
361            // Covers higher dimension array declarations and initializations
362            previousElement = getPreviousElementOfMultiDimArray(ast);
363        }
364        else {
365            // first array index, is preceded with identifier or type
366            final DetailAST parent = ast.getParent();
367            switch (parent.getType()) {
368                // generics
369                case TokenTypes.TYPE_UPPER_BOUNDS:
370                case TokenTypes.TYPE_LOWER_BOUNDS:
371                    previousElement = ast.getPreviousSibling();
372                    break;
373                case TokenTypes.LITERAL_NEW:
374                case TokenTypes.TYPE_ARGUMENT:
375                case TokenTypes.DOT:
376                    previousElement = getTypeLastNode(ast);
377                    break;
378                // mundane array declaration, can be either java style or C style
379                case TokenTypes.TYPE:
380                    previousElement = getPreviousNodeWithParentOfTypeAst(ast, parent);
381                    break;
382                // java 8 method reference
383                case TokenTypes.METHOD_REF:
384                    final DetailAST ident = getIdentLastToken(ast);
385                    if (ident == null) {
386                        // i.e. int[]::new
387                        previousElement = ast.getParent().getFirstChild();
388                    }
389                    else {
390                        previousElement = ident;
391                    }
392                    break;
393                default:
394                    throw new IllegalStateException("unexpected ast syntax " + parent);
395            }
396        }
397        return previousElement;
398    }

400    /**
401     * Gets the previous element of a second or higher dimension of an
402     * array declaration or initialization.
403     *
404     * @param leftBracket the token to get previous element of
405     * @return the previous element
406     */
407    private static DetailAST getPreviousElementOfMultiDimArray(DetailAST leftBracket) {
408        final DetailAST previousRightBracket = leftBracket.getPreviousSibling().getLastChild();
409
410        DetailAST ident = null;
411        // This will get us past the type ident, to the actual identifier
412        DetailAST parent = leftBracket.getParent().getParent();
413        while (ident == null) {
414            ident = parent.findFirstToken(TokenTypes.IDENT);
415            parent = parent.getParent();
416        }
417
418        final DetailAST previousElement;
419        if (ident.getColumnNo() > previousRightBracket.getColumnNo()
420                && ident.getColumnNo() < leftBracket.getColumnNo()) {
421            // C style and Java style ' int[] arr []' in same construct
422            previousElement = ident;
423        }
424        else {
425            // 'int[][] arr' or 'int arr[][]'
426            previousElement = previousRightBracket;
427        }
428        return previousElement;
429    }

431    /**
432     * Gets previous node for {@link TokenTypes#INDEX_OP INDEX_OP} token
433     * for usage in getPositionAfter method, it is a simplified copy of
434     * getArrayDeclaratorPreviousElement method.
435     *
436     * @param ast
437     *        , {@link TokenTypes#INDEX_OP INDEX_OP} node.
438     * @return previous node by text order.
439     */
440    private static DetailAST getIndexOpPreviousElement(DetailAST ast) {
441        final DetailAST result;
442        final DetailAST firstChild = ast.getFirstChild();
443        if (firstChild.getType() == TokenTypes.INDEX_OP) {
444            // second or higher array index
445            result = firstChild.findFirstToken(TokenTypes.RBRACK);
446        }
447        else if (firstChild.getType() == TokenTypes.IDENT) {
448            result = firstChild;
449        }
450        else {
451            final DetailAST ident = getIdentLastToken(ast);
452            if (ident == null) {
453                final DetailAST rparen = ast.findFirstToken(TokenTypes.RPAREN);
454                // construction like new int[]{1}[0]
455                if (rparen == null) {
456                    final DetailAST lastChild = firstChild.getLastChild();
457                    result = lastChild.findFirstToken(TokenTypes.RCURLY);
458                }
459                // construction like ((byte[]) pixels)[0]
460                else {
461                    result = rparen;
462                }
463            }
464            else {
465                result = ident;
466            }
467        }
468        return result;
469    }

471    /**
472     * Searches parameter node for a type node.
473     * Returns it or its last node if it has an extended structure.
474     *
475     * @param ast
476     *        , subject node.
477     * @return type node.
478     */
479    private static DetailAST getTypeLastNode(DetailAST ast) {
480        final DetailAST typeLastNode;
481        final DetailAST parent = ast.getParent();
482        final boolean isPrecededByTypeArgs =
483                parent.findFirstToken(TokenTypes.TYPE_ARGUMENTS) != null;
484        final Optional<DetailAST> objectArrayType = Optional.ofNullable(getIdentLastToken(ast));
485
486        if (isPrecededByTypeArgs) {
487            typeLastNode = parent.findFirstToken(TokenTypes.TYPE_ARGUMENTS)
488                    .findFirstToken(TokenTypes.GENERIC_END);
489        }
490        else if (objectArrayType.isPresent()) {
491            typeLastNode = objectArrayType.get();
492        }
493        else {
494            typeLastNode = parent.getFirstChild();
495        }
496
497        return typeLastNode;
498    }

500    /**
501     * Finds previous node by text order for an array declarator,
502     * which parent type is {@link TokenTypes#TYPE TYPE}.
503     *
504     * @param ast
505     *        , array declarator node.
506     * @param parent
507     *        , its parent node.
508     * @return previous node by text order.
509     */
510    private static DetailAST getPreviousNodeWithParentOfTypeAst(DetailAST ast, DetailAST parent) {
511        final DetailAST previousElement;
512        final DetailAST ident = getIdentLastToken(parent.getParent());
513        final DetailAST lastTypeNode = getTypeLastNode(ast);
514        // sometimes there are ident-less sentences
515        // i.e. "(Object[]) null", but in casual case should be
516        // checked whether ident or lastTypeNode has preceding position
517        // determining if it is java style or C style
518
519        if (ident == null || ident.getLineNo() > ast.getLineNo()) {
520            previousElement = lastTypeNode;
521        }
522        else if (ident.getLineNo() < ast.getLineNo()) {
523            previousElement = ident;
524        }
525        // ident and lastTypeNode lay on one line
526        else {
527            final int instanceOfSize = 13;
528            // +2 because ast has `[]` after the ident
529            if (ident.getColumnNo() >= ast.getColumnNo() + 2
530                // +13 because ident (at most 1 character) is followed by
531                // ' instanceof ' (12 characters)
532                || lastTypeNode.getColumnNo() >= ident.getColumnNo() + instanceOfSize) {
533                previousElement = lastTypeNode;
534            }
535            else {
536                previousElement = ident;
537            }
538        }
539        return previousElement;
540    }

542    /**
543     * Gets leftmost token of identifier.
544     *
545     * @param ast
546     *        , token possibly possessing an identifier.
547     * @return leftmost token of identifier.
548     */
549    private static DetailAST getIdentLastToken(DetailAST ast) {
550        final DetailAST result;
551        final Optional<DetailAST> dot = getPrecedingDot(ast);
552        // method call case
553        if (dot.isEmpty() || ast.getFirstChild().getType() == TokenTypes.METHOD_CALL) {
554            final DetailAST methodCall = ast.findFirstToken(TokenTypes.METHOD_CALL);
555            if (methodCall == null) {
556                result = ast.findFirstToken(TokenTypes.IDENT);
557            }
558            else {
559                result = methodCall.findFirstToken(TokenTypes.RPAREN);
560            }
561        }
562        // qualified name case
563        else {
564            result = dot.get().getFirstChild().getNextSibling();
565        }
566        return result;
567    }

569    /**
570     * Gets the dot preceding a class member array index operation or class
571     * reference.
572     *
573     * @param leftBracket the ast we are checking
574     * @return dot preceding the left bracket
575     */
576    private static Optional<DetailAST> getPrecedingDot(DetailAST leftBracket) {
577        final DetailAST referencedMemberDot = leftBracket.findFirstToken(TokenTypes.DOT);
578        final Optional<DetailAST> result = Optional.ofNullable(referencedMemberDot);
579        return result.or(() -> getReferencedClassDot(leftBracket));
580    }
581
582    /**
583     * Gets the dot preceding a class reference.
584     *
585     * @param leftBracket the ast we are checking
586     * @return dot preceding the left bracket
587     */
588    private static Optional<DetailAST> getReferencedClassDot(DetailAST leftBracket) {
589        final DetailAST parent = leftBracket.getParent();
590        Optional<DetailAST> classDot = Optional.empty();
591        if (parent.getType() != TokenTypes.ASSIGN) {
592            classDot = Optional.ofNullable(parent.findFirstToken(TokenTypes.DOT));
593        }
594        return classDot;
595    }

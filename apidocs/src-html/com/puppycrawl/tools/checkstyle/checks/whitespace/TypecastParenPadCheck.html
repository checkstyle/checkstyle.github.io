020package com.puppycrawl.tools.checkstyle.checks.whitespace;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * <p>Checks the padding of parentheses for typecasts. That is whether a space
027 * is required after a left parenthesis and before a right parenthesis, or such
028 * spaces are forbidden.
029 * </p>
030 * <p>
031 * The policy to verify is specified using the {@link PadOption} class and
032 * defaults to {@link PadOption#NOSPACE}.
033 * </p>
034 * <p>
035 * An example of how to configure the check is:
036 * </p>
037 * <pre>
038 * &lt;module name="TypecastParenPad"/&gt;
039 * </pre>
040 * <p>
041 * An example of how to configure the check to require spaces for the
042 * parentheses of constructor, method, and super constructor invocations is:
043 * </p>
044 * <pre>
045 * &lt;module name="TypecastParenPad"&gt;
046 *     &lt;property name="option" value="space"/&gt;
047 * &lt;/module&gt;
048 * </pre>
049 * @author Oliver Burn
050 */
051public class TypecastParenPadCheck extends AbstractParenPadCheck {
052
053    @Override
054    public int[] getRequiredTokens() {
055        return new int[] {TokenTypes.RPAREN, TokenTypes.TYPECAST};
056    }
057
058    @Override
059    public int[] getDefaultTokens() {
060        return getRequiredTokens();
061    }
062
063    @Override
064    public int[] getAcceptableTokens() {
065        return getRequiredTokens();
066    }
067
068    @Override
069    public void visitToken(DetailAST ast) {
070        // Strange logic in this method to guard against checking RPAREN tokens
071        // that are not associated with a TYPECAST token.
072        if (ast.getType() == TokenTypes.TYPECAST) {
073            processLeft(ast);
074        }
075        else if (ast.getParent().getType() == TokenTypes.TYPECAST
076                 && ast.getParent().findFirstToken(TokenTypes.RPAREN) == ast) {
077            processRight(ast);
078        }
079    }
080
081}

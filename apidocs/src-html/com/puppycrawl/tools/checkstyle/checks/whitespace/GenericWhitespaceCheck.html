020package com.puppycrawl.tools.checkstyle.checks.whitespace;
021
022import com.puppycrawl.tools.checkstyle.FileStatefulCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.TokenTypes;
026import com.puppycrawl.tools.checkstyle.utils.CommonUtil;

028/**
029 * <p>
030 * Checks that the whitespace around the Generic tokens (angle brackets)
031 * "&lt;" and "&gt;" are correct to the <i>typical</i> convention.
032 * The convention is not configurable.
033 * </p>
034 * <p>
035 * Left angle bracket ("&lt;"):
036 * </p>
037 * <ul>
038 * <li> should be preceded with whitespace only
039 *   in generic methods definitions.</li>
040 * <li> should not be preceded with whitespace
041 *   when it is precede method name or constructor.</li>
042 * <li> should not be preceded with whitespace when following type name.</li>
043 * <li> should not be followed with whitespace in all cases.</li>
044 * </ul>
045 * <p>
046 * Right angle bracket ("&gt;"):
047 * </p>
048 * <ul>
049 * <li> should not be preceded with whitespace in all cases.</li>
050 * <li> should be followed with whitespace in almost all cases,
051 *   except diamond operators and when preceding method name or constructor.</li></ul>
052 * <p>
053 * To configure the check:
054 * </p>
055 * <pre>
056 * &lt;module name=&quot;GenericWhitespace&quot;/&gt;
057 * </pre>
058 * <p>
059 * Examples with correct spacing:
060 * </p>
061 * <pre>
062 * // Generic methods definitions
063 * public void &lt;K, V extends Number&gt; boolean foo(K, V) {}
064 * // Generic type definition
065 * class name&lt;T1, T2, ..., Tn&gt; {}
066 * // Generic type reference
067 * OrderedPair&lt;String, Box&lt;Integer&gt;&gt; p;
068 * // Generic preceded method name
069 * boolean same = Util.&lt;Integer, String&gt;compare(p1, p2);
070 * // Diamond operator
071 * Pair&lt;Integer, String&gt; p1 = new Pair&lt;&gt;(1, "apple");
072 * // Method reference
073 * List&lt;T&gt; list = ImmutableList.Builder&lt;T&gt;::new;
074 * // Method reference
075 * sort(list, Comparable::&lt;String&gt;compareTo);
076 * // Constructor call
077 * MyClass obj = new &lt;String&gt;MyClass();
078 * </pre>
079 * <p>
080 * Examples with incorrect spacing:
081 * </p>
082 * <pre>
083 * List&lt; String&gt; l; // violation, "&lt;" followed by whitespace
084 * Box b = Box. &lt;String&gt;of("foo"); // violation, "&lt;" preceded with whitespace
085 * public&lt;T&gt; void foo() {} // violation, "&lt;" not preceded with whitespace
086 *
087 * List a = new ArrayList&lt;&gt; (); // violation, "&gt;" followed by whitespace
088 * Map&lt;Integer, String&gt;m; // violation, "&gt;" not followed by whitespace
089 * Pair&lt;Integer, Integer &gt; p; // violation, "&gt;" preceded with whitespace
090 * </pre>
091 *
092 * @since 5.0
093 */
094@FileStatefulCheck
095public class GenericWhitespaceCheck extends AbstractCheck {
096
097    /**
098     * A key is pointing to the warning message text in "messages.properties"
099     * file.
100     */
101    public static final String MSG_WS_PRECEDED = "ws.preceded";
102
103    /**
104     * A key is pointing to the warning message text in "messages.properties"
105     * file.
106     */
107    public static final String MSG_WS_FOLLOWED = "ws.followed";
108
109    /**
110     * A key is pointing to the warning message text in "messages.properties"
111     * file.
112     */
113    public static final String MSG_WS_NOT_PRECEDED = "ws.notPreceded";
114
115    /**
116     * A key is pointing to the warning message text in "messages.properties"
117     * file.
118     */
119    public static final String MSG_WS_ILLEGAL_FOLLOW = "ws.illegalFollow";
120
121    /** Open angle bracket literal. */
122    private static final String OPEN_ANGLE_BRACKET = "<";
123
124    /** Close angle bracket literal. */
125    private static final String CLOSE_ANGLE_BRACKET = ">";
126
127    /** Used to count the depth of a Generic expression. */
128    private int depth;
129
130    @Override
131    public int[] getDefaultTokens() {
132        return getRequiredTokens();
133    }
134
135    @Override
136    public int[] getAcceptableTokens() {
137        return getRequiredTokens();
138    }
139
140    @Override
141    public int[] getRequiredTokens() {
142        return new int[] {TokenTypes.GENERIC_START, TokenTypes.GENERIC_END};
143    }
144
145    @Override
146    public void beginTree(DetailAST rootAST) {
147        // Reset for each tree, just increase there are violations in preceding
148        // trees.
149        depth = 0;
150    }
151
152    @Override
153    public void visitToken(DetailAST ast) {
154        switch (ast.getType()) {
155            case TokenTypes.GENERIC_START:
156                processStart(ast);
157                depth++;
158                break;
159            case TokenTypes.GENERIC_END:
160                processEnd(ast);
161                depth--;
162                break;
163            default:
164                throw new IllegalArgumentException("Unknown type " + ast);
165        }
166    }
167
168    /**
169     * Checks the token for the end of Generics.
170     * @param ast the token to check
171     */
172    private void processEnd(DetailAST ast) {
173        final String line = getLine(ast.getLineNo() - 1);
174        final int before = ast.getColumnNo() - 1;
175        final int after = ast.getColumnNo() + 1;
176
177        if (before >= 0 && Character.isWhitespace(line.charAt(before))
178                && !containsWhitespaceBefore(before, line)) {
179            log(ast, MSG_WS_PRECEDED, CLOSE_ANGLE_BRACKET);
180        }
181
182        if (after < line.length()) {
183            // Check if the last Generic, in which case must be a whitespace
184            // or a '(),[.'.
185            if (depth == 1) {
186                processSingleGeneric(ast, line, after);
187            }
188            else {
189                processNestedGenerics(ast, line, after);
190            }
191        }
192    }
193
194    /**
195     * Process Nested generics.
196     * @param ast token
197     * @param line line content
198     * @param after position after
199     */
200    private void processNestedGenerics(DetailAST ast, String line, int after) {
201        // In a nested Generic type, so can only be a '>' or ',' or '&'
202
203        // In case of several extends definitions:
204        //
205        //   class IntEnumValueType<E extends Enum<E> & IntEnum>
206        //                                          ^
207        //   should be whitespace if followed by & -+
208        //
209        final int indexOfAmp = line.indexOf('&', after);
210        if (indexOfAmp >= 1
211            && containsWhitespaceBetween(after, indexOfAmp, line)) {
212            if (indexOfAmp - after == 0) {
213                log(ast, MSG_WS_NOT_PRECEDED, "&");
214            }
215            else if (indexOfAmp - after != 1) {
216                log(ast, MSG_WS_FOLLOWED, CLOSE_ANGLE_BRACKET);
217            }
218        }
219        else if (line.charAt(after) == ' ') {
220            log(ast, MSG_WS_FOLLOWED, CLOSE_ANGLE_BRACKET);
221        }
222    }
223
224    /**
225     * Process Single-generic.
226     * @param ast token
227     * @param line line content
228     * @param after position after
229     */
230    private void processSingleGeneric(DetailAST ast, String line, int after) {
231        final char charAfter = line.charAt(after);
232        if (isGenericBeforeMethod(ast) || isGenericBeforeCtor(ast)) {
233            if (Character.isWhitespace(charAfter)) {
234                log(ast, MSG_WS_FOLLOWED, CLOSE_ANGLE_BRACKET);
235            }
236        }
237        else if (!isCharacterValidAfterGenericEnd(charAfter)) {
238            log(ast, MSG_WS_ILLEGAL_FOLLOW, CLOSE_ANGLE_BRACKET);
239        }
240    }
241
242    /**
243     * Checks if generic is before constructor invocation.
244     * @param ast ast
245     * @return true if generic before a constructor invocation
246     */
247    private static boolean isGenericBeforeCtor(DetailAST ast) {
248        final DetailAST parent = ast.getParent();
249        return parent.getParent().getType() == TokenTypes.LITERAL_NEW
250                && (parent.getNextSibling().getType() == TokenTypes.IDENT
251                    || parent.getNextSibling().getType() == TokenTypes.DOT);
252    }
253
254    /**
255     * Is generic before method reference.
256     * @param ast ast
257     * @return true if generic before a method ref
258     */
259    private static boolean isGenericBeforeMethod(DetailAST ast) {
260        return ast.getParent().getParent().getType() == TokenTypes.DOT
261                && ast.getParent().getParent().getParent().getType() == TokenTypes.METHOD_CALL
262                || isAfterMethodReference(ast);
263    }
264
265    /**
266     * Checks if current generic end ('>') is located after
267     * {@link TokenTypes#METHOD_REF method reference operator}.
268     * @param genericEnd {@link TokenTypes#GENERIC_END}
269     * @return true if '>' follows after method reference.
270     */
271    private static boolean isAfterMethodReference(DetailAST genericEnd) {
272        return genericEnd.getParent().getParent().getType() == TokenTypes.METHOD_REF;
273    }
274
275    /**
276     * Checks the token for the start of Generics.
277     * @param ast the token to check
278     */
279    private void processStart(DetailAST ast) {
280        final String line = getLine(ast.getLineNo() - 1);
281        final int before = ast.getColumnNo() - 1;
282        final int after = ast.getColumnNo() + 1;
283
284        // Need to handle two cases as in:
285        //
286        //   public static <T> Callable<T> callable(Runnable task, T result)
287        //                 ^           ^
288        //      ws reqd ---+           +--- whitespace NOT required
289        //
290        if (before >= 0) {
291            // Detect if the first case
292            final DetailAST parent = ast.getParent();
293            final DetailAST grandparent = parent.getParent();
294            if (grandparent.getType() == TokenTypes.CTOR_DEF
295                    || grandparent.getType() == TokenTypes.METHOD_DEF
296                    || isGenericBeforeCtor(ast)) {
297                // Require whitespace
298                if (!Character.isWhitespace(line.charAt(before))) {
299                    log(ast, MSG_WS_NOT_PRECEDED, OPEN_ANGLE_BRACKET);
300                }
301            }
302            // Whitespace not required
303            else if (Character.isWhitespace(line.charAt(before))
304                && !containsWhitespaceBefore(before, line)) {
305                log(ast, MSG_WS_PRECEDED, OPEN_ANGLE_BRACKET);
306            }
307        }
308
309        if (after < line.length()
310                && Character.isWhitespace(line.charAt(after))) {
311            log(ast, MSG_WS_FOLLOWED, OPEN_ANGLE_BRACKET);
312        }
313    }
314
315    /**
316     * Returns whether the specified string contains only whitespace between
317     * specified indices.
318     *
319     * @param fromIndex the index to start the search from. Inclusive
320     * @param toIndex the index to finish the search. Exclusive
321     * @param line the line to check
322     * @return whether there are only whitespaces (or nothing)
323     */
324    private static boolean containsWhitespaceBetween(int fromIndex, int toIndex, String line) {
325        boolean result = true;
326        for (int i = fromIndex; i < toIndex; i++) {
327            if (!Character.isWhitespace(line.charAt(i))) {
328                result = false;
329                break;
330            }
331        }
332        return result;
333    }
334
335    /**
336     * Returns whether the specified string contains only whitespace up to specified index.
337     *
338     * @param before the index to start the search from. Inclusive
339     * @param line   the index to finish the search. Exclusive
340     * @return {@code true} if there are only whitespaces,
341     *     false if there is nothing before or some other characters
342     */
343    private static boolean containsWhitespaceBefore(int before, String line) {
344        return before != 0 && CommonUtil.hasWhitespaceBefore(before, line);
345    }
346
347    /**
348     * Checks whether given character is valid to be right after generic ends.
349     * @param charAfter character to check
350     * @return checks if given character is valid
351     */
352    private static boolean isCharacterValidAfterGenericEnd(char charAfter) {
353        return charAfter == '(' || charAfter == ')'
354            || charAfter == ',' || charAfter == '['
355            || charAfter == '.' || charAfter == ':'
356            || charAfter == ';'
357            || Character.isWhitespace(charAfter);
358    }

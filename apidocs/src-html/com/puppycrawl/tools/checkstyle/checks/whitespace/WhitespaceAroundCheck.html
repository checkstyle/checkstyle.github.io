028/**
029 * <p>
030 * Checks that a token is surrounded by whitespace. Empty constructor,
031 * method, class, enum, interface, loop bodies (blocks), lambdas of the form
032 * </p>
033 * <pre>
034 * public MyClass() {}      // empty constructor
035 * public void func() {}    // empty method
036 * public interface Foo {} // empty interface
037 * public class Foo {} // empty class
038 * public enum Foo {} // empty enum
039 * MyClass c = new MyClass() {}; // empty anonymous class
040 * while (i = 1) {} // empty while loop
041 * for (int i = 1; i &gt; 1; i++) {} // empty for loop
042 * do {} while (i = 1); // empty do-while loop
043 * Runnable noop = () -&gt; {}; // empty lambda
044 * public @interface Beta {} // empty annotation type
045 * </pre>
046 * <p>
047 * may optionally be exempted from the policy using the {@code allowEmptyMethods},
048 * {@code allowEmptyConstructors}, {@code allowEmptyTypes}, {@code allowEmptyLoops},
049 * {@code allowEmptyLambdas} and {@code allowEmptyCatches} properties.
050 * </p>
051 * <p>
052 * This check does not flag as violation double brace initialization like:
053 * </p>
054 * <pre>
055 * new Properties() {{
056 *     setProperty("key", "value");
057 * }};
058 * </pre>
059 * <p>
060 * Parameter allowEmptyCatches allows to suppress violations when token list
061 * contains SLIST to check if beginning of block is surrounded by whitespace
062 * and catch block is empty, for example:
063 * </p>
064 * <pre>
065 * try {
066 *     k = 5 / i;
067 * } catch (ArithmeticException ex) {}
068 * </pre>
069 * <p>
070 * With this property turned off, this raises violation because the beginning
071 * of the catch block (left curly bracket) is not separated from the end
072 * of the catch block (right curly bracket).
073 * </p>
074 * <ul>
075 * <li>
076 * Property {@code allowEmptyConstructors} - Allow empty constructor bodies.
077 * Type is {@code boolean}.
078 * Default value is {@code false}.
079 * </li>
080 * <li>
081 * Property {@code allowEmptyMethods} - Allow empty method bodies.
082 * Type is {@code boolean}.
083 * Default value is {@code false}.
084 * </li>
085 * <li>
086 * Property {@code allowEmptyTypes} - Allow empty class, interface and enum bodies.
087 * Type is {@code boolean}.
088 * Default value is {@code false}.
089 * </li>
090 * <li>
091 * Property {@code allowEmptyLoops} - Allow empty loop bodies.
092 * Type is {@code boolean}.
093 * Default value is {@code false}.
094 * </li>
095 * <li>
096 * Property {@code allowEmptyLambdas} - Allow empty lambda bodies.
097 * Type is {@code boolean}.
098 * Default value is {@code false}.
099 * </li>
100 * <li>
101 * Property {@code allowEmptyCatches} - Allow empty catch bodies.
102 * Type is {@code boolean}.
103 * Default value is {@code false}.
104 * </li>
105 * <li>
106 * Property {@code ignoreEnhancedForColon} - Ignore whitespace around colon in
107 * <a href="https://docs.oracle.com/javase/specs/jls/se11/html/jls-14.html#jls-14.14.2">
108 * enhanced for</a> loop.
109 * Type is {@code boolean}.
110 * Default value is {@code true}.

221 * <p>To configure the check:
222 * </p>
223 * <pre>
224 * &lt;module name=&quot;WhitespaceAround&quot;/&gt;
225 * </pre>
226 * <p>Example:
227 * </p>
228 * <pre>
229 * class Test {
230 *     public Test(){} // 2 violations, '{' is not followed and preceded by whitespace.
231 *     public static void main(String[] args) {
232 *         if (foo) { // ok
233 *             // body
234 *         }
235 *         else{ // violation
236 *             // body
237 *         }
238 *
239 *         for (int i = 1; i &gt; 1; i++) {} // violation, '{' is not followed by whitespace.
240 *
241 *         Runnable noop = () -&gt;{}; // 2 violations,
242 *                                     // '{' is not followed and preceded by whitespace.
243 *         try {
244 *             // body
245 *         } catch (Exception e){} // 2 violations,
246 *                                 // '{' is not followed and preceded by whitespace.
247 *
248 *         char[] vowels = {'a', 'e', 'i', 'o', 'u'};
249 *         for (char item: vowels) { // ok, because ignoreEnhancedForColon is true by default
250 *             // body
251 *         }
252 *     }
253 * }
254 * </pre>
255 * <p>To configure the check for whitespace only around
256 * assignment operators:
257 * </p>
258 * <pre>
259 * &lt;module name=&quot;WhitespaceAround&quot;&gt;
260 *   &lt;property name=&quot;tokens&quot;
261 *     value=&quot;ASSIGN,DIV_ASSIGN,PLUS_ASSIGN,MINUS_ASSIGN,STAR_ASSIGN,
262 *            MOD_ASSIGN,SR_ASSIGN,BSR_ASSIGN,SL_ASSIGN,BXOR_ASSIGN,
263 *            BOR_ASSIGN,BAND_ASSIGN&quot;/&gt;
264 * &lt;/module&gt;
265 * </pre>
266 * <p>Example:
267 * </p>
268 * <pre>
269 * class Test {
270 *     public static void main(String[] args) {
271 *         int b=10; // violation
272 *         int c = 10; // ok
273 *         b+=10; // violation
274 *         b += 10; // ok
275 *         c*=10; // violation
276 *         c *= 10; // ok
277 *         c-=5; // violation
278 *         c -= 5; // ok
279 *         c/=2; // violation
280 *         c /= 2; // ok
281 *         c%=1; // violation
282 *         c %= 1; // ok
283 *         c&gt;&gt;=1; // violation
284 *         c &gt;&gt;= 1; // ok
285 *         c&gt;&gt;&gt;=1; // violation
286 *         c &gt;&gt;&gt;= 1; // ok
287 *     }
288 *     public void myFunction() {
289 *         c^=1; // violation
290 *         c ^= 1; // ok
291 *         c|=1; // violation
292 *         c |= 1; // ok
293 *         c&amp;=1; // violation
294 *         c &amp;= 1; // ok
295 *         c&lt;&lt;=1; // violation
296 *         c &lt;&lt;= 1; // ok
297 *     }
298 * }
299 * </pre>
300 * <p>To configure the check for whitespace only around curly braces:
301 * </p>
302 * <pre>
303 * &lt;module name=&quot;WhitespaceAround&quot;&gt;
304 *   &lt;property name=&quot;tokens&quot; value=&quot;LCURLY,RCURLY&quot;/&gt;
305 * &lt;/module&gt;
306 * </pre>
307 * <p>Example:
308 * </p>
309 * <pre>
310 * class Test {
311 *     public void myFunction() {} // violation

313 * }
314 * </pre>
315 * <p>
316 * To configure the check to allow empty method bodies:
317 * </p>
318 * <pre>

323 * <p>Example:

435 * <pre>

693    @Override
694    public void visitToken(DetailAST ast) {
695        final int currentType = ast.getType();
696        if (!isNotRelevantSituation(ast, currentType)) {
697            final int[] line = getLineCodePoints(ast.getLineNo() - 1);
698            final int before = ast.getColumnNo() - 1;
699            final int after = ast.getColumnNo() + ast.getText().length();
700
701            if (before >= 0 && shouldCheckSeparationFromPreviousToken(ast)
702                        && !CommonUtil.isCodePointWhitespace(line, before)) {
703                log(ast, MSG_WS_NOT_PRECEDED, ast.getText());
704            }
705
706            if (after < line.length) {
707                final char nextChar = Character.toChars(line[after])[0];
708                if (shouldCheckSeparationFromNextToken(ast, nextChar)
709                        && !Character.isWhitespace(nextChar)) {
710                    log(ast, MSG_WS_NOT_FOLLOWED, ast.getText());
711                }
712            }
713        }
714    }

716    /**
717     * Is ast not a target of Check.
718     *
719     * @param ast ast
720     * @param currentType type of ast
721     * @return true is ok to skip validation
722     */
723    private boolean isNotRelevantSituation(DetailAST ast, int currentType) {
724        final int parentType = ast.getParent().getType();
725        final boolean starImport = currentType == TokenTypes.STAR
726                && parentType == TokenTypes.DOT;
727        final boolean insideCaseGroup = parentType == TokenTypes.CASE_GROUP;
728
729        final boolean starImportOrSlistInsideCaseGroup = starImport || insideCaseGroup;
730        final boolean colonOfCaseOrDefaultOrForEach =
731                isColonOfCaseOrDefault(parentType)
732                        || isColonOfForEach(parentType);
733        final boolean emptyBlockOrType =
734                isEmptyBlock(ast, parentType)
735                    || allowEmptyTypes && isEmptyType(ast);
736
737        return starImportOrSlistInsideCaseGroup
738                || colonOfCaseOrDefaultOrForEach
739                || emptyBlockOrType
740                || isArrayInitialization(currentType, parentType);
741    }

743    /**
744     * Check if it should be checked if previous token is separated from current by
745     * whitespace.
746     * This function is needed to recognise double brace initialization as valid,
747     * unfortunately it's not possible to implement this functionality
748     * in isNotRelevantSituation method, because in this method when we return
749     * true(is not relevant) ast is later doesn't check at all. For example:
750     * new Properties() {{setProperty("double curly braces", "are not a style violation");
751     * }};
752     * For second left curly brace in first line when we would return true from
753     * isNotRelevantSituation it wouldn't later check that the next token(setProperty)
754     * is not separated from previous token.
755     *
756     * @param ast current AST.
757     * @return true if it should be checked if previous token is separated by whitespace,
758     *      false otherwise.
759     */
760    private static boolean shouldCheckSeparationFromPreviousToken(DetailAST ast) {
761        return !isPartOfDoubleBraceInitializerForPreviousToken(ast);
762    }
763
764    /**
765     * Check if it should be checked if next token is separated from current by
766     * whitespace. Explanation why this method is needed is identical to one
767     * included in shouldCheckSeparationFromPreviousToken method.
768     *
769     * @param ast current AST.
770     * @param nextChar next character.
771     * @return true if it should be checked if next token is separated by whitespace,
772     *      false otherwise.
773     */
774    private boolean shouldCheckSeparationFromNextToken(DetailAST ast, char nextChar) {
775        return !isEmptyCtorBlockCheckedFromSlist(ast)
776                && !(ast.getType() == TokenTypes.LITERAL_RETURN
777                && ast.getFirstChild().getType() == TokenTypes.SEMI)
778                && ast.getType() != TokenTypes.ARRAY_INIT
779                && !isAnonymousInnerClassEnd(ast.getType(), nextChar)
780                && !isPartOfDoubleBraceInitializerForNextToken(ast);
781    }

783    /**
784     * Check for "})" or "};" or "},". Happens with anon-inners
785     *
786     * @param currentType token
787     * @param nextChar next symbol
788     * @return true is that is end of anon inner class
789     */
790    private static boolean isAnonymousInnerClassEnd(int currentType, char nextChar) {
791        return currentType == TokenTypes.RCURLY
792                && (nextChar == ')'
793                        || nextChar == ';'
794                        || nextChar == ','
795                        || nextChar == '.');
796    }

799     * Is empty block.
800     *
801     * @param ast ast
802     * @param parentType parent
803     * @return true is block is empty
804     */
805    private boolean isEmptyBlock(DetailAST ast, int parentType) {
806        return isEmptyMethodBlock(ast, parentType)
807                || isEmptyCtorBlockCheckedFromRcurly(ast)
808                || isEmptyLoop(ast, parentType)
809                || isEmptyLambda(ast, parentType)
810                || isEmptyCatch(ast, parentType);

813    /**
814     * Tests if a given {@code DetailAST} is part of an empty block.
815     * An example empty block might look like the following
816     * <pre>   public void myMethod(int val) {}</pre>
817     * In the above, the method body is an empty block ("{}").
818     *
819     * @param ast the {@code DetailAST} to test.
820     * @param parentType the token type of {@code ast}'s parent.
821     * @param match the parent token type we're looking to match.
822     * @return {@code true} if {@code ast} makes up part of an
823     *         empty block contained under a {@code match} token type
824     *         node.
825     */
826    private static boolean isEmptyBlock(DetailAST ast, int parentType, int match) {
827        final boolean result;
828        final int type = ast.getType();
829        if (type == TokenTypes.RCURLY) {
830            final DetailAST parent = ast.getParent();
831            final DetailAST grandParent = ast.getParent().getParent();
832            result = parent.getFirstChild().getType() == TokenTypes.RCURLY
833                    && grandParent.getType() == match;
834        }
835        else {
836            result = type == TokenTypes.SLIST
837                && parentType == match
838                && ast.getFirstChild().getType() == TokenTypes.RCURLY;
839        }
840        return result;
841    }

986    /**
987     * Check if given ast is part of double brace initializer and if it
988     * should omit checking if previous token is separated by whitespace.
989     *
990     * @param ast ast to check
991     * @return true if it should omit checking for previous token, false otherwise
992     */
993    private static boolean isPartOfDoubleBraceInitializerForPreviousToken(DetailAST ast) {
994        final boolean initializerBeginsAfterClassBegins =
995                ast.getParent().getType() == TokenTypes.INSTANCE_INIT;
996        final boolean classEndsAfterInitializerEnds = ast.getPreviousSibling() != null
997                && ast.getPreviousSibling().getType() == TokenTypes.INSTANCE_INIT;
998        return initializerBeginsAfterClassBegins || classEndsAfterInitializerEnds;
999    }

1001    /**
1002     * Check if given ast is part of double brace initializer and if it
1003     * should omit checking if next token is separated by whitespace.
1004     * See <a href="https://github.com/checkstyle/checkstyle/pull/2845">
1005     * PR#2845</a> for more information why this function was needed.
1006     *
1007     * @param ast ast to check
1008     * @return true if it should omit checking for next token, false otherwise
1009     */
1010    private static boolean isPartOfDoubleBraceInitializerForNextToken(DetailAST ast) {
1011        final boolean classBeginBeforeInitializerBegin = ast.getType() == TokenTypes.LCURLY
1012            && ast.getNextSibling().getType() == TokenTypes.INSTANCE_INIT;
1013        final boolean initializerEndsBeforeClassEnds =
1014            ast.getParent().getParent().getType() == TokenTypes.INSTANCE_INIT
1015            && ast.getParent().getParent().getNextSibling().getType() == TokenTypes.RCURLY;
1016        return classBeginBeforeInitializerBegin || initializerEndsBeforeClassEnds;
1017    }

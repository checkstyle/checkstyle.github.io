028/**
029 * Checks that a token is surrounded by whitespace.
030 *
031 * <p>By default the check will check the following operators:
032 *  {@link TokenTypes#LITERAL_ASSERT ASSERT},
033 *  {@link TokenTypes#ASSIGN ASSIGN},
034 *  {@link TokenTypes#BAND BAND},
035 *  {@link TokenTypes#BAND_ASSIGN BAND_ASSIGN},
036 *  {@link TokenTypes#BOR BOR},
037 *  {@link TokenTypes#BOR_ASSIGN BOR_ASSIGN},
038 *  {@link TokenTypes#BSR BSR},
039 *  {@link TokenTypes#BSR_ASSIGN BSR_ASSIGN},
040 *  {@link TokenTypes#BXOR BXOR},
041 *  {@link TokenTypes#BXOR_ASSIGN BXOR_ASSIGN},
042 *  {@link TokenTypes#COLON COLON},
043 *  {@link TokenTypes#DIV DIV},
044 *  {@link TokenTypes#DIV_ASSIGN DIV_ASSIGN},
045 *  {@link TokenTypes#DO_WHILE DO_WHILE},
046 *  {@link TokenTypes#EQUAL EQUAL},
047 *  {@link TokenTypes#GE GE},
048 *  {@link TokenTypes#GT GT},
049 *  {@link TokenTypes#LAND LAND},
050 *  {@link TokenTypes#LCURLY LCURLY},
051 *  {@link TokenTypes#LE LE},
052 *  {@link TokenTypes#LITERAL_CATCH LITERAL_CATCH},
053 *  {@link TokenTypes#LITERAL_DO LITERAL_DO},
054 *  {@link TokenTypes#LITERAL_ELSE LITERAL_ELSE},
055 *  {@link TokenTypes#LITERAL_FINALLY LITERAL_FINALLY},
056 *  {@link TokenTypes#LITERAL_FOR LITERAL_FOR},
057 *  {@link TokenTypes#LITERAL_IF LITERAL_IF},
058 *  {@link TokenTypes#LITERAL_RETURN LITERAL_RETURN},
059 *  {@link TokenTypes#LITERAL_SWITCH LITERAL_SWITCH},
060 *  {@link TokenTypes#LITERAL_SYNCHRONIZED LITERAL_SYNCHRONIZED},
061 *  {@link TokenTypes#LITERAL_TRY LITERAL_TRY},
062 *  {@link TokenTypes#LITERAL_WHILE LITERAL_WHILE},
063 *  {@link TokenTypes#LOR LOR},
064 *  {@link TokenTypes#LT LT},
065 *  {@link TokenTypes#MINUS MINUS},
066 *  {@link TokenTypes#MINUS_ASSIGN MINUS_ASSIGN},
067 *  {@link TokenTypes#MOD MOD},
068 *  {@link TokenTypes#MOD_ASSIGN MOD_ASSIGN},
069 *  {@link TokenTypes#NOT_EQUAL NOT_EQUAL},
070 *  {@link TokenTypes#PLUS PLUS},
071 *  {@link TokenTypes#PLUS_ASSIGN PLUS_ASSIGN},
072 *  {@link TokenTypes#QUESTION QUESTION},
073 *  {@link TokenTypes#RCURLY RCURLY},
074 *  {@link TokenTypes#SL SL},
075 *  {@link TokenTypes#SLIST SLIST},
076 *  {@link TokenTypes#SL_ASSIGN SL_ASSIGN},
077 *  {@link TokenTypes#SR SR},
078 *  {@link TokenTypes#SR_ASSIGN SR_ASSIGN},
079 *  {@link TokenTypes#STAR STAR},
080 *  {@link TokenTypes#STAR_ASSIGN STAR_ASSIGN},
081 *  {@link TokenTypes#LITERAL_ASSERT LITERAL_ASSERT},
082 *  {@link TokenTypes#TYPE_EXTENSION_AND TYPE_EXTENSION_AND}.
083 *
084 * <p>An example of how to configure the check is:
085 *
086 * <pre>
087 * &lt;module name="WhitespaceAround"/&gt;
088 * </pre>
089 *
090 * <p>An example of how to configure the check for whitespace only around
091 * assignment operators is:
092 *
093 * <pre>
094 * &lt;module name="WhitespaceAround"&gt;
095 *     &lt;property name="tokens"
096 *               value="ASSIGN,DIV_ASSIGN,PLUS_ASSIGN,MINUS_ASSIGN,STAR_ASSIGN,
097 *                      MOD_ASSIGN,SR_ASSIGN,BSR_ASSIGN,SL_ASSIGN,BXOR_ASSIGN,
098 *                      BOR_ASSIGN,BAND_ASSIGN"/&gt;
099 * &lt;/module&gt;
100 * </pre>
101 *
102 * <p>An example of how to configure the check for whitespace only around
103 * curly braces is:
104 * <pre>
105 * &lt;module name="WhitespaceAround"&gt;
106 *     &lt;property name="tokens"
107 *               value="LCURLY,RCURLY"/&gt;
108 * &lt;/module&gt;
109 * </pre>
110 *
111 * <p>In addition, this check can be configured to allow empty methods, types,
112 * for, while, do-while loops, lambdas and constructor bodies.
113 * For example:
114 *
115 * <pre>{@code
116 * public MyClass() {}      // empty constructor
117 * public void func() {}    // empty method
118 * public interface Foo {} // empty interface
119 * public class Foo {} // empty class
120 * public enum Foo {} // empty enum
121 * MyClass c = new MyClass() {}; // empty anonymous class
122 * while (i = 1) {} // empty while loop
123 * for (int i = 1; i &gt; 1; i++) {} // empty for loop
124 * do {} while (i = 1); // empty do-while loop
125 * Runnable noop = () -> {}; // empty lambda
126 * public @interface Beta {} // empty annotation type
127 * }</pre>
128 *
129 * <p>This check does not flag as violation double brace initialization like:</p>
130 * <pre>
131 *   new Properties() {{
132 *     setProperty("key", "value");
133 *   }};
134 * </pre>
135 *
136 * <p>To configure the check to allow empty method blocks use
137 *
138 * <pre>   &lt;property name="allowEmptyMethods" value="true" /&gt;</pre>
139 *
140 * <p>To configure the check to allow empty constructor blocks use
141 *
142 * <pre>   &lt;property name="allowEmptyConstructors" value="true" /&gt;</pre>
143 *
144 * <p>To configure the check to allow empty type blocks use
145 *
146 * <pre>   &lt;property name="allowEmptyTypes" value="true" /&gt;</pre>
147 *
148 * <p>To configure the check to allow empty loop blocks use
149 *
150 * <pre>   &lt;property name="allowEmptyLoops" value="true" /&gt;</pre>
151 *
152 * <p>To configure the check to allow empty lambdas blocks use
153 *
154 * <pre>   &lt;property name="allowEmptyLambdas" value="true" /&gt;</pre>
155 *
156 * <p>Also, this check can be configured to ignore the colon in an enhanced for
157 * loop. The colon in an enhanced for loop is ignored by default
158 *
159 * <p>To configure the check to ignore the colon
160 *
161 * <pre>   &lt;property name="ignoreEnhancedForColon" value="true" /&gt;</pre>
162 *
163 */
164@StatelessCheck
165public class WhitespaceAroundCheck extends AbstractCheck {

375    @Override
376    public void visitToken(DetailAST ast) {
377        final int currentType = ast.getType();
378        if (!isNotRelevantSituation(ast, currentType)) {
379            final String line = getLine(ast.getLineNo() - 1);
380            final int before = ast.getColumnNo() - 1;
381            final int after = ast.getColumnNo() + ast.getText().length();
382
383            if (before >= 0) {
384                final char prevChar = line.charAt(before);
385                if (shouldCheckSeparationFromPreviousToken(ast)
386                        && !Character.isWhitespace(prevChar)) {
387                    log(ast, MSG_WS_NOT_PRECEDED, ast.getText());
388                }
389            }
390
391            if (after < line.length()) {
392                final char nextChar = line.charAt(after);
393                if (shouldCheckSeparationFromNextToken(ast, nextChar)
394                        && !Character.isWhitespace(nextChar)) {
395                    log(ast.getLineNo(), ast.getColumnNo() + ast.getText().length(),
396                            MSG_WS_NOT_FOLLOWED, ast.getText());
397                }
398            }
399        }
400    }

402    /**
403     * Is ast not a target of Check.
404     * @param ast ast
405     * @param currentType type of ast
406     * @return true is ok to skip validation
407     */
408    private boolean isNotRelevantSituation(DetailAST ast, int currentType) {
409        final int parentType = ast.getParent().getType();
410        final boolean starImport = currentType == TokenTypes.STAR
411                && parentType == TokenTypes.DOT;
412        final boolean slistInsideCaseGroup = currentType == TokenTypes.SLIST
413                && parentType == TokenTypes.CASE_GROUP;
414
415        final boolean starImportOrSlistInsideCaseGroup = starImport || slistInsideCaseGroup;
416        final boolean colonOfCaseOrDefaultOrForEach =
417                isColonOfCaseOrDefault(currentType, parentType)
418                        || isColonOfForEach(currentType, parentType);
419        final boolean emptyBlockOrType =
420                isEmptyBlock(ast, parentType)
421                    || allowEmptyTypes && isEmptyType(ast);
422
423        return starImportOrSlistInsideCaseGroup
424                || colonOfCaseOrDefaultOrForEach
425                || emptyBlockOrType
426                || isArrayInitialization(currentType, parentType);
427    }

429    /**
430     * Check if it should be checked if previous token is separated from current by
431     * whitespace.
432     * This function is needed to recognise double brace initialization as valid,
433     * unfortunately its not possible to implement this functionality
434     * in isNotRelevantSituation method, because in this method when we return
435     * true(is not relevant) ast is later doesn't check at all. For example:
436     * new Properties() {{setProperty("double curly braces", "are not a style error");
437     * }};
438     * For second left curly brace in first line when we would return true from
439     * isNotRelevantSituation it wouldn't later check that the next token(setProperty)
440     * is not separated from previous token.
441     * @param ast current AST.
442     * @return true if it should be checked if previous token is separated by whitespace,
443     *      false otherwise.
444     */
445    private static boolean shouldCheckSeparationFromPreviousToken(DetailAST ast) {
446        return !isPartOfDoubleBraceInitializerForPreviousToken(ast);
447    }
448
449    /**
450     * Check if it should be checked if next token is separated from current by
451     * whitespace. Explanation why this method is needed is identical to one
452     * included in shouldCheckSeparationFromPreviousToken method.
453     * @param ast current AST.
454     * @param nextChar next character.
455     * @return true if it should be checked if next token is separated by whitespace,
456     *      false otherwise.
457     */
458    private static boolean shouldCheckSeparationFromNextToken(DetailAST ast, char nextChar) {
459        return !(ast.getType() == TokenTypes.LITERAL_RETURN
460                    && ast.getFirstChild().getType() == TokenTypes.SEMI)
461                && ast.getType() != TokenTypes.ARRAY_INIT
462                && !isAnonymousInnerClassEnd(ast.getType(), nextChar)
463                && !isPartOfDoubleBraceInitializerForNextToken(ast);
464    }

466    /**
467     * Check for "})" or "};" or "},". Happens with anon-inners
468     * @param currentType token
469     * @param nextChar next symbol
470     * @return true is that is end of anon inner class
471     */
472    private static boolean isAnonymousInnerClassEnd(int currentType, char nextChar) {
473        return currentType == TokenTypes.RCURLY
474                && (nextChar == ')'
475                        || nextChar == ';'
476                        || nextChar == ','
477                        || nextChar == '.');
478    }

480    /**
481     * Is empty block.
482     * @param ast ast
483     * @param parentType parent
484     * @return true is block is empty
485     */
486    private boolean isEmptyBlock(DetailAST ast, int parentType) {
487        return isEmptyMethodBlock(ast, parentType)
488                || isEmptyCtorBlock(ast, parentType)
489                || isEmptyLoop(ast, parentType)
490                || isEmptyLambda(ast, parentType)
491                || isEmptyCatch(ast, parentType);
492    }

494    /**
495     * Tests if a given {@code DetailAST} is part of an empty block.
496     * An example empty block might look like the following
497     * <p>
498     * <pre>   public void myMethod(int val) {}</pre>
499     * </p>
500     * In the above, the method body is an empty block ("{}").
501     *
502     * @param ast the {@code DetailAST} to test.
503     * @param parentType the token type of {@code ast}'s parent.
504     * @param match the parent token type we're looking to match.
505     * @return {@code true} if {@code ast} makes up part of an
506     *         empty block contained under a {@code match} token type
507     *         node.
508     */
509    private static boolean isEmptyBlock(DetailAST ast, int parentType, int match) {
510        final boolean result;
511        final int type = ast.getType();
512        if (type == TokenTypes.RCURLY) {
513            final DetailAST parent = ast.getParent();
514            final DetailAST grandParent = ast.getParent().getParent();
515            result = parentType == TokenTypes.SLIST
516                    && parent.getFirstChild().getType() == TokenTypes.RCURLY
517                    && grandParent.getType() == match;
518        }
519        else {
520            result = type == TokenTypes.SLIST
521                && parentType == match
522                && ast.getFirstChild().getType() == TokenTypes.RCURLY;
523        }
524        return result;
525    }

527    /**
528     * Whether colon belongs to cases or defaults.
529     * @param currentType current
530     * @param parentType parent
531     * @return true if current token in colon of case or default tokens
532     */
533    private static boolean isColonOfCaseOrDefault(int currentType, int parentType) {
534        return currentType == TokenTypes.COLON
535                && (parentType == TokenTypes.LITERAL_DEFAULT
536                        || parentType == TokenTypes.LITERAL_CASE);
537    }

545    private boolean isColonOfForEach(int currentType, int parentType) {
546        return currentType == TokenTypes.COLON
547                && parentType == TokenTypes.FOR_EACH_CLAUSE
548                && ignoreEnhancedForColon;

627    /**
628     * Test if the given {@code DetailAST} is part of an empty block.
629     * An example empty block might look like the following
630     * <p>
631     * <pre>   class Foo {}</pre>
632     * </p>
633     *
634     * @param ast ast the {@code DetailAST} to test.
635     * @return {@code true} if {@code ast} makes up part of an
636     *         empty block contained under a {@code match} token type
637     *         node.
638     */
639    private static boolean isEmptyType(DetailAST ast) {
640        final int type = ast.getType();
641        final DetailAST nextSibling = ast.getNextSibling();
642        final DetailAST previousSibling = ast.getPreviousSibling();
643        return type == TokenTypes.LCURLY
644                    && nextSibling.getType() == TokenTypes.RCURLY
645                || type == TokenTypes.RCURLY
646                    && previousSibling != null
647                    && previousSibling.getType() == TokenTypes.LCURLY;
648    }
649
650    /**
651     * Check if given ast is part of double brace initializer and if it
652     * should omit checking if previous token is separated by whitespace.
653     * @param ast ast to check
654     * @return true if it should omit checking for previous token, false otherwise
655     */
656    private static boolean isPartOfDoubleBraceInitializerForPreviousToken(DetailAST ast) {
657        final boolean initializerBeginsAfterClassBegins = ast.getType() == TokenTypes.SLIST
658                && ast.getParent().getType() == TokenTypes.INSTANCE_INIT;
659        final boolean classEndsAfterInitializerEnds = ast.getType() == TokenTypes.RCURLY
660                && ast.getPreviousSibling() != null
661                && ast.getPreviousSibling().getType() == TokenTypes.INSTANCE_INIT;
662        return initializerBeginsAfterClassBegins || classEndsAfterInitializerEnds;
663    }
664
665    /**
666     * Check if given ast is part of double brace initializer and if it
667     * should omit checking if next token is separated by whitespace.
668     * See <a href="https://github.com/checkstyle/checkstyle/pull/2845">
669     * PR#2845</a> for more information why this function was needed.
670     * @param ast ast to check
671     * @return true if it should omit checking for next token, false otherwise
672     */
673    private static boolean isPartOfDoubleBraceInitializerForNextToken(DetailAST ast) {
674        final boolean classBeginBeforeInitializerBegin = ast.getType() == TokenTypes.LCURLY
675            && ast.getNextSibling().getType() == TokenTypes.INSTANCE_INIT;
676        final boolean initializerEndsBeforeClassEnds = ast.getType() == TokenTypes.RCURLY
677            && ast.getParent().getType() == TokenTypes.SLIST
678            && ast.getParent().getParent().getType() == TokenTypes.INSTANCE_INIT
679            && ast.getParent().getParent().getNextSibling().getType() == TokenTypes.RCURLY;
680        return classBeginBeforeInitializerBegin || initializerEndsBeforeClassEnds;
681    }

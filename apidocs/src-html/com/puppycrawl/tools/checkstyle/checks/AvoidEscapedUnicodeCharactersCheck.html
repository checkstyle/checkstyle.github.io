035/**
036 * <p>
037 * Restricts using
038 * <a href = "https://docs.oracle.com/javase/specs/jls/se11/html/jls-3.html#jls-3.3">
039 * Unicode escapes</a>
040 * (such as &#92;u221e). It is possible to allow using escapes for
041 * <a href="https://en.wiktionary.org/wiki/Appendix:Control_characters">
042 * non-printable, control characters</a>.
043 * Also, this check can be configured to allow using escapes
044 * if trail comment is present. By the option it is possible to
045 * allow using escapes if literal contains only them.
046 * </p>
047 * <ul>
048 * <li>
049 * Property {@code allowEscapesForControlCharacters} - Allow use escapes for
050 * non-printable, control characters.
051 * Type is {@code boolean}.
052 * Default value is {@code false}.
053 * </li>
054 * <li>
055 * Property {@code allowByTailComment} - Allow use escapes if trail comment is present.
056 * Type is {@code boolean}.
057 * Default value is {@code false}.
058 * </li>
059 * <li>
060 * Property {@code allowIfAllCharactersEscaped} - Allow if all characters in literal are escaped.
061 * Type is {@code boolean}.
062 * Default value is {@code false}.
063 * </li>
064 * <li>
065 * Property {@code allowNonPrintableEscapes} - Allow use escapes for
066 * non-printable, whitespace characters.
067 * Type is {@code boolean}.
068 * Default value is {@code false}.
069 * </li>
070 * </ul>
071 * <p>
072 * To configure the check:
073 * </p>
074 * <pre>
075 * &lt;module name="AvoidEscapedUnicodeCharacters"/&gt;
076 * </pre>
077 * <p>
078 * Examples of using Unicode:</p>
079 * <pre>
080 * String unitAbbrev = "Î¼s";      // Best: perfectly clear even without a comment.
081 * String unitAbbrev = "&#92;u03bcs"; // Poor: the reader has no idea what this is.
082 * </pre>
083 * <p>
084 * An example of non-printable, control characters.
085 * </p>
086 * <pre>
087 * return '&#92;ufeff' + content; // byte order mark
088 * </pre>
089 * <p>
090 * An example of how to configure the check to allow using escapes
091 * for non-printable, control characters:
092 * </p>
093 * <pre>
094 * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
095 *   &lt;property name="allowEscapesForControlCharacters" value="true"/&gt;
096 * &lt;/module&gt;
097 * </pre>
098 * <p>
099 * Example of using escapes with trail comment:
100 * </p>
101 * <pre>
102 * String unitAbbrev = "&#92;u03bcs"; // Greek letter mu, "s"
103 * String textBlockUnitAbbrev = """
104 *          &#92;u03bcs"""; // Greek letter mu, "s"
105 * </pre>
106 * <p>An example of how to configure the check to allow using escapes
107 * if trail comment is present:
108 * </p>
109 * <pre>
110 * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
111 *   &lt;property name="allowByTailComment" value="true"/&gt;
112 * &lt;/module&gt;
113 * </pre>
114 * <p>Example of using escapes if literal contains only them:
115 * </p>
116 * <pre>
117 * String unitAbbrev = "&#92;u03bc&#92;u03bc&#92;u03bc";
118 * </pre>
119 * <p>An example of how to configure the check to allow escapes
120 * if literal contains only them:
121 * </p>
122 * <pre>
123 * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
124 *   &lt;property name="allowIfAllCharactersEscaped" value="true"/&gt;
125 * &lt;/module&gt;
126 * </pre>
127 * <p>An example of how to configure the check to allow using escapes
128 * for non-printable, whitespace characters:
129 * </p>
130 * <pre>
131 * &lt;module name="AvoidEscapedUnicodeCharacters"&gt;
132 *   &lt;property name="allowNonPrintableEscapes" value="true"/&gt;
133 * &lt;/module&gt;
134 * </pre>
135 * <p>
136 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
137 * </p>
138 * <p>
139 * Violation Message Keys:
140 * </p>
141 * <ul>
142 * <li>
143 * {@code forbid.escaped.unicode.char}
144 * </li>
145 * </ul>
146 *
147 * @since 5.8
148 */
149@FileStatefulCheck
150public class AvoidEscapedUnicodeCharactersCheck
151    extends AbstractCheck {
152
153    /**
154     * A key is pointing to the warning message text in "messages.properties"
155     * file.
156     */
157    public static final String MSG_KEY = "forbid.escaped.unicode.char";
158
159    /** Regular expression for Unicode chars. */
160    private static final Pattern UNICODE_REGEXP = Pattern.compile("\\\\u[a-fA-F0-9]{4}");
161
162    /**
163     * Regular expression Unicode control characters.
164     *
165     * @see <a href="https://en.wiktionary.org/wiki/Appendix:Control_characters">
166     *     Appendix:Control characters</a>
167     */
168    private static final Pattern UNICODE_CONTROL = Pattern.compile("\\\\[uU]"
169            + "(00[0-1][0-9A-Fa-f]"
170            + "|00[8-9][0-9A-Fa-f]"
171            + "|00[aA][dD]"
172            + "|034[fF]"
173            + "|070[fF]"
174            + "|180[eE]"
175            + "|200[b-fB-F]"
176            + "|202[a-eA-E]"
177            + "|206[0-4a-fA-F]"
178            + "|[fF]{3}[9a-bA-B]"
179            + "|[fF][eE][fF]{2})");
180
181    /**
182     * Regular expression for all escaped chars.
183     * See "EscapeSequence" at
184     * https://docs.oracle.com/javase/specs/jls/se15/html/jls-3.html#jls-3.10.7
185     */
186    private static final Pattern ALL_ESCAPED_CHARS = Pattern.compile("^((\\\\u)[a-fA-F0-9]{4}"
187            + "|\""
188            + "|'"
189            + "|\\\\"
190            + "|\\\\b"
191            + "|\\\\f"
192            + "|\\\\n"
193            + "|\\\\r"
194            + "|\\\\s"
195            + "|\\\\t"
196            + ")+$");
197
198    /** Regular expression for escaped backslash. */
199    private static final Pattern ESCAPED_BACKSLASH = Pattern.compile("\\\\\\\\");
200

273    /** Cpp style comments. */
274    private Map<Integer, TextBlock> singlelineComments;
275    /** C style comments. */
276    private Map<Integer, List<TextBlock>> blockComments;
277
278    /** Allow use escapes for non-printable, control characters. */
279    private boolean allowEscapesForControlCharacters;
280
281    /** Allow use escapes if trail comment is present. */
282    private boolean allowByTailComment;
283
284    /** Allow if all characters in literal are escaped. */
285    private boolean allowIfAllCharactersEscaped;
286
287    /** Allow use escapes for non-printable, whitespace characters. */
288    private boolean allowNonPrintableEscapes;
289
290    /**
291     * Setter to allow use escapes for non-printable, control characters.
292     *
293     * @param allow user's value.
294     */
295    public final void setAllowEscapesForControlCharacters(boolean allow) {
296        allowEscapesForControlCharacters = allow;
297    }
298
299    /**
300     * Setter to allow use escapes if trail comment is present.
301     *
302     * @param allow user's value.
303     */
304    public final void setAllowByTailComment(boolean allow) {
305        allowByTailComment = allow;
306    }
307
308    /**
309     * Setter to allow if all characters in literal are escaped.
310     *
311     * @param allow user's value.
312     */
313    public final void setAllowIfAllCharactersEscaped(boolean allow) {
314        allowIfAllCharactersEscaped = allow;
315    }
316
317    /**
318     * Setter to allow use escapes for non-printable, whitespace characters.

322    public final void setAllowNonPrintableEscapes(boolean allow) {
323        allowNonPrintableEscapes = allow;

326    @Override
327    public int[] getDefaultTokens() {
328        return getRequiredTokens();
329    }
330
331    @Override
332    public int[] getAcceptableTokens() {
333        return getRequiredTokens();
334    }
335
336    @Override
337    public int[] getRequiredTokens() {
338        return new int[] {
339            TokenTypes.STRING_LITERAL,
340            TokenTypes.CHAR_LITERAL,
341            TokenTypes.TEXT_BLOCK_CONTENT,
342        };
343    }
344
345    @Override
346    public void beginTree(DetailAST rootAST) {
347        singlelineComments = getFileContents().getSingleLineComments();
348        blockComments = getFileContents().getBlockComments();
349    }
350
351    @Override
352    public void visitToken(DetailAST ast) {
353        final String literal =
354            CheckUtil.stripIndentAndInitialNewLineFromTextBlock(ast.getText());
355
356        if (hasUnicodeChar(literal) && !(allowByTailComment && hasTrailComment(ast)
357                || isAllCharactersEscaped(literal)
358                || allowEscapesForControlCharacters
359                        && isOnlyUnicodeValidChars(literal, UNICODE_CONTROL)
360                || allowNonPrintableEscapes
361                        && isOnlyUnicodeValidChars(literal, NON_PRINTABLE_CHARS))) {
362            log(ast, MSG_KEY);
363        }
364    }
365
366    /**
367     * Checks if literal has Unicode chars.
368     *
369     * @param literal String literal.
370     * @return true if literal has Unicode chars.
371     */
372    private static boolean hasUnicodeChar(String literal) {
373        final String literalWithoutEscapedBackslashes =
374                ESCAPED_BACKSLASH.matcher(literal).replaceAll("");
375        return UNICODE_REGEXP.matcher(literalWithoutEscapedBackslashes).find();
376    }
377
378    /**
379     * Check if String literal contains Unicode control chars.
380     *
381     * @param literal String literal.
382     * @param pattern RegExp for valid characters.
383     * @return true, if String literal contains Unicode control chars.
384     */
385    private static boolean isOnlyUnicodeValidChars(String literal, Pattern pattern) {
386        final int unicodeMatchesCounter =
387                countMatches(UNICODE_REGEXP, literal);
388        final int unicodeValidMatchesCounter =
389                countMatches(pattern, literal);
390        return unicodeMatchesCounter - unicodeValidMatchesCounter == 0;
391    }
392
393    /**
394     * Check if trail comment is present after ast token.
395     *
396     * @param ast current token.
397     * @return true if trail comment is present after ast token.
398     */
399    private boolean hasTrailComment(DetailAST ast) {
400        int lineNo = ast.getLineNo();
401
402        // Since the trailing comment in the case of text blocks must follow the """ delimiter,
403        // we need to look for it after TEXT_BLOCK_LITERAL_END.
404        if (ast.getType() == TokenTypes.TEXT_BLOCK_CONTENT) {
405            lineNo = ast.getNextSibling().getLineNo();
406        }
407        boolean result = false;
408        if (singlelineComments.containsKey(lineNo)) {
409            result = true;
410        }
411        else {
412            final List<TextBlock> commentList = blockComments.get(lineNo);
413            if (commentList != null) {
414                final TextBlock comment = commentList.get(commentList.size() - 1);
415                final String line = getLines()[lineNo - 1];
416                result = isTrailingBlockComment(comment, line);
417            }
418        }
419        return result;
420    }
421
422    /**
423     * Whether the C style comment is trailing.
424     *
425     * @param comment the comment to check.
426     * @param line the line where the comment starts.
427     * @return true if the comment is trailing.
428     */
429    private static boolean isTrailingBlockComment(TextBlock comment, String line) {
430        return comment.getText().length != 1
431            || CommonUtil.isBlank(line.substring(comment.getEndColNo() + 1));
432    }
433
434    /**
435     * Count regexp matches into String literal.
436     *
437     * @param pattern pattern.
438     * @param target String literal.
439     * @return count of regexp matches.
440     */
441    private static int countMatches(Pattern pattern, String target) {
442        int matcherCounter = 0;
443        final Matcher matcher = pattern.matcher(target);
444        while (matcher.find()) {
445            matcherCounter++;
446        }
447        return matcherCounter;
448    }
449
450    /**
451     * Checks if all characters in String literal is escaped.
452     *
453     * @param literal current literal.
454     * @return true if all characters in String literal is escaped.
455     */
456    private boolean isAllCharactersEscaped(String literal) {
457        return allowIfAllCharactersEscaped
458                && ALL_ESCAPED_CHARS.matcher(literal).find();
459    }

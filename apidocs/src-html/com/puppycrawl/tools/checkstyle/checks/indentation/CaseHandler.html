020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for case statements.
027 *
028 */
029public class CaseHandler extends AbstractExpressionHandler {
030
031    /**
032     * The child elements of a case expression.
033     */
034    private static final int[] CASE_CHILDREN = {
035        TokenTypes.LITERAL_CASE,
036        TokenTypes.LITERAL_DEFAULT,
037    };
038
039    /**
040     * Construct an instance of this handler with the given indentation check,
041     * abstract syntax tree, and parent handler.
042     *
043     * @param indentCheck   the indentation check
044     * @param expr          the abstract syntax tree
045     * @param parent        the parent handler
046     */
047    public CaseHandler(IndentationCheck indentCheck,
048        DetailAST expr, AbstractExpressionHandler parent) {
049        super(indentCheck, "case", expr, parent);
050    }
051
052    @Override
053    protected IndentLevel getIndentImpl() {
054        return new IndentLevel(getParent().getIndent(),
055                               getIndentCheck().getCaseIndent());
056    }
057
058    /**
059     * Check the indentation of the case statement.
060     */
061    private void checkCase() {
062        checkChildren(getMainAst(), CASE_CHILDREN, getIndent(), true, false);
063    }
064
065    @Override
066    public IndentLevel getSuggestedChildIndent(AbstractExpressionHandler child) {
067        return getIndent();
068    }
069
070    @Override
071    public void checkIndentation() {
072        checkCase();
073    }
074
075}

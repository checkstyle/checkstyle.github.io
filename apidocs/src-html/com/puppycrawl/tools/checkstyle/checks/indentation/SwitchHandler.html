020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for switch statements.
027 *
028 */
029public class SwitchHandler extends BlockParentHandler {
030
031    /**
032     * Construct an instance of this handler with the given indentation check,
033     * abstract syntax tree, and parent handler.
034     *
035     * @param indentCheck   the indentation check
036     * @param ast           the abstract syntax tree
037     * @param parent        the parent handler
038     */
039    public SwitchHandler(IndentationCheck indentCheck,
040        DetailAST ast, AbstractExpressionHandler parent) {
041        super(indentCheck, "switch", ast, parent);
042    }
043
044    @Override
045    protected DetailAST getLeftCurly() {
046        return getMainAst().findFirstToken(TokenTypes.LCURLY);
047    }
048
049    @Override
050    protected DetailAST getRightCurly() {
051        return getMainAst().findFirstToken(TokenTypes.RCURLY);
052    }
053
054    @Override
055    protected DetailAST getListChild() {
056        // all children should be taken care of by case handler (plus
057        // there is no parent of just the cases, if checking is needed
058        // here in the future, an additional way beyond checkChildren()
059        // will have to be devised to get children)
060        return null;
061    }
062
063    @Override
064    protected DetailAST getNonListChild() {
065        return null;
066    }
067
068    /**
069     * Check the indentation of the switch expression.
070     */
071    private void checkSwitchExpr() {
072        checkExpressionSubtree(
073            getMainAst().findFirstToken(TokenTypes.LPAREN).getNextSibling(),
074            getIndent(),
075            false,
076            false);
077    }
078
079    @Override
080    public void checkIndentation() {
081        checkSwitchExpr();
082        super.checkIndentation();
083    }
084
085}

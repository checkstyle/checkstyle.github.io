020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for inner classes.
027 *
028 * @author jrichard
029 */
030public class ObjectBlockHandler extends BlockParentHandler {
031
032    /**
033     * Construct an instance of this handler with the given indentation check,
034     * abstract syntax tree, and parent handler.
035     *
036     * @param indentCheck   the indentation check
037     * @param ast           the abstract syntax tree
038     * @param parent        the parent handler
039     */
040    public ObjectBlockHandler(IndentationCheck indentCheck,
041        DetailAST ast, AbstractExpressionHandler parent) {
042        super(indentCheck, "object def", ast, parent);
043    }
044
045    @Override
046    protected DetailAST getTopLevelAst() {
047        return null;
048    }
049
050    @Override
051    protected DetailAST getLeftCurly() {
052        return getMainAst().findFirstToken(TokenTypes.LCURLY);
053    }
054
055    @Override
056    protected DetailAST getRightCurly() {
057        return getMainAst().findFirstToken(TokenTypes.RCURLY);
058    }
059
060    @Override
061    protected DetailAST getListChild() {
062        return getMainAst();
063    }
064
065    @Override
066    protected IndentLevel getIndentImpl() {
067        final DetailAST parentAST = getMainAst().getParent();
068        IndentLevel indent = getParent().getIndent();
069        if (parentAST.getType() == TokenTypes.LITERAL_NEW) {
070            indent.addAcceptedIndent(super.getIndentImpl());
071        }
072        else if (parentAST.getType() == TokenTypes.ENUM_CONSTANT_DEF) {
073            indent = super.getIndentImpl();
074        }
075        return indent;
076    }
077
078    @Override
079    public void checkIndentation() {
080        // if we have a class or interface as a parent, don't do anything,
081        // as this is checked by class def; so
082        // only do this if we have a new for a parent (anonymous inner
083        // class)
084        final DetailAST parentAST = getMainAst().getParent();
085        if (parentAST.getType() == TokenTypes.LITERAL_NEW) {
086            super.checkIndentation();
087        }
088    }
089
090    @Override
091    protected IndentLevel curlyIndent() {
092        final IndentLevel indent = super.curlyIndent();
093        indent.addAcceptedIndent(indent.getFirstIndentLevel() + getLineWrappingIndentation());
094        return indent;
095    }
096
097    /**
098     * A shortcut for {@code IndentationCheck} property.
099     * @return value of lineWrappingIndentation property
100     *         of {@code IndentationCheck}
101     */
102    private int getLineWrappingIndentation() {
103        return getIndentCheck().getLineWrappingIndentation();
104    }
105
106}

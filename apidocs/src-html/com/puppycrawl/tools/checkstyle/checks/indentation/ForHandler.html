020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for for loops.
027 *
028 */
029public class ForHandler extends BlockParentHandler {
030
031    /**
032     * Construct an instance of this handler with the given indentation check,
033     * abstract syntax tree, and parent handler.
034     *
035     * @param indentCheck   the indentation check
036     * @param ast           the abstract syntax tree
037     * @param parent        the parent handler
038     */
039    public ForHandler(IndentationCheck indentCheck,
040        DetailAST ast, AbstractExpressionHandler parent) {
041        super(indentCheck, "for", ast, parent);
042    }
043
044    /**
045     * Check the indentation of the parameters of the 'for' loop.
046     */
047    private void checkForParams() {
048        final IndentLevel expected =
049            new IndentLevel(getIndent(), getBasicOffset());
050        final DetailAST init = getMainAst().findFirstToken(TokenTypes.FOR_INIT);
051
052        if (init == null) {
053            // for each
054            final DetailAST forEach =
055                    getMainAst().findFirstToken(TokenTypes.FOR_EACH_CLAUSE);
056            checkExpressionSubtree(forEach, expected, false, false);
057        }
058        else {
059            checkExpressionSubtree(init, expected, false, false);
060
061            final DetailAST cond =
062                    getMainAst().findFirstToken(TokenTypes.FOR_CONDITION);
063            checkExpressionSubtree(cond, expected, false, false);
064
065            final DetailAST forIterator =
066                    getMainAst().findFirstToken(TokenTypes.FOR_ITERATOR);
067            checkExpressionSubtree(forIterator, expected, false, false);
068        }
069    }
070
071    @Override
072    public void checkIndentation() {
073        checkForParams();
074        super.checkIndentation();
075        checkWrappingIndentation(getMainAst(), getForLoopRightParen(getMainAst()));
076    }
077
078    /**
079     * Returns right parenthesis of for-loop statement.
080     *
081     * @param literalForAst
082     *          literal-for ast node(TokenTypes.LITERAL_FOR)
083     * @return right parenthesis of for-loop statement.
084     */
085    private static DetailAST getForLoopRightParen(DetailAST literalForAst) {
086        return literalForAst.findFirstToken(TokenTypes.RPAREN);
087    }
088
089}

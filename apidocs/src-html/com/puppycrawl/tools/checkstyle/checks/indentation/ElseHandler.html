020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.api.TokenTypes;
024
025/**
026 * Handler for else blocks.
027 *
028 * @author jrichard
029 */
030public class ElseHandler extends BlockParentHandler {
031
032    /**
033     * Construct an instance of this handler with the given indentation check,
034     * abstract syntax tree, and parent handler.
035     *
036     * @param indentCheck   the indentation check
037     * @param ast           the abstract syntax tree
038     * @param parent        the parent handler
039     */
040    public ElseHandler(IndentationCheck indentCheck,
041        DetailAST ast, AbstractExpressionHandler parent) {
042        super(indentCheck, "else", ast, parent);
043    }
044
045    @Override
046    protected void checkTopLevelToken() {
047        // check if else is nested with rcurly of if:
048        //
049        //  } else ...
050
051        final DetailAST ifAST = getMainAst().getParent();
052        final DetailAST slist = ifAST.findFirstToken(TokenTypes.SLIST);
053        if (slist == null) {
054            super.checkTopLevelToken();
055        }
056        else {
057            final DetailAST lcurly = slist.getLastChild();
058            // indentation checked as part of LITERAL IF check
059            if (lcurly.getLineNo() != getMainAst().getLineNo()) {
060                super.checkTopLevelToken();
061            }
062        }
063    }
064
065    @Override
066    protected DetailAST getNonListChild() {
067        return getMainAst().getFirstChild();
068    }
069
070}

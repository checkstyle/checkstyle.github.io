020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022import com.puppycrawl.tools.checkstyle.api.DetailAST;
023import com.puppycrawl.tools.checkstyle.utils.TokenUtil;
024
025/**
026 * Handler for yield expression.
027 */
028public class YieldHandler extends AbstractExpressionHandler {
029
030    /**
031     * Construct an instance of this handler with the given indentation check,
032     * abstract syntax tree, and parent handler.
033     *
034     * @param indentCheck the indentation check
035     * @param ast         the abstract syntax tree
036     * @param parent      the parent handler
037     */
038    public YieldHandler(IndentationCheck indentCheck,
039                      DetailAST ast,
040                      AbstractExpressionHandler parent) {
041        super(indentCheck, "yield", ast, parent);
042    }
043
044    @Override
045    public void checkIndentation() {
046        checkYield();
047        final DetailAST expression = getMainAst().getFirstChild();
048        if (!TokenUtil.areOnSameLine(getMainAst(), expression)) {
049            checkExpressionSubtree(expression, getIndent(), false, false);
050        }
051    }
052
053    /**
054     * Check the indentation of the yield keyword.
055     */
056    private void checkYield() {
057        final DetailAST yieldKey = getMainAst();
058        final int columnNo = expandedTabsColumnNo(yieldKey);
059        if (isOnStartOfLine(yieldKey) && !getIndent().isAcceptable(columnNo)) {
060            logError(yieldKey, "", columnNo);
061        }
062    }
063}

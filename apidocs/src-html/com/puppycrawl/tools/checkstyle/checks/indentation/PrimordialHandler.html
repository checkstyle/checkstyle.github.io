020package com.puppycrawl.tools.checkstyle.checks.indentation;
021
022/**
023 * A default no-op handler.
024 *
025 */
026public class PrimordialHandler extends AbstractExpressionHandler {
027
028    /**
029     * Construct an instance of this handler with the given indentation check.
030     *
031     * @param indentCheck   the indentation check
032     */
033    public PrimordialHandler(IndentationCheck indentCheck) {
034        super(indentCheck, null, null, null);
035    }
036
037    @Override
038    public void checkIndentation() {
039        // nothing to check
040    }
041
042    @Override
043    public IndentLevel getSuggestedChildIndent(AbstractExpressionHandler child) {
044        return getIndentImpl();
045    }
046
047    @Override
048    protected IndentLevel getIndentImpl() {
049        return new IndentLevel(0);
050    }
051
052}

020package com.puppycrawl.tools.checkstyle.checks.modifier;
021
022import com.puppycrawl.tools.checkstyle.StatelessCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.Scope;
026import com.puppycrawl.tools.checkstyle.api.TokenTypes;
027import com.puppycrawl.tools.checkstyle.utils.ScopeUtil;
028
029/**
030 * <p>
031 * Checks for implicit modifiers on nested types in classes and records.
032 * </p>
033 * <p>
034 * This check is effectively the opposite of
035 * <a href="https://checkstyle.org/checks/modifier/redundantmodifier.html#RedundantModifier">
036 * RedundantModifier</a>.
037 * It checks the modifiers on nested types in classes and records, ensuring that certain modifiers
038 * are explicitly specified even though they are actually redundant.
039 * </p>
040 * <p>
041 * Nested enums, interfaces, and records within a class are always {@code static} and as such the
042 * compiler does not require the {@code static} modifier. This check provides the ability to enforce
043 * that the {@code static} modifier is explicitly coded and not implicitly added by the compiler.
044 * </p>
045 * <pre>
046 * public final class Person {
047 *   enum Age {  // violation
048 *     CHILD, ADULT
049 *   }
050 * }
051 * </pre>
052 * <p>
053 * Rationale for this check: Nested enums, interfaces, and records are treated differently from
054 * nested classes as they are only allowed to be {@code static}. Developers should not need to
055 * remember this rule, and this check provides the means to enforce that the modifier is coded
056 * explicitly.
057 * </p>
058 * <ul>
059 * <li>
060 * Property {@code violateImpliedStaticOnNestedEnum} - Control whether to enforce that
061 * {@code static} is explicitly coded on nested enums in classes and records.
062 * Type is {@code boolean}.
063 * Default value is {@code true}.
064 * </li>
065 * <li>
066 * Property {@code violateImpliedStaticOnNestedInterface} - Control whether to enforce that
067 * {@code static} is explicitly coded on nested interfaces in classes and records.
068 * Type is {@code boolean}.
069 * Default value is {@code true}.
070 * </li>
071 * <li>
072 * Property {@code violateImpliedStaticOnNestedRecord} - Control whether to enforce that
073 * {@code static} is explicitly coded on nested records in classes and records.
074 * Type is {@code boolean}.
075 * Default value is {@code true}.
076 * </li>
077 * </ul>
078 * <p>
079 * To configure the check so that it checks that all implicit modifiers on nested interfaces, enums,
080 * and records are explicitly specified in classes and records.
081 * </p>
082 * <p>
083 * Configuration:
084 * </p>
085 * <pre>
086 * &lt;module name="ClassMemberImpliedModifier" /&gt;
087 * </pre>
088 * <p>
089 * Code:
090 * </p>
091 * <pre>
092 * public final class Person {
093 *   static interface Address1 {  // valid
094 *   }
095 *
096 *   interface Address2 {  // violation
097 *   }
098 *
099 *   static enum Age1 {  // valid
100 *     CHILD, ADULT
101 *   }
102 *
103 *   enum Age2 {  // violation
104 *     CHILD, ADULT
105 *   }
106 *
107 *   public static record GoodRecord() {} // valid
108 *   public record BadRecord() {} // violation
109 *
110 *   public static record OuterRecord() {
111 *     static record InnerRecord1(){} // valid
112 *     record InnerRecord2(){} // violation
113 *   }
114 * }
115 * </pre>
116 * <p>
117 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
118 * </p>
119 * <p>
120 * Violation Message Keys:
121 * </p>
122 * <ul>
123 * <li>
124 * {@code class.implied.modifier}
125 * </li>
126 * </ul>
127 *
128 * @since 8.16
129 */
130@StatelessCheck
131public class ClassMemberImpliedModifierCheck
132    extends AbstractCheck {
133
134    /**
135     * A key is pointing to the warning message text in "messages.properties" file.
136     */
137    public static final String MSG_KEY = "class.implied.modifier";
138
139    /** Name for 'static' keyword. */
140    private static final String STATIC_KEYWORD = "static";
141
142    /**
143     * Control whether to enforce that {@code static} is explicitly coded
144     * on nested enums in classes and records.
145     */
146    private boolean violateImpliedStaticOnNestedEnum = true;
147
148    /**
149     * Control whether to enforce that {@code static} is explicitly coded
150     * on nested interfaces in classes and records.
151     */
152    private boolean violateImpliedStaticOnNestedInterface = true;
153
154    /**
155     * Control whether to enforce that {@code static} is explicitly coded
156     * on nested records in classes and records.
157     */
158    private boolean violateImpliedStaticOnNestedRecord = true;
159
160    /**
161     * Setter to control whether to enforce that {@code static} is explicitly coded
162     * on nested enums in classes and records.
163     *
164     * @param violateImplied
165     *        True to perform the check, false to turn the check off.
166     * @since 8.16
167     */
168    public void setViolateImpliedStaticOnNestedEnum(boolean violateImplied) {
169        violateImpliedStaticOnNestedEnum = violateImplied;
170    }
171
172    /**
173     * Setter to control whether to enforce that {@code static} is explicitly coded
174     * on nested interfaces in classes and records.
175     *
176     * @param violateImplied
177     *        True to perform the check, false to turn the check off.
178     * @since 8.16
179     */
180    public void setViolateImpliedStaticOnNestedInterface(boolean violateImplied) {
181        violateImpliedStaticOnNestedInterface = violateImplied;
182    }
183
184    /**
185     * Setter to control whether to enforce that {@code static} is explicitly coded
186     * on nested records in classes and records.
187     *
188     * @param violateImplied
189     *        True to perform the check, false to turn the check off.
190     * @since 8.36
191     */
192    public void setViolateImpliedStaticOnNestedRecord(boolean violateImplied) {
193        violateImpliedStaticOnNestedRecord = violateImplied;
194    }
195
196    @Override
197    public int[] getDefaultTokens() {
198        return getAcceptableTokens();
199    }
200
201    @Override
202    public int[] getRequiredTokens() {
203        return getAcceptableTokens();
204    }
205
206    @Override
207    public int[] getAcceptableTokens() {
208        return new int[] {
209            TokenTypes.INTERFACE_DEF,
210            TokenTypes.ENUM_DEF,
211            TokenTypes.RECORD_DEF,
212        };
213    }
214
215    @Override
216    public void visitToken(DetailAST ast) {
217        if (isInTypeBlock(ast)) {
218            final DetailAST modifiers = ast.findFirstToken(TokenTypes.MODIFIERS);
219            switch (ast.getType()) {
220                case TokenTypes.ENUM_DEF:
221                    if (violateImpliedStaticOnNestedEnum
222                            && modifiers.findFirstToken(TokenTypes.LITERAL_STATIC) == null) {
223                        log(ast, MSG_KEY, STATIC_KEYWORD);
224                    }
225                    break;
226                case TokenTypes.INTERFACE_DEF:
227                    if (violateImpliedStaticOnNestedInterface
228                            && modifiers.findFirstToken(TokenTypes.LITERAL_STATIC) == null) {
229                        log(ast, MSG_KEY, STATIC_KEYWORD);
230                    }
231                    break;
232                case TokenTypes.RECORD_DEF:
233                    if (violateImpliedStaticOnNestedRecord
234                            && modifiers.findFirstToken(TokenTypes.LITERAL_STATIC) == null) {
235                        log(ast, MSG_KEY, STATIC_KEYWORD);
236                    }
237                    break;
238                default:
239                    throw new IllegalStateException(ast.toString());
240            }
241        }
242    }
243
244    /**
245     * Checks if ast is in a class, enum, anon class or record block.
246     *
247     * @param ast the current ast
248     * @return true if ast is in a class, enum, anon class or record
249     */
250    private static boolean isInTypeBlock(DetailAST ast) {
251        return ScopeUtil.isInScope(ast, Scope.ANONINNER)
252                || ScopeUtil.isInClassBlock(ast)
253                || ScopeUtil.isInEnumBlock(ast)
254                || ScopeUtil.isInRecordBlock(ast);
255    }
256
257}

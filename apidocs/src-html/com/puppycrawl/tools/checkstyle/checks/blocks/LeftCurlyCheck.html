031/**
032 * <p>
033 * Checks for the placement of left curly braces (<code>'{'</code>) for code blocks.
034 * </p>
035 * <ul>
036 * <li>
037 * Property {@code option} - Specify the policy on placement of a left curly brace
038 * (<code>'{'</code>).
039 * Type is {@code com.puppycrawl.tools.checkstyle.checks.blocks.LeftCurlyOption}.
040 * Default value is {@code eol}.
041 * </li>
042 * <li>
043 * Property {@code ignoreEnums} - Allow to ignore enums when left curly brace policy is EOL.
044 * Type is {@code boolean}.
045 * Default value is {@code true}.
046 * </li>
047 * <li>
048 * Property {@code tokens} - tokens to check
049 * Type is {@code java.lang.String[]}.
050 * Validation type is {@code tokenSet}.
051 * Default value is:
052 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_DEF">
053 * ANNOTATION_DEF</a>,
054 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
055 * CLASS_DEF</a>,
056 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
057 * CTOR_DEF</a>,
058 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_CONSTANT_DEF">
059 * ENUM_CONSTANT_DEF</a>,
060 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
061 * ENUM_DEF</a>,
062 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
063 * INTERFACE_DEF</a>,
064 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LAMBDA">
065 * LAMBDA</a>,
066 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_CASE">
067 * LITERAL_CASE</a>,
068 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_CATCH">
069 * LITERAL_CATCH</a>,
070 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_DEFAULT">
071 * LITERAL_DEFAULT</a>,
072 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_DO">
073 * LITERAL_DO</a>,
074 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_ELSE">
075 * LITERAL_ELSE</a>,
076 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_FINALLY">
077 * LITERAL_FINALLY</a>,
078 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_FOR">
079 * LITERAL_FOR</a>,
080 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_IF">
081 * LITERAL_IF</a>,
082 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_SWITCH">
083 * LITERAL_SWITCH</a>,
084 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_SYNCHRONIZED">
085 * LITERAL_SYNCHRONIZED</a>,
086 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_TRY">
087 * LITERAL_TRY</a>,
088 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#LITERAL_WHILE">
089 * LITERAL_WHILE</a>,
090 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
091 * METHOD_DEF</a>,
092 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#OBJBLOCK">
093 * OBJBLOCK</a>,
094 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#STATIC_INIT">
095 * STATIC_INIT</a>,
096 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#RECORD_DEF">
097 * RECORD_DEF</a>,
098 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#COMPACT_CTOR_DEF">
099 * COMPACT_CTOR_DEF</a>.
100 * </li>
101 * </ul>
102 * <p>
103 * To configure the check:
104 * </p>
105 * <pre>
106 * &lt;module name="LeftCurly"/&gt;
107 * </pre>
108 * <pre>
109 * class Test
110 * { // Violation - '{' should be on the previous line
111 *   private interface TestInterface
112 *   { // Violation - '{' should be on the previous line
113 *   }
114 *
115 *   private
116 *   class
117 *   MyClass { // OK
118 *   }
119 *
120 *   enum Colors {RED, // OK
121 *     BLUE,
122 *     GREEN;
123 *   }
124 * }
125 * </pre>
126 * <p>
127 * To configure the check to apply the {@code nl} policy to type blocks:
128 * </p>
129 * <pre>
130 * &lt;module name=&quot;LeftCurly&quot;&gt;
131 *   &lt;property name=&quot;option&quot; value=&quot;nl&quot;/&gt;
132 *   &lt;property name=&quot;tokens&quot; value=&quot;CLASS_DEF,INTERFACE_DEF&quot;/&gt;
133 * &lt;/module&gt;
134 * </pre>
135 * <pre>
136 * class Test
137 * { // OK
138 *   private interface TestInterface
139 *   { // OK
140 *   }
141 *
142 *   private
143 *   class
144 *   MyClass { // Violation - '{' should be on a new line
145 *   }
146 *
147 *   enum Colors {RED, // OK
148 *     BLUE,
149 *     GREEN;
150 *   }
151 * }
152 * </pre>
153 * <p>
154 * To configure the check to apply the {@code nlow} policy to type blocks:
155 * </p>
156 * <pre>
157 * &lt;module name=&quot;LeftCurly&quot;&gt;
158 *   &lt;property name=&quot;option&quot; value=&quot;nlow&quot;/&gt;
159 *   &lt;property name=&quot;tokens&quot; value=&quot;CLASS_DEF,INTERFACE_DEF&quot;/&gt;
160 * &lt;/module&gt;
161 * </pre>
162 * <pre>
163 * class Test
164 * { // Violation - '{' should be on the previous line
165 *   private interface TestInterface { // OK
166 *   }
167 *
168 *   private
169 *   class
170 *   MyClass { // Violation - '{' should be on a new line
171 *   }
172 *
173 *   enum Colors {RED, // OK
174 *     BLUE,
175 *     GREEN;
176 *   }
177 * }
178 * </pre>
179 * <p>
180 * An example of how to configure the check to validate enum definitions:
181 * </p>
182 * <pre>
183 * &lt;module name=&quot;LeftCurly&quot;&gt;
184 *   &lt;property name=&quot;ignoreEnums&quot; value=&quot;false&quot;/&gt;
185 * &lt;/module&gt;
186 * </pre>
187 * <pre>
188 * class Test
189 * { // Violation - '{' should be on the previous line
190 *   private interface TestInterface
191 *   { // Violation - '{' should be on the previous line
192 *   }
193 *
194 *   private
195 *   class
196 *   MyClass { // OK
197 *   }
198 *
199 *   enum Colors {RED, // Violation - '{' should have line break after
200 *   BLUE,
201 *   GREEN;
202 *   }
203 * }
204 * </pre>
205 * <p>
206 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
207 * </p>
208 * <p>
209 * Violation Message Keys:
210 * </p>
211 * <ul>
212 * <li>
213 * {@code line.break.after}
214 * </li>
215 * <li>
216 * {@code line.new}
217 * </li>
218 * <li>
219 * {@code line.previous}
220 * </li>
221 * </ul>
222 *
223 * @since 3.0
224 */
225@StatelessCheck
226public class LeftCurlyCheck
227    extends AbstractCheck {
228
229    /**
230     * A key is pointing to the warning message text in "messages.properties"
231     * file.
232     */
233    public static final String MSG_KEY_LINE_NEW = "line.new";
234
235    /**
236     * A key is pointing to the warning message text in "messages.properties"
237     * file.
238     */
239    public static final String MSG_KEY_LINE_PREVIOUS = "line.previous";
240
241    /**
242     * A key is pointing to the warning message text in "messages.properties"
243     * file.
244     */
245    public static final String MSG_KEY_LINE_BREAK_AFTER = "line.break.after";
246
247    /** Open curly brace literal. */
248    private static final String OPEN_CURLY_BRACE = "{";
249
250    /** Allow to ignore enums when left curly brace policy is EOL. */
251    private boolean ignoreEnums = true;
252
253    /**
254     * Specify the policy on placement of a left curly brace (<code>'{'</code>).
255     * */
256    private LeftCurlyOption option = LeftCurlyOption.EOL;
257
258    /**
259     * Setter to specify the policy on placement of a left curly brace (<code>'{'</code>).
260     *
261     * @param optionStr string to decode option from
262     * @throws IllegalArgumentException if unable to decode
263     * @since 3.0
264     */
265    public void setOption(String optionStr) {
266        option = LeftCurlyOption.valueOf(optionStr.trim().toUpperCase(Locale.ENGLISH));
267    }
268
269    /**
270     * Setter to allow to ignore enums when left curly brace policy is EOL.
271     *
272     * @param ignoreEnums check's option for ignoring enums.
273     * @since 6.9
274     */
275    public void setIgnoreEnums(boolean ignoreEnums) {
276        this.ignoreEnums = ignoreEnums;
277    }

319    /**
320     * Visits token.
321     *
322     * @param ast the token to process
323     * @noinspection SwitchStatementWithTooManyBranches
324     * @noinspectionreason SwitchStatementWithTooManyBranches - we cannot reduce
325     *      the number of branches in this switch statement, since many tokens
326     *      require specific methods to find the first left curly
327     */
328    @Override
329    public void visitToken(DetailAST ast) {
330        final DetailAST startToken;
331        final DetailAST brace;
332
333        switch (ast.getType()) {
334            case TokenTypes.CTOR_DEF:
335            case TokenTypes.METHOD_DEF:
336            case TokenTypes.COMPACT_CTOR_DEF:
337                startToken = skipModifierAnnotations(ast);
338                brace = ast.findFirstToken(TokenTypes.SLIST);
339                break;
340            case TokenTypes.INTERFACE_DEF:
341            case TokenTypes.CLASS_DEF:
342            case TokenTypes.ANNOTATION_DEF:
343            case TokenTypes.ENUM_DEF:
344            case TokenTypes.ENUM_CONSTANT_DEF:
345            case TokenTypes.RECORD_DEF:
346                startToken = skipModifierAnnotations(ast);
347                brace = ast.findFirstToken(TokenTypes.OBJBLOCK);
348                break;
349            case TokenTypes.LITERAL_WHILE:
350            case TokenTypes.LITERAL_CATCH:
351            case TokenTypes.LITERAL_SYNCHRONIZED:
352            case TokenTypes.LITERAL_FOR:
353            case TokenTypes.LITERAL_TRY:
354            case TokenTypes.LITERAL_FINALLY:
355            case TokenTypes.LITERAL_DO:
356            case TokenTypes.LITERAL_IF:
357            case TokenTypes.STATIC_INIT:
358            case TokenTypes.LAMBDA:
359                startToken = ast;
360                brace = ast.findFirstToken(TokenTypes.SLIST);
361                break;
362            case TokenTypes.LITERAL_ELSE:
363                startToken = ast;
364                brace = getBraceAsFirstChild(ast);
365                break;
366            case TokenTypes.LITERAL_CASE:
367            case TokenTypes.LITERAL_DEFAULT:
368                startToken = ast;
369                brace = getBraceFromSwitchMember(ast);
370                break;
371            default:
372                // ATTENTION! We have default here, but we expect case TokenTypes.METHOD_DEF,
373                // TokenTypes.LITERAL_FOR, TokenTypes.LITERAL_WHILE, TokenTypes.LITERAL_DO only.
374                // It has been done to improve coverage to 100%. I couldn't replace it with
375                // if-else-if block because code was ugly and didn't pass pmd check.
376
377                startToken = ast;
378                brace = ast.findFirstToken(TokenTypes.LCURLY);
379                break;
380        }
381
382        if (brace != null) {
383            verifyBrace(brace, startToken);
384        }
385    }
386
387    /**
388     * Gets the brace of a switch statement/ expression member.
389     *
390     * @param ast {@code DetailAST}.
391     * @return {@code DetailAST} if the first child is {@code TokenTypes.SLIST},
392     *     {@code null} otherwise.
393     */
394    private static DetailAST getBraceFromSwitchMember(DetailAST ast) {
395        final DetailAST brace;
396        final DetailAST parent = ast.getParent();
397        if (parent.getType() == TokenTypes.SWITCH_RULE) {
398            brace = parent.findFirstToken(TokenTypes.SLIST);
399        }
400        else {
401            brace = getBraceAsFirstChild(ast.getNextSibling());
402        }
403        return brace;
404    }
405
406    /**
407     * Gets a SLIST if it is the first child of the AST.
408     *
409     * @param ast {@code DetailAST}.
410     * @return {@code DetailAST} if the first child is {@code TokenTypes.SLIST},
411     *     {@code null} otherwise.
412     */
413    private static DetailAST getBraceAsFirstChild(DetailAST ast) {
414        DetailAST brace = null;
415        if (ast != null) {
416            final DetailAST candidate = ast.getFirstChild();
417            if (candidate != null && candidate.getType() == TokenTypes.SLIST) {
418                brace = candidate;
419            }
420        }
421        return brace;
422    }
423
424    /**
425     * Skip all {@code TokenTypes.ANNOTATION}s to the first non-annotation.
426     *
427     * @param ast {@code DetailAST}.
428     * @return {@code DetailAST}.
429     */
430    private static DetailAST skipModifierAnnotations(DetailAST ast) {
431        DetailAST resultNode = ast;
432        final DetailAST modifiers = ast.findFirstToken(TokenTypes.MODIFIERS);
433
434        if (modifiers != null) {
435            final DetailAST lastAnnotation = findLastAnnotation(modifiers);
436
437            if (lastAnnotation != null) {
438                if (lastAnnotation.getNextSibling() == null) {
439                    resultNode = modifiers.getNextSibling();
440                }
441                else {
442                    resultNode = lastAnnotation.getNextSibling();
443                }
444            }
445        }
446        return resultNode;
447    }
448
449    /**
450     * Find the last token of type {@code TokenTypes.ANNOTATION}
451     * under the given set of modifiers.
452     *
453     * @param modifiers {@code DetailAST}.
454     * @return {@code DetailAST} or null if there are no annotations.
455     */
456    private static DetailAST findLastAnnotation(DetailAST modifiers) {
457        DetailAST annotation = modifiers.findFirstToken(TokenTypes.ANNOTATION);
458        while (annotation != null && annotation.getNextSibling() != null
459               && annotation.getNextSibling().getType() == TokenTypes.ANNOTATION) {
460            annotation = annotation.getNextSibling();
461        }
462        return annotation;
463    }
464
465    /**
466     * Verifies that a specified left curly brace is placed correctly
467     * according to policy.
468     *
469     * @param brace token for left curly brace
470     * @param startToken token for start of expression
471     */
472    private void verifyBrace(final DetailAST brace,
473                             final DetailAST startToken) {
474        final String braceLine = getLine(brace.getLineNo() - 1);
475
476        // Check for being told to ignore, or have '{}' which is a special case
477        if (braceLine.length() <= brace.getColumnNo() + 1
478                || braceLine.charAt(brace.getColumnNo() + 1) != '}') {
479            if (option == LeftCurlyOption.NL) {
480                if (!CommonUtil.hasWhitespaceBefore(brace.getColumnNo(), braceLine)) {
481                    log(brace, MSG_KEY_LINE_NEW, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
482                }
483            }
484            else if (option == LeftCurlyOption.EOL) {
485                validateEol(brace, braceLine);
486            }
487            else if (!TokenUtil.areOnSameLine(startToken, brace)) {
488                validateNewLinePosition(brace, startToken, braceLine);
489            }
490        }
491    }
492
493    /**
494     * Validate EOL case.
495     *
496     * @param brace brace AST
497     * @param braceLine line content
498     */
499    private void validateEol(DetailAST brace, String braceLine) {
500        if (CommonUtil.hasWhitespaceBefore(brace.getColumnNo(), braceLine)) {
501            log(brace, MSG_KEY_LINE_PREVIOUS, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
502        }
503        if (!hasLineBreakAfter(brace)) {
504            log(brace, MSG_KEY_LINE_BREAK_AFTER, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
505        }
506    }
507
508    /**
509     * Validate token on new Line position.
510     *
511     * @param brace brace AST
512     * @param startToken start Token
513     * @param braceLine content of line with Brace
514     */
515    private void validateNewLinePosition(DetailAST brace, DetailAST startToken, String braceLine) {
516        // not on the same line
517        if (startToken.getLineNo() + 1 == brace.getLineNo()) {
518            if (CommonUtil.hasWhitespaceBefore(brace.getColumnNo(), braceLine)) {
519                log(brace, MSG_KEY_LINE_PREVIOUS, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
520            }
521            else {
522                log(brace, MSG_KEY_LINE_NEW, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
523            }
524        }
525        else if (!CommonUtil.hasWhitespaceBefore(brace.getColumnNo(), braceLine)) {
526            log(brace, MSG_KEY_LINE_NEW, OPEN_CURLY_BRACE, brace.getColumnNo() + 1);
527        }
528    }
529
530    /**
531     * Checks if left curly has line break after.
532     *
533     * @param leftCurly
534     *        Left curly token.
535     * @return
536     *        True, left curly has line break after.
537     */
538    private boolean hasLineBreakAfter(DetailAST leftCurly) {
539        DetailAST nextToken = null;
540        if (leftCurly.getType() == TokenTypes.SLIST) {
541            nextToken = leftCurly.getFirstChild();
542        }
543        else {
544            if (!ignoreEnums
545                    && leftCurly.getParent().getParent().getType() == TokenTypes.ENUM_DEF) {
546                nextToken = leftCurly.getNextSibling();
547            }
548        }
549        return nextToken == null
550                || nextToken.getType() == TokenTypes.RCURLY
551                || !TokenUtil.areOnSameLine(leftCurly, nextToken);
552    }

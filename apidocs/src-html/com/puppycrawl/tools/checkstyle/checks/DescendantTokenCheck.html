031/**
032 * <p>
033 * Checks for restricted tokens beneath other tokens.
034 * </p>
035 * <p>
036 * WARNING: This is a very powerful and flexible check, but, at the same time,
037 * it is low-level and very implementation-dependent because its results depend
038 * on the grammar we use to build abstract syntax trees. Thus we recommend using
039 * other checks when they provide the desired functionality. Essentially, this
040 * check just works on the level of an abstract syntax tree and knows nothing
041 * about language structures.
042 * </p>
043 * <ul>
044 * <li>
045 * Property {@code limitedTokens} - Specify set of tokens with limited occurrences as descendants.
046 * Type is {@code java.lang.String[]}.
047 * Validation type is {@code tokenSet}.
048 * Default value is {@code ""}.
049 * </li>
050 * <li>
051 * Property {@code minimumDepth} - Specify the minimum depth for descendant counts.
052 * Type is {@code int}.
053 * Default value is {@code 0}.
054 * </li>
055 * <li>
056 * Property {@code maximumDepth} - Specify the maximum depth for descendant counts.
057 * Type is {@code int}.
058 * Default value is {@code 2147483647}.
059 * </li>
060 * <li>
061 * Property {@code minimumNumber} - Specify a minimum count for descendants.
062 * Type is {@code int}.
063 * Default value is {@code 0}.
064 * </li>
065 * <li>
066 * Property {@code maximumNumber} - Specify a maximum count for descendants.
067 * Type is {@code int}.
068 * Default value is {@code 2147483647}.
069 * </li>
070 * <li>
071 * Property {@code sumTokenCounts} - Control whether the number of tokens found
072 * should be calculated from the sum of the individual token counts.
073 * Type is {@code boolean}.
074 * Default value is {@code false}.
075 * </li>
076 * <li>
077 * Property {@code minimumMessage} - Define the violation message
078 * when the minimum count is not reached.
079 * Type is {@code java.lang.String}.
080 * Default value is {@code null}.
081 * </li>
082 * <li>
083 * Property {@code maximumMessage} - Define the violation message
084 * when the maximum count is exceeded.
085 * Type is {@code java.lang.String}.
086 * Default value is {@code null}.
087 * </li>
088 * </ul>
089 * <p>
090 * To configure the check to produce a violation on a switch statement with no default case:
091 * </p>
092 * <pre>
093 * &lt;module name=&quot;DescendantToken&quot;&gt;
094 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
095 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
096 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_DEFAULT&quot;/&gt;
097 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
098 * &lt;/module&gt;
099 * </pre>
100 * <p>
101 * To configure the check to produce a violation on a condition in {@code for}
102 * which performs no check:
103 * </p>
104 * <pre>
105 * &lt;module name=&quot;DescendantToken&quot;&gt;
106 *   &lt;property name=&quot;tokens&quot; value=&quot;FOR_CONDITION&quot;/&gt;
107 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
108 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
109 * &lt;/module&gt;
110 * </pre>
111 * <p>
112 * To configure the check to produce a violation on comparing {@code this} with
113 * {@code null}(i.e. {@code this == null} and {@code this != null}):
114 * </p>
115 * <pre>
116 * &lt;module name=&quot;DescendantToken&quot;&gt;
117 *   &lt;property name=&quot;tokens&quot; value=&quot;EQUAL,NOT_EQUAL&quot;/&gt;
118 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_THIS,LITERAL_NULL&quot;/&gt;
119 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
120 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;1&quot;/&gt;
121 *   &lt;property name=&quot;sumTokenCounts&quot; value=&quot;true&quot;/&gt;
122 * &lt;/module&gt;
123 * </pre>
124 * <p>
125 * To configure the check to produce a violation on a {@code String} literal equality check:
126 * </p>
127 * <pre>
128 * &lt;module name=&quot;DescendantToken&quot;&gt;
129 *   &lt;property name=&quot;tokens&quot; value=&quot;EQUAL,NOT_EQUAL&quot;/&gt;
130 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;STRING_LITERAL&quot;/&gt;
131 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
132 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;1&quot;/&gt;
133 * &lt;/module&gt;
134 * </pre>
135 * <p>
136 * To configure the check to produce a violation on an assert statement that may
137 * have side effects (formatted for browser display):
138 * </p>
139 * <pre>
140 * &lt;module name=&quot;DescendantToken&quot;&gt;
141 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_ASSERT&quot;/&gt;
142 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;ASSIGN,DEC,INC,POST_DEC,
143 *     POST_INC,PLUS_ASSIGN,MINUS_ASSIGN,STAR_ASSIGN,DIV_ASSIGN,MOD_ASSIGN,
144 *     BSR_ASSIGN,SR_ASSIGN,SL_ASSIGN,BAND_ASSIGN,BXOR_ASSIGN,BOR_ASSIGN,
145 *     METHOD_CALL&quot;/&gt;
146 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
147 * &lt;/module&gt;
148 * </pre>
149 * <p>
150 * To configure the check to produce a violation on an initializer in {@code for}
151 * performs no setup (where a {@code while} statement could be used instead):
152 * </p>
153 * <pre>
154 * &lt;module name=&quot;DescendantToken&quot;&gt;
155 *   &lt;property name=&quot;tokens&quot; value=&quot;FOR_INIT&quot;/&gt;
156 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
157 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
158 * &lt;/module&gt;
159 * </pre>
160 * <p>
161 * To configure the check to produce a violation on a switch that is nested in another switch:
162 * </p>
163 * <pre>
164 * &lt;module name=&quot;DescendantToken&quot;&gt;
165 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
166 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
167 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
168 *   &lt;property name=&quot;minimumDepth&quot; value=&quot;1&quot;/&gt;
169 * &lt;/module&gt;
170 * </pre>
171 * <p>
172 * To configure the check to produce a violation on a return statement from
173 * within a catch or finally block:
174 * </p>
175 * <pre>
176 * &lt;module name=&quot;DescendantToken&quot;&gt;
177 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_FINALLY,LITERAL_CATCH&quot;/&gt;
178 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_RETURN&quot;/&gt;
179 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
180 * &lt;/module&gt;
181 * </pre>
182 * <p>
183 * To configure the check to produce a violation on a try statement within a catch or finally block:
184 * </p>
185 * <pre>
186 * &lt;module name=&quot;DescendantToken&quot;&gt;
187 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_CATCH,LITERAL_FINALLY&quot;/&gt;
188 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_TRY&quot;/&gt;
189 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
190 * &lt;/module&gt;
191 * </pre>
192 * <p>
193 * To configure the check to produce a violation on a switch with too many cases:
194 * </p>
195 * <pre>
196 * &lt;module name=&quot;DescendantToken&quot;&gt;
197 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
198 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_CASE&quot;/&gt;
199 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
200 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;10&quot;/&gt;
201 * &lt;/module&gt;
202 * </pre>
203 * <p>
204 * To configure the check to produce a violation on a method with too many local variables:
205 * </p>
206 * <pre>
207 * &lt;module name=&quot;DescendantToken&quot;&gt;
208 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
209 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;VARIABLE_DEF&quot;/&gt;
210 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
211 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;10&quot;/&gt;
212 * &lt;/module&gt;
213 * </pre>
214 * <p>
215 * To configure the check to produce a violation on a method with too many returns:
216 * </p>
217 * <pre>
218 * &lt;module name=&quot;DescendantToken&quot;&gt;
219 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
220 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_RETURN&quot;/&gt;
221 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;3&quot;/&gt;
222 * &lt;/module&gt;
223 * </pre>
224 * <p>
225 * To configure the check to produce a violation on an interface with too many fields:
226 * </p>
227 * <pre>
228 * &lt;module name=&quot;DescendantToken&quot;&gt;
229 *   &lt;property name=&quot;tokens&quot; value=&quot;INTERFACE_DEF&quot;/&gt;
230 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;VARIABLE_DEF&quot;/&gt;
231 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
232 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
233 * &lt;/module&gt;
234 * </pre>
235 * <p>
236 * To configure the check to produce a violation on a method which throws too many exceptions:
237 * </p>
238 * <pre>
239 * &lt;module name=&quot;DescendantToken&quot;&gt;
240 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_THROWS&quot;/&gt;
241 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;IDENT&quot;/&gt;
242 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
243 * &lt;/module&gt;
244 * </pre>
245 * <p>
246 * To configure the check to produce a violation on a method with too many expressions:
247 * </p>
248 * <pre>
249 * &lt;module name=&quot;DescendantToken&quot;&gt;
250 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
251 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
252 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;200&quot;/&gt;
253 * &lt;/module&gt;
254 * </pre>
255 * <p>
256 * To configure the check to produce a violation on an empty statement:
257 * </p>
258 * <pre>
259 * &lt;module name=&quot;DescendantToken&quot;&gt;
260 *   &lt;property name=&quot;tokens&quot; value=&quot;EMPTY_STAT&quot;/&gt;
261 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EMPTY_STAT&quot;/&gt;
262 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
263 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;0&quot;/&gt;
264 *   &lt;property name=&quot;maximumMessage&quot;
265 *     value=&quot;Empty statement is not allowed.&quot;/&gt;
266 * &lt;/module&gt;
267 * </pre>
268 * <p>
269 * To configure the check to produce a violation on a class with too many fields:
270 * </p>
271 * <pre>
272 * &lt;module name=&quot;DescendantToken&quot;&gt;
273 *   &lt;property name=&quot;tokens&quot; value=&quot;CLASS_DEF&quot;/&gt;
274 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;VARIABLE_DEF&quot;/&gt;

278 * </pre>
279 * <p>

284 * </p>

302@FileStatefulCheck
303public class DescendantTokenCheck extends AbstractCheck {
304
305    /**
306     * A key is pointing to the warning message text in "messages.properties"
307     * file.
308     */
309    public static final String MSG_KEY_MIN = "descendant.token.min";
310
311    /**
312     * A key is pointing to the warning message text in "messages.properties"
313     * file.
314     */
315    public static final String MSG_KEY_MAX = "descendant.token.max";
316
317    /**
318     * A key is pointing to the warning message text in "messages.properties"
319     * file.
320     */
321    public static final String MSG_KEY_SUM_MIN = "descendant.token.sum.min";
322
323    /**
324     * A key is pointing to the warning message text in "messages.properties"
325     * file.
326     */
327    public static final String MSG_KEY_SUM_MAX = "descendant.token.sum.max";
328
329    /** Specify the minimum depth for descendant counts. */
330    private int minimumDepth;
331    /** Specify the maximum depth for descendant counts. */
332    private int maximumDepth = Integer.MAX_VALUE;
333    /** Specify a minimum count for descendants. */
334    private int minimumNumber;
335    /** Specify a maximum count for descendants. */
336    private int maximumNumber = Integer.MAX_VALUE;
337    /**
338     * Control whether the number of tokens found should be calculated from
339     * the sum of the individual token counts.
340     */
341    private boolean sumTokenCounts;
342    /** Specify set of tokens with limited occurrences as descendants. */
343    private int[] limitedTokens = CommonUtil.EMPTY_INT_ARRAY;
344    /** Define the violation message when the minimum count is not reached. */
345    private String minimumMessage;
346    /** Define the violation message when the maximum count is exceeded. */
347    private String maximumMessage;
348
349    /**
350     * Counts of descendant tokens.
351     * Indexed by (token ID - 1) for performance.
352     */
353    private int[] counts = CommonUtil.EMPTY_INT_ARRAY;
354
355    @Override
356    public int[] getDefaultTokens() {
357        return getRequiredTokens();
358    }
359
360    @Override
361    public int[] getRequiredTokens() {
362        return CommonUtil.EMPTY_INT_ARRAY;
363    }
364
365    @Override
366    public void visitToken(DetailAST ast) {
367        // reset counts
368        Arrays.fill(counts, 0);
369        countTokens(ast, 0);
370
371        if (sumTokenCounts) {
372            logAsTotal(ast);
373        }
374        else {
375            logAsSeparated(ast);
376        }
377    }
378
379    /**
380     * Log violations for each Token.
381     *
382     * @param ast token
383     */
384    private void logAsSeparated(DetailAST ast) {
385        // name of this token
386        final String name = TokenUtil.getTokenName(ast.getType());
387
388        for (int element : limitedTokens) {
389            final int tokenCount = counts[element - 1];
390            if (tokenCount < minimumNumber) {
391                final String descendantName = TokenUtil.getTokenName(element);
392
393                if (minimumMessage == null) {
394                    minimumMessage = MSG_KEY_MIN;
395                }
396                log(ast,
397                        minimumMessage,
398                        String.valueOf(tokenCount),
399                        String.valueOf(minimumNumber),
400                        name,
401                        descendantName);
402            }
403            if (tokenCount > maximumNumber) {
404                final String descendantName = TokenUtil.getTokenName(element);
405
406                if (maximumMessage == null) {
407                    maximumMessage = MSG_KEY_MAX;
408                }
409                log(ast,
410                        maximumMessage,
411                        String.valueOf(tokenCount),
412                        String.valueOf(maximumNumber),
413                        name,
414                        descendantName);
415            }
416        }
417    }
418
419    /**
420     * Log validation as one violation.
421     *
422     * @param ast current token
423     */
424    private void logAsTotal(DetailAST ast) {
425        // name of this token
426        final String name = TokenUtil.getTokenName(ast.getType());
427
428        int total = 0;
429        for (int element : limitedTokens) {
430            total += counts[element - 1];
431        }
432        if (total < minimumNumber) {
433            if (minimumMessage == null) {
434                minimumMessage = MSG_KEY_SUM_MIN;
435            }
436            log(ast,
437                    minimumMessage,
438                    String.valueOf(total),
439                    String.valueOf(minimumNumber), name);
440        }
441        if (total > maximumNumber) {
442            if (maximumMessage == null) {
443                maximumMessage = MSG_KEY_SUM_MAX;
444            }
445            log(ast,
446                    maximumMessage,
447                    String.valueOf(total),
448                    String.valueOf(maximumNumber), name);
449        }
450    }
451
452    /**
453     * Counts the number of occurrences of descendant tokens.
454     *
455     * @param ast the root token for descendants.
456     * @param depth the maximum depth of the counted descendants.
457     */
458    private void countTokens(DetailAST ast, int depth) {
459        if (depth <= maximumDepth) {
460            // update count
461            if (depth >= minimumDepth) {
462                final int type = ast.getType();
463                if (type <= counts.length) {
464                    counts[type - 1]++;
465                }
466            }
467            DetailAST child = ast.getFirstChild();
468            final int nextDepth = depth + 1;
469            while (child != null) {
470                countTokens(child, nextDepth);
471                child = child.getNextSibling();
472            }
473        }
474    }
475
476    @Override
477    public int[] getAcceptableTokens() {
478        // Any tokens set by property 'tokens' are acceptable
479        final Set<String> tokenNames = getTokenNames();
480        final int[] result = new int[tokenNames.size()];
481        int index = 0;
482        for (String name : tokenNames) {
483            result[index] = TokenUtil.getTokenId(name);
484            index++;
485        }
486        return result;
487    }
488
489    /**
490     * Setter to specify set of tokens with limited occurrences as descendants.
491     *
492     * @param limitedTokensParam - list of tokens to ignore.
493     */
494    public void setLimitedTokens(String... limitedTokensParam) {
495        limitedTokens = new int[limitedTokensParam.length];
496
497        int maxToken = 0;
498        for (int i = 0; i < limitedTokensParam.length; i++) {
499            limitedTokens[i] = TokenUtil.getTokenId(limitedTokensParam[i]);
500            if (limitedTokens[i] >= maxToken + 1) {
501                maxToken = limitedTokens[i];
502            }
503        }
504        counts = new int[maxToken];
505    }

032/**
033 * <p>
034 * Checks for restricted tokens beneath other tokens.
035 * </p>
036 * <p>
037 * WARNING: This is a very powerful and flexible check, but, at the same time,
038 * it is low-level and very implementation-dependent because its results depend
039 * on the grammar we use to build abstract syntax trees. Thus, we recommend using
040 * other checks when they provide the desired functionality. Essentially, this
041 * check just works on the level of an abstract syntax tree and knows nothing
042 * about language structures.
043 * </p>
044 * <ul>
045 * <li>
046 * Property {@code limitedTokens} - Specify set of tokens with limited occurrences as descendants.
047 * Type is {@code java.lang.String[]}.
048 * Validation type is {@code tokenTypesSet}.
049 * Default value is {@code ""}.
050 * </li>
051 * <li>
052 * Property {@code minimumDepth} - Specify the minimum depth for descendant counts.
053 * Type is {@code int}.
054 * Default value is {@code 0}.
055 * </li>
056 * <li>
057 * Property {@code maximumDepth} - Specify the maximum depth for descendant counts.
058 * Type is {@code int}.
059 * Default value is {@code 2147483647}.
060 * </li>
061 * <li>
062 * Property {@code minimumNumber} - Specify a minimum count for descendants.
063 * Type is {@code int}.
064 * Default value is {@code 0}.
065 * </li>
066 * <li>
067 * Property {@code maximumNumber} - Specify a maximum count for descendants.
068 * Type is {@code int}.
069 * Default value is {@code 2147483647}.
070 * </li>
071 * <li>
072 * Property {@code sumTokenCounts} - Control whether the number of tokens found
073 * should be calculated from the sum of the individual token counts.
074 * Type is {@code boolean}.
075 * Default value is {@code false}.
076 * </li>
077 * <li>
078 * Property {@code minimumMessage} - Define the violation message
079 * when the minimum count is not reached.
080 * Type is {@code java.lang.String}.
081 * Default value is {@code null}.
082 * </li>
083 * <li>
084 * Property {@code maximumMessage} - Define the violation message
085 * when the maximum count is exceeded.
086 * Type is {@code java.lang.String}.
087 * Default value is {@code null}.
088 * </li>
089 * <li>
090 * Property {@code tokens} - tokens to check
091 * Type is {@code anyTokenTypesSet}.
092 * Default value is {@code ""}.
093 * </li>
094 * </ul>
095 * <p>
096 * To configure the check to produce a violation on a switch statement with no default case:
097 * </p>
098 * <pre>
099 * &lt;module name=&quot;DescendantToken&quot;&gt;
100 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
101 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
102 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_DEFAULT&quot;/&gt;
103 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
104 * &lt;/module&gt;
105 * </pre>
106 * <p>
107 * Example:
108 * </p>
109 * <pre>
110 * class Test {
111 *  public static void main(String[] args) {
112 *    int x = 1;
113 *    switch (x) { // ok
114 *      case 1:
115 *        System.out.println("hi");
116 *        break;
117 *      default:
118 *        System.out.println("Default");
119 *        break;
120 *    }
121 *
122 *    int y = 1;
123 *    switch (y) { // violation
124 *      case 1:
125 *        System.out.println("hi");
126 *        break;
127 *    }
128 *  }
129 * }
130 * </pre>
131 * <p>
132 * To configure the check to produce a violation on a switch with too many cases:
133 * </p>
134 * <pre>
135 * &lt;module name=&quot;DescendantToken&quot;&gt;
136 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
137 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_CASE&quot;/&gt;
138 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
139 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
140 * &lt;/module&gt;
141 * </pre>
142 * <p>
143 * Example:
144 * </p>
145 * <pre>
146 * class Test {
147 *   public void foo() {
148 *     int x = 1;
149 *     switch (x) { // ok
150 *       case 1:
151 *         // Some code
152 *         break;
153 *       default:
154 *         // Some code
155 *         break;
156 *     }
157 *
158 *     switch (x) { // violation
159 *       case 1:
160 *         // Some code
161 *         break;
162 *       case 2:
163 *         // Some code
164 *         break;
165 *       default:
166 *         // Some code
167 *         break;
168 *     }
169 *   }
170 * }
171 * </pre>
172 * <p>
173 * To configure the check to produce a violation on a switch that is nested in another switch:
174 * </p>
175 * <pre>
176 * &lt;module name=&quot;DescendantToken&quot;&gt;
177 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
178 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_SWITCH&quot;/&gt;
179 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
180 *   &lt;property name=&quot;minimumDepth&quot; value=&quot;1&quot;/&gt;
181 * &lt;/module&gt;
182 * </pre>
183 * <p>
184 * Example:
185 * </p>
186 * <pre>
187 * class Test {
188 *   public void foo() {
189 *     int x = 1;
190 *     int y = 2;
191 *     switch (x) { // ok
192 *       case 1:
193 *         System.out.println("xyz");
194 *         break;
195 *     }
196 *     switch (y) { // violation
197 *       case 1:
198 *         switch (y) {
199 *           case 2:
200 *             System.out.println("xyz");
201 *             break;
202 *         }
203 *         break;
204 *     }
205 *   }
206 * }
207 * </pre>
208 * <p>
209 * To configure the check to produce a violation on a condition in {@code for}
210 * which performs no check:
211 * </p>
212 * <pre>
213 * &lt;module name=&quot;DescendantToken&quot;&gt;
214 *   &lt;property name=&quot;tokens&quot; value=&quot;FOR_CONDITION&quot;/&gt;
215 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
216 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
217 * &lt;/module&gt;
218 * </pre>
219 * <p>
220 * Example:
221 * </p>
222 * <pre>
223 * class Test {
224 *   public static void main(String[] args) {
225 *     for (int i = 0; i != 10; i++) { // ok
226 *       System.out.println(i);
227 *     }
228 *     int k = 0;
229 *     for (; ; ) { // violation
230 *       System.out.println(k);
231 *     }
232 *   }
233 * }
234 * </pre>
235 * <p>
236 * To configure the check to produce a violation on an initializer in {@code for}
237 * performs no setup (where a {@code while} statement could be used instead):
238 * </p>
239 * <pre>
240 * &lt;module name=&quot;DescendantToken&quot;&gt;
241 *   &lt;property name=&quot;tokens&quot; value=&quot;FOR_INIT&quot;/&gt;
242 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
243 *   &lt;property name=&quot;minimumNumber&quot; value=&quot;1&quot;/&gt;
244 * &lt;/module&gt;
245 * </pre>
246 * <p>
247 * Example:
248 * </p>
249 * <pre>
250 * class Test {
251 *   public void foo() {
252 *     int[] array = new int[] {1, 2, 3, 4, 5};
253 *
254 *     for (int i = 0; i != array.length; i++) { // ok
255 *       System.out.println(i);
256 *     }
257 *
258 *     int j = 0;
259 *     for (; j != array.length;) { // violation
260 *       System.out.println(j);
261 *       j++;
262 *     }
263 *   }
264 * }
265 * </pre>
266 * <p>
267 * To configure the check to produce a violation on a return statement from
268 * within a catch or finally block:
269 * </p>
270 * <pre>
271 * &lt;module name=&quot;DescendantToken&quot;&gt;
272 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_FINALLY,LITERAL_CATCH&quot;/&gt;
273 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_RETURN&quot;/&gt;
274 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
275 * &lt;/module&gt;
276 * </pre>
277 * <p>
278 * Example:
279 * </p>
280 * <pre>
281 * class Test {
282 *   public void foo() {
283 *     try {
284 *       // Some code
285 *     } catch (Exception e) {
286 *         System.out.println("xyz");
287 *         return; // violation
288 *     } finally {
289 *         System.out.println("xyz");
290 *     }
291 *   }
292 * }
293 * </pre>
294 * <p>
295 * To configure the check to produce a violation on a try statement within a catch or finally block:
296 * </p>
297 * <pre>
298 * &lt;module name=&quot;DescendantToken&quot;&gt;
299 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_CATCH,LITERAL_FINALLY&quot;/&gt;
300 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_TRY&quot;/&gt;
301 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
302 * &lt;/module&gt;
303 * </pre>
304 * <p>
305 * Example:
306 * </p>
307 * <pre>
308 * class Test {
309 *   public void foo() {
310 *     try {
311 *         // Some code
312 *     } catch (Exception e) { // ok
313 *         System.out.println("xyz");
314 *         return;
315 *     } finally { // ok
316 *         System.out.println("xyz");
317 *     }
318 *     try {
319 *         // Some code
320 *     } catch (Exception e) {
321 *         try { // violation
322 *           // Some code
323 *       } catch (Exception ex) {
324 *           // handle exception
325 *       }
326 *     } finally {
327 *         try { // violation
328 *           // Some code
329 *       } catch (Exception e) {
330 *           // handle exception
331 *       }
332 *     }
333 *   }
334 * }
335 * </pre>
336 * <p>
337 * To configure the check to produce a violation on a method with too many local variables:
338 * </p>
339 * <pre>
340 * &lt;module name=&quot;DescendantToken&quot;&gt;
341 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
342 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;VARIABLE_DEF&quot;/&gt;
343 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
344 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
345 * &lt;/module&gt;
346 * </pre>
347 * <p>
348 * Example:
349 * </p>
350 * <pre>
351 * class Test {
352 *   public void foo() { // ok
353 *     int var1 = 1;
354 *   }
355 *   public void boo() { // violation
356 *     int var1 = 1;
357 *     int var2 = 2;
358 *   }
359 * }
360 * </pre>
361 * <p>
362 * To configure the check to produce a violation on a method with too many returns:
363 * </p>
364 * <pre>
365 * &lt;module name=&quot;DescendantToken&quot;&gt;
366 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
367 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_RETURN&quot;/&gt;
368 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;2&quot;/&gt;
369 * &lt;/module&gt;
370 * </pre>
371 * <p>
372 * Example:
373 * </p>
374 * <pre>
375 * class Test {
376 *   public int foo(int x) { // ok
377 *     if (x == -1) {
378 *         return -1;
379 *     } else if (x == 0) {
380 *           return 0;
381 *     }
382 *   }
383 *   public int boo(int x) { // violation
384 *     if (x == -1) {
385 *         return -1;
386 *     } else if (x == 0) {
387 *         return 0;
388 *     } else {
389 *         return x;
390 *     }
391 *   }
392 * }
393 * </pre>
394 * <p>
395 * To configure the check to produce a violation on a method which throws too many exceptions:
396 * </p>
397 * <pre>
398 * &lt;module name=&quot;DescendantToken&quot;&gt;
399 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_THROWS&quot;/&gt;
400 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;IDENT&quot;/&gt;
401 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
402 * &lt;/module&gt;
403 * </pre>
404 * <p>
405 * Example:
406 * </p>
407 * <pre>
408 * class Foo {
409 *   public void foo() throws ArithmeticException { // ok
410 *       // ...
411 *   }
412 * }
413 * class Boo {
414 *   public void boo() throws ArithmeticException, ArithmeticException { // violation
415 *       // ...
416 *   }
417 * }
418 * </pre>
419 * <p>
420 * To configure the check to produce a violation on a method with too many expressions:
421 * </p>
422 * <pre>
423 * &lt;module name=&quot;DescendantToken&quot;&gt;
424 *   &lt;property name=&quot;tokens&quot; value=&quot;METHOD_DEF&quot;/&gt;
425 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EXPR&quot;/&gt;
426 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;2&quot;/&gt;
427 * &lt;/module&gt;
428 * </pre>
429 * <p>
430 * Example:
431 * </p>
432 * <pre>
433 * class Foo {
434 *   public void foo() { // ok
435 *       int x = 1;
436 *       int z = x + 2;
437 *   }
438 * }
439 * class Boo {
440 *   public void boo() { // violation
441 *       int x = 1;
442 *       int y = 2;
443 *       int z = x + y;
444 *   }
445 * }
446 * </pre>
447 * <p>
448 * To configure the check to produce a violation on an empty statement:
449 * </p>
450 * <pre>
451 * &lt;module name=&quot;DescendantToken&quot;&gt;
452 *   &lt;property name=&quot;tokens&quot; value=&quot;EMPTY_STAT&quot;/&gt;
453 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;EMPTY_STAT&quot;/&gt;
454 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
455 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;0&quot;/&gt;
456 *   &lt;property name=&quot;maximumMessage&quot;
457 *     value=&quot;Empty statement is not allowed.&quot;/&gt;
458 * &lt;/module&gt;
459 * </pre>
460 * <p>
461 * Example:
462 * </p>
463 * <pre>
464 * class Test {
465 *   public void foo() { // ok
466 *       System.out.println("Hello");
467 *   }
468 *   public void boo() {
469 *       ; // violation
470 *   }
471 * }
472 * </pre>
473 * <p>
474 * To configure the check to produce a violation on a class or interface with too many fields:
475 * </p>
476 * <pre>
477 * &lt;module name=&quot;DescendantToken&quot;&gt;
478 *   &lt;property name=&quot;tokens&quot; value=&quot;CLASS_DEF,INTERFACE_DEF&quot;/&gt;
479 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;VARIABLE_DEF&quot;/&gt;
480 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;2&quot;/&gt;
481 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
482 * &lt;/module&gt;
483 * </pre>
484 * <p>
485 * Example:
486 * </p>
487 * <pre>
488 * class A { // ok
489 *   private int field1;
490 *
491 *   // Some code
492 * }
493 * class B { // violation
494 *   private int field1;
495 *   private int field2;
496 *
497 *   // Some code
498 * }
499 * interface C { // ok
500 *   int FIELD_1 = 1;
501 * }
502 * interface D { // violation
503 *   int FIELD_1 = 1;
504 *   int FIELD_2 = 2;
505 * }
506 * </pre>
507 * <p>
508 * To configure the check to produce a violation on comparing {@code this} with
509 * {@code null}:
510 * </p>
511 * <pre>
512 * &lt;module name=&quot;DescendantToken&quot;&gt;
513 *   &lt;property name=&quot;tokens&quot; value=&quot;EQUAL,NOT_EQUAL&quot;/&gt;
514 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;LITERAL_THIS,LITERAL_NULL&quot;/&gt;
515 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;1&quot;/&gt;
516 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;1&quot;/&gt;
517 *   &lt;property name=&quot;sumTokenCounts&quot; value=&quot;true&quot;/&gt;
518 * &lt;/module&gt;
519 * </pre>
520 * <p>
521 * Example:
522 * </p>
523 * <pre>
524 * class Test {
525 *   public void foo() {
526 *     if (this == null) { // violation
527 *       System.out.println("xyz");
528 *     }
529 *     if (this != null) { // violation
530 *       System.out.println("xyz");
531 *     }
532 *
533 *     Object obj = new Object();
534 *     if (obj == null) { // ok
535 *       System.out.println("xyz");
536 *     }
537 *     if (obj != null) { // ok
538 *       System.out.println("xyz");
539 *     }
540 *   }
541 * }
542 * </pre>
543 * <p>
544 * To configure the check to produce a violation on a {@code String} literal equality check:
545 * </p>
546 * <pre>
547 * &lt;module name=&quot;DescendantToken&quot;&gt;
548 *   &lt;property name=&quot;tokens&quot; value=&quot;EQUAL,NOT_EQUAL&quot;/&gt;
549 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;STRING_LITERAL&quot;/&gt;
550 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
551 *   &lt;property name=&quot;maximumDepth&quot; value=&quot;1&quot;/&gt;
552 * &lt;/module&gt;
553 * </pre>
554 * <p>
555 * Example:
556 * </p>
557 * <pre>
558 * class Test {
559 *  public void foo() {
560 *    String str = "abc";
561 *    if (str.equals("abc")) { // ok
562 *      System.out.println("equal.");
563 *   }
564 *   if (str == "abc") { // violation
565 *     System.out.println("equal.");
566 *   }
567 *  }
568 * }
569 * </pre>
570 * <p>
571 * To configure the check to produce a violation on an assert statement that may
572 * have side effects:
573 * </p>
574 * <pre>
575 * &lt;module name=&quot;DescendantToken&quot;&gt;
576 *   &lt;property name=&quot;tokens&quot; value=&quot;LITERAL_ASSERT&quot;/&gt;
577 *   &lt;property name=&quot;limitedTokens&quot; value=&quot;ASSIGN,DEC,INC,POST_DEC,
578 *     POST_INC,PLUS_ASSIGN,MINUS_ASSIGN,STAR_ASSIGN,DIV_ASSIGN,MOD_ASSIGN,
579 *     BSR_ASSIGN,SR_ASSIGN,SL_ASSIGN,BAND_ASSIGN,BXOR_ASSIGN,BOR_ASSIGN,
580 *     METHOD_CALL&quot;/&gt;
581 *   &lt;property name=&quot;maximumNumber&quot; value=&quot;0&quot;/&gt;
582 * &lt;/module&gt;
583 * </pre>
584 * <p>
585 * Example:
586 * </p>
587 * <pre>
588 * class Test {
589 *   public void foo() {
590 *     int a = 5;
591 *     assert a++ == 0 : "is not"; // violation
592 *     System.out.println(a);
593 *     assert a == 0 : "is not"; // ok
594 *   }
595 * }
596 * </pre>
597 * <p>
598 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
599 * </p>
600 * <p>
601 * Violation Message Keys:
602 * </p>
603 * <ul>
604 * <li>
605 * {@code descendant.token.max}
606 * </li>
607 * <li>
608 * {@code descendant.token.min}
609 * </li>
610 * <li>
611 * {@code descendant.token.sum.max}
612 * </li>
613 * <li>
614 * {@code descendant.token.sum.min}
615 * </li>
616 * </ul>
617 *
618 * @since 3.2
619 */
620@FileStatefulCheck
621public class DescendantTokenCheck extends AbstractCheck {
622
623    /**
624     * A key is pointing to the warning message text in "messages.properties"
625     * file.
626     */
627    public static final String MSG_KEY_MIN = "descendant.token.min";
628
629    /**
630     * A key is pointing to the warning message text in "messages.properties"
631     * file.
632     */
633    public static final String MSG_KEY_MAX = "descendant.token.max";
634
635    /**
636     * A key is pointing to the warning message text in "messages.properties"
637     * file.
638     */
639    public static final String MSG_KEY_SUM_MIN = "descendant.token.sum.min";
640
641    /**
642     * A key is pointing to the warning message text in "messages.properties"
643     * file.
644     */
645    public static final String MSG_KEY_SUM_MAX = "descendant.token.sum.max";
646
647    /** Specify the minimum depth for descendant counts. */
648    private int minimumDepth;
649    /** Specify the maximum depth for descendant counts. */
650    private int maximumDepth = Integer.MAX_VALUE;
651    /** Specify a minimum count for descendants. */
652    private int minimumNumber;
653    /** Specify a maximum count for descendants. */
654    private int maximumNumber = Integer.MAX_VALUE;
655    /**
656     * Control whether the number of tokens found should be calculated from
657     * the sum of the individual token counts.
658     */
659    private boolean sumTokenCounts;
660    /** Specify set of tokens with limited occurrences as descendants. */
661    @XdocsPropertyType(PropertyType.TOKEN_ARRAY)
662    private int[] limitedTokens = CommonUtil.EMPTY_INT_ARRAY;
663    /** Define the violation message when the minimum count is not reached. */
664    private String minimumMessage;
665    /** Define the violation message when the maximum count is exceeded. */
666    private String maximumMessage;
667
668    /**
669     * Counts of descendant tokens.
670     * Indexed by (token ID - 1) for performance.
671     */
672    private int[] counts = CommonUtil.EMPTY_INT_ARRAY;
673
674    @Override
675    public int[] getAcceptableTokens() {
676        return TokenUtil.getAllTokenIds();
677    }
678
679    @Override
680    public int[] getDefaultTokens() {
681        return getRequiredTokens();
682    }
683
684    @Override
685    public int[] getRequiredTokens() {
686        return CommonUtil.EMPTY_INT_ARRAY;
687    }
688
689    @Override
690    public void visitToken(DetailAST ast) {
691        // reset counts
692        Arrays.fill(counts, 0);
693        countTokens(ast, 0);
694
695        if (sumTokenCounts) {
696            logAsTotal(ast);
697        }
698        else {
699            logAsSeparated(ast);
700        }
701    }

703    /**
704     * Log violations for each Token.
705     *
706     * @param ast token
707     */
708    private void logAsSeparated(DetailAST ast) {
709        // name of this token
710        final String name = TokenUtil.getTokenName(ast.getType());
711
712        for (int element : limitedTokens) {
713            final int tokenCount = counts[element - 1];
714            if (tokenCount < minimumNumber) {
715                final String descendantName = TokenUtil.getTokenName(element);
716
717                if (minimumMessage == null) {
718                    minimumMessage = MSG_KEY_MIN;
719                }
720                log(ast,
721                        minimumMessage,
722                        String.valueOf(tokenCount),
723                        String.valueOf(minimumNumber),
724                        name,
725                        descendantName);
726            }
727            if (tokenCount > maximumNumber) {
728                final String descendantName = TokenUtil.getTokenName(element);
729
730                if (maximumMessage == null) {
731                    maximumMessage = MSG_KEY_MAX;
732                }
733                log(ast,
734                        maximumMessage,
735                        String.valueOf(tokenCount),
736                        String.valueOf(maximumNumber),
737                        name,
738                        descendantName);
739            }
740        }
741    }

743    /**
744     * Log validation as one violation.
745     *
746     * @param ast current token
747     */
748    private void logAsTotal(DetailAST ast) {
749        // name of this token
750        final String name = TokenUtil.getTokenName(ast.getType());
751
752        int total = 0;
753        for (int element : limitedTokens) {
754            total += counts[element - 1];
755        }
756        if (total < minimumNumber) {
757            if (minimumMessage == null) {
758                minimumMessage = MSG_KEY_SUM_MIN;
759            }
760            log(ast,
761                    minimumMessage,
762                    String.valueOf(total),
763                    String.valueOf(minimumNumber), name);
764        }
765        if (total > maximumNumber) {
766            if (maximumMessage == null) {
767                maximumMessage = MSG_KEY_SUM_MAX;
768            }
769            log(ast,
770                    maximumMessage,
771                    String.valueOf(total),
772                    String.valueOf(maximumNumber), name);
773        }
774    }

776    /**
777     * Counts the number of occurrences of descendant tokens.
778     *
779     * @param ast the root token for descendants.
780     * @param depth the maximum depth of the counted descendants.
781     */
782    private void countTokens(DetailAST ast, int depth) {
783        if (depth <= maximumDepth) {
784            // update count
785            if (depth >= minimumDepth) {
786                final int type = ast.getType();
787                if (type <= counts.length) {
788                    counts[type - 1]++;
789                }
790            }
791            DetailAST child = ast.getFirstChild();
792            final int nextDepth = depth + 1;
793            while (child != null) {
794                countTokens(child, nextDepth);
795                child = child.getNextSibling();
796            }
797        }
798    }

800    /**
801     * Setter to specify set of tokens with limited occurrences as descendants.
802     *
803     * @param limitedTokensParam tokens to ignore.
804     */
805    public void setLimitedTokens(String... limitedTokensParam) {
806        limitedTokens = new int[limitedTokensParam.length];
807
808        int maxToken = 0;
809        for (int i = 0; i < limitedTokensParam.length; i++) {
810            limitedTokens[i] = TokenUtil.getTokenId(limitedTokensParam[i]);
811            if (limitedTokens[i] >= maxToken + 1) {
812                maxToken = limitedTokens[i];
813            }
814        }
815        counts = new int[maxToken];
816    }

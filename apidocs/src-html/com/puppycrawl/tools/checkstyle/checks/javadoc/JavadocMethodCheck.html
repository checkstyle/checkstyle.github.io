094    /** Compiled regexp to match Javadoc tags that take an argument. */
095    private static final Pattern MATCH_JAVADOC_ARG = CommonUtil.createPattern(
096            "^\\s*(?>\\*|\\/\\*\\*)?\\s*@(throws|exception|param)\\s+(\\S+)\\s+\\S*");
097
098    /** Compiled regexp to match first part of multilineJavadoc tags. */
099    private static final Pattern MATCH_JAVADOC_ARG_MULTILINE_START = CommonUtil.createPattern(
100            "^\\s*(?>\\*|\\/\\*\\*)?\\s*@(throws|exception|param)\\s+(\\S+)\\s*$");
101
102    /** Compiled regexp to look for a continuation of the comment. */
103    private static final Pattern MATCH_JAVADOC_MULTILINE_CONT =
104            CommonUtil.createPattern("(\\*/|@|[^\\s\\*])");

108    /** Multiline finished at next Javadoc. */

318    @Override
319    protected final void processAST(DetailAST ast) {
320        final Scope theScope = calculateScope(ast);
321        if (shouldCheck(ast, theScope)) {
322            final FileContents contents = getFileContents();
323            final TextBlock textBlock = contents.getJavadocBefore(ast.getLineNo());
324
325            if (textBlock != null) {
326                checkComment(ast, textBlock);
327            }
328        }
329    }

338    /**
339     * Whether we should check this node.
340     *
341     * @param ast a given node.
342     * @param nodeScope the scope of the node.
343     * @return whether we should check a given node.
344     */
345    private boolean shouldCheck(final DetailAST ast, final Scope nodeScope) {
346        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
347
348        return (excludeScope == null
349                || nodeScope != excludeScope
350                && surroundingScope != excludeScope)
351            && nodeScope.isIn(scope)
352            && surroundingScope.isIn(scope);
353    }

355    /**
356     * Checks the Javadoc for a method.
357     *
358     * @param ast the token for the method
359     * @param comment the Javadoc comment
360     */
361    private void checkComment(DetailAST ast, TextBlock comment) {
362        final List<JavadocTag> tags = getMethodTags(comment);
363
364        if (!hasShortCircuitTag(ast, tags)) {
365            if (ast.getType() == TokenTypes.ANNOTATION_FIELD_DEF) {
366                checkReturnTag(tags, ast.getLineNo(), true);
367            }
368            else {
369                final Iterator<JavadocTag> it = tags.iterator();
370                // Check for inheritDoc
371                boolean hasInheritDocTag = false;
372                while (!hasInheritDocTag && it.hasNext()) {
373                    hasInheritDocTag = it.next().isInheritDocTag();
374                }
375                final boolean reportExpectedTags = !hasInheritDocTag
376                    && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
377
378                checkParamTags(tags, ast, reportExpectedTags);
379                checkThrowsTags(tags, getThrows(ast), reportExpectedTags);
380                if (CheckUtil.isNonVoidMethod(ast)) {
381                    checkReturnTag(tags, ast.getLineNo(), reportExpectedTags);
382                }
383            }
384
385            // Dump out all unused tags
386            tags.stream().filter(javadocTag -> !javadocTag.isSeeOrInheritDocTag())
387                .forEach(javadocTag -> log(javadocTag.getLineNo(), MSG_UNUSED_TAG_GENERAL));
388        }
389    }

391    /**
392     * Validates whether the Javadoc has a short circuit tag. Currently this is
393     * the inheritTag. Any violations are logged.
394     *
395     * @param ast the construct being checked
396     * @param tags the list of Javadoc tags associated with the construct
397     * @return true if the construct has a short circuit tag.
398     */
399    private boolean hasShortCircuitTag(final DetailAST ast, final List<JavadocTag> tags) {
400        boolean result = true;
401        // Check if it contains {@inheritDoc} tag
402        if (tags.size() == 1
403                && tags.get(0).isInheritDocTag()) {
404            // Invalid if private, a constructor, or a static method
405            if (!JavadocTagInfo.INHERIT_DOC.isValidOn(ast)) {
406                log(ast, MSG_INVALID_INHERIT_DOC);
407            }
408        }
409        else {
410            result = false;
411        }
412        return result;
413    }

415    /**
416     * Returns the scope for the method/constructor at the specified AST. If
417     * the method is in an interface or annotation block, the scope is assumed
418     * to be public.
419     *
420     * @param ast the token of the method/constructor
421     * @return the scope of the method/constructor
422     */
423    private static Scope calculateScope(final DetailAST ast) {
424        final Scope scope;
425
426        if (ScopeUtil.isInInterfaceOrAnnotationBlock(ast)) {
427            scope = Scope.PUBLIC;
428        }
429        else {
430            final DetailAST mods = ast.findFirstToken(TokenTypes.MODIFIERS);
431            scope = ScopeUtil.getScopeFromMods(mods);
432        }
433        return scope;
434    }

436    /**
437     * Returns the tags in a javadoc comment. Only finds throws, exception,
438     * param, return and see tags.
439     *
440     * @param comment the Javadoc comment
441     * @return the tags found
442     */
443    private static List<JavadocTag> getMethodTags(TextBlock comment) {
444        final String[] lines = comment.getText();
445        final List<JavadocTag> tags = new ArrayList<>();
446        int currentLine = comment.getStartLineNo() - 1;
447        final int startColumnNumber = comment.getStartColNo();
448
449        for (int i = 0; i < lines.length; i++) {
450            currentLine++;
451            final Matcher javadocArgMatcher =
452                MATCH_JAVADOC_ARG.matcher(lines[i]);
453            final Matcher javadocNoargMatcher =
454                MATCH_JAVADOC_NOARG.matcher(lines[i]);
455            final Matcher noargCurlyMatcher =
456                MATCH_JAVADOC_NOARG_CURLY.matcher(lines[i]);
457            final Matcher argMultilineStart =
458                MATCH_JAVADOC_ARG_MULTILINE_START.matcher(lines[i]);
459            final Matcher noargMultilineStart =
460                MATCH_JAVADOC_NOARG_MULTILINE_START.matcher(lines[i]);
461
462            if (javadocArgMatcher.find()) {
463                final int col = calculateTagColumn(javadocArgMatcher, i, startColumnNumber);
464                tags.add(new JavadocTag(currentLine, col, javadocArgMatcher.group(1),
465                        javadocArgMatcher.group(2)));
466            }
467            else if (javadocNoargMatcher.find()) {
468                final int col = calculateTagColumn(javadocNoargMatcher, i, startColumnNumber);
469                tags.add(new JavadocTag(currentLine, col, javadocNoargMatcher.group(1)));
470            }
471            else if (noargCurlyMatcher.find()) {
472                final int col = calculateTagColumn(noargCurlyMatcher, i, startColumnNumber);
473                tags.add(new JavadocTag(currentLine, col, noargCurlyMatcher.group(1)));
474            }
475            else if (argMultilineStart.find()) {
476                final int col = calculateTagColumn(argMultilineStart, i, startColumnNumber);
477                tags.addAll(getMultilineArgTags(argMultilineStart, col, lines, i, currentLine));
478            }
479            else if (noargMultilineStart.find()) {
480                tags.addAll(getMultilineNoArgTags(noargMultilineStart, lines, i, currentLine));
481            }
482        }
483        return tags;
484    }

486    /**
487     * Calculates column number using Javadoc tag matcher.
488     * @param javadocTagMatcher found javadoc tag matcher
489     * @param lineNumber line number of Javadoc tag in comment
490     * @param startColumnNumber column number of Javadoc comment beginning
491     * @return column number
492     */
493    private static int calculateTagColumn(Matcher javadocTagMatcher,
494            int lineNumber, int startColumnNumber) {
495        int col = javadocTagMatcher.start(1) - 1;
496        if (lineNumber == 0) {
497            col += startColumnNumber;
498        }
499        return col;
500    }

502    /**
503     * Gets multiline Javadoc tags with arguments.
504     * @param argMultilineStart javadoc tag Matcher
505     * @param column column number of Javadoc tag
506     * @param lines comment text lines
507     * @param lineIndex line number that contains the javadoc tag
508     * @param tagLine javadoc tag line number in file
509     * @return javadoc tags with arguments
510     */
511    private static List<JavadocTag> getMultilineArgTags(final Matcher argMultilineStart,
512            final int column, final String[] lines, final int lineIndex, final int tagLine) {
513        final List<JavadocTag> tags = new ArrayList<>();
514        final String param1 = argMultilineStart.group(1);
515        final String param2 = argMultilineStart.group(2);
516        for (int remIndex = lineIndex + 1; remIndex < lines.length; remIndex++) {
517            final Matcher multilineCont = MATCH_JAVADOC_MULTILINE_CONT.matcher(lines[remIndex]);
518            if (multilineCont.find()) {
519                final String lFin = multilineCont.group(1);
520                if (!lFin.equals(NEXT_TAG)
521                    && !lFin.equals(END_JAVADOC)) {
522                    tags.add(new JavadocTag(tagLine, column, param1, param2));
523                }
524                break;
525            }
526        }
527
528        return tags;
529    }

531    /**
532     * Gets multiline Javadoc tags with no arguments.
533     * @param noargMultilineStart javadoc tag Matcher
534     * @param lines comment text lines
535     * @param lineIndex line number that contains the javadoc tag
536     * @param tagLine javadoc tag line number in file
537     * @return javadoc tags with no arguments
538     */
539    private static List<JavadocTag> getMultilineNoArgTags(final Matcher noargMultilineStart,
540            final String[] lines, final int lineIndex, final int tagLine) {
541        int remIndex = lineIndex;
542        Matcher multilineCont;
543
544        do {
545            remIndex++;
546            multilineCont = MATCH_JAVADOC_MULTILINE_CONT.matcher(lines[remIndex]);
547        } while (!multilineCont.find());
548
549        final List<JavadocTag> tags = new ArrayList<>();
550        final String lFin = multilineCont.group(1);
551        if (!lFin.equals(NEXT_TAG)
552            && !lFin.equals(END_JAVADOC)) {
553            final String param1 = noargMultilineStart.group(1);
554            final int col = noargMultilineStart.start(1) - 1;
555
556            tags.add(new JavadocTag(tagLine, col, param1));
557        }
558
559        return tags;
560    }

562    /**
563     * Computes the parameter nodes for a method.
564     *
565     * @param ast the method node.
566     * @return the list of parameter nodes for ast.
567     */
568    private static List<DetailAST> getParameters(DetailAST ast) {
569        final DetailAST params = ast.findFirstToken(TokenTypes.PARAMETERS);
570        final List<DetailAST> returnValue = new ArrayList<>();
571
572        DetailAST child = params.getFirstChild();
573        while (child != null) {
574            if (child.getType() == TokenTypes.PARAMETER_DEF) {
575                final DetailAST ident = child.findFirstToken(TokenTypes.IDENT);
576                if (ident != null) {
577                    returnValue.add(ident);
578                }
579            }
580            child = child.getNextSibling();
581        }
582        return returnValue;
583    }

585    /**
586     * Computes the exception nodes for a method.
587     *
588     * @param ast the method node.
589     * @return the list of exception nodes for ast.
590     */
591    private List<ExceptionInfo> getThrows(DetailAST ast) {
592        final List<ExceptionInfo> returnValue = new ArrayList<>();
593        final DetailAST throwsAST = ast
594                .findFirstToken(TokenTypes.LITERAL_THROWS);
595        if (throwsAST != null) {
596            DetailAST child = throwsAST.getFirstChild();
597            while (child != null) {
598                if (child.getType() == TokenTypes.IDENT
599                        || child.getType() == TokenTypes.DOT) {
600                    final FullIdent ident = FullIdent.createFullIdent(child);
601                    final ExceptionInfo exceptionInfo = new ExceptionInfo(
602                            createClassInfo(new Token(ident), getCurrentClassName()));
603                    returnValue.add(exceptionInfo);
604                }
605                child = child.getNextSibling();
606            }
607        }
608        return returnValue;
609    }

611    /**
612     * Checks a set of tags for matching parameters.
613     *
614     * @param tags the tags to check
615     * @param parent the node which takes the parameters
616     * @param reportExpectedTags whether we should report if do not find
617     *            expected tag
618     */
619    private void checkParamTags(final List<JavadocTag> tags,
620            final DetailAST parent, boolean reportExpectedTags) {
621        final List<DetailAST> params = getParameters(parent);
622        final List<DetailAST> typeParams = CheckUtil
623                .getTypeParameters(parent);
624
625        // Loop over the tags, checking to see they exist in the params.
626        final ListIterator<JavadocTag> tagIt = tags.listIterator();
627        while (tagIt.hasNext()) {
628            final JavadocTag tag = tagIt.next();
629
630            if (!tag.isParamTag()) {
631                continue;
632            }
633
634            tagIt.remove();
635
636            final String arg1 = tag.getFirstArg();
637            boolean found = removeMatchingParam(params, arg1);
638
639            if (CommonUtil.startsWithChar(arg1, '<') && CommonUtil.endsWithChar(arg1, '>')) {
640                found = searchMatchingTypeParameter(typeParams,
641                        arg1.substring(1, arg1.length() - 1));
642            }
643
644            // Handle extra JavadocTag
645            if (!found) {
646                log(tag.getLineNo(), tag.getColumnNo(), MSG_UNUSED_TAG,
647                        "@param", arg1);
648            }
649        }
650
651        // Now dump out all type parameters/parameters without tags :- unless
652        // the user has chosen to suppress these problems
653        if (!allowMissingParamTags && reportExpectedTags) {
654            for (DetailAST param : params) {
655                log(param, MSG_EXPECTED_TAG,
656                    JavadocTagInfo.PARAM.getText(), param.getText());
657            }
658
659            for (DetailAST typeParam : typeParams) {
660                log(typeParam, MSG_EXPECTED_TAG,
661                    JavadocTagInfo.PARAM.getText(),
662                    "<" + typeParam.findFirstToken(TokenTypes.IDENT).getText()
663                    + ">");
664            }
665        }
666    }

668    /**
669     * Returns true if required type found in type parameters.
670     * @param typeParams
671     *            list of type parameters
672     * @param requiredTypeName
673     *            name of required type
674     * @return true if required type found in type parameters.
675     */
676    private static boolean searchMatchingTypeParameter(List<DetailAST> typeParams,
677            String requiredTypeName) {
678        // Loop looking for matching type param
679        final Iterator<DetailAST> typeParamsIt = typeParams.iterator();
680        boolean found = false;
681        while (typeParamsIt.hasNext()) {
682            final DetailAST typeParam = typeParamsIt.next();
683            if (typeParam.findFirstToken(TokenTypes.IDENT).getText()
684                    .equals(requiredTypeName)) {
685                found = true;
686                typeParamsIt.remove();
687                break;
688            }
689        }
690        return found;
691    }

693    /**
694     * Remove parameter from params collection by name.
695     * @param params collection of DetailAST parameters
696     * @param paramName name of parameter
697     * @return true if parameter found and removed
698     */
699    private static boolean removeMatchingParam(List<DetailAST> params, String paramName) {
700        boolean found = false;
701        final Iterator<DetailAST> paramIt = params.iterator();
702        while (paramIt.hasNext()) {
703            final DetailAST param = paramIt.next();
704            if (param.getText().equals(paramName)) {
705                found = true;
706                paramIt.remove();
707                break;
708            }
709        }
710        return found;
711    }

713    /**
714     * Checks for only one return tag. All return tags will be removed from the
715     * supplied list.
716     *
717     * @param tags the tags to check
718     * @param lineNo the line number of the expected tag
719     * @param reportExpectedTags whether we should report if do not find
720     *            expected tag
721     */
722    private void checkReturnTag(List<JavadocTag> tags, int lineNo,
723        boolean reportExpectedTags) {
724        // Loop over tags finding return tags. After the first one, report an
725        // violation.
726        boolean found = false;
727        final ListIterator<JavadocTag> it = tags.listIterator();
728        while (it.hasNext()) {
729            final JavadocTag javadocTag = it.next();
730            if (javadocTag.isReturnTag()) {
731                if (found) {
732                    log(javadocTag.getLineNo(), javadocTag.getColumnNo(),
733                            MSG_DUPLICATE_TAG,
734                            JavadocTagInfo.RETURN.getText());
735                }
736                found = true;
737                it.remove();
738            }
739        }
740
741        // Handle there being no @return tags :- unless
742        // the user has chosen to suppress these problems
743        if (!found && !allowMissingReturnTag && reportExpectedTags) {
744            log(lineNo, MSG_RETURN_EXPECTED);
745        }
746    }

748    /**
749     * Checks a set of tags for matching throws.
750     *
751     * @param tags the tags to check
752     * @param throwsList the throws to check
753     * @param reportExpectedTags whether we should report if do not find
754     *            expected tag
755     */
756    private void checkThrowsTags(List<JavadocTag> tags,
757            List<ExceptionInfo> throwsList, boolean reportExpectedTags) {
758        // Loop over the tags, checking to see they exist in the throws.
759        // The foundThrows used for performance only
760        final Set<String> foundThrows = new HashSet<>();
761        final ListIterator<JavadocTag> tagIt = tags.listIterator();
762        while (tagIt.hasNext()) {
763            final JavadocTag tag = tagIt.next();
764
765            if (!tag.isThrowsTag()) {
766                continue;
767            }
768            tagIt.remove();
769
770            // Loop looking for matching throw
771            final String documentedEx = tag.getFirstArg();
772            final Token token = new Token(tag.getFirstArg(), tag.getLineNo(), tag
773                    .getColumnNo());
774            final AbstractClassInfo documentedClassInfo = createClassInfo(token,
775                    getCurrentClassName());
776            final boolean found = foundThrows.contains(documentedEx)
777                    || isInThrows(throwsList, documentedClassInfo, foundThrows);
778
779            // Handle extra JavadocTag.
780            if (!found) {
781                boolean reqd = true;
782                if (allowUndeclaredRTE) {
783                    reqd = !isUnchecked(documentedClassInfo.getClazz());
784                }
785
786                if (reqd && validateThrows) {
787                    log(tag.getLineNo(), tag.getColumnNo(),
788                        MSG_UNUSED_TAG,
789                        JavadocTagInfo.THROWS.getText(), tag.getFirstArg());
790                }
791            }
792        }
793        // Now dump out all throws without tags :- unless
794        // the user has chosen to suppress these problems
795        if (!allowMissingThrowsTags && reportExpectedTags) {
796            throwsList.stream().filter(exceptionInfo -> !exceptionInfo.isFound())
797                .forEach(exceptionInfo -> {
798                    final Token token = exceptionInfo.getName();
799                    log(token.getLineNo(), token.getColumnNo(),
800                        MSG_EXPECTED_TAG,
801                        JavadocTagInfo.THROWS.getText(), token.getText());
802                });
803        }
804    }

806    /**
807     * Verifies that documented exception is in throws.
808     *
809     * @param throwsList list of throws
810     * @param documentedClassInfo documented exception class info
811     * @param foundThrows previously found throws
812     * @return true if documented exception is in throws.
813     */
814    private boolean isInThrows(List<ExceptionInfo> throwsList,
815            AbstractClassInfo documentedClassInfo, Set<String> foundThrows) {
816        boolean found = false;
817        ExceptionInfo foundException = null;
818
819        // First look for matches on the exception name
820        for (ExceptionInfo exceptionInfo : throwsList) {
821            if (exceptionInfo.getName().getText().equals(
822                    documentedClassInfo.getName().getText())) {
823                found = true;
824                foundException = exceptionInfo;
825                break;
826            }
827        }
828
829        // Now match on the exception type
830        final ListIterator<ExceptionInfo> exceptionInfoIt = throwsList.listIterator();
831        while (!found && exceptionInfoIt.hasNext()) {
832            final ExceptionInfo exceptionInfo = exceptionInfoIt.next();
833
834            if (documentedClassInfo.getClazz() == exceptionInfo.getClazz()) {
835                found = true;
836                foundException = exceptionInfo;
837            }
838            else if (allowThrowsTagsForSubclasses) {
839                found = isSubclass(documentedClassInfo.getClazz(), exceptionInfo.getClazz());
840            }
841        }
842
843        if (foundException != null) {
844            foundException.setFound();
845            foundThrows.add(documentedClassInfo.getName().getText());
846        }
847
848        return found;
849    }

851    /** Stores useful information about declared exception. */
852    private static class ExceptionInfo {
853
854        /** Class information associated with this exception. */
855        private final AbstractClassInfo classInfo;
856        /** Does the exception have throws tag associated with. */
857        private boolean found;
858
859        /**
860         * Creates new instance for {@code FullIdent}.
861         *
862         * @param classInfo class info
863         */
864        /* package */ ExceptionInfo(AbstractClassInfo classInfo) {
865            this.classInfo = classInfo;
866        }
867
868        /** Mark that the exception has associated throws tag. */
869        private void setFound() {
870            found = true;
871        }
872
873        /**
874         * Checks that the exception has throws tag associated with it.
875         * @return whether the exception has throws tag associated with
876         */
877        private boolean isFound() {
878            return found;
879        }
880
881        /**
882         * Gets exception name.
883         * @return exception's name
884         */
885        private Token getName() {
886            return classInfo.getName();
887        }
888
889        /**
890         * Gets exception class.
891         * @return class for this exception
892         */
893        private Class<?> getClazz() {
894            return classInfo.getClazz();
895        }

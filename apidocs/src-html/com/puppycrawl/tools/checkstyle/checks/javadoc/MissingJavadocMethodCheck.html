038/**
039 * <p>
040 * Checks for missing Javadoc comments for a method or constructor. The scope to verify is
041 * specified using the {@code Scope} class and defaults to {@code Scope.PUBLIC}. To verify
042 * another scope, set property scope to a different
043 * <a href="https://checkstyle.org/property_types.html#Scope">scope</a>.
044 * </p>
045 * <p>
046 * Javadoc is not required on a method that is tagged with the {@code @Override} annotation.
047 * However, under Java 5 it is not possible to mark a method required for an interface (this
048 * was <i>corrected</i> under Java 6). Hence, Checkstyle supports using the convention of using
049 * a single {@code {@inheritDoc}} tag instead of all the other tags.
050 * </p>
051 * <p>
052 * For getters and setters for the property {@code allowMissingPropertyJavadoc}, the methods must
053 * match exactly the structures below.
054 * </p>
055 * <pre>
056 * public void setNumber(final int number)
057 * {
058 *     mNumber = number;
059 * }
060 *
061 * public int getNumber()
062 * {
063 *     return mNumber;
064 * }
065 *
066 * public boolean isSomething()
067 * {
068 *     return false;
069 * }
070 * </pre>
071 * <ul>
072 * <li>
073 * Property {@code minLineCount} - Control the minimal amount of lines in method to allow no
074 * documentation.
075 * Type is {@code int}.
076 * Default value is {@code -1}.
077 * </li>
078 * <li>
079 * Property {@code allowedAnnotations} - Configure annotations that allow missed
080 * documentation.
081 * Type is {@code java.lang.String[]}.
082 * Default value is {@code Override}.
083 * </li>
084 * <li>
085 * Property {@code scope} - Specify the visibility scope where Javadoc comments are checked.
086 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
087 * Default value is {@code public}.
088 * </li>
089 * <li>
090 * Property {@code excludeScope} - Specify the visibility scope where Javadoc comments are
091 * not checked.
092 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
093 * Default value is {@code null}.
094 * </li>
095 * <li>
096 * Property {@code allowMissingPropertyJavadoc} - Control whether to allow missing Javadoc on
097 * accessor methods for properties (setters and getters).
098 * Type is {@code boolean}.
099 * Default value is {@code false}.
100 * </li>
101 * <li>
102 * Property {@code ignoreMethodNamesRegex} - ignore method whose names are matching specified
103 * regex.
104 * Type is {@code java.util.regex.Pattern}.
105 * Default value is {@code null}.
106 * </li>
107 * <li>
108 * Property {@code tokens} - tokens to check
109 * Type is {@code java.lang.String[]}.
110 * Validation type is {@code tokenSet}.
111 * Default value is:
112 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
113 * METHOD_DEF</a>,
114 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
115 * CTOR_DEF</a>,
116 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_FIELD_DEF">
117 * ANNOTATION_FIELD_DEF</a>,
118 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#COMPACT_CTOR_DEF">
119 * COMPACT_CTOR_DEF</a>.
120 * </li>
121 * </ul>
122 * <p>
123 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
124 * </p>
125 * <p>
126 * Violation Message Keys:
127 * </p>
128 * <ul>
129 * <li>
130 * {@code javadoc.missing}
131 * </li>
132 * </ul>
133 *
134 * @since 8.21
135 */
136@FileStatefulCheck
137public class MissingJavadocMethodCheck extends AbstractCheck {
138
139    /**
140     * A key is pointing to the warning message text in "messages.properties"
141     * file.
142     */
143    public static final String MSG_JAVADOC_MISSING = "javadoc.missing";
144
145    /** Default value of minimal amount of lines in method to allow no documentation.*/
146    private static final int DEFAULT_MIN_LINE_COUNT = -1;
147
148    /** Specify the visibility scope where Javadoc comments are checked. */
149    private Scope scope = Scope.PUBLIC;
150
151    /** Specify the visibility scope where Javadoc comments are not checked. */
152    private Scope excludeScope;
153
154    /** Control the minimal amount of lines in method to allow no documentation.*/
155    private int minLineCount = DEFAULT_MIN_LINE_COUNT;
156
157    /**
158     * Control whether to allow missing Javadoc on accessor methods for
159     * properties (setters and getters).
160     */
161    private boolean allowMissingPropertyJavadoc;
162
163    /** Ignore method whose names are matching specified regex. */
164    private Pattern ignoreMethodNamesRegex;
165
166    /** Configure annotations that allow missed documentation. */
167    private Set<String> allowedAnnotations = Set.of("Override");
168
169    /**
170     * Setter to configure annotations that allow missed documentation.
171     *
172     * @param userAnnotations user's value.
173     * @since 8.21
174     */
175    public void setAllowedAnnotations(String... userAnnotations) {
176        allowedAnnotations = Set.of(userAnnotations);
177    }
178
179    /**
180     * Setter to ignore method whose names are matching specified regex.
181     *
182     * @param pattern a pattern.
183     * @since 8.21
184     */
185    public void setIgnoreMethodNamesRegex(Pattern pattern) {
186        ignoreMethodNamesRegex = pattern;
187    }
188
189    /**
190     * Setter to control the minimal amount of lines in method to allow no documentation.
191     *
192     * @param value user's value.
193     * @since 8.21
194     */
195    public void setMinLineCount(int value) {
196        minLineCount = value;
197    }
198
199    /**
200     * Setter to control whether to allow missing Javadoc on accessor methods for properties
201     * (setters and getters).
202     *
203     * @param flag a {@code Boolean} value
204     * @since 8.21
205     */
206    public void setAllowMissingPropertyJavadoc(final boolean flag) {
207        allowMissingPropertyJavadoc = flag;
208    }
209
210    /**
211     * Setter to specify the visibility scope where Javadoc comments are checked.
212     *
213     * @param scope a scope.
214     * @since 8.21
215     */
216    public void setScope(Scope scope) {
217        this.scope = scope;
218    }
219
220    /**
221     * Setter to specify the visibility scope where Javadoc comments are not checked.
222     *
223     * @param excludeScope a scope.
224     * @since 8.21
225     */
226    public void setExcludeScope(Scope excludeScope) {
227        this.excludeScope = excludeScope;
228    }
229
230    @Override
231    public final int[] getRequiredTokens() {
232        return CommonUtil.EMPTY_INT_ARRAY;
233    }
234
235    @Override
236    public int[] getDefaultTokens() {
237        return getAcceptableTokens();
238    }
239
240    @Override
241    public int[] getAcceptableTokens() {
242        return new int[] {
243            TokenTypes.METHOD_DEF,
244            TokenTypes.CTOR_DEF,
245            TokenTypes.ANNOTATION_FIELD_DEF,
246            TokenTypes.COMPACT_CTOR_DEF,
247        };
248    }
249
250    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
251    @SuppressWarnings("deprecation")
252    @Override
253    public final void visitToken(DetailAST ast) {
254        final Scope theScope = ScopeUtil.getScope(ast);
255        if (shouldCheck(ast, theScope)) {
256            final FileContents contents = getFileContents();
257            final TextBlock textBlock = contents.getJavadocBefore(ast.getLineNo());
258
259            if (textBlock == null && !isMissingJavadocAllowed(ast)) {
260                log(ast, MSG_JAVADOC_MISSING);
261            }
262        }
263    }
264
265    /**
266     * Some javadoc.
267     *
268     * @param methodDef Some javadoc.
269     * @return Some javadoc.
270     */
271    private static int getMethodsNumberOfLine(DetailAST methodDef) {
272        final int numberOfLines;
273        final DetailAST lcurly = methodDef.getLastChild();
274        final DetailAST rcurly = lcurly.getLastChild();
275
276        if (lcurly.getFirstChild() == rcurly) {
277            numberOfLines = 1;
278        }
279        else {
280            numberOfLines = rcurly.getLineNo() - lcurly.getLineNo() - 1;
281        }
282        return numberOfLines;
283    }
284
285    /**
286     * Checks if a missing Javadoc is allowed by the check's configuration.
287     *
288     * @param ast the tree node for the method or constructor.
289     * @return True if this method or constructor doesn't need Javadoc.
290     */
291    private boolean isMissingJavadocAllowed(final DetailAST ast) {
292        return allowMissingPropertyJavadoc
293                && (CheckUtil.isSetterMethod(ast) || CheckUtil.isGetterMethod(ast))
294            || matchesSkipRegex(ast)
295            || isContentsAllowMissingJavadoc(ast);
296    }
297
298    /**
299     * Checks if the Javadoc can be missing if the method or constructor is
300     * below the minimum line count or has a special annotation.
301     *
302     * @param ast the tree node for the method or constructor.
303     * @return True if this method or constructor doesn't need Javadoc.
304     */
305    private boolean isContentsAllowMissingJavadoc(DetailAST ast) {
306        return (ast.getType() == TokenTypes.METHOD_DEF
307                || ast.getType() == TokenTypes.CTOR_DEF
308                || ast.getType() == TokenTypes.COMPACT_CTOR_DEF)
309                && (getMethodsNumberOfLine(ast) <= minLineCount
310                    || AnnotationUtil.containsAnnotation(ast, allowedAnnotations));
311    }
312
313    /**
314     * Checks if the given method name matches the regex. In that case
315     * we skip enforcement of javadoc for this method
316     *
317     * @param methodDef {@link TokenTypes#METHOD_DEF METHOD_DEF}
318     * @return true if given method name matches the regex.
319     */
320    private boolean matchesSkipRegex(DetailAST methodDef) {
321        boolean result = false;
322        if (ignoreMethodNamesRegex != null) {
323            final DetailAST ident = methodDef.findFirstToken(TokenTypes.IDENT);
324            final String methodName = ident.getText();
325
326            final Matcher matcher = ignoreMethodNamesRegex.matcher(methodName);
327            if (matcher.matches()) {
328                result = true;
329            }
330        }
331        return result;
332    }
333
334    /**
335     * Whether we should check this node.
336     *
337     * @param ast a given node.
338     * @param nodeScope the scope of the node.
339     * @return whether we should check a given node.
340     */
341    private boolean shouldCheck(final DetailAST ast, final Scope nodeScope) {
342        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
343
344        return (excludeScope == null
345                || nodeScope != excludeScope
346                && surroundingScope != excludeScope)
347            && nodeScope.isIn(scope)
348            && surroundingScope.isIn(scope);
349    }

020package com.puppycrawl.tools.checkstyle.checks.javadoc;
021
022import java.util.HashSet;
023import java.util.Set;
024
025/**
026 * Utility class to resolve a class name to an actual class. Note that loaded
027 * classes are not initialized.
028 * <p>Limitations: this does not handle inner classes very well.</p>
029 *
030 * @author Oliver Burn
031 */
032public class ClassResolver {
033
034    /** Period literal. */
035    private static final String PERIOD = ".";
036    /** Dollar sign literal. */
037    private static final String DOLLAR_SIGN = "$";
038
039    /** Name of the package to check if the class belongs to. **/
040    private final String pkg;
041    /** Set of imports to check against. **/
042    private final Set<String> imports;
043    /** Use to load classes. **/
044    private final ClassLoader loader;
045
046    /**
047     * Creates a new {@code ClassResolver} instance.
048     *
049     * @param loader the ClassLoader to load classes with.
050     * @param pkg the name of the package the class may belong to
051     * @param imports set of imports to check if the class belongs to
052     */
053    public ClassResolver(ClassLoader loader, String pkg, Set<String> imports) {
054        this.loader = loader;
055        this.pkg = pkg;
056        this.imports = new HashSet<>(imports);
057        this.imports.add("java.lang.*");
058    }
059
060    /**
061     * Attempts to resolve the Class for a specified name. The algorithm is
062     * to check:
063     * - fully qualified name
064     * - explicit imports
065     * - enclosing package
066     * - star imports
067     * @param name name of the class to resolve
068     * @param currentClass name of current class (for inner classes).
069     * @return the resolved class
070     * @throws ClassNotFoundException if unable to resolve the class
071     */
072    // -@cs[ForbidWildcardAsReturnType] This method can return any type, so no way to avoid wildcard
073    public Class<?> resolve(String name, String currentClass)
074            throws ClassNotFoundException {
075        // See if the class is full qualified
076        Class<?> clazz = resolveQualifiedName(name);
077        if (clazz == null) {
078            // try matching explicit imports
079            clazz = resolveMatchingExplicitImport(name);
080
081            if (clazz == null) {
082                // See if in the package
083                clazz = resolveInPackage(name);
084
085                if (clazz == null) {
086                    // see if inner class of this class
087                    clazz = resolveInnerClass(name, currentClass);
088
089                    if (clazz == null) {
090                        clazz = resolveByStarImports(name);
091                        // -@cs[NestedIfDepth] it is better to have single return point from method
092                        if (clazz == null) {
093                            throw new ClassNotFoundException(name);
094                        }
095                    }
096                }
097            }
098        }
099        return clazz;
100    }
101
102    /**
103     * Try to find class by search in package.
104     * @param name class name
105     * @return class object
106     */
107    private Class<?> resolveInPackage(String name) {
108        Class<?> clazz = null;
109        if (pkg != null && !pkg.isEmpty()) {
110            final Class<?> classFromQualifiedName = resolveQualifiedName(pkg + PERIOD + name);
111            if (classFromQualifiedName != null) {
112                clazz = classFromQualifiedName;
113            }
114        }
115        return clazz;
116    }
117
118    /**
119     * Try to find class by matching explicit Import.
120     * @param name class name
121     * @return class object
122     */
123    private Class<?> resolveMatchingExplicitImport(String name) {
124        Class<?> clazz = null;
125        for (String imp : imports) {
126            // Very important to add the "." in the check below. Otherwise you
127            // when checking for "DataException", it will match on
128            // "SecurityDataException". This has been the cause of a very
129            // difficult bug to resolve!
130            if (imp.endsWith(PERIOD + name)) {
131                clazz = resolveQualifiedName(imp);
132                if (clazz != null) {
133                    break;
134                }
135            }
136        }
137        return clazz;
138    }
139
140    /**
141     * See if inner class of this class.
142     * @param name name of the search Class to search
143     * @param currentClass class where search in
144     * @return class if found , or null if not resolved
145     * @throws ClassNotFoundException  if an error occurs
146     */
147    private Class<?> resolveInnerClass(String name, String currentClass)
148            throws ClassNotFoundException {
149        Class<?> clazz = null;
150        if (!currentClass.isEmpty()) {
151            String innerClass = currentClass + DOLLAR_SIGN + name;
152
153            if (!pkg.isEmpty()) {
154                innerClass = pkg + PERIOD + innerClass;
155            }
156
157            if (isLoadable(innerClass)) {
158                clazz = safeLoad(innerClass);
159            }
160        }
161        return clazz;
162    }
163
164    /**
165     * Try star imports.
166     * @param name name of the Class to search
167     * @return  class if found , or null if not resolved
168     */
169    private Class<?> resolveByStarImports(String name) {
170        Class<?> clazz = null;
171        for (String imp : imports) {
172            if (imp.endsWith(".*")) {
173                final String fqn = imp.substring(0, imp.lastIndexOf('.') + 1) + name;
174                clazz = resolveQualifiedName(fqn);
175                if (clazz != null) {
176                    break;
177                }
178            }
179        }
180        return clazz;
181    }
182
183    /**
184     * Checks if the given class name can be loaded.
185     * @param name name of the class to check
186     * @return whether a specified class is loadable with safeLoad().
187     */
188    public boolean isLoadable(String name) {
189        boolean result;
190        try {
191            safeLoad(name);
192            result = true;
193        }
194        catch (final ClassNotFoundException | NoClassDefFoundError ignored) {
195            result = false;
196        }
197        return result;
198    }
199
200    /**
201     * Will load a specified class is such a way that it will NOT be
202     * initialised.
203     * @param name name of the class to load
204     * @return the {@code Class} for the specified class
205     * @throws ClassNotFoundException if an error occurs
206     * @throws NoClassDefFoundError if an error occurs
207     */
208    // -@cs[ForbidWildcardAsReturnType] The class is deprecated and will be removed soon.
209    private Class<?> safeLoad(String name) throws ClassNotFoundException, NoClassDefFoundError {
210        // The next line will load the class using the specified class
211        // loader. The magic is having the "false" parameter. This means the
212        // class will not be initialised. Very, very important.
213        return Class.forName(name, false, loader);
214    }
215
216    /**
217     * Tries to resolve a class for fully-specified name.
218     * @param name a given name of class.
219     * @return Class object for the given name or null.
220     */
221    private Class<?> resolveQualifiedName(final String name) {
222        Class<?> classObj = null;
223        try {
224            if (isLoadable(name)) {
225                classObj = safeLoad(name);
226            }
227            else {
228                //Perhaps it's fully-qualified inner class
229                final int dot = name.lastIndexOf('.');
230                if (dot != -1) {
231                    final String innerName =
232                        name.substring(0, dot) + DOLLAR_SIGN + name.substring(dot + 1);
233                    classObj = resolveQualifiedName(innerName);
234                }
235            }
236        }
237        catch (final ClassNotFoundException ex) {
238            // we shouldn't get this exception here,
239            // so this is unexpected runtime exception
240            throw new IllegalStateException(ex);
241        }
242        return classObj;
243    }

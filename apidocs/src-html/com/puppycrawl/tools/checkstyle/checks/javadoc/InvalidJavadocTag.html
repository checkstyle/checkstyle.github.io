020package com.puppycrawl.tools.checkstyle.checks.javadoc;
021
022/**
023 * Value object for storing data about an invalid Javadoc validTags.
024 */
025public final class InvalidJavadocTag {
026
027    /** The line in which the invalid tag occurs. */
028    private final int line;
029    /** The column in which the invalid tag occurs. */
030    private final int col;
031    /** The name of the invalid tag. */
032    private final String name;
033
034    /**
035     * Creates an instance.
036     *
037     * @param line the line of the tag
038     * @param col the column of the tag
039     * @param name the name of the invalid tag
040     */
041    public InvalidJavadocTag(int line, int col, String name) {
042        this.line = line;
043        this.col = col;
044        this.name = name;
045    }
046
047    /**
048     *  Getter for line field.
049     *
050     *  @return line field
051     */
052    public int getLine() {
053        return line;
054    }
055
056    /**
057     *  Getter for col field.
058     *
059     *  @return col field
060     */
061    public int getCol() {
062        return col;
063    }
064
065    /**
066     *  Getter for name field.
067     *
068     *  @return name field
069     */
070    public String getName() {
071        return name;
072    }
073
074}

046/**
047 * <p>
048 * Checks the Javadoc of a method or constructor.
049 * </p>
050 * <p>
051 * Violates parameters and type parameters for which no param tags are present can
052 * be suppressed by defining property {@code allowMissingParamTags}.
053 * </p>
054 * <p>
055 * Violates methods which return non-void but for which no return tag is present can
056 * be suppressed by defining property {@code allowMissingReturnTag}.
057 * </p>
058 * <p>
059 * Violates exceptions which are declared to be thrown (by {@code throws} in the method
060 * signature or by {@code throw new} in the method body), but for which no throws tag is
061 * present by activation of property {@code validateThrows}.
062 * Note that {@code throw new} is not checked in the following places:
063 * </p>
064 * <ul>
065 * <li>
066 * Inside a try block (with catch). It is not possible to determine if the thrown
067 * exception can be caught by the catch block as there is no knowledge of the
068 * inheritance hierarchy, so the try block is ignored entirely. However, catch
069 * and finally blocks, as well as try blocks without catch, are still checked.
070 * </li>
071 * <li>
072 * Local classes, anonymous classes and lambda expressions. It is not known when the
073 * throw statements inside such classes are going to be evaluated, so they are ignored.
074 * </li>
075 * </ul>
076 * <p>
077 * ATTENTION: Checkstyle does not have information about hierarchy of exception types
078 * so usage of base class is considered as separate exception type.
079 * As workaround, you need to specify both types in javadoc (parent and exact type).
080 * </p>
081 * <p>
082 * Javadoc is not required on a method that is tagged with the {@code @Override}
083 * annotation. However, under Java 5 it is not possible to mark a method required
084 * for an interface (this was <i>corrected</i> under Java 6). Hence, Checkstyle
085 * supports using the convention of using a single {@code {@inheritDoc}} tag
086 * instead of all the other tags.
087 * </p>
088 * <p>
089 * Note that only inheritable items will allow the {@code {@inheritDoc}}
090 * tag to be used in place of comments. Static methods at all visibilities,
091 * private non-static methods and constructors are not inheritable.
092 * </p>
093 * <p>
094 * For example, if the following method is implementing a method required by
095 * an interface, then the Javadoc could be done as:
096 * </p>
097 * <pre>
098 * &#47;** {&#64;inheritDoc} *&#47;
099 * public int checkReturnTag(final int aTagIndex,
100 *                           JavadocTag[] aTags,
101 *                           int aLineNo)
102 * </pre>
103 * <ul>
104 * <li>
105 * Property {@code allowedAnnotations} - Specify annotations that allow missed documentation.
106 * Type is {@code java.lang.String[]}.
107 * Default value is {@code Override}.
108 * </li>
109 * <li>
110 * Property {@code validateThrows} - Control whether to validate {@code throws} tags.
111 * Type is {@code boolean}.
112 * Default value is {@code false}.
113 * </li>
114 * <li>
115 * Property {@code accessModifiers} - Specify the access modifiers where Javadoc comments are
116 * checked.
117 * Type is {@code com.puppycrawl.tools.checkstyle.checks.naming.AccessModifierOption[]}.
118 * Default value is {@code public, protected, package, private}.
119 * </li>
120 * <li>
121 * Property {@code allowMissingParamTags} - Control whether to ignore violations
122 * when a method has parameters but does not have matching {@code param} tags in the javadoc.
123 * Type is {@code boolean}.
124 * Default value is {@code false}.
125 * </li>
126 * <li>
127 * Property {@code allowMissingReturnTag} - Control whether to ignore violations
128 * when a method returns non-void type and does not have a {@code return} tag in the javadoc.
129 * Type is {@code boolean}.
130 * Default value is {@code false}.
131 * </li>
132 * <li>
133 * Property {@code tokens} - tokens to check
134 * Type is {@code java.lang.String[]}.
135 * Validation type is {@code tokenSet}.
136 * Default value is:
137 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#METHOD_DEF">
138 * METHOD_DEF</a>,
139 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CTOR_DEF">
140 * CTOR_DEF</a>,
141 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_FIELD_DEF">
142 * ANNOTATION_FIELD_DEF</a>,
143 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#COMPACT_CTOR_DEF">
144 * COMPACT_CTOR_DEF</a>.
145 * </li>
146 * </ul>
147 * <p>
148 * To configure the default check:
149 * </p>
150 * <pre>
151 * &lt;module name="JavadocMethod"/&gt;
152 * </pre>
153 * <p>Example:</p>
154 * <pre>
155 * public class Test {
156 *
157 *  &#47;**
158 *   *
159 *   *&#47;
160 *  Test(int x) {            // violation, param tag missing for x
161 *  }
162 *
163 *  &#47;**
164 *   *
165 *   *&#47;
166 *  public int foo(int p1) { // violation, param tag missing for p1
167 *    return p1;             // violation, return tag missing
168 *  }
169 *
170 *  &#47;**
171 *   *
172 *   * &#64;param p1 The first number
173 *   *&#47;
174 *  &#64;Deprecated
175 *  private int boo(int p1) {
176 *    return p1;             // violation, return tag missing
177 *  }
178 *
179 *  &#47;**
180 *   *
181 *   *&#47;
182 *  void bar(int p1) {       // violation, param tag missing for p1
183 *  }                        // ok, no return tag for void method
184 * }
185 * </pre>
186 * <p>
187 * To configure the check for only {@code public} modifier, ignoring any missing param tags is:
188 * </p>
189 * <pre>
190 * &lt;module name="JavadocMethod"&gt;
191 *   &lt;property name="accessModifiers" value="public"/&gt;
192 *   &lt;property name="allowMissingParamTags" value="true"/&gt;
193 * &lt;/module&gt;
194 * </pre>
195 * <p>Example:</p>
196 * <pre>
197 * public class Test {
198 *
199 *  &#47;**
200 *   *
201 *   *&#47;
202 *  Test(int x) {            // ok, only public methods checked
203 *  }
204 *
205 *  &#47;**
206 *   *
207 *   *&#47;
208 *  public int foo(int p1) { // ok, missing param tags allowed
209 *    return p1;             // violation, return tag missing
210 *  }
211 *
212 *  &#47;**
213 *   *
214 *   * &#64;param p1 The first number
215 *   *&#47;
216 *  &#64;Deprecated
217 *  private int boo(int p1) {
218 *    return p1;             // ok, only public methods checked
219 *  }
220 *
221 *  &#47;**
222 *   *
223 *   *&#47;
224 *  void bar(int p1) {       // ok, missing param tags allowed
225 *  }                        // ok, no return tag for void method
226 * }
227 * </pre>
228 * <p>
229 * To configure the check for methods which are in {@code private} and {@code package},
230 * but not any other modifier:
231 * </p>
232 * <pre>
233 * &lt;module name="JavadocMethod"&gt;
234 *   &lt;property name="accessModifiers" value="private, package"/&gt;
235 * &lt;/module&gt;
236 * </pre>
237 * <p>Example:</p>
238 * <pre>
239 * class Test {
240 *
241 *  &#47;**
242 *   *
243 *   *&#47;
244 *  Test(int x) {            // violation, param tag missing for x
245 *  }
246 *
247 *  &#47;**
248 *   *
249 *   *&#47;
250 *  public int foo(int p1) { // ok, public methods not checked
251 *    return p1;
252 *  }
253 *
254 *  &#47;**
255 *   *
256 *   * &#64;param p1 The first number
257 *   *&#47;
258 *  &#64;Deprecated
259 *  private int boo(int p1) {
260 *    return p1;             // violation, return tag missing
261 *  }
262 *
263 *  &#47;**
264 *   *
265 *   *&#47;
266 *  void bar(int p1) {       // violation, param tag missing for p1
267 *  }                        // ok, no return tag for void method
268 * }
269 * </pre>
270 * <p>
271 * To configure the check to ignore any missing return tags:
272 * </p>
273 * <pre>
274 * &lt;module name="JavadocMethod"&gt;
275 *   &lt;property name="allowMissingReturnTag" value="true"/&gt;
276 * &lt;/module&gt;
277 * </pre>
278 * <p>Example:</p>
279 * <pre>
280 * public class Test {
281 *
282 *  &#47;**
283 *   *
284 *   *&#47;
285 *  Test(int x) {            // violation, param tag missing for x
286 *  }
287 *
288 *  &#47;**
289 *   *
290 *   *&#47;
291 *  public int foo(int p1) { // violation, param tag missing for p1
292 *    return p1;             // ok, missing return tag allowed
293 *  }
294 *
295 *  &#47;**
296 *   *
297 *   * &#64;param p1 The first number
298 *   *&#47;
299 *  &#64;Deprecated
300 *  private int boo(int p1) {
301 *    return p1;             // ok, missing return tag allowed
302 *  }
303 *
304 *  &#47;**
305 *   *
306 *   *&#47;
307 *  void bar(int p1) {       // violation, param tag missing for p1
308 *  }                        // ok, no return tag for void method
309 * }
310 * </pre>
311 * <p>
312 *  To configure the check to ignore Methods with annotation {@code Deprecated}:
313 * </p>
314 * <pre>
315 * &lt;module name="JavadocMethod"&gt;
316 *   &lt;property name="allowedAnnotations" value="Deprecated"/&gt;
317 * &lt;/module&gt;
318 * </pre>
319 * <p>Example:</p>
320 * <pre>
321 * public class Test {
322 *
323 *  &#47;**
324 *   *
325 *   *&#47;
326 *  Test(int x) {            // violation, param tag missing for x
327 *  }
328 *
329 *  &#47;**
330 *   *
331 *   *&#47;
332 *  public int foo(int p1) { // violation, param tag missing for p1
333 *    return p1;             // violation, return tag missing
334 *  }
335 *
336 *  &#47;**
337 *   *
338 *   * &#64;param p1 The first number
339 *   *&#47;
340 *  &#64;Deprecated
341 *  private int boo(int p1) {
342 *    return p1;             // ok, Deprecated methods not checked
343 *  }
344 *
345 *  &#47;**
346 *   *
347 *   *&#47;
348 *  void bar(int p1) {       // violation, param tag missing for p1
349 *  }                        // ok, no return tag for void method
350 * }
351 * </pre>
352 * <p>
353 *     To configure the check only for tokens which are Constructor Definitions:
354 * </p>
355 * <pre>
356 * &lt;module name="JavadocMethod"&gt;
357 *   &lt;property name="tokens" value="CTOR_DEF"/&gt;
358 * &lt;/module&gt;
359 * </pre>
360 * <p>Example:</p>
361 * <pre>
362 * public class Test {
363 *
364 *  &#47;**
365 *   *
366 *   *&#47;
367 *  Test(int x) {            // violation, param tag missing for x
368 *  }
369 *
370 *  &#47;**
371 *   *
372 *   *&#47;
373 *  public int foo(int p1) { // ok, method not checked
374 *    return p1;             // ok, method not checked
375 *  }
376 *
377 *  &#47;**
378 *   *
379 *   * &#64;param p1 The first number
380 *   *&#47;
381 *  &#64;Deprecated
382 *  private int boo(int p1) {
383 *    return p1;             // ok, method not checked
384 *  }
385 *
386 *  &#47;**
387 *   *
388 *   *&#47;
389 *  void bar(int p1) {       // ok, method not checked
390 *  }
391 * }
392 * </pre>
393 * <p>
394 * To configure the check to validate {@code throws} tags, you can use following config.
395 * </p>
396 * <pre>
397 * &lt;module name="JavadocMethod"&gt;
398 *   &lt;property name="validateThrows" value="true"/&gt;
399 * &lt;/module&gt;
400 * </pre>
401 * <p>Example:</p>
402 * <pre>
403 * &#47;**
404 *  * Actual exception thrown is child class of class that is declared in throws.
405 *  * It is limitation of checkstyle (as checkstyle does not know type hierarchy).
406 *  * Javadoc is valid not declaring FileNotFoundException
407 *  * BUT checkstyle can not distinguish relationship between exceptions.
408 *  * &#64;param file some file
409 *  * &#64;throws IOException if some problem
410 *  *&#47;
411 * public void doSomething8(File file) throws IOException {
412 *     if (file == null) {
413 *         throw new FileNotFoundException(); // violation
414 *     }
415 * }
416 *
417 * &#47;**
418 *  * Exact throw type referencing in javadoc even first is parent of second type.
419 *  * It is a limitation of checkstyle (as checkstyle does not know type hierarchy).
420 *  * This javadoc is valid for checkstyle and for javadoc tool.
421 *  * &#64;param file some file
422 *  * &#64;throws IOException if some problem
423 *  * &#64;throws FileNotFoundException if file is not found
424 *  *&#47;
425 * public void doSomething9(File file) throws IOException {
426 *     if (file == null) {
427 *         throw new FileNotFoundException();
428 *     }
429 * }
430 *
431 * &#47;**
432 *  * Ignore try block, but keep catch and finally blocks.
433 *  *
434 *  * &#64;param s String to parse
435 *  * &#64;return A positive integer
436 *  *&#47;
437 * public int parsePositiveInt(String s) {
438 *     try {
439 *         int value = Integer.parseInt(s);
440 *         if (value &lt;= 0) {
441 *             throw new NumberFormatException(value + " is negative/zero"); // ok, try
442 *         }
443 *         return value;
444 *     } catch (NumberFormatException ex) {
445 *         throw new IllegalArgumentException("Invalid number", ex); // violation, catch
446 *     } finally {
447 *         throw new IllegalStateException("Should never reach here"); // violation, finally
448 *     }
449 * }
450 *
451 * &#47;**
452 *  * Try block without catch is not ignored.
453 *  *
454 *  * &#64;return a String from standard input, if there is one
455 *  *&#47;
456 * public String readLine() {
457 *     try (Scanner sc = new Scanner(System.in)) {
458 *         if (!sc.hasNext()) {
459 *             throw new IllegalStateException("Empty input"); // violation, not caught
460 *         }
461 *         return sc.next();
462 *     }
463 * }
464 *
465 * &#47;**
466 *  * Lambda expressions are ignored as we do not know when the exception will be thrown.
467 *  *
468 *  * &#64;param s a String to be printed at some point in the future
469 *  * &#64;return a Runnable to be executed when the string is to be printed
470 *  *&#47;
471 * public Runnable printLater(String s) {
472 *     return () -&gt; {
473 *         if (s == null) {
474 *             throw new NullPointerException(); // ok
475 *         }
476 *         System.out.println(s);
477 *     };
478 * }
479 * </pre>
480 * <p>
481 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
482 * </p>
483 * <p>
484 * Violation Message Keys:
485 * </p>
486 * <ul>
487 * <li>
488 * {@code javadoc.classInfo}
489 * </li>
490 * <li>
491 * {@code javadoc.duplicateTag}
492 * </li>
493 * <li>
494 * {@code javadoc.expectedTag}
495 * </li>
496 * <li>
497 * {@code javadoc.invalidInheritDoc}
498 * </li>
499 * <li>
500 * {@code javadoc.return.expected}
501 * </li>
502 * <li>
503 * {@code javadoc.unusedTag}
504 * </li>
505 * <li>
506 * {@code javadoc.unusedTagGeneral}
507 * </li>
508 * </ul>
509 *
510 * @since 3.0
511 */

702    @Override
703    public final void visitToken(DetailAST ast) {
704        if (ast.getType() == TokenTypes.CLASS_DEF
705                 || ast.getType() == TokenTypes.INTERFACE_DEF
706                 || ast.getType() == TokenTypes.ENUM_DEF
707                 || ast.getType() == TokenTypes.RECORD_DEF) {
708            processClass(ast);
709        }
710        else {
711            processAST(ast);
712        }

746    /**
747     * Whether we should check this node.
748     *
749     * @param ast a given node.
750     * @return whether we should check a given node.
751     */
752    private boolean shouldCheck(final DetailAST ast) {
753        final AccessModifierOption surroundingAccessModifier = CheckUtil
754                .getSurroundingAccessModifier(ast);
755        final AccessModifierOption accessModifier = CheckUtil
756                .getAccessModifierFromModifiersToken(ast);
757        return surroundingAccessModifier != null
758                && Arrays.stream(accessModifiers)
759                        .anyMatch(modifier -> modifier == surroundingAccessModifier)
760                && Arrays.stream(accessModifiers).anyMatch(modifier -> modifier == accessModifier);
761    }

763    /**
764     * Checks the Javadoc for a method.
765     *
766     * @param ast the token for the method
767     * @param comment the Javadoc comment
768     */
769    private void checkComment(DetailAST ast, TextBlock comment) {
770        final List<JavadocTag> tags = getMethodTags(comment);
771
772        if (!hasShortCircuitTag(ast, tags)) {
773            if (ast.getType() == TokenTypes.ANNOTATION_FIELD_DEF) {
774                checkReturnTag(tags, ast.getLineNo(), true);
775            }
776            else {
777                final Iterator<JavadocTag> it = tags.iterator();
778                // Check for inheritDoc
779                boolean hasInheritDocTag = false;
780                while (!hasInheritDocTag && it.hasNext()) {
781                    hasInheritDocTag = it.next().isInheritDocTag();
782                }
783                final boolean reportExpectedTags = !hasInheritDocTag
784                    && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
785
786                // COMPACT_CTOR_DEF has no parameters
787                if (ast.getType() != TokenTypes.COMPACT_CTOR_DEF) {
788                    checkParamTags(tags, ast, reportExpectedTags);
789                }
790                final List<ExceptionInfo> throwed =
791                    combineExceptionInfo(getThrows(ast), getThrowed(ast));
792                checkThrowsTags(tags, throwed, reportExpectedTags);
793                if (CheckUtil.isNonVoidMethod(ast)) {
794                    checkReturnTag(tags, ast.getLineNo(), reportExpectedTags);
795                }
796
797            }
798
799            // Dump out all unused tags
800            tags.stream().filter(javadocTag -> !javadocTag.isSeeOrInheritDocTag())
801                .forEach(javadocTag -> log(javadocTag.getLineNo(), MSG_UNUSED_TAG_GENERAL));
802        }
803    }

805    /**
806     * Validates whether the Javadoc has a short circuit tag. Currently, this is
807     * the inheritTag. Any violations are logged.
808     *
809     * @param ast the construct being checked
810     * @param tags the list of Javadoc tags associated with the construct
811     * @return true if the construct has a short circuit tag.
812     */
813    private boolean hasShortCircuitTag(final DetailAST ast, final List<JavadocTag> tags) {
814        boolean result = true;
815        // Check if it contains {@inheritDoc} tag
816        if (tags.size() == 1
817                && tags.get(0).isInheritDocTag()) {
818            // Invalid if private, a constructor, or a static method
819            if (!JavadocTagInfo.INHERIT_DOC.isValidOn(ast)) {
820                log(ast, MSG_INVALID_INHERIT_DOC);
821            }
822        }
823        else {
824            result = false;
825        }
826        return result;
827    }

829    /**
830     * Returns the tags in a javadoc comment. Only finds throws, exception,
831     * param, return and see tags.
832     *
833     * @param comment the Javadoc comment
834     * @return the tags found
835     */
836    private static List<JavadocTag> getMethodTags(TextBlock comment) {
837        final String[] lines = comment.getText();
838        final List<JavadocTag> tags = new ArrayList<>();
839        int currentLine = comment.getStartLineNo() - 1;
840        final int startColumnNumber = comment.getStartColNo();
841
842        for (int i = 0; i < lines.length; i++) {
843            currentLine++;
844            final Matcher javadocArgMatcher =
845                MATCH_JAVADOC_ARG.matcher(lines[i]);
846            final Matcher javadocArgMissingDescriptionMatcher =
847                MATCH_JAVADOC_ARG_MISSING_DESCRIPTION.matcher(lines[i]);
848            final Matcher javadocNoargMatcher =
849                MATCH_JAVADOC_NOARG.matcher(lines[i]);
850            final Matcher noargCurlyMatcher =
851                MATCH_JAVADOC_NOARG_CURLY.matcher(lines[i]);
852            final Matcher noargMultilineStart =
853                MATCH_JAVADOC_NOARG_MULTILINE_START.matcher(lines[i]);
854
855            if (javadocArgMatcher.find()) {
856                final int col = calculateTagColumn(javadocArgMatcher, i, startColumnNumber);
857                tags.add(new JavadocTag(currentLine, col, javadocArgMatcher.group(1),
858                        javadocArgMatcher.group(2)));
859            }
860            else if (javadocArgMissingDescriptionMatcher.find()) {
861                final int col = calculateTagColumn(javadocArgMissingDescriptionMatcher, i,
862                    startColumnNumber);
863                tags.add(new JavadocTag(currentLine, col,
864                    javadocArgMissingDescriptionMatcher.group(1),
865                    javadocArgMissingDescriptionMatcher.group(2)));
866            }
867            else if (javadocNoargMatcher.find()) {
868                final int col = calculateTagColumn(javadocNoargMatcher, i, startColumnNumber);
869                tags.add(new JavadocTag(currentLine, col, javadocNoargMatcher.group(1)));
870            }
871            else if (noargCurlyMatcher.find()) {
872                final int col = calculateTagColumn(noargCurlyMatcher, i, startColumnNumber);
873                tags.add(new JavadocTag(currentLine, col, noargCurlyMatcher.group(1)));
874            }
875            else if (noargMultilineStart.find()) {
876                tags.addAll(getMultilineNoArgTags(noargMultilineStart, lines, i, currentLine));
877            }
878        }
879        return tags;
880    }

882    /**
883     * Calculates column number using Javadoc tag matcher.
884     *
885     * @param javadocTagMatchResult found javadoc tag match result
886     * @param lineNumber line number of Javadoc tag in comment
887     * @param startColumnNumber column number of Javadoc comment beginning
888     * @return column number
889     */
890    private static int calculateTagColumn(MatchResult javadocTagMatchResult,
891            int lineNumber, int startColumnNumber) {
892        int col = javadocTagMatchResult.start(1) - 1;
893        if (lineNumber == 0) {
894            col += startColumnNumber;
895        }
896        return col;
897    }

899    /**
900     * Gets multiline Javadoc tags with no arguments.
901     *
902     * @param noargMultilineStart javadoc tag Matcher
903     * @param lines comment text lines
904     * @param lineIndex line number that contains the javadoc tag
905     * @param tagLine javadoc tag line number in file
906     * @return javadoc tags with no arguments
907     */
908    private static List<JavadocTag> getMultilineNoArgTags(final Matcher noargMultilineStart,
909            final String[] lines, final int lineIndex, final int tagLine) {
910        int remIndex = lineIndex;
911        Matcher multilineCont;
912
913        do {
914            remIndex++;
915            multilineCont = MATCH_JAVADOC_MULTILINE_CONT.matcher(lines[remIndex]);
916        } while (!multilineCont.find());
917
918        final List<JavadocTag> tags = new ArrayList<>();
919        final String lFin = multilineCont.group(1);
920        if (!NEXT_TAG.equals(lFin)
921            && !END_JAVADOC.equals(lFin)) {
922            final String param1 = noargMultilineStart.group(1);
923            final int col = noargMultilineStart.start(1) - 1;
924
925            tags.add(new JavadocTag(tagLine, col, param1));
926        }
927
928        return tags;
929    }

979     *

981     * @return list of ExceptionInfo
982     */
983    private static List<ExceptionInfo> getThrowed(DetailAST methodAst) {
984        final List<ExceptionInfo> returnValue = new ArrayList<>();
985        final DetailAST blockAst = methodAst.findFirstToken(TokenTypes.SLIST);

993                        final DetailAST child = newAst.getFirstChild();
994                        returnValue.add(getExceptionInfo(child));
995                    }
996                }
997            }
998        }
999        return returnValue;

1029    /**
1030     * Checks if a 'throw' usage is contained within a block that should be ignored.
1031     * Such blocks consist of try (with catch) blocks, local classes, anonymous classes,
1032     * and lambda expressions. Note that a try block without catch is not considered.
1033     *
1034     * @param methodBodyAst DetailAST node representing the method body
1035     * @param throwAst DetailAST node representing the 'throw' literal
1036     * @return true if throwAst is inside a block that should be ignored
1037     */
1038    private static boolean isInIgnoreBlock(DetailAST methodBodyAst, DetailAST throwAst) {
1039        DetailAST ancestor = throwAst.getParent();
1040        while (ancestor != methodBodyAst) {
1041            if (ancestor.getType() == TokenTypes.LITERAL_TRY
1042                    && ancestor.findFirstToken(TokenTypes.LITERAL_CATCH) != null
1043                    || ancestor.getType() == TokenTypes.LAMBDA
1044                    || ancestor.getType() == TokenTypes.OBJBLOCK) {
1045                // throw is inside a try block, and there is a catch block,
1046                // or throw is inside a lambda expression/anonymous class/local class
1047                break;
1048            }
1049            if (ancestor.getType() == TokenTypes.LITERAL_CATCH
1050                    || ancestor.getType() == TokenTypes.LITERAL_FINALLY) {
1051                // if the throw is inside a catch or finally block,
1052                // skip the immediate ancestor (try token)
1053                ancestor = ancestor.getParent();
1054            }
1055            ancestor = ancestor.getParent();
1056        }
1057        return ancestor != methodBodyAst;
1058    }

1078    /**
1079     * Finds node of specified type among root children, siblings, siblings children
1080     * on any deep level.
1081     *
1082     * @param root    DetailAST
1083     * @param astType value of TokenType
1084     * @return {@link List} of {@link DetailAST} nodes which matches the predicate.
1085     */
1086    public static List<DetailAST> findTokensInAstByType(DetailAST root, int astType) {
1087        final List<DetailAST> result = new ArrayList<>();
1088        // iterative preorder depth-first search
1089        DetailAST curNode = root;
1090        do {
1091            // process curNode
1092            if (curNode.getType() == astType) {
1093                result.add(curNode);
1094            }
1095            // process children (if any)
1096            if (curNode.hasChildren()) {
1097                curNode = curNode.getFirstChild();
1098                continue;
1099            }
1100            // backtrack to parent if last child, stopping at root
1101            while (curNode != root && curNode.getNextSibling() == null) {
1102                curNode = curNode.getParent();
1103            }
1104            // explore siblings if not root
1105            if (curNode != root) {
1106                curNode = curNode.getNextSibling();
1107            }
1108        } while (curNode != root);
1109        return result;
1110    }

1112    /**
1113     * Checks a set of tags for matching parameters.
1114     *
1115     * @param tags the tags to check
1116     * @param parent the node which takes the parameters
1117     * @param reportExpectedTags whether we should report if do not find
1118     *            expected tag
1119     */
1120    private void checkParamTags(final List<JavadocTag> tags,
1121            final DetailAST parent, boolean reportExpectedTags) {
1122        final List<DetailAST> params = getParameters(parent);
1123        final List<DetailAST> typeParams = CheckUtil
1124                .getTypeParameters(parent);
1125
1126        // Loop over the tags, checking to see they exist in the params.
1127        final ListIterator<JavadocTag> tagIt = tags.listIterator();
1128        while (tagIt.hasNext()) {
1129            final JavadocTag tag = tagIt.next();
1130
1131            if (!tag.isParamTag()) {
1132                continue;
1133            }
1134
1135            tagIt.remove();
1136
1137            final String arg1 = tag.getFirstArg();
1138            boolean found = removeMatchingParam(params, arg1);
1139
1140            if (CommonUtil.startsWithChar(arg1, '<') && CommonUtil.endsWithChar(arg1, '>')) {
1141                found = searchMatchingTypeParameter(typeParams,
1142                        arg1.substring(1, arg1.length() - 1));
1143            }
1144
1145            // Handle extra JavadocTag
1146            if (!found) {
1147                log(tag.getLineNo(), tag.getColumnNo(), MSG_UNUSED_TAG,
1148                        "@param", arg1);
1149            }
1150        }
1151
1152        // Now dump out all type parameters/parameters without tags :- unless
1153        // the user has chosen to suppress these problems
1154        if (!allowMissingParamTags && reportExpectedTags) {
1155            for (DetailAST param : params) {
1156                log(param, MSG_EXPECTED_TAG,
1157                    JavadocTagInfo.PARAM.getText(), param.getText());
1158            }
1159
1160            for (DetailAST typeParam : typeParams) {
1161                log(typeParam, MSG_EXPECTED_TAG,
1162                    JavadocTagInfo.PARAM.getText(),
1163                    "<" + typeParam.findFirstToken(TokenTypes.IDENT).getText()
1164                    + ">");
1165            }
1166        }
1167    }

1251    /**
1252     * Checks a set of tags for matching throws.
1253     *
1254     * @param tags the tags to check
1255     * @param throwsList the throws to check
1256     * @param reportExpectedTags whether we should report if do not find
1257     *            expected tag
1258     */
1259    private void checkThrowsTags(List<JavadocTag> tags,
1260            List<ExceptionInfo> throwsList, boolean reportExpectedTags) {
1261        // Loop over the tags, checking to see they exist in the throws.
1262        // The foundThrows used for performance only
1263        final Set<String> foundThrows = new HashSet<>();
1264        final ListIterator<JavadocTag> tagIt = tags.listIterator();
1265        while (tagIt.hasNext()) {
1266            final JavadocTag tag = tagIt.next();
1267
1268            if (!tag.isThrowsTag()) {
1269                continue;
1270            }
1271            tagIt.remove();
1272
1273            // Loop looking for matching throw
1274            final Token token = new Token(tag.getFirstArg(), tag.getLineNo(), tag
1275                    .getColumnNo());
1276            final ClassInfo documentedClassInfo = new ClassInfo(token);
1277            processThrows(throwsList, documentedClassInfo, foundThrows);
1278        }
1279        // Now dump out all throws without tags :- unless
1280        // the user has chosen to suppress these problems
1281        if (validateThrows && reportExpectedTags) {
1282            throwsList.stream().filter(exceptionInfo -> !exceptionInfo.isFound())
1283                .forEach(exceptionInfo -> {
1284                    final Token token = exceptionInfo.getName();
1285                    log(exceptionInfo.getAst(),
1286                        MSG_EXPECTED_TAG,
1287                        JavadocTagInfo.THROWS.getText(), token.getText());
1288                });
1289        }
1290    }

1292    /**
1293     * Verifies that documented exception is in throws.
1294     *
1295     * @param throwsIterable collection of throws
1296     * @param documentedClassInfo documented exception class info
1297     * @param foundThrows previously found throws
1298     */
1299    private static void processThrows(Iterable<ExceptionInfo> throwsIterable,
1300                                      ClassInfo documentedClassInfo, Set<String> foundThrows) {
1301        ExceptionInfo foundException = null;
1302
1303        // First look for matches on the exception name
1304        for (ExceptionInfo exceptionInfo : throwsIterable) {
1305            if (isClassNamesSame(exceptionInfo.getName().getText(),
1306                    documentedClassInfo.getName().getText())) {
1307                foundException = exceptionInfo;
1308                break;
1309            }
1310        }
1311
1312        if (foundException != null) {
1313            foundException.setFound();
1314            foundThrows.add(documentedClassInfo.getName().getText());
1315        }
1316    }

1330    /**
1331     * Check that class names are same by short name of class. If some class name is fully
1332     * qualified it is cut to short name.
1333     *
1334     * @param class1 class name
1335     * @param class2 class name
1336     * @return true is ExceptionInfo object have the same name
1337     */
1338    private static boolean isClassNamesSame(String class1, String class2) {
1339        boolean result = false;
1340        if (class1.equals(class2)) {
1341            result = true;
1342        }
1343        else {
1344            final String separator = ".";
1345            if (class1.contains(separator) || class2.contains(separator)) {
1346                final String class1ShortName = class1
1347                        .substring(class1.lastIndexOf('.') + 1);
1348                final String class2ShortName = class2
1349                        .substring(class2.lastIndexOf('.') + 1);
1350                result = class1ShortName.equals(class2ShortName);
1351            }
1352        }
1353        return result;
1354    }

1360     */
1361    private void processClass(DetailAST ast) {
1362        final DetailAST ident = ast.findFirstToken(TokenTypes.IDENT);
1363        String innerClass = ident.getText();
1364
1365        innerClass = "$" + innerClass;
1366        currentClassName += innerClass;
1367    }

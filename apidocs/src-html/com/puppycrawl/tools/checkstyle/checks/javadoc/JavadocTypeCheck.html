041/**
042 * <p>
043 * Checks the Javadoc comments for annotation/enum/class/interface definitions. By default, does
044 * not check for author or version tags. The scope to verify is specified using the {@code Scope}
045 * class and defaults to {@code Scope.PRIVATE}. To verify another scope, set property
046 * scope to one of the {@code Scope} constants. To define the format for an author
047 * tag or a version tag, set property authorFormat or versionFormat respectively to a
048 * <a href="https://docs.oracle.com/javase/7/docs/api/java/util/regex/Pattern.html">
049 * regular expression</a>.
050 * </p>
051 * <p>
052 * Does not perform checks for author and version tags for inner classes,
053 * as they should be redundant because of outer class.
054 * </p>
055 * <p>
056 * Error messages about type parameters for which no param tags are present
057 * can be suppressed by defining property {@code allowMissingParamTags}.
058 * </p>
059 * <ul>
060 * <li>
061 * Property {@code scope} - Specify the visibility scope where Javadoc comments are checked.
062 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
063 * Default value is {@code private}.
064 * </li>
065 * <li>
066 * Property {@code excludeScope} - Specify the visibility scope where Javadoc
067 * comments are not checked.
068 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
069 * Default value is {@code null}.
070 * </li>
071 * <li>
072 * Property {@code authorFormat} - Specify the pattern for {@code @author} tag.
073 * Type is {@code java.util.regex.Pattern}.
074 * Default value is {@code null}.
075 * </li>
076 * <li>
077 * Property {@code versionFormat} - Specify the pattern for {@code @version} tag.
078 * Type is {@code java.util.regex.Pattern}.
079 * Default value is {@code null}.
080 * </li>
081 * <li>
082 * Property {@code allowMissingParamTags} - Control whether to ignore violations
083 * when a class has type parameters but does not have matching param tags in the Javadoc.
084 * Type is {@code boolean}.
085 * Default value is {@code false}.
086 * </li>
087 * <li>
088 * Property {@code allowUnknownTags} - Control whether to ignore violations when
089 * a Javadoc tag is not recognised.
090 * Type is {@code boolean}.
091 * Default value is {@code false}.
092 * </li>
093 * <li>
094 * Property {@code allowedAnnotations} - Specify the list of annotations that allow
095 * missed documentation. Only short names are allowed, e.g. {@code Generated}.
096 * Type is {@code java.lang.String[]}.
097 * Default value is {@code Generated}.
098 * </li>
099 * <li>
100 * Property {@code tokens} - tokens to check
101 * Type is {@code java.lang.String[]}.
102 * Validation type is {@code tokenSet}.
103 * Default value is:
104 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
105 * INTERFACE_DEF</a>,
106 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
107 * CLASS_DEF</a>,
108 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
109 * ENUM_DEF</a>,
110 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_DEF">
111 * ANNOTATION_DEF</a>,
112 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#RECORD_DEF">
113 * RECORD_DEF</a>.
114 * </li>
115 * </ul>
116 * <p>
117 * To configure the default check:
118 * </p>
119 * <pre>
120 * &lt;module name="JavadocType"/&gt;
121 * </pre>
122 * <p>
123 * To configure the check for {@code public} scope:
124 * </p>
125 * <pre>
126 * &lt;module name="JavadocType"&gt;
127 *   &lt;property name="scope" value="public"/&gt;
128 * &lt;/module&gt;
129 * </pre>
130 * <p>
131 * To configure the check for an {@code @author} tag:
132 * </p>
133 * <pre>
134 * &lt;module name="JavadocType"&gt;
135 *   &lt;property name="authorFormat" value="\S"/&gt;
136 * &lt;/module&gt;
137 * </pre>
138 * <p>
139 * To configure the check for a CVS revision version tag:
140 * </p>
141 * <pre>
142 * &lt;module name="JavadocType"&gt;
143 *   &lt;property name="versionFormat" value="\$Revision.*\$"/&gt;
144 * &lt;/module&gt;
145 * </pre>
146 * <p>
147 * To configure the check for {@code private} classes only:
148 * </p>
149 * <pre>
150 * &lt;module name="JavadocType"&gt;
151 *   &lt;property name="scope" value="private"/&gt;
152 *   &lt;property name="excludeScope" value="package"/&gt;
153 * &lt;/module&gt;
154 * </pre>
155 * <p>
156 * Example that allows missing comments for classes annotated with
157 * {@code @SpringBootApplication} and {@code @Configuration}:
158 * </p>
159 * <pre>
160 * &#64;SpringBootApplication // no violations about missing comment on class
161 * public class Application {}
162 *
163 * &#64;Configuration // no violations about missing comment on class
164 * class DatabaseConfiguration {}
165 * </pre>
166 * <p>
167 * Use following configuration:
168 * </p>
169 * <pre>
170 * &lt;module name="JavadocType"&gt;
171 *   &lt;property name="allowedAnnotations" value="SpringBootApplication,Configuration"/&gt;
172 * &lt;/module&gt;
173 * </pre>
174 * <p>
175 * Parent is {@code com.puppycrawl.tools.checkstyle.TreeWalker}
176 * </p>
177 * <p>
178 * Violation Message Keys:
179 * </p>
180 * <ul>
181 * <li>
182 * {@code javadoc.unknownTag}
183 * </li>
184 * <li>
185 * {@code javadoc.unusedTag}
186 * </li>
187 * <li>
188 * {@code javadoc.unusedTagGeneral}
189 * </li>
190 * <li>
191 * {@code type.missingTag}
192 * </li>
193 * <li>
194 * {@code type.tagFormat}
195 * </li>
196 * </ul>
197 *
198 * @since 3.0
199 *
200 */
201@StatelessCheck
202public class JavadocTypeCheck
203    extends AbstractCheck {

235    /** Open angle bracket literal. */
236    private static final String OPEN_ANGLE_BRACKET = "<";
237
238    /** Close angle bracket literal. */
239    private static final String CLOSE_ANGLE_BRACKET = ">";
240
241    /** Pattern to match type name within angle brackets in javadoc param tag. */
242    private static final Pattern TYPE_NAME_IN_JAVADOC_TAG =
243            Pattern.compile("\\s*<([^>]+)>.*");
244
245    /** Pattern to split type name field in javadoc param tag. */
246    private static final Pattern TYPE_NAME_IN_JAVADOC_TAG_SPLITTER =
247            Pattern.compile("\\s+");
248
249    /** Specify the visibility scope where Javadoc comments are checked. */
250    private Scope scope = Scope.PRIVATE;
251    /** Specify the visibility scope where Javadoc comments are not checked. */
252    private Scope excludeScope;
253    /** Specify the pattern for {@code @author} tag. */
254    private Pattern authorFormat;
255    /** Specify the pattern for {@code @version} tag. */
256    private Pattern versionFormat;

258     * Control whether to ignore violations when a class has type parameters but
259     * does not have matching param tags in the Javadoc.

261    private boolean allowMissingParamTags;
262    /** Control whether to ignore violations when a Javadoc tag is not recognised. */
263    private boolean allowUnknownTags;

357    @Override
358    public void visitToken(DetailAST ast) {
359        if (shouldCheck(ast)) {
360            final FileContents contents = getFileContents();
361            final int lineNo = ast.getLineNo();
362            final TextBlock textBlock = contents.getJavadocBefore(lineNo);
363            if (textBlock != null) {
364                final List<JavadocTag> tags = getJavadocTags(textBlock);
365                if (ScopeUtil.isOuterMostType(ast)) {
366                    // don't check author/version for inner classes
367                    checkTag(ast, tags, JavadocTagInfo.AUTHOR.getName(),
368                            authorFormat);
369                    checkTag(ast, tags, JavadocTagInfo.VERSION.getName(),
370                            versionFormat);
371                }
372
373                final List<String> typeParamNames =
374                    CheckUtil.getTypeParameterNames(ast);
375
376                if (!allowMissingParamTags) {
377                    // Check type parameters that should exist, do
378                    for (final String typeParamName : typeParamNames) {
379                        checkTypeParamTag(
380                            ast, tags, typeParamName);
381                    }
382                }
383
384                checkUnusedTypeParamTags(tags, typeParamNames);
385            }
386        }
387    }
388
389    /**
390     * Whether we should check this node.
391     *
392     * @param ast a given node.
393     * @return whether we should check a given node.
394     */
395    private boolean shouldCheck(final DetailAST ast) {
396        final Scope customScope;
397
398        if (ScopeUtil.isInInterfaceOrAnnotationBlock(ast)) {
399            customScope = Scope.PUBLIC;
400        }
401        else {
402            final DetailAST mods = ast.findFirstToken(TokenTypes.MODIFIERS);
403            customScope = ScopeUtil.getScopeFromMods(mods);
404        }
405        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
406
407        return customScope.isIn(scope)
408            && (surroundingScope == null || surroundingScope.isIn(scope))
409            && (excludeScope == null
410                || !customScope.isIn(excludeScope)
411                || surroundingScope != null
412                && !surroundingScope.isIn(excludeScope))
413            && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
414    }
415
416    /**
417     * Gets all standalone tags from a given javadoc.
418     *
419     * @param textBlock the Javadoc comment to process.
420     * @return all standalone tags from the given javadoc.
421     */
422    private List<JavadocTag> getJavadocTags(TextBlock textBlock) {
423        final JavadocTags tags = JavadocUtil.getJavadocTags(textBlock,
424            JavadocUtil.JavadocTagType.BLOCK);
425        if (!allowUnknownTags) {
426            for (final InvalidJavadocTag tag : tags.getInvalidTags()) {
427                log(tag.getLine(), tag.getCol(), MSG_UNKNOWN_TAG,
428                    tag.getName());
429            }
430        }
431        return tags.getValidTags();
432    }
433
434    /**
435     * Verifies that a type definition has a required tag.
436     *
437     * @param ast the AST node for the type definition.
438     * @param tags tags from the Javadoc comment for the type definition.
439     * @param tagName the required tag name.
440     * @param formatPattern regexp for the tag value.
441     */
442    private void checkTag(DetailAST ast, List<JavadocTag> tags, String tagName,
443                          Pattern formatPattern) {
444        if (formatPattern != null) {
445            boolean hasTag = false;
446            final String tagPrefix = "@";
447            for (int i = tags.size() - 1; i >= 0; i--) {
448                final JavadocTag tag = tags.get(i);
449                if (tag.getTagName().equals(tagName)) {
450                    hasTag = true;
451                    if (!formatPattern.matcher(tag.getFirstArg()).find()) {
452                        log(ast, MSG_TAG_FORMAT, tagPrefix + tagName, formatPattern.pattern());
453                    }
454                }
455            }
456            if (!hasTag) {
457                log(ast, MSG_MISSING_TAG, tagPrefix + tagName);
458            }
459        }
460    }
461
462    /**
463     * Verifies that a type definition has the specified param tag for
464     * the specified type parameter name.
465     *
466     * @param ast the AST node for the type definition.
467     * @param tags tags from the Javadoc comment for the type definition.
468     * @param typeParamName the name of the type parameter
469     */
470    private void checkTypeParamTag(final DetailAST ast,
471            final List<JavadocTag> tags, final String typeParamName) {
472        boolean found = false;
473        for (int i = tags.size() - 1; i >= 0; i--) {
474            final JavadocTag tag = tags.get(i);
475            if (tag.isParamTag()
476                && tag.getFirstArg().indexOf(OPEN_ANGLE_BRACKET
477                        + typeParamName + CLOSE_ANGLE_BRACKET) == 0) {
478                found = true;
479                break;
480            }
481        }
482        if (!found) {
483            log(ast, MSG_MISSING_TAG, JavadocTagInfo.PARAM.getText()
484                + " " + OPEN_ANGLE_BRACKET + typeParamName + CLOSE_ANGLE_BRACKET);
485        }
486    }
487
488    /**
489     * Checks for unused param tags for type parameters.
490     *
491     * @param tags tags from the Javadoc comment for the type definition.
492     * @param typeParamNames names of type parameters
493     */
494    private void checkUnusedTypeParamTags(
495        final List<JavadocTag> tags,
496        final List<String> typeParamNames) {
497        for (int i = tags.size() - 1; i >= 0; i--) {
498            final JavadocTag tag = tags.get(i);
499            if (tag.isParamTag()) {
500                final String typeParamName = extractTypeParamNameFromTag(tag);
501
502                if (!typeParamNames.contains(typeParamName)) {
503                    log(tag.getLineNo(), tag.getColumnNo(),
504                            MSG_UNUSED_TAG,
505                            JavadocTagInfo.PARAM.getText(),
506                            OPEN_ANGLE_BRACKET + typeParamName + CLOSE_ANGLE_BRACKET);
507                }
508            }
509        }
510    }
511
512    /**
513     * Extracts type parameter name from tag.
514     *
515     * @param tag javadoc tag to extract parameter name
516     * @return extracts type parameter name from tag
517     */
518    private static String extractTypeParamNameFromTag(JavadocTag tag) {
519        final String typeParamName;
520        final Matcher matchInAngleBrackets =
521                TYPE_NAME_IN_JAVADOC_TAG.matcher(tag.getFirstArg());
522        if (matchInAngleBrackets.find()) {
523            typeParamName = matchInAngleBrackets.group(1).trim();
524        }
525        else {
526            typeParamName = TYPE_NAME_IN_JAVADOC_TAG_SPLITTER.split(tag.getFirstArg())[0];
527        }
528        return typeParamName;
529    }

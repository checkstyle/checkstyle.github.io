043/**
044 * <p>
045 * Checks the Javadoc comments for type definitions. By default, does
046 * not check for author or version tags. The scope to verify is specified using the {@code Scope}
047 * class and defaults to {@code Scope.PRIVATE}. To verify another scope, set property
048 * scope to one of the {@code Scope} constants. To define the format for an author
049 * tag or a version tag, set property authorFormat or versionFormat respectively to a
050 * <a href="https://docs.oracle.com/javase/7/docs/api/java/util/regex/Pattern.html">
051 * pattern</a>.
052 * </p>
053 * <p>
054 * Does not perform checks for author and version tags for inner classes,
055 * as they should be redundant because of outer class.
056 * </p>
057 * <p>
058 * Does not perform checks for type definitions that do not have any Javadoc comments.
059 * </p>
060 * <p>
061 * Error messages about type parameters and record components for which no param tags are present
062 * can be suppressed by defining property {@code allowMissingParamTags}.
063 * </p>
064 * <ul>
065 * <li>
066 * Property {@code scope} - Specify the visibility scope where Javadoc comments are checked.
067 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
068 * Default value is {@code private}.
069 * </li>
070 * <li>
071 * Property {@code excludeScope} - Specify the visibility scope where Javadoc
072 * comments are not checked.
073 * Type is {@code com.puppycrawl.tools.checkstyle.api.Scope}.
074 * Default value is {@code null}.
075 * </li>
076 * <li>
077 * Property {@code authorFormat} - Specify the pattern for {@code @author} tag.
078 * Type is {@code java.util.regex.Pattern}.
079 * Default value is {@code null}.
080 * </li>
081 * <li>
082 * Property {@code versionFormat} - Specify the pattern for {@code @version} tag.
083 * Type is {@code java.util.regex.Pattern}.
084 * Default value is {@code null}.
085 * </li>
086 * <li>
087 * Property {@code allowMissingParamTags} - Control whether to ignore violations
088 * when a class has type parameters but does not have matching param tags in the Javadoc.
089 * Type is {@code boolean}.
090 * Default value is {@code false}.
091 * </li>
092 * <li>
093 * Property {@code allowUnknownTags} - Control whether to ignore violations when
094 * a Javadoc tag is not recognised.
095 * Type is {@code boolean}.
096 * Default value is {@code false}.
097 * </li>
098 * <li>
099 * Property {@code allowedAnnotations} - Specify annotations that allow
100 * skipping validation at all. Only short names are allowed, e.g. {@code Generated}.
101 * Type is {@code java.lang.String[]}.
102 * Default value is {@code Generated}.
103 * </li>
104 * <li>
105 * Property {@code tokens} - tokens to check
106 * Type is {@code java.lang.String[]}.
107 * Validation type is {@code tokenSet}.
108 * Default value is:
109 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#INTERFACE_DEF">
110 * INTERFACE_DEF</a>,
111 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#CLASS_DEF">
112 * CLASS_DEF</a>,
113 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ENUM_DEF">
114 * ENUM_DEF</a>,
115 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#ANNOTATION_DEF">
116 * ANNOTATION_DEF</a>,
117 * <a href="https://checkstyle.org/apidocs/com/puppycrawl/tools/checkstyle/api/TokenTypes.html#RECORD_DEF">
118 * RECORD_DEF</a>.
119 * </li>
120 * </ul>
121 * <p>
122 * To configure the default check:
123 * </p>
124 * <pre>
125 * &lt;module name="JavadocType"/&gt;
126 * </pre>
127 * <p>
128 * Example:
129 * </p>
130 * <pre>
131 * &#47;**
132 *  * &#64;author a
133 *  * &#64;version $Revision1$
134 *  *&#47;
135 * public class ClassA { // OK
136 *     &#47;** *&#47;
137 *     private class ClassB {} // OK
138 * }
139 *
140 * &#47;**
141 *  * &#64;author
142 *  * &#64;version abc
143 *  * &#64;unknownTag value // violation
144 *  *&#47;
145 * public class ClassC {} // OK
146 *
147 * &#47;** *&#47;
148 * public class ClassD&lt;T&gt; {} // violation, as param tag for &lt;T&gt; is missing
149 *
150 * &#47;** *&#47;
151 * private class ClassE&lt;T&gt; {} // violation, as param tag for &lt;T&gt; is missing
152 *
153 * &#47;** *&#47;
154 * &#64;Generated
155 * public class ClassF&lt;T&gt; {} // OK
156 * </pre>
157 * <p>
158 * To configure the check for {@code public} scope:
159 * </p>
160 * <pre>
161 * &lt;module name="JavadocType"&gt;
162 *   &lt;property name="scope" value="public"/&gt;
163 * &lt;/module&gt;
164 * </pre>
165 * <p>
166 * Example:
167 * </p>
168 * <pre>

256 *  * &#64;version abc
257 *  * &#64;unknownTag value // violation
258 *  *&#47;
259 * public class ClassC {} // violation, as version format is invalid
260 *
261 * &#47;** *&#47;
262 * public class ClassD {} // violation, as version tag is missing
263 *
264 * &#47;** *&#47;
265 * private class ClassE {} // violation, as version tag is missing
266 *
267 * &#47;** *&#47;
268 * &#64;Generated
269 * public class ClassF&lt;T&gt; {} // OK
270 * </pre>
271 * <p>
272 * To configure the check for {@code private} classes only:
273 * </p>
274 * <pre>
275 * &lt;module name="JavadocType"&gt;

279 * </pre>
280 * <p>
281 * Example:

283 * <pre>
284 * &#47;**
285 *  * &#64;author a

288 * public class ClassA { // OK

291 * }
292 *
293 * &#47;**
294 *  * &#64;author
295 *  * &#64;version abc

297 *  *&#47;

299 *
300 * &#47;** *&#47;

302 *
303 * &#47;** *&#47;
304 * private class ClassE&lt;T&gt; {} // violation, as param tag for &lt;T&gt; is missing
305 *
306 * &#47;** *&#47;

309 * </pre>
310 * <p>
311 * To configure the check that allows missing {@code @param} tags:

318 * <p>

600    // suppress deprecation until https://github.com/checkstyle/checkstyle/issues/11166
601    @SuppressWarnings("deprecation")
602    @Override
603    public void visitToken(DetailAST ast) {
604        if (shouldCheck(ast)) {
605            final FileContents contents = getFileContents();
606            final int lineNo = ast.getLineNo();
607            final TextBlock textBlock = contents.getJavadocBefore(lineNo);
608            if (textBlock != null) {
609                final List<JavadocTag> tags = getJavadocTags(textBlock);
610                if (ScopeUtil.isOuterMostType(ast)) {
611                    // don't check author/version for inner classes
612                    checkTag(ast, tags, JavadocTagInfo.AUTHOR.getName(),
613                            authorFormat);
614                    checkTag(ast, tags, JavadocTagInfo.VERSION.getName(),
615                            versionFormat);
616                }
617
618                final List<String> typeParamNames =
619                    CheckUtil.getTypeParameterNames(ast);
620                final List<String> recordComponentNames =
621                    getRecordComponentNames(ast);
622
623                if (!allowMissingParamTags) {
624
625                    typeParamNames.forEach(typeParamName -> {
626                        checkTypeParamTag(ast, tags, typeParamName);
627                    });
628
629                    recordComponentNames.forEach(componentName -> {
630                        checkComponentParamTag(ast, tags, componentName);
631                    });
632                }
633
634                checkUnusedParamTags(tags, typeParamNames, recordComponentNames);
635            }
636        }
637    }

639    /**
640     * Whether we should check this node.
641     *
642     * @param ast a given node.
643     * @return whether we should check a given node.
644     */
645    private boolean shouldCheck(DetailAST ast) {
646        final Scope customScope = ScopeUtil.getScope(ast);
647        final Scope surroundingScope = ScopeUtil.getSurroundingScope(ast);
648
649        return customScope.isIn(scope)
650            && (surroundingScope == null || surroundingScope.isIn(scope))
651            && (excludeScope == null
652                || !customScope.isIn(excludeScope)
653                || surroundingScope != null
654                && !surroundingScope.isIn(excludeScope))
655            && !AnnotationUtil.containsAnnotation(ast, allowedAnnotations);
656    }

658    /**
659     * Gets all standalone tags from a given javadoc.
660     *
661     * @param textBlock the Javadoc comment to process.
662     * @return all standalone tags from the given javadoc.
663     */
664    private List<JavadocTag> getJavadocTags(TextBlock textBlock) {
665        final JavadocTags tags = JavadocUtil.getJavadocTags(textBlock,
666            JavadocUtil.JavadocTagType.BLOCK);
667        if (!allowUnknownTags) {
668            for (final InvalidJavadocTag tag : tags.getInvalidTags()) {
669                log(tag.getLine(), tag.getCol(), MSG_UNKNOWN_TAG,
670                    tag.getName());
671            }
672        }
673        return tags.getValidTags();
674    }

676    /**
677     * Verifies that a type definition has a required tag.
678     *
679     * @param ast the AST node for the type definition.
680     * @param tags tags from the Javadoc comment for the type definition.
681     * @param tagName the required tag name.
682     * @param formatPattern regexp for the tag value.
683     */
684    private void checkTag(DetailAST ast, Iterable<JavadocTag> tags, String tagName,
685                          Pattern formatPattern) {
686        if (formatPattern != null) {
687            boolean hasTag = false;
688            final String tagPrefix = "@";
689
690            for (final JavadocTag tag :tags) {
691                if (tag.getTagName().equals(tagName)) {
692                    hasTag = true;
693                    if (!formatPattern.matcher(tag.getFirstArg()).find()) {
694                        log(ast, MSG_TAG_FORMAT, tagPrefix + tagName, formatPattern.pattern());
695                    }
696                }
697            }
698            if (!hasTag) {
699                log(ast, MSG_MISSING_TAG, tagPrefix + tagName);
700            }
701        }
702    }

704    /**
705     * Verifies that a record definition has the specified param tag for
706     * the specified record component name.
707     *
708     * @param ast the AST node for the record definition.
709     * @param tags tags from the Javadoc comment for the record definition.
710     * @param recordComponentName the name of the type parameter
711     */
712    private void checkComponentParamTag(DetailAST ast,
713                                        Collection<JavadocTag> tags,
714                                        String recordComponentName) {
715
716        final boolean found = tags
717            .stream()
718            .filter(JavadocTag::isParamTag)
719            .anyMatch(tag -> tag.getFirstArg().indexOf(recordComponentName) == 0);
720
721        if (!found) {
722            log(ast, MSG_MISSING_TAG, JavadocTagInfo.PARAM.getText()
723                + SPACE + recordComponentName);
724        }
725    }

727    /**
728     * Verifies that a type definition has the specified param tag for
729     * the specified type parameter name.
730     *
731     * @param ast the AST node for the type definition.
732     * @param tags tags from the Javadoc comment for the type definition.
733     * @param typeParamName the name of the type parameter
734     */
735    private void checkTypeParamTag(DetailAST ast,
736            Collection<JavadocTag> tags, String typeParamName) {
737        final String typeParamNameWithBrackets =
738            OPEN_ANGLE_BRACKET + typeParamName + CLOSE_ANGLE_BRACKET;
739
740        final boolean found = tags
741            .stream()
742            .filter(JavadocTag::isParamTag)
743            .anyMatch(tag -> tag.getFirstArg().indexOf(typeParamNameWithBrackets) == 0);
744
745        if (!found) {
746            log(ast, MSG_MISSING_TAG, JavadocTagInfo.PARAM.getText()
747                + SPACE + typeParamNameWithBrackets);
748        }
749    }
750
751    /**
752     * Checks for unused param tags for type parameters and record components.
753     *
754     * @param tags tags from the Javadoc comment for the type definition
755     * @param typeParamNames names of type parameters
756     * @param recordComponentNames record component names in this definition
757     */
758    private void checkUnusedParamTags(
759        List<JavadocTag> tags,
760        List<String> typeParamNames,
761        List<String> recordComponentNames) {
762
763        for (final JavadocTag tag: tags) {
764            if (tag.isParamTag()) {
765                final String paramName = extractParamNameFromTag(tag);
766                final boolean found = typeParamNames.contains(paramName)
767                        || recordComponentNames.contains(paramName);
768
769                if (!found) {
770                    final String actualParamName =
771                        TYPE_NAME_IN_JAVADOC_TAG_SPLITTER.split(tag.getFirstArg())[0];
772                    log(tag.getLineNo(), tag.getColumnNo(),
773                        MSG_UNUSED_TAG,
774                        JavadocTagInfo.PARAM.getText(), actualParamName);
775                }
776            }
777        }
778
779    }

781    /**
782     * Extracts parameter name from tag.
783     *
784     * @param tag javadoc tag to extract parameter name
785     * @return extracts type parameter name from tag
786     */
787    private static String extractParamNameFromTag(JavadocTag tag) {
788        final String typeParamName;
789        final Matcher matchInAngleBrackets =
790                TYPE_NAME_IN_JAVADOC_TAG.matcher(tag.getFirstArg());
791        if (matchInAngleBrackets.find()) {
792            typeParamName = matchInAngleBrackets.group(1).trim();
793        }
794        else {
795            typeParamName = TYPE_NAME_IN_JAVADOC_TAG_SPLITTER.split(tag.getFirstArg())[0];
796        }
797        return typeParamName;
798    }

800    /**
801     * Collects the record components in a record definition.
802     *
803     * @param node the possible record definition ast.
804     * @return the record components in this record definition.
805     */
806    private static List<String> getRecordComponentNames(DetailAST node) {
807        final DetailAST components = node.findFirstToken(TokenTypes.RECORD_COMPONENTS);
808        final List<String> componentList = new ArrayList<>();
809
810        if (components != null) {
811            TokenUtil.forEachChild(components,
812                TokenTypes.RECORD_COMPONENT_DEF, component -> {
813                    final DetailAST ident = component.findFirstToken(TokenTypes.IDENT);
814                    componentList.add(ident.getText());
815                });
816        }
817
818        return componentList;
819    }

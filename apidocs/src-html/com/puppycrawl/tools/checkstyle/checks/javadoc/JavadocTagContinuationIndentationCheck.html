020package com.puppycrawl.tools.checkstyle.checks.javadoc;
021
022import java.util.ArrayList;
023import java.util.List;
024
025import com.puppycrawl.tools.checkstyle.StatelessCheck;
026import com.puppycrawl.tools.checkstyle.api.DetailNode;
027import com.puppycrawl.tools.checkstyle.api.JavadocTokenTypes;
028import com.puppycrawl.tools.checkstyle.utils.CommonUtil;
029import com.puppycrawl.tools.checkstyle.utils.JavadocUtil;
030
031/**
032 * <p>
033 * Checks the indentation of the continuation lines in at-clauses.
034 * </p>
035 * <ul>
036 * <li>
037 * Property {@code violateExecutionOnNonTightHtml} - Control when to print violations
038 * if the Javadoc being examined by this check violates the tight html rules defined at
039 * <a href="https://checkstyle.org/writingjavadocchecks.html#Tight-HTML_rules">Tight-HTML Rules</a>.
040 * Default value is {@code false}.
041 * </li>
042 * <li>
043 * Property {@code offset} - Specify how many spaces to use for new indentation level.
044 * Default value is {@code 4}.
045 * </li>
046 * </ul>
047 * <p>
048 * To configure the default check:
049 * </p>
050 * <pre>
051 * &lt;module name="JavadocTagContinuationIndentation"/&gt;
052 * </pre>
053 * <p>
054 * Example:
055 * </p>
056 * <pre>
057 * &#47;**
058 *  * @tag comment
059 *  *  Indentation spacing is 1. Line with violation
060 *  *   Indentation spacing is 2. Line with violation
061 *  *     Indentation spacing is 4. OK
062 *  *&#47;
063 * public class Test {
064 * }
065 * </pre>
066 * <p>
067 * To configure the check with two spaces indentation:
068 * </p>
069 * <pre>
070 * &lt;module name="JavadocTagContinuationIndentation"&gt;
071 *   &lt;property name="offset" value="2"/&gt;
072 * &lt;/module&gt;
073 * </pre>
074 * <p>
075 * Example:
076 * </p>
077 * <pre>
078 * &#47;**
079 *  * @tag comment
080 *  * Indentation spacing is 0. Line with violation
081 *  *   Indentation spacing is 2. OK
082 *  *  Indentation spacing is 1. Line with violation
083 *  *&#47;
084 * public class Test {
085 * }
086 * </pre>
087 * <p>
088 * To configure the check to show violations for Tight-HTML Rules:
089 * </p>
090 * <pre>
091 * &lt;module name="JavadocTagContinuationIndentation"&gt;
092 *   &lt;property name="violateExecutionOnNonTightHtml" value="true"/&gt;
093 * &lt;/module&gt;
094 * </pre>
095 * <p>
096 * Example:
097 * </p>
098 * <pre>
099 * &#47;**
100 *  * &lt;p&gt; 'p' tag is unclosed. Line with violation, this html tag needs closing tag.
101 *  * &lt;p&gt; 'p' tag is closed&lt;/p&gt;. OK
102 *  *&#47;
103 * public class Test {
104 * }
105 * </pre>
106 *
107 * @since 6.0
108 *
109 */
110@StatelessCheck
111public class JavadocTagContinuationIndentationCheck extends AbstractJavadocCheck {
112
113    /**
114     * A key is pointing to the warning message text in "messages.properties"
115     * file.
116     */
117    public static final String MSG_KEY = "tag.continuation.indent";
118
119    /** Default tag continuation indentation. */
120    private static final int DEFAULT_INDENTATION = 4;
121
122    /**
123     * Specify how many spaces to use for new indentation level.
124     */
125    private int offset = DEFAULT_INDENTATION;
126
127    /**
128     * Setter to specify how many spaces to use for new indentation level.
129     *
130     * @param offset custom value.
131     */
132    public void setOffset(int offset) {
133        this.offset = offset;
134    }
135
136    @Override
137    public int[] getDefaultJavadocTokens() {
138        return new int[] {JavadocTokenTypes.DESCRIPTION };
139    }
140
141    @Override
142    public int[] getRequiredJavadocTokens() {
143        return getAcceptableJavadocTokens();
144    }
145
146    @Override
147    public void visitJavadocToken(DetailNode ast) {
148        if (!isInlineDescription(ast)) {
149            final List<DetailNode> textNodes = getAllNewlineNodes(ast);
150            for (DetailNode newlineNode : textNodes) {
151                final DetailNode textNode = JavadocUtil.getNextSibling(JavadocUtil
152                        .getNextSibling(newlineNode));
153                if (textNode != null && textNode.getType() == JavadocTokenTypes.TEXT) {
154                    final String text = textNode.getText();
155                    if (!CommonUtil.isBlank(text.trim())
156                            && (text.length() <= offset
157                                    || !text.substring(1, offset + 1).trim().isEmpty())) {
158                        log(textNode.getLineNumber(), MSG_KEY, offset);
159                    }
160                }
161            }
162        }
163    }
164
165    /**
166     * Finds and collects all NEWLINE nodes inside DESCRIPTION node.
167     *
168     * @param descriptionNode DESCRIPTION node.
169     * @return List with NEWLINE nodes.
170     */
171    private static List<DetailNode> getAllNewlineNodes(DetailNode descriptionNode) {
172        final List<DetailNode> textNodes = new ArrayList<>();
173        DetailNode node = JavadocUtil.getFirstChild(descriptionNode);
174        while (JavadocUtil.getNextSibling(node) != null) {
175            if (node.getType() == JavadocTokenTypes.NEWLINE) {
176                textNodes.add(node);
177            }
178            node = JavadocUtil.getNextSibling(node);
179        }
180        return textNodes;
181    }
182
183    /**
184     * Checks, if description node is a description of in-line tag.
185     *
186     * @param description DESCRIPTION node.
187     * @return true, if description node is a description of in-line tag.
188     */
189    private static boolean isInlineDescription(DetailNode description) {
190        boolean isInline = false;
191        DetailNode inlineTag = description.getParent();
192        while (inlineTag != null) {
193            if (inlineTag.getType() == JavadocTokenTypes.JAVADOC_INLINE_TAG) {
194                isInline = true;
195                break;
196            }
197            inlineTag = inlineTag.getParent();
198        }
199        return isInline;
200    }
201
202}

020package com.puppycrawl.tools.checkstyle.checks.annotation;
021
022import com.puppycrawl.tools.checkstyle.StatelessCheck;
023import com.puppycrawl.tools.checkstyle.api.AbstractCheck;
024import com.puppycrawl.tools.checkstyle.api.DetailAST;
025import com.puppycrawl.tools.checkstyle.api.TokenTypes;
026import com.puppycrawl.tools.checkstyle.utils.AnnotationUtility;
027
028/**
029 * This check makes sure that all package annotations are in the
030 * package-info.java file.
031 *
032 * <p>
033 * According to the Java Language Specification.
034 * </p>
035 *
036 * <p>
037 * The JLS does not enforce the placement of package annotations.
038 * This placement may vary based on implementation. The JLS
039 * does highly recommend that all package annotations are
040 * placed in the package-info.java file.
041 *
042 * See <a
043 * href="https://docs.oracle.com/javase/specs/jls/se8/html/jls-7.html#jls-7.4.1">
044 * Java Language Specification, section 7.4.1</a>.
045 * </p>
046 * @author Travis Schneeberger
047 */
048@StatelessCheck
049public class PackageAnnotationCheck extends AbstractCheck {
050
051    /**
052     * A key is pointing to the warning message text in "messages.properties"
053     * file.
054     */
055    public static final String MSG_KEY = "annotation.package.location";
056
057    @Override
058    public int[] getDefaultTokens() {
059        return getRequiredTokens();
060    }
061
062    @Override
063    public int[] getRequiredTokens() {
064        return new int[] {
065            TokenTypes.PACKAGE_DEF,
066        };
067    }
068
069    @Override
070    public int[] getAcceptableTokens() {
071        return getRequiredTokens();
072    }
073
074    @Override
075    public void visitToken(final DetailAST ast) {
076        final boolean containsAnnotation =
077            AnnotationUtility.containsAnnotation(ast);
078        final boolean inPackageInfo =
079            getFileContents().inPackageInfo();
080
081        if (containsAnnotation && !inPackageInfo) {
082            log(ast.getLine(), MSG_KEY);
083        }
084    }
085
086}

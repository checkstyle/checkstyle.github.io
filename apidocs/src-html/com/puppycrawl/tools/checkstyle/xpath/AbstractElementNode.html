020package com.puppycrawl.tools.checkstyle.xpath;
021
022import java.util.Collections;
023import java.util.List;
024import java.util.Optional;
025
026import com.puppycrawl.tools.checkstyle.xpath.iterators.DescendantIterator;
027import com.puppycrawl.tools.checkstyle.xpath.iterators.FollowingIterator;
028import com.puppycrawl.tools.checkstyle.xpath.iterators.PrecedingIterator;
029import com.puppycrawl.tools.checkstyle.xpath.iterators.ReverseListIterator;
030import net.sf.saxon.om.AxisInfo;
031import net.sf.saxon.om.NodeInfo;
032import net.sf.saxon.tree.iter.ArrayIterator;
033import net.sf.saxon.tree.iter.AxisIterator;
034import net.sf.saxon.tree.iter.EmptyIterator;
035import net.sf.saxon.tree.iter.SingleNodeIterator;
036import net.sf.saxon.tree.util.Navigator;
037import net.sf.saxon.type.Type;
038
039/**
040 * Represents element node of Xpath-tree.
041 */
042public abstract class AbstractElementNode extends AbstractNode {
043
044    /** String literal for text attribute. */
045    protected static final String TEXT_ATTRIBUTE_NAME = "text";
046
047    /** Constant for optimization. */
048    private static final AbstractNode[] EMPTY_ABSTRACT_NODE_ARRAY = new AbstractNode[0];
049
050    /** Holder value for lazy creation of attribute node. */
051    private static final AttributeNode ATTRIBUTE_NODE_UNINITIALIZED = new AttributeNode(null, null);
052
053    /** The root node. */
054    private final AbstractNode root;
055
056    /** The parent of the current node. */
057    private final AbstractNode parent;
058
059    /** Depth of the node. */
060    private final int depth;
061
062    /** Represents index among siblings. */
063    private final int indexAmongSiblings;
064
065    /** The text attribute node. */
066    private AttributeNode attributeNode = ATTRIBUTE_NODE_UNINITIALIZED;
067
068    /**
069     * Creates a new {@code AbstractElementNode} instance.
070     *
071     * @param root {@code Node} root of the tree
072     * @param parent {@code Node} parent of the current node
073     * @param depth the current node depth in the hierarchy
074     * @param indexAmongSiblings the current node index among the parent children nodes
075     */
076    protected AbstractElementNode(AbstractNode root, AbstractNode parent,
077            int depth, int indexAmongSiblings) {
078        super(root.getTreeInfo());
079        this.parent = parent;
080        this.root = root;
081        this.depth = depth;
082        this.indexAmongSiblings = indexAmongSiblings;
083    }
084
085    /**
086     * Creates {@code AttributeNode} object and returns it.
087     *
088     * @return attribute node if possible, otherwise the {@code null} value
089     */
090    protected abstract AttributeNode createAttributeNode();
091
092    /**
093     * Compares current object with specified for order.
094     *
095     * @param other another {@code NodeInfo} object
096     * @return number representing order of current object to specified one
097     */
098    @Override
099    public int compareOrder(NodeInfo other) {
100        int result = 0;
101        if (other instanceof AbstractNode) {
102            result = Integer.compare(depth, ((AbstractNode) other).getDepth());
103            if (result == 0) {
104                result = compareCommonAncestorChildrenOrder(this, other);
105            }
106        }
107        return result;
108    }
109
110    /**
111     * Walks up the hierarchy until a common ancestor is found.
112     * Then compares topmost sibling nodes.
113     *
114     * @param first {@code NodeInfo} to compare
115     * @param second {@code NodeInfo} to compare
116     * @return the value {@code 0} if {@code first == second};
117     *         a value less than {@code 0} if {@code first} should be first;
118     *         a value greater than {@code 0} if {@code second} should be first.
119     */
120    private static int compareCommonAncestorChildrenOrder(NodeInfo first, NodeInfo second) {
121        NodeInfo child1 = first;
122        NodeInfo child2 = second;
123        while (!child1.getParent().equals(child2.getParent())) {
124            child1 = child1.getParent();
125            child2 = child2.getParent();
126        }
127        final int index1 = ((AbstractElementNode) child1).indexAmongSiblings;
128        final int index2 = ((AbstractElementNode) child2).indexAmongSiblings;
129        return Integer.compare(index1, index2);
130    }
131
132    /**
133     * Getter method for node depth.
134     *
135     * @return depth
136     */
137    @Override
138    public int getDepth() {
139        return depth;
140    }
141
142    /**
143     * Returns attribute value.
144     *
145     * @param namespace namespace
146     * @param localPart actual name of the attribute
147     * @return attribute value or null if the attribute was not found
148     */
149    @Override
150    public String getAttributeValue(String namespace, String localPart) {
151        final String result;
152        if (TEXT_ATTRIBUTE_NAME.equals(localPart)) {
153            result = Optional.ofNullable(getAttributeNode())
154                .map(AttributeNode::getStringValue)
155                .orElse(null);
156        }
157        else {
158            result = null;
159        }
160        return result;
161    }
162
163    /**
164     * Returns type of the node.
165     *
166     * @return node kind
167     */
168    @Override
169    public int getNodeKind() {
170        return Type.ELEMENT;
171    }
172
173    /**
174     * Returns parent.
175     *
176     * @return parent
177     */
178    @Override
179    public NodeInfo getParent() {
180        return parent;
181    }
182
183    /**
184     * Returns root.
185     *
186     * @return root
187     */
188    @Override
189    public AbstractNode getRoot() {
190        return root;
191    }
192
193    /**
194     * Determines axis iteration algorithm. Throws {@code UnsupportedOperationException} in case,
195     * when there is no axis iterator for given axisNumber.
196     *
197     * <p>Reason of suppression for resource, IOResourceOpenedButNotSafelyClosed:
198     * {@link AxisIterator} implements {@link java.io.Closeable} interface,
199     * but none of the subclasses of the {@link AxisIterator}
200     * class has non-empty {@code close()} method.
201     *
202     * @param axisNumber element from {@code AxisInfo}
203     * @return {@code AxisIterator} object
204     */
205    @Override
206    public AxisIterator iterateAxis(int axisNumber) {
207        final AxisIterator result;
208        switch (axisNumber) {
209            case AxisInfo.ANCESTOR:
210                result = new Navigator.AncestorEnumeration(this, false);
211                break;
212            case AxisInfo.ANCESTOR_OR_SELF:
213                result = new Navigator.AncestorEnumeration(this, true);
214                break;
215            case AxisInfo.ATTRIBUTE:
216                result = SingleNodeIterator.makeIterator(getAttributeNode());
217                break;
218            case AxisInfo.CHILD:
219                if (hasChildNodes()) {
220                    result = new ArrayIterator.OfNodes<>(
221                            getChildren().toArray(EMPTY_ABSTRACT_NODE_ARRAY));
222                }
223                else {
224                    result = EmptyIterator.ofNodes();
225                }
226                break;
227            case AxisInfo.DESCENDANT:
228                if (hasChildNodes()) {
229                    result = new DescendantIterator(this, DescendantIterator.StartWith.CHILDREN);
230                }
231                else {
232                    result = EmptyIterator.ofNodes();
233                }
234                break;
235            case AxisInfo.DESCENDANT_OR_SELF:
236                result = new DescendantIterator(this, DescendantIterator.StartWith.CURRENT_NODE);
237                break;
238            case AxisInfo.PARENT:
239                result = SingleNodeIterator.makeIterator(parent);
240                break;
241            case AxisInfo.SELF:
242                result = SingleNodeIterator.makeIterator(this);
243                break;
244            case AxisInfo.FOLLOWING_SIBLING:
245                result = getFollowingSiblingsIterator();
246                break;
247            case AxisInfo.PRECEDING_SIBLING:
248                result = getPrecedingSiblingsIterator();
249                break;
250            case AxisInfo.FOLLOWING:
251                result = new FollowingIterator(this);
252                break;
253            case AxisInfo.PRECEDING:
254                result = new PrecedingIterator(this);
255                break;
256            default:
257                throw throwUnsupportedOperationException();
258        }
259
260        return result;
261    }
262
263    /**
264     * Returns preceding sibling axis iterator.
265     *
266     * <p>Reason of suppression for resource, IOResourceOpenedButNotSafelyClosed:
267     * {@link AxisIterator} implements {@link java.io.Closeable} interface,
268     * but none of the subclasses of the {@link AxisIterator}
269     * class has non-empty {@code close()} method.
270     *
271     * @return iterator
272     */
273    private AxisIterator getPrecedingSiblingsIterator() {
274        final AxisIterator result;
275        if (indexAmongSiblings == 0) {
276            result = EmptyIterator.ofNodes();
277        }
278        else {
279            result = new ReverseListIterator(getPrecedingSiblings());
280        }
281        return result;
282    }
283
284    /**
285     * Returns following sibling axis iterator.
286     *
287     * <p>Reason of suppression for resource, IOResourceOpenedButNotSafelyClosed:
288     * {@link AxisIterator} implements {@link java.io.Closeable} interface,
289     * but none of the subclasses of the {@link AxisIterator}
290     * class has non-empty {@code close()} method.
291     *
292     * @return iterator
293     */
294    private AxisIterator getFollowingSiblingsIterator() {
295        final AxisIterator result;
296        if (indexAmongSiblings == parent.getChildren().size() - 1) {
297            result = EmptyIterator.ofNodes();
298        }
299        else {
300            result = new ArrayIterator.OfNodes<>(
301                    getFollowingSiblings().toArray(EMPTY_ABSTRACT_NODE_ARRAY));
302        }
303        return result;
304    }
305
306    /**
307     * Returns following siblings of the current node.
308     *
309     * @return siblings
310     */
311    private List<AbstractNode> getFollowingSiblings() {
312        final List<AbstractNode> siblings = parent.getChildren();
313        return siblings.subList(indexAmongSiblings + 1, siblings.size());
314    }
315
316    /**
317     * Returns preceding siblings of the current node.
318     *
319     * @return siblings
320     */
321    private List<AbstractNode> getPrecedingSiblings() {
322        final List<AbstractNode> siblings = parent.getChildren();
323        return Collections.unmodifiableList(siblings.subList(0, indexAmongSiblings));
324    }
325
326    /**
327     * Checks if token type supports {@code @text} attribute,
328     * extracts its value, creates {@code AttributeNode} object and returns it.
329     * Value can be accessed using {@code @text} attribute.
330     *
331     * @return attribute node if possible, otherwise the {@code null} value
332     */
333    private AttributeNode getAttributeNode() {
334        if (attributeNode == ATTRIBUTE_NODE_UNINITIALIZED) {
335            attributeNode = createAttributeNode();
336        }
337        return attributeNode;
338    }
339
340    /**
341     * Returns UnsupportedOperationException exception.
342     *
343     * @return UnsupportedOperationException exception
344     */
345    private static UnsupportedOperationException throwUnsupportedOperationException() {
346        return new UnsupportedOperationException("Operation is not supported");
347    }

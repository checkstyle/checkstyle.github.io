020package com.puppycrawl.tools.checkstyle.xpath;
021
022import java.util.List;
023
024import net.sf.saxon.om.NodeInfo;
025import net.sf.saxon.tree.iter.AxisIterator;
026import net.sf.saxon.type.Type;
027
028/**
029 * Represents attribute of the element.
030 *
031 */
032public class AttributeNode extends AbstractNode {
033
034    /** The name of the attribute. */
035    private final String name;
036
037    /** The value of the attribute. */
038    private final String value;
039
040    /**
041     * Creates a new {@code AttributeNode} instance.
042     *
043     * @param name name of the attribute
044     * @param value value of the attribute
045     */
046    public AttributeNode(String name, String value) {
047        super(null);
048        this.name = name;
049        this.value = value;
050    }
051
052    /**
053     * Compares current object with specified for order.
054     * Throws {@code UnsupportedOperationException} because functionality not required here.
055     *
056     * @param nodeInfo another {@code NodeInfo} object
057     * @return number representing order of current object to specified one
058     */
059    @Override
060    public int compareOrder(NodeInfo nodeInfo) {
061        throw throwUnsupportedOperationException();
062    }
063
064    /**
065     * Returns attribute value. Throws {@code UnsupportedOperationException} because attribute node
066     * has no attributes.
067     *
068     * @param namespace namespace
069     * @param localPart actual name of the attribute
070     * @return attribute value
071     */
072    @Override
073    public String getAttributeValue(String namespace, String localPart) {
074        throw throwUnsupportedOperationException();
075    }
076
077    /**
078     * Returns local part.
079     *
080     * @return local part
081     */
082    @Override
083    public String getLocalPart() {
084        return name;
085    }
086
087    /**
088     * Returns type of the node.
089     *
090     * @return node kind
091     */
092    @Override
093    public int getNodeKind() {
094        return Type.ATTRIBUTE;
095    }
096
097    /**
098     * Returns parent.  Never called for attribute node, throws
099     * {@code UnsupportedOperationException}.
100     * has no attributes.
101     *
102     * @return parent
103     */
104    @Override
105    public NodeInfo getParent() {
106        throw throwUnsupportedOperationException();
107    }
108
109    /**
110     * Returns root. Never called for attribute node, throws
111     * {@code UnsupportedOperationException}.
112     *
113     * @return root
114     */
115    @Override
116    public NodeInfo getRoot() {
117        throw throwUnsupportedOperationException();
118    }
119
120    /**
121     * Returns string value.
122     *
123     * @return string value
124     */
125    @Override
126    public String getStringValue() {
127        return value;
128    }
129
130    /**
131     * Determines axis iteration algorithm. Attribute node can not be iterated, throws
132     * {@code UnsupportedOperationException}.
133     *
134     * @param axisNumber element from {@code AxisInfo}
135     * @return {@code AxisIterator} object
136     */
137    @Override
138    public AxisIterator iterateAxis(int axisNumber) {
139        throw throwUnsupportedOperationException();
140    }
141
142    /**
143     * Returns line number. Attribute node has no line number, throws
144     * {@code UnsupportedOperationException}.
145     *
146     * @return line number
147     */
148    @Override
149    public int getLineNumber() {
150        throw throwUnsupportedOperationException();
151    }
152
153    /**
154     * Returns column number. Attribute node has no column number, throws
155     * {@code UnsupportedOperationException}.
156     *
157     * @return column number
158     */
159    @Override
160    public int getColumnNumber() {
161        throw throwUnsupportedOperationException();
162    }
163
164    /**
165     * Getter method for token type. Attribute node has no token type, throws
166     * {@code UnsupportedOperationException}.
167     *
168     * @return token type
169     */
170    @Override
171    public int getTokenType() {
172        throw throwUnsupportedOperationException();
173    }
174
175    /**
176     * Returns underlying node. Attribute node has no underlying node, throws
177     * {@code UnsupportedOperationException}.
178     *
179     * @return underlying node
180     */
181    @Override
182    public Object getUnderlyingNode() {
183        throw throwUnsupportedOperationException();
184    }
185
186    /**
187     * Getter method for node depth. This method is not applicable to attribute nodes,
188     * throws unsupported exception.
189     *
190     * @return never
191     */
192    @Override
193    public int getDepth() {
194        throw throwUnsupportedOperationException();
195    }
196
197    /**
198     * Creates nodes for children. Attribute node has no children, so
199     * this method throws unsupported exception.
200     *
201     * @return never
202     */
203    @Override
204    protected List<AbstractNode> createChildren() {
205        throw throwUnsupportedOperationException();
206    }
207
208    /**
209     * Determine whether the node has any children.
210     *
211     * @return always {@code false}
212     */
213    @Override
214    public boolean hasChildNodes() {
215        return false;
216    }
217
218    /**
219     * Returns UnsupportedOperationException exception.
220     *
221     * @return UnsupportedOperationException exception
222     */
223    private static UnsupportedOperationException throwUnsupportedOperationException() {
224        return new UnsupportedOperationException("Operation is not supported");
225    }
226
227}

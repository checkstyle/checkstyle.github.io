020package com.puppycrawl.tools.checkstyle;
021
022import java.lang.annotation.ElementType;
023import java.lang.annotation.Inherited;
024import java.lang.annotation.Retention;
025import java.lang.annotation.RetentionPolicy;
026import java.lang.annotation.Target;
027
028/**
029 * This annotation means that the check does not contain mutable state and
030 * may be safely used from others threads at the same time.
031 * The check does not contain mutable state, if it does not change any field values at the
032 * execution stage (but it still can set properties at the initialization stage).
033 * Checker guarantees that there will be exactly one check instance during the audit.
034 * This also means, that all files will be processed by the same check instance.
035 *
036 * @noinspection ClassIndependentOfModule
037 * @noinspectionreason ClassIndependentOfModule - we keep this annotation at top level by design
038 */
039@Retention(RetentionPolicy.RUNTIME)
040@Target(ElementType.TYPE)
041@Inherited
042public @interface StatelessCheck {
043
044    // this annotation does not have properties
045
046}
